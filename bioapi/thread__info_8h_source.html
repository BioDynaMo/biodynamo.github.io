62  void Renew() {
63  max_threads_ = omp_get_max_threads();
64  numa_nodes_ = numa_num_configured_nodes();
65 
66  thread_numa_mapping_.clear();
67  numa_thread_id_.clear();
68  threads_in_numa_.clear();
69 
70  thread_numa_mapping_.resize(max_threads_, 0);
71  numa_thread_id_.resize(max_threads_, 0);
72  threads_in_numa_.resize(numa_nodes_, 0);
73 
74 // (openmp thread id -> numa node)
75 #pragma omp parallel
76  {
77  int tid = omp_get_thread_num();
78  thread_numa_mapping_[tid] = numa_node_of_cpu(sched_getcpu());
79  }
80 
81  // (numa -> number of associated threads), and
82  // (omp_thread_id -> thread id in numa)
83  for (uint16_t n = 0; n < numa_nodes_; n++) {
84  uint64_t cnt = 0;
85  for (uint64_t t = 0; t < max_threads_; t++) {
86  int numa = thread_numa_mapping_[t];
87  if (n == numa) {
88  numa_thread_id_[t] = cnt;
89  cnt++;
90  }
91  }
92  threads_in_numa_[n] = cnt;
93  }
94  }
