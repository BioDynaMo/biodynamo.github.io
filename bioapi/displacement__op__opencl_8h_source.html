43  void operator()() {
44  auto* sim = Simulation::GetActive();
45  auto* grid = sim->GetGrid();
46  auto* param = sim->GetParam();
47  auto* rm = sim->GetResourceManager();
48 
49  // Check the number of NUMA domains on the system. Currently only 1 is
50  // supported for GPU execution.
51  if (ThreadInfo::GetInstance()->GetNumaNodes() > 1) {
52  Log::Fatal(
53  "DisplacementOpOpenCL",
54  "\nThe GPU execution only supports systems with 1 NUMA domain.");
55  return;
56  }
57 
58  uint32_t num_objects = rm->GetNumSimObjects();
59 
60  auto* ocl_state = OpenCLState::GetInstance();
61  auto context = ocl_state->GetOpenCLContext();
62  auto queue = ocl_state->GetOpenCLCommandQueue();
63  auto programs = ocl_state->GetOpenCLProgramList();
64 
65  // Cannot use Double3 here, because the `data()` function returns a const
66  // pointer to the underlying array, whereas the CUDA kernal will cast it to
67  // a void pointer. The conversion of `const double *` to `void *` is
68  // illegal.
69  std::vector<std::array<cl_double, 3>> cell_movements(num_objects);
70  std::vector<std::array<cl_double, 3>> cell_positions(num_objects);
71  std::vector<cl_double> cell_diameters(num_objects);
72  std::vector<cl_double> cell_adherence(num_objects);
73  std::vector<std::array<cl_double, 3>> cell_tractor_force(num_objects);
74  std::vector<cl_uint> cell_boxid(num_objects);
75  std::vector<cl_double> mass(num_objects);
76  std::vector<cl_uint> starts;
77  std::vector<cl_ushort> lengths;
78  std::vector<cl_uint> successors(num_objects);
79  cl_uint box_length;
80  std::array<cl_uint, 3> num_boxes_axis;
81  std::array<cl_int, 3> grid_dimensions;
82  cl_double squared_radius =
83  grid->GetLargestObjectSize() * grid->GetLargestObjectSize();
84 
85  bool is_non_spherical_object = false;
86 
87  rm->ApplyOnAllElements([&](SimObject* so, SoHandle soh) {
88  // Check if there are any non-spherical objects in our simulation, because
89  // GPU accelerations currently supports only sphere-sphere interactions
90  IsNonSphericalObjectPresent(so, &is_non_spherical_object);
91  if (is_non_spherical_object) {
92  Log::Fatal("DisplacementOpOpenCL",
93  "\nWe detected a non-spherical object during the GPU "
94  "execution. This is currently not supported.");
95  return;
96  }
97  auto* cell = bdm_static_cast<Cell*>(so);
98  auto idx = soh.GetElementIdx();
99  mass[idx] = cell->GetMass();
100  cell_diameters[idx] = cell->GetDiameter();
101  cell_adherence[idx] = cell->GetAdherence();
102  cell_boxid[idx] = cell->GetBoxIdx();
103  cell_tractor_force[idx][0] = cell->GetTractorForce()[0];
104  cell_tractor_force[idx][1] = cell->GetTractorForce()[1];
105  cell_tractor_force[idx][2] = cell->GetTractorForce()[2];
106  cell_positions[idx][0] = cell->GetPosition()[0];
107  cell_positions[idx][1] = cell->GetPosition()[1];
108  cell_positions[idx][2] = cell->GetPosition()[2];
109  });
110 
111  uint16_t numa_node = 0; // GPU code only supports 1 NUMA domain currently
112  for (size_t i = 0; i < grid->successors_.size(numa_node); i++) {
113  auto sh = SoHandle(numa_node, i);
114  successors[i] = grid->successors_[sh].GetElementIdx();
115  }
116 
117  starts.resize(grid->boxes_.size());
118  lengths.resize(grid->boxes_.size());
119  size_t i = 0;
120  for (auto& box : grid->boxes_) {
121  starts[i] = box.start_.GetElementIdx();
122  lengths[i] = box.length_;
123  i++;
124  }
125  grid->GetGridInfo(&box_length, &num_boxes_axis, &grid_dimensions);
126 
127  // Allocate GPU buffers
128  cl::Buffer positions_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
129  num_objects * 3 * sizeof(cl_double),
130  cell_positions.data()->data());
131  cl::Buffer diameters_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
132  num_objects * sizeof(cl_double),
133  cell_diameters.data());
134  cl::Buffer tractor_force_arg(
135  *context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
136  num_objects * 3 * sizeof(cl_double), cell_tractor_force.data()->data());
137  cl::Buffer adherence_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
138  num_objects * sizeof(cl_double),
139  cell_adherence.data());
140  cl::Buffer box_id_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
141  num_objects * sizeof(cl_uint), cell_boxid.data());
142  cl::Buffer mass_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
143  num_objects * sizeof(cl_double), mass.data());
144  cl::Buffer cell_movements_arg(
145  *context, CL_MEM_READ_WRITE | CL_MEM_USE_HOST_PTR,
146  num_objects * 3 * sizeof(cl_double), cell_movements.data()->data());
147  cl::Buffer starts_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
148  starts.size() * sizeof(cl_uint), starts.data());
149  cl::Buffer lengths_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
150  lengths.size() * sizeof(cl_short), lengths.data());
151  cl::Buffer successors_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
152  successors.size() * sizeof(cl_uint),
153  successors.data());
154  cl::Buffer nba_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
155  3 * sizeof(cl_uint), num_boxes_axis.data());
156  cl::Buffer gd_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
157  3 * sizeof(cl_int), grid_dimensions.data());
158 
159  // Create the kernel object from our program
160  // TODO(ahmad): generalize the program selection, in case we have more than
161  // one. We can maintain an unordered map of programs maybe
162  cl::Kernel collide((*programs)[0], "collide");
163 
164  // Set kernel parameters
165  collide.setArg(0, positions_arg);
166  collide.setArg(1, diameters_arg);
167  collide.setArg(2, tractor_force_arg);
168  collide.setArg(3, adherence_arg);
169  collide.setArg(4, box_id_arg);
170  collide.setArg(5, mass_arg);
171  collide.setArg(6, param->simulation_time_step_);
172  collide.setArg(7, param->simulation_max_displacement_);
173  collide.setArg(8, squared_radius);
174 
175  collide.setArg(9, static_cast<cl_int>(num_objects));
176  collide.setArg(10, starts_arg);
177  collide.setArg(11, lengths_arg);
178  collide.setArg(12, successors_arg);
179  collide.setArg(13, box_length);
180  collide.setArg(14, nba_arg);
181  collide.setArg(15, gd_arg);
182  collide.setArg(16, cell_movements_arg);
183 
184  // The amount of threads for each work group (similar to CUDA thread block)
185  int block_size = 256;
186 
187  try {
188  // The global size determines the total number of threads that will be
189  // spawned on the GPU, in groups of local_size
190  cl::NDRange global_size =
191  cl::NDRange(num_objects + (block_size - (num_objects % block_size)));
192  cl::NDRange local_size = cl::NDRange(block_size);
193  queue->enqueueNDRangeKernel(collide, cl::NullRange, global_size,
194  local_size);
195  } catch (const cl::Error& err) {
196  Log::Error("DisplacementOpOpenCL", err.what(), "(", err.err(), ") = ",
197  ocl_state->GetErrorString(err.err()));
198  throw;
199  }
200 
201  try {
202  queue->enqueueReadBuffer(cell_movements_arg, CL_TRUE, 0,
203  num_objects * 3 * sizeof(cl_double),
204  cell_movements.data()->data());
205  } catch (const cl::Error& err) {
206  Log::Error("DisplacementOpOpenCL", err.what(), "(", err.err(), ") = ",
207  ocl_state->GetErrorString(err.err()));
208  throw;
209  }
210 
211  // set new positions after all updates have been calculated
212  // otherwise some cells would see neighbors with already updated positions
213  // which would lead to inconsistencies
214  rm->ApplyOnAllElements([&](SimObject* so, SoHandle soh) {
215  auto* cell = dynamic_cast<Cell*>(so);
216  auto idx = soh.GetElementIdx();
217  Double3 new_pos;
218  new_pos[0] = cell_movements[idx][0];
219  new_pos[1] = cell_movements[idx][1];
220  new_pos[2] = cell_movements[idx][2];
221  cell->UpdatePosition(new_pos);
222  if (param->bound_space_) {
223  ApplyBoundingBox(so, param->min_bound_, param->max_bound_);
224  }
225  });
226  }
227 };
