219  void push_back(const T& element) { // NOLINT
220  if (size_ < N) {
221  data_[size_++] = element;
222  } else {
223  // allocate heap memory
224  assert(size_ != std::numeric_limits<uint16_t>::max() &&
225  "Maxium number of elements exceeded");
226  if (size_ == capacity()) {
227  uint64_t tmp = static_cast<uint64_t>(capacity()) * kGrowFactor;
228  uint16_t new_capacity = static_cast<uint16_t>(tmp);
229  if (tmp > std::numeric_limits<uint16_t>::max()) {
230  new_capacity = std::numeric_limits<uint16_t>::max();
231  }
232  reserve(new_capacity);
233  }
234  heap_data_[HeapSize()] = element;
