243  void RetractTerminalEnd(double speed) {
244  // check if is a terminal branch
245  if (daughter_left_ != nullptr) {
246  return;
247  }
248  // scaling for integration step
249  auto* core_param = Simulation::GetActive()->GetParam();
250  speed *= core_param->simulation_time_step_;
251 
252  auto* mother_soma = dynamic_cast<NeuronSoma*>(mother_.Get());
253  auto* mother_neurite = dynamic_cast<NeuriteElement*>(mother_.Get());
254 
255  if (actual_length_ > speed + 0.1) {
256  // if actual_length_ > length : retraction keeping the same tension
257  // (putting a limit on how short a branch can be is absolutely necessary
258  // otherwise the tension might explode)
259 
260  double new_actual_length = actual_length_ - speed;
261  double factor = new_actual_length / actual_length_;
262  SetActualLength(new_actual_length);
263  // cf removeproximalCylinder()
264  resting_length_ =
265  spring_constant_ * actual_length_ / (tension_ + spring_constant_);
266  SetSpringAxis(spring_axis_ * factor);
267 
268  SetMassLocation(mother_->OriginOf(Base::GetUid()) + spring_axis_);
269  UpdatePosition();
270  UpdateVolume(); // and update concentration of internal stuff.
271  } else if (mother_soma) {
272  mother_->RemoveDaughter(Base::GetSoPtr<NeuriteElement>());
273  this->RemoveFromSimulation();
274  } else if (mother_neurite &&
275  mother_neurite->GetDaughterRight() == nullptr) {
276  // if actual_length_ < length and mother is a neurite element with no
277  // other daughter : merge with mother
278  RemoveProximalNeuriteElement(); // also updates volume_...
279  RetractTerminalEnd(speed / core_param->simulation_time_step_);
280  } else {
281  // if mother is neurite element with other daughter or is not a neurite
282  // segment: disappear.
283  mother_->RemoveDaughter(Base::GetSoPtr<NeuriteElement>());
284  this->RemoveFromSimulation();
285

287  }

619  Double3 CalculateDisplacement(double squared_radius, double dt) override {
620  Double3 force_on_my_point_mass{0, 0, 0};
621  Double3 force_on_my_mothers_point_mass{0, 0, 0};
622 
623  // 1) Spring force
624  // Only the spring of this cylinder. The daughters spring also act on this
625  // mass, but they are treated in point (2)
626  double factor = -tension_ / actual_length_; // the minus sign is important
627  // because the spring axis goes
628  // in the opposite direction
629  force_on_my_point_mass = force_on_my_point_mass + (spring_axis_ * factor);
630 
631  // 2) Force transmitted by daugthers (if they exist)
632  if (daughter_left_ != nullptr) {
633  auto force_from_daughter =
634  daughter_left_->ForceTransmittedFromDaugtherToMother(*this);
635  force_on_my_point_mass = force_on_my_point_mass + force_from_daughter;
636  }
637  if (daughter_right_ != nullptr) {
638  auto force_from_daughter =
639  daughter_right_->ForceTransmittedFromDaugtherToMother(*this);
640  force_on_my_point_mass = force_on_my_point_mass + force_from_daughter;
641  }
642 
643  Double3 force_from_neighbors = {0, 0, 0};
644 
645  auto* core_param = Simulation::GetActive()->GetParam();
646  // this value will be used to reduce force for neurite/neurite interactions
647  double h_over_m = 0.01;
648 
649  // 3) Object avoidance force
650  bool has_neurite_neighbor = false;
651  // (We check for every neighbor object if they touch us, i.e. push us away)
652  auto calculate_neighbor_forces = [this, &force_from_neighbors,
653  &force_on_my_mothers_point_mass,
654  &h_over_m, &has_neurite_neighbor](
655  const SimObject* neighbor) {
656  // if neighbor is a NeuriteElement
657  // use shape to determine if neighbor is a NeuriteElement
658  // this is much faster than using a dynamic_cast
659  if (neighbor->GetShape() == Shape::kCylinder) {
660  // if it is a direct relative, or sister branch, we don't take it into
661  // account
662  if (this->GetDaughterLeft() == *neighbor ||
663  this->GetDaughterRight() == *neighbor ||
664  this->GetMother() ==
665  bdm_static_cast<const NeuriteElement*>(neighbor)->GetMother() ||
666  (this->GetMother() == *neighbor)) {
667  return;
668  }
669  } else if (auto* neighbor_soma =
670  dynamic_cast<const NeuronSoma*>(neighbor)) {
671  // if neighbor is NeuronSoma
672  // if it is a direct relative, we don't take it into account
673  if (this->GetMother() == *neighbor_soma) {
674  return;
675  }
676  }
677 
678  DefaultForce force;
679  Double4 force_from_neighbor = force.GetForce(this, neighbor);
680 
681  // hack: if the neighbour is a neurite, we need to reduce the force from
682  // that neighbour in order to avoid kink behaviour
683  if (neighbor->GetShape() == Shape::kCylinder) {
684  force_from_neighbor = force_from_neighbor * h_over_m;
685  has_neurite_neighbor = true;
686  }
687 
688  if (std::abs(force_from_neighbor[3]) <
689  1E-10) { // TODO(neurites) hard coded value
690  // (if all the force is transmitted to the (distal end) point mass)
691  force_from_neighbors[0] += force_from_neighbor[0];
692  force_from_neighbors[1] += force_from_neighbor[1];
693  force_from_neighbors[2] += force_from_neighbor[2];
694  } else {
695  // (if there is a part transmitted to the proximal end)
696  double part_for_point_mass = 1.0 - force_from_neighbor[3];
697  force_from_neighbors[0] += force_from_neighbor[0] * part_for_point_mass;
698  force_from_neighbors[1] += force_from_neighbor[1] * part_for_point_mass;
699  force_from_neighbors[2] += force_from_neighbor[2] * part_for_point_mass;
700  force_on_my_mothers_point_mass[0] +=
701  force_from_neighbor[0] * force_from_neighbor[3];
702  force_on_my_mothers_point_mass[1] +=
703  force_from_neighbor[1] * force_from_neighbor[3];
704  force_on_my_mothers_point_mass[2] +=
705  force_from_neighbor[2] * force_from_neighbor[3];
706  }
707  };
708 
709  auto* ctxt = Simulation::GetActive()->GetExecutionContext();
710  ctxt->ForEachNeighborWithinRadius(calculate_neighbor_forces, *this,
711  squared_radius);
712  // hack: if the neighbour is a neurite, and as we reduced the force from
713  // that neighbour, we also need to reduce my internal force (from internal
714  // tension and daughters)
715  if (has_neurite_neighbor) {
716  force_on_my_point_mass = force_on_my_point_mass * h_over_m;
717  }
718 
719  force_on_my_point_mass = force_on_my_point_mass + force_from_neighbors;
720 
721  // 5) define the force that will be transmitted to the mother
722  force_to_transmit_to_proximal_mass_ = force_on_my_mothers_point_mass;
723  // 6.1) Define movement scale
724  double force_norm = force_on_my_point_mass.Norm();
725  // 6.2) If is F not strong enough -> no movements
726  if (force_norm < adherence_) {
727  return {0, 0, 0};
728  }
729 
730  // So, what follows is only executed if we do actually move :
731 
732  // 6.3) Since there's going be a move, we calculate it
733  auto& displacement = force_on_my_point_mass;
734  double& displacement_norm = force_norm;
735 
736  // 6.4) There is an upper bound for the movement.
737  if (displacement_norm > core_param->simulation_max_displacement_) {
738  displacement = displacement * (core_param->simulation_max_displacement_ /
739  displacement_norm);
740  }
741  return displacement;
742  }
