226  NeighborIterator& ForwardToNonEmptyBox(uint64_t grid_timestamp) {
227  // increment box id until non empty box has been found
228  while (++box_idx_ < neighbor_boxes_.size()) {
229  // box is empty or uninitialized (padding box) -> continue
230  if (neighbor_boxes_[box_idx_]->IsEmpty(grid_timestamp)) {

282  void UpdateGrid() {
283  auto* rm = Simulation::GetActive()->GetResourceManager();
284 
285  if (rm->GetNumSimObjects() != 0) {
286  ClearGrid();
287  timestamp_++;
288 
289  auto inf = Math::kInfinity;
290  std::array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
291  CalculateGridDimensions(&tmp_dim);
292  RoundOffGridDimensions(tmp_dim);
293 
294  auto los = ceil(largest_object_size_);
295  assert(los > 0 &&
296  "The largest object size was found to be 0. Please check if your "
297  "cells are correctly initialized.");
298  box_length_ = los;
299 
300  for (int i = 0; i < 3; i++) {
301  int dimension_length =
302  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
303  int r = dimension_length % box_length_;
304  // If the grid is not perfectly divisible along each dimension by the
305  // resolution, extend the grid so that it is
306  if (r != 0) {
307  // std::abs for the case that box_length_ > dimension_length
308  grid_dimensions_[2 * i + 1] += (box_length_ - r);
309  } else {
310  // Else extend the grid dimension with one row, because the outmost
311  // object lies exactly on the border
312  grid_dimensions_[2 * i + 1] += box_length_;
313  }
314  }
315 
316  // Pad the grid to avoid out of bounds check when search neighbors
317  for (int i = 0; i < 3; i++) {
318  grid_dimensions_[2 * i] -= box_length_;
319  grid_dimensions_[2 * i + 1] += box_length_;
320  }
321 
322  // Calculate how many boxes fit along each dimension
323  for (int i = 0; i < 3; i++) {
324  int dimension_length =
325  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
326  assert((dimension_length % box_length_ == 0) &&
327  "The grid dimensions are not a multiple of its box length");
328  num_boxes_axis_[i] = dimension_length / box_length_;
329  }
330 
331  num_boxes_xy_ = num_boxes_axis_[0] * num_boxes_axis_[1];
332  auto total_num_boxes = num_boxes_xy_ * num_boxes_axis_[2];
333 
334  CheckGridGrowth();
335 
336  // resize boxes_
337  if (boxes_.size() != total_num_boxes) {
338  if (boxes_.capacity() < total_num_boxes) {
339  boxes_.reserve(total_num_boxes * 2);
340  }
341  boxes_.resize(total_num_boxes);
342  }
343 
344  successors_.reserve();
345 
346  // Assign simulation objects to boxes
347  rm->ApplyOnAllElementsParallelDynamic(
348  1000, [this](SimObject* sim_object, SoHandle soh) {
349  const auto& position = sim_object->GetPosition();
350  auto idx = this->GetBoxIndex(position);
351  auto box = this->GetBoxPointer(idx);
352  box->AddObject(soh, &successors_, this);
353  sim_object->SetBoxIdx(idx);
354  });
355  auto* param = Simulation::GetActive()->GetParam();
356  if (param->bound_space_) {
357  int min = param->min_bound_;
358  int max = param->max_bound_;
359  threshold_dimensions_ = {min, max};
360  }
361 
362  if (nb_mutex_builder_ != nullptr) {
363  nb_mutex_builder_->Update();
364  }
365  } else {
366  // There are no sim objects in this simulation
367  auto* param = Simulation::GetActive()->GetParam();
368 
369  bool uninitialized = boxes_.size() == 0;
370  if (uninitialized && param->bound_space_) {
371  // Simulation has never had any simulation objects
372  // Initialize grid dimensions with `Param::min_bound_` and
373  // `Param::max_bound_`
374  // This is required for the DiffusionGrid
375  int min = param->min_bound_;
376  int max = param->max_bound_;
377  grid_dimensions_ = {min, max, min, max, min, max};
378  threshold_dimensions_ = {min, max};
379  has_grown_ = true;
380  } else if (!uninitialized) {
381  // all simulation objects have been removed in the last iteration
382  // grid state remains the same, but we have to set has_grown_ to false
383  // otherwise the DiffusionGrid will attempt to resize
384  has_grown_ = false;
385  } else {
386  Log::Fatal(
387  "Grid",
388  "You tried to initialize an empty simulation without bound space. "
389  "Therefore we cannot determine the size of the simulation space. "
390  "Please add simulation objects, or set Param::bound_space_, "
391  "Param::min_bound_, and Param::max_bound_.");
392  }
393  }

404  inline double SquaredEuclideanDistance(const Double3& pos1,
405  const Double3& pos2) const {
406  const double dx = pos2[0] - pos1[0];
407  const double dy = pos2[1] - pos1[1];
408  const double dz = pos2[2] - pos1[2];
409  return (dx * dx + dy * dy + dz * dz);
410  }

412  inline bool WithinSquaredEuclideanDistance(double squared_radius,
413  const Double3& pos1,
414  const Double3& pos2) const {
415  const double dx = pos2[0] - pos1[0];
416  const double dx2 = dx * dx;
417  if (dx2 > squared_radius) {
418  return false;
419  }
420 
421  const double dy = pos2[1] - pos1[1];
422  const double dy2_plus_dx2 = dy * dy + dx2;
423  if (dy2_plus_dx2 > squared_radius) {
424  return false;
425  }
426 
427  const double dz = pos2[2] - pos1[2];
428  const double distance = dz * dz + dy2_plus_dx2;
429  return distance < squared_radius;
430  }

505  void ForEachNeighbor(
506  const std::function<void(const SimObject*, double)>& lambda,
507  const SimObject& query) {
508  const auto& position = query.GetPosition();
509  auto idx = query.GetBoxIdx();
510 
511  FixedSizeVector<const Box*, 27> neighbor_boxes;
512  GetMooreBoxes(&neighbor_boxes, idx);
513 
514  auto* rm = Simulation::GetActive()->GetResourceManager();
515 
516  NeighborIterator ni(neighbor_boxes, timestamp_);
517  const unsigned batch_size = 64;
518  uint64_t size = 0;
519  SimObject* sim_objects[batch_size] __attribute__((aligned(64)));
520  double x[batch_size] __attribute__((aligned(64)));
521  double y[batch_size] __attribute__((aligned(64)));
522  double z[batch_size] __attribute__((aligned(64)));
523  double squared_distance[batch_size] __attribute__((aligned(64)));
524 
525  auto process_batch = [&]() {
526 #pragma omp simd
527  for (uint64_t i = 0; i < size; ++i) {
528  const double dx = x[i] - position[0];
529  const double dy = y[i] - position[1];
530  const double dz = z[i] - position[2];
531 
532  squared_distance[i] = dx * dx + dy * dy + dz * dz;
533  }
534 
535  for (uint64_t i = 0; i < size; ++i) {
536  lambda(sim_objects[i], squared_distance[i]);
537  }
538  size = 0;
539  };
540 
541  while (!ni.IsAtEnd()) {
542  auto soh = *ni;
543  // increment iterator already here to hide memory latency
544  ++ni;
545  auto* sim_object = rm->GetSimObjectWithSoHandle(soh);
546  if (sim_object != &query) {
547  sim_objects[size] = sim_object;
548  const auto& pos = sim_object->GetPosition();
549  x[size] = pos[0];
550  y[size] = pos[1];
551  z[size] = pos[2];
552  size++;
553  if (size == batch_size) {
554  process_batch();
555  }
556  }
557  }
558  process_batch();
559  }

787  auto max_gd =

806  auto* rm = Simulation::GetActive()->GetResourceManager();
807 
808  const auto max_threads = omp_get_max_threads();
809  // allocate version for each thread - avoid false sharing by padding them
810  // assumes 64 byte cache lines (8 * sizeof(double))
811  std::vector<std::array<double, 8>> xmin(max_threads, {{Math::kInfinity}});
812  std::vector<std::array<double, 8>> ymin(max_threads, {{Math::kInfinity}});
813  std::vector<std::array<double, 8>> zmin(max_threads, {{Math::kInfinity}});
814 
815  std::vector<std::array<double, 8>> xmax(max_threads, {{-Math::kInfinity}});
816  std::vector<std::array<double, 8>> ymax(max_threads, {{-Math::kInfinity}});
817  std::vector<std::array<double, 8>> zmax(max_threads, {{-Math::kInfinity}});
818 
819  std::vector<std::array<double, 8>> largest(max_threads, {{0}});
820 
821  rm->ApplyOnAllElementsParallelDynamic(1000, [&](SimObject* so, SoHandle) {
822  auto tid = omp_get_thread_num();
823  const auto& position = so->GetPosition();
824  // x
825  if (position[0] < xmin[tid][0]) {
826  xmin[tid][0] = position[0];
827  }
828  if (position[0] > xmax[tid][0]) {
829  xmax[tid][0] = position[0];
830  }
831  // y
832  if (position[1] < ymin[tid][0]) {
833  ymin[tid][0] = position[1];
834  }
835  if (position[1] > ymax[tid][0]) {
836  ymax[tid][0] = position[1];
837  }
838  // z
839  if (position[2] < zmin[tid][0]) {
840  zmin[tid][0] = position[2];
841  }
842  if (position[2] > zmax[tid][0]) {
843  zmax[tid][0] = position[2];
844  }
845  // larget object
846  auto diameter = so->GetDiameter();
847  if (diameter > largest[tid][0]) {
848  largest[tid][0] = diameter;
849  }
850  });
851 
852  // reduce partial results into global one
853  double& gxmin = (*ret_grid_dimensions)[0];
854  double& gxmax = (*ret_grid_dimensions)[1];
855  double& gymin = (*ret_grid_dimensions)[2];
856  double& gymax = (*ret_grid_dimensions)[3];
857  double& gzmin = (*ret_grid_dimensions)[4];
858  double& gzmax = (*ret_grid_dimensions)[5];
859  for (int tid = 0; tid < max_threads; tid++) {
860  // x
861  if (xmin[tid][0] < gxmin) {
862  gxmin = xmin[tid][0];
863  }
864  if (xmax[tid][0] > gxmax) {
865  gxmax = xmax[tid][0];
866  }
867  // y
868  if (ymin[tid][0] < gymin) {
869  gymin = ymin[tid][0];
870  }
871  if (ymax[tid][0] > gymax) {
872  gymax = ymax[tid][0];
873  }
874  // z
875  if (zmin[tid][0] < gzmin) {
876  gzmin = zmin[tid][0];
877  }
878  if (zmax[tid][0] > gzmax) {
879  gzmax = zmax[tid][0];
880  }
881  // larget object
882  if (largest[tid][0] > largest_object_size_) {
883  largest_object_size_ = largest[tid][0];
884  }
885  }
886  }
