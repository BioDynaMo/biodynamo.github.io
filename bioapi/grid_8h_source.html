227  NeighborIterator& ForwardToNonEmptyBox(uint64_t grid_timestamp) {
228  // increment box id until non empty box has been found
229  while (++box_idx_ < neighbor_boxes_.size()) {
230  // box is empty or uninitialized (padding box) -> continue
231  if (neighbor_boxes_[box_idx_]->IsEmpty(grid_timestamp)) {
232  continue;

298  void UpdateGrid() {
299  auto* rm = Simulation::GetActive()->GetResourceManager();
300 
301  if (rm->GetNumSimObjects() != 0) {
302  ClearGrid();
303  timestamp_++;
304 
305  auto inf = Math::kInfinity;
306  std::array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
307  CalculateGridDimensions(&tmp_dim);
308  RoundOffGridDimensions(tmp_dim);
309 
310  auto los = ceil(largest_object_size_);
311  assert(los > 0 &&
312  "The largest object size was found to be 0. Please check if your "
313  "cells are correctly initialized.");
314  box_length_ = los;
315 
316  for (int i = 0; i < 3; i++) {
317  int dimension_length =
318  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
319  int r = dimension_length % box_length_;
320  // If the grid is not perfectly divisible along each dimension by the
321  // resolution, extend the grid so that it is
322  if (r != 0) {
323  // std::abs for the case that box_length_ > dimension_length
324  grid_dimensions_[2 * i + 1] += (box_length_ - r);
325  } else {
326  // Else extend the grid dimension with one row, because the outmost
327  // object lies exactly on the border
328  grid_dimensions_[2 * i + 1] += box_length_;
329  }
330  }
331 
332  // Pad the grid to avoid out of bounds check when search neighbors
333  for (int i = 0; i < 3; i++) {
334  grid_dimensions_[2 * i] -= box_length_;
335  grid_dimensions_[2 * i + 1] += box_length_;
336  }
337 
338  // Calculate how many boxes fit along each dimension
339  for (int i = 0; i < 3; i++) {
340  int dimension_length =
341  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
342  assert((dimension_length % box_length_ == 0) &&
343  "The grid dimensions are not a multiple of its box length");
344  num_boxes_axis_[i] = dimension_length / box_length_;
345  }
346 
347  num_boxes_xy_ = num_boxes_axis_[0] * num_boxes_axis_[1];
348  auto total_num_boxes = num_boxes_xy_ * num_boxes_axis_[2];
349 
350  CheckGridGrowth();
351 
352  // resize boxes_
353  if (boxes_.size() != total_num_boxes) {
354  if (boxes_.capacity() < total_num_boxes) {
355  boxes_.reserve(total_num_boxes * 2);
356  }
357  boxes_.resize(total_num_boxes);
358  }
359 
360  successors_.reserve();
361 
362  // Assign simulation objects to boxes
363  AssignToBoxesFunctor functor(this);
364  rm->ApplyOnAllElementsParallelDynamic(1000, functor);
365  auto* param = Simulation::GetActive()->GetParam();
366  if (param->bound_space_) {
367  int min = param->min_bound_;
368  int max = param->max_bound_;
369  threshold_dimensions_ = {min, max};
370  }
371 
372  if (param->thread_safety_mechanism_ ==
373  Param::ThreadSafetyMechanism::kAutomatic) {
374  nb_mutex_builder_->Update();
375  }
376  } else {
377  // There are no sim objects in this simulation
378  auto* param = Simulation::GetActive()->GetParam();
379 
380  bool uninitialized = boxes_.size() == 0;
381  if (uninitialized && param->bound_space_) {
382  // Simulation has never had any simulation objects
383  // Initialize grid dimensions with `Param::min_bound_` and
384  // `Param::max_bound_`
385  // This is required for the DiffusionGrid
386  int min = param->min_bound_;
387  int max = param->max_bound_;
388  grid_dimensions_ = {min, max, min, max, min, max};
389  threshold_dimensions_ = {min, max};
390  has_grown_ = true;
391  } else if (!uninitialized) {
392  // all simulation objects have been removed in the last iteration
393  // grid state remains the same, but we have to set has_grown_ to false
394  // otherwise the DiffusionGrid will attempt to resize
395  has_grown_ = false;
396  } else {
397  Log::Fatal(
398  "Grid",
399  "You tried to initialize an empty simulation without bound space. "
400  "Therefore we cannot determine the size of the simulation space. "
401  "Please add simulation objects, or set Param::bound_space_, "
402  "Param::min_bound_, and Param::max_bound_.");
403  }
404  }
405  }

415  inline double SquaredEuclideanDistance(const Double3& pos1,
416  const Double3& pos2) const {
417  const double dx = pos2[0] - pos1[0];
418  const double dy = pos2[1] - pos1[1];
419  const double dz = pos2[2] - pos1[2];
420  return (dx * dx + dy * dy + dz * dz);
421  }

423  inline bool WithinSquaredEuclideanDistance(double squared_radius,
424  const Double3& pos1,
425  const Double3& pos2) const {
426  const double dx = pos2[0] - pos1[0];
427  const double dx2 = dx * dx;
428  if (dx2 > squared_radius) {
429  return false;
430  }
431 
432  const double dy = pos2[1] - pos1[1];
433  const double dy2_plus_dx2 = dy * dy + dx2;
434  if (dy2_plus_dx2 > squared_radius) {
435  return false;
436  }
437 
438  const double dz = pos2[2] - pos1[2];
439  const double distance = dz * dz + dy2_plus_dx2;
440  return distance < squared_radius;
441  }

443  void UpdateBoxZOrder() {
444  // iterate boxes in Z-order / morton order
445  // TODO(lukas) this is a very quick attempt to test an idea
446  // improve performance of this brute force solution
447  zorder_sorted_boxes_.resize(boxes_.size());
448  const uint32_t nx = num_boxes_axis_[0];
449  const uint32_t ny = num_boxes_axis_[1];
450  const uint32_t nz = num_boxes_axis_[2];
451 #pragma omp parallel for collapse(3)
452  for (uint32_t x = 0; x < nx; x++) {
453  for (uint32_t y = 0; y < ny; y++) {
454  for (uint32_t z = 0; z < nz; z++) {
455  auto box_idx = GetBoxIndex(std::array<uint32_t, 3>{x, y, z});
456  auto morton = libmorton::morton3D_64_encode(x, y, z);
457  zorder_sorted_boxes_[box_idx] =
458  std::pair<uint32_t, const Box*>{morton, &boxes_[box_idx]};
459  }
460  }
461  }
462 #ifdef LINUX
463  __gnu_parallel::sort(
464  zorder_sorted_boxes_.begin(), zorder_sorted_boxes_.end(),
465  [](const auto& lhs, const auto& rhs) { return lhs.first < rhs.first; });
466 #else
467  std::sort(
468  zorder_sorted_boxes_.begin(), zorder_sorted_boxes_.end(),
469  [](const auto& lhs, const auto& rhs) { return lhs.first < rhs.first; });
470 #endif // LINUX

519  void ForEachNeighbor(Functor<void, const SimObject*, double>& lambda,
520  const SimObject& query) {
521  const auto& position = query.GetPosition();
522  auto idx = query.GetBoxIdx();
523 
524  FixedSizeVector<const Box*, 27> neighbor_boxes;
525  GetMooreBoxes(&neighbor_boxes, idx);
526 
527  auto* rm = Simulation::GetActive()->GetResourceManager();
528 
529  NeighborIterator ni(neighbor_boxes, timestamp_);
530  const unsigned batch_size = 64;
531  uint64_t size = 0;
532  SimObject* sim_objects[batch_size] __attribute__((aligned(64)));
533  double x[batch_size] __attribute__((aligned(64)));
534  double y[batch_size] __attribute__((aligned(64)));
535  double z[batch_size] __attribute__((aligned(64)));
536  double squared_distance[batch_size] __attribute__((aligned(64)));
537 
538  auto process_batch = [&]() {
539 #pragma omp simd
540  for (uint64_t i = 0; i < size; ++i) {
541  const double dx = x[i] - position[0];
542  const double dy = y[i] - position[1];
543  const double dz = z[i] - position[2];
544 
545  squared_distance[i] = dx * dx + dy * dy + dz * dz;
546  }
547 
548  for (uint64_t i = 0; i < size; ++i) {
549  lambda(sim_objects[i], squared_distance[i]);
550  }
551  size = 0;
552  };
553 
554  while (!ni.IsAtEnd()) {
555  auto soh = *ni;
556  // increment iterator already here to hide memory latency
557  ++ni;
558  auto* sim_object = rm->GetSimObjectWithSoHandle(soh);
559  if (sim_object != &query) {
560  sim_objects[size] = sim_object;
561  const auto& pos = sim_object->GetPosition();
562  x[size] = pos[0];
563  y[size] = pos[1];
564  z[size] = pos[2];
565  size++;
566  if (size == batch_size) {
567  process_batch();
568  }
569  }
570  }
571  process_batch();
572  }

899  // x
900  if (xmin[tid][0] < gxmin) {
901  gxmin = xmin[tid][0];
902  }
903  if (xmax[tid][0] > gxmax) {
904  gxmax = xmax[tid][0];

911  gymax = ymax[tid][0];
912  }

915  gzmin = zmin[tid][0];
916  }
917  if (zmax[tid][0] > gzmax) {
918  gzmax = zmax[tid][0];
919  }
