76  void Run(SimObject* sim_object) override {
77  auto* sim = Simulation::GetActive();
78  auto* param = sim->GetParam();
79  auto* scheduler = sim->GetScheduler();
80 
81  const auto& timestep = param->simulation_time_step_;
82  uint64_t simulated_steps = scheduler->GetSimulatedSteps();
83  const auto absolute_time = simulated_steps * timestep;
84 
85  if (param->numerical_ode_solver_ == Param::NumericalODESolver::kEuler) {
86  // Euler
87  for (uint64_t i = 0; i < first_derivatives_.size(); i++) {
88  double slope = first_derivatives_[i](absolute_time, concentrations_[i]);
89  concentrations_[i] += slope * timestep;
90  }
91  } else if (param->numerical_ode_solver_ ==
92  Param::NumericalODESolver::kRK4) {
93  // Runge-Kutta 4
94  for (uint64_t i = 0; i < first_derivatives_.size(); i++) {
95  double interval_midpoint = absolute_time + timestep / 2.0;
96  double interval_endpoint = absolute_time + timestep;
97 
98  double k1 = first_derivatives_[i](absolute_time, concentrations_[i]);
99  double k2 = first_derivatives_[i](
100  interval_midpoint, concentrations_[i] + timestep * k1 / 2.0);
101  double k3 = first_derivatives_[i](
102  interval_midpoint, concentrations_[i] + timestep * k2 / 2.0);
103  double k4 = first_derivatives_[i](interval_endpoint,
104  concentrations_[i] + timestep * k3);
105 
106  concentrations_[i] += timestep / 6.0 * (k1 + 2 * k2 + 2 * k3 + k4);
107  }
108  }
109  }
