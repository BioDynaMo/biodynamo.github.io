91  void Run(SimObject* sim_object) override {
92  auto* sim = Simulation::GetActive();
93  auto* param = sim->GetParam();
94  auto* scheduler = sim->GetScheduler();
95 
96  const auto& timestep = param->simulation_time_step_;
97  uint64_t simulated_steps = scheduler->GetSimulatedSteps();
98  const auto absolute_time = simulated_steps * timestep;
99 
100  if (param->numerical_ode_solver_ == Param::NumericalODESolver::kEuler) {
101  // Euler
102  for (uint64_t i = 0; i < first_derivatives_.size(); i++) {
103  double slope = first_derivatives_[i](absolute_time, concentrations_[i]);
104  concentrations_[i] += slope * timestep;
105  }
106  } else if (param->numerical_ode_solver_ ==
107  Param::NumericalODESolver::kRK4) {
108  // Runge-Kutta 4
109  for (uint64_t i = 0; i < first_derivatives_.size(); i++) {
110  double interval_midpoint = absolute_time + timestep / 2.0;
111  double interval_endpoint = absolute_time + timestep;
112 
113  double k1 = first_derivatives_[i](absolute_time, concentrations_[i]);
114  double k2 = first_derivatives_[i](
115  interval_midpoint, concentrations_[i] + timestep * k1 / 2.0);
116  double k3 = first_derivatives_[i](
117  interval_midpoint, concentrations_[i] + timestep * k2 / 2.0);
118  double k4 = first_derivatives_[i](interval_endpoint,
119  concentrations_[i] + timestep * k3);
120 
121  concentrations_[i] += timestep / 6.0 * (k1 + 2 * k2 + 2 * k3 + k4);
122  }
123  }
124  }
