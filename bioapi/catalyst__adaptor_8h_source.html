103  void Visualize(uint64_t total_steps, bool last_iteration) {
104  if (!initialized_) {
105  Initialize();

107  }
108 
109  auto* param = Simulation::GetActive()->GetParam();

112  }
113 
114  if (param->live_visualization_) {

118  if (param->export_visualization_) {

121  }

146  void Initialize() {
147  auto* sim = Simulation::GetActive();
148  auto* param = sim->GetParam();
149 
150  exclusive_export_viz_ =
151  param->export_visualization_ && !param->live_visualization_;
152  if (param->live_visualization_ || param->export_visualization_) {
153  if (g_processor_ == nullptr) {
154  g_processor_ = vtkCPProcessor::New();
155  g_processor_->Initialize();
156  }
157 
158  if (param->live_visualization_ &&
159  g_processor_->GetNumberOfPipelines() != 0) {
160  Log::Fatal("CatalystAdaptor",
161  "Live visualization does not support multiple "
162  "simulations. Turning off live visualization for ",
163  sim->GetUniqueName());
164  } else if (param->python_catalyst_pipeline_) {
165  vtkNew<vtkCPPythonScriptPipeline> pipeline;
166  pipeline->Initialize(python_script_.c_str());
167  g_processor_->AddPipeline(pipeline.GetPointer());
168  } else if (!exclusive_export_viz_) {
169  pipeline_ = new InSituPipeline();
170  g_processor_->AddPipeline(pipeline_);
171  }
172 
173  vtkNew<vtkCPDataDescription> data_description;
174  data_description->SetTimeData(0, 0);
175 
176  auto* param = Simulation::GetActive()->GetParam();
177  for (auto& pair : param->visualize_sim_objects_) {
178  vtk_so_grids_[pair.first.c_str()] =
179  new VtkSoGrid(pair.first.c_str(), data_description);
180  }
181  for (auto& entry : param->visualize_diffusion_) {
182  vtk_dgrids_[entry.name_] =
183  new VtkDiffusionGrid(entry.name_, data_description);
184  }
185  }

247  // Process a single simulation object
248  void ProcessSimObject(const SimObject* so,
249  const vtkNew<vtkCPDataDescription>& data_description) {
250  auto* param = Simulation::GetActive()->GetParam();
251  auto so_name = so->GetTypeName();
252 
253  if (param->visualize_sim_objects_.find(so_name) !=
254  param->visualize_sim_objects_.end()) {
255  // assert(vtk_so_grids_.find(so->GetTypeName()) != vtk_so_grids_.end() &&
256  // Concat("VtkSoGrid for ", so->GetTypeName(), " not found!");
257  auto* vsg = vtk_so_grids_[so->GetTypeName()];
258  if (!vsg->initialized_) {
259  vsg->Init(so);
260  }
261 
262  // If we segfault at here it probably means that there is a problem
263  // with the pipeline (either the C++ pipeline or Python pipeline)
264  // We do not need to RequestDataDescription in Export Mode, because
265  // we do not make use of Catalyst CoProcessing capabilities
266  if (exclusive_export_viz_ ||
267  (g_processor_->RequestDataDescription(
268  data_description.GetPointer())) != 0) {
269  CatalystSoVisitor visitor(vsg);
270  so->ForEachDataMemberIn(vsg->vis_data_members_, &visitor);
271  }
272  }

288  void ProcessDiffusionGrid(
289  const DiffusionGrid* grid,
290  const vtkNew<vtkCPDataDescription>& data_description) {
291  auto* param = Simulation::GetActive()->GetParam();
292  auto name = grid->GetSubstanceName();
293 
294  // get visualization config
295  const Param::VisualizeDiffusion* vd = nullptr;
296  for (auto& entry : param->visualize_diffusion_) {
297  if (entry.name_ == name) {
298  vd = &entry;
299  }
300  }
301 
302  if (vd != nullptr) {
303  auto* vdg = vtk_dgrids_[grid->GetSubstanceName()];
304  if (!vdg->used_) {
305  vdg->Init();
306  }
307 
308  // If we segfault at here it probably means that there is a problem
309  // with the pipeline (either the C++ pipeline or Python pipeline)
310  // We do not need to RequestDataDescription in Export Mode, because
311  // we do not make use of Catalyst CoProcessing capabilities
312  if (exclusive_export_viz_ ||
313  (g_processor_->RequestDataDescription(
314  data_description.GetPointer())) != 0) {
315  auto num_boxes = grid->GetNumBoxesArray();
316  auto grid_dimensions = grid->GetDimensions();
317  auto box_length = grid->GetBoxLength();
318  auto total_boxes = grid->GetNumBoxes();
319 
320  double origin_x = grid_dimensions[0];
321  double origin_y = grid_dimensions[2];
322  double origin_z = grid_dimensions[4];
323  vdg->data_->SetOrigin(origin_x, origin_y, origin_z);
324  vdg->data_->SetDimensions(num_boxes[0], num_boxes[1], num_boxes[2]);
325  vdg->data_->SetSpacing(box_length, box_length, box_length);
326 
327  if (vdg->concentration_) {
328  auto* co_ptr = const_cast<double*>(grid->GetAllConcentrations());
329  vdg->concentration_->SetArray(co_ptr,
330  static_cast<vtkIdType>(total_boxes), 1);
331  }
332  if (vdg->gradient_) {
333  auto gr_ptr = const_cast<double*>(grid->GetAllGradients());
334  vdg->gradient_->SetArray(gr_ptr,
335  static_cast<vtkIdType>(total_boxes * 3), 1);
336  }
337  }
338  }

386  static void GenerateSimulationInfoJson(
387  const std::unordered_map<std::string, VtkSoGrid*>& vtk_so_grids,
388  const std::unordered_map<std::string, VtkDiffusionGrid*>& vtk_dgrids) {
389  auto* sim = Simulation::GetActive();
390  auto* param = sim->GetParam();
391  // simulation objects
392  std::stringstream sim_objects;
393  uint64_t num_sim_objects = param->visualize_sim_objects_.size();
394  uint64_t counter = 0;
395  for (const auto& entry : param->visualize_sim_objects_) {
396  std::string so_name = entry.first;
397 
398  auto search = vtk_so_grids.find(so_name);
399  if (search == vtk_so_grids.end()) {
400  continue;
401  }
402  auto* so_grid = search->second;
403  // user wanted to export this simulation object type, but not a single
404  // instance has been created during the entire simulation
405  if (!so_grid->initialized_) {
406  Log::Warning("Visualize Simulation Objects",
407  "You are trying to visualize simulation object type ",
408  so_name,
409  ", but not a single instance has been created during the "
410  "entire simulation. Please make sure the names in the "
411  "configuration file match the ones in the simulation.");
412  num_sim_objects--;
413  continue;
414  }
415 
416  sim_objects << " {" << std::endl
417  << " \"name\":\"" << so_name << "\"," << std::endl;
418  if (so_grid->shape_ == Shape::kSphere) {
419  sim_objects << " \"glyph\":\"Glyph\"," << std::endl
420  << " \"shape\":\"Sphere\"," << std::endl
421  << " \"scaling_attribute\":\"diameter_\"" << std::endl;
422  } else if (so_grid->shape_ == kCylinder) {
423  sim_objects << " \"glyph\":\"BDMGlyph\"," << std::endl
424  << " \"shape\":\"Cylinder\"," << std::endl
425  << " \"x_scaling_attribute\":\"diameter_\","
426  << std::endl
427  << " \"y_scaling_attribute\":\"actual_length_\","
428  << std::endl
429  << " \"z_scaling_attribute\":\"diameter_\","
430  << std::endl
431  << " \"Vectors\":\"spring_axis_\"," << std::endl
432  << " \"MassLocation\":\"mass_location_\"" << std::endl;
433  }
434  sim_objects << " }";
435  if (counter != num_sim_objects - 1) {
436  sim_objects << ",";
437  }
438  sim_objects << std::endl;
439  counter++;
440  }
441 
442  // extracellular substances
443  std::stringstream substances;
444  uint64_t num_substances = param->visualize_diffusion_.size();
445  for (uint64_t i = 0; i < num_substances; i++) {
446  auto& name = param->visualize_diffusion_[i].name_;
447 
448  auto search = vtk_dgrids.find(name);
449  if (search == vtk_dgrids.end()) {
450  continue;
451  }
452  auto* dgrid = search->second;
453  // user wanted to export this substance, but it did not exist during
454  // the entire simulation
455  if (!dgrid->used_) {
456  Log::Warning(
457  "Visualize Diffusion Grids",
458  "You are trying to visualize diffusion grid ", name,
459  ", but it has not been created during the entire simulation. "
460  "Please make sure the names in the "
461  "configuration file match the ones in the simulation.");
462  num_substances--;
463  continue;
464  }
465  substances << " { \"name\":\"" << name << "\", ";
466  std::string has_gradient =
467  param->visualize_diffusion_[i].gradient_ ? "true" : "false";
468  substances << "\"has_gradient\":\"" << has_gradient << "\" }";
469 
470  if (i != num_substances - 1) {
471  substances << "," << std::endl;
472  }
473  }
474 
475  // write to file
476  std::ofstream ofstr;
477  ofstr.open(Concat(sim->GetOutputDir(), "/", kSimulationInfoJson));
478  ofstr << "{" << std::endl
479  << " \"simulation\": {" << std::endl
480  << " \"name\":\"" << sim->GetUniqueName() << "\"," << std::endl
481  << " \"result_dir\":\"" << sim->GetOutputDir() << "\"" << std::endl
482  << " }," << std::endl
483  << " \"sim_objects\": [" << std::endl
484  << sim_objects.str() << " ]," << std::endl
485  << " \"extracellular_substances\": [" << std::endl
486  << substances.str() << std::endl
487  << " ]" << std::endl
488  << "}" << std::endl;
489  ofstr.close();
490  }
