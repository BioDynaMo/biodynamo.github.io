24 inline void ApplyBoundingBox(SimObject* sim_object, double lb, double rb) {
25  // Need to create a small distance from the positive edge of each dimension;
26  // otherwise it will fall out of the boundary of the simulation space
27  double eps = 1e-10;
28  auto pos = sim_object->GetPosition();
29  bool updated = false;
30  for (int i = 0; i < 3; i++) {
31  if (pos[i] < lb) {
32  pos[i] = lb;
33  updated = true;
34  } else if (pos[i] >= rb) {
35  pos[i] = rb - eps;
36  updated = true;
37  }
38  }
39  if (updated) {
40  sim_object->SetPosition(pos);
41  }
42 }
