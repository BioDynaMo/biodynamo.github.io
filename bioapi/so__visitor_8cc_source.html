34 TDataArray* GetDataArray(const std::string& dm_name, VtkSoGrid* so_grid,
35  int components = 1) {
36  TDataArray* vtk_array = nullptr;
37  auto& data_arrays = so_grid->data_arrays_;
38  auto search = data_arrays.find(dm_name);
39  if (search != data_arrays.end()) {
40  auto& da_wrapper = search->second;
41  vtk_array = static_cast<TDataArray*>(da_wrapper.data_);
42 
43  // reset
44  auto* scheduler = Simulation::GetActive()->GetScheduler();
45  if (da_wrapper.time_step_ != scheduler->GetSimulatedSteps()) {
46  vtk_array->Reset();
47  da_wrapper.time_step_ = scheduler->GetSimulatedSteps();
48  }
49 
50  } else {
51  // create
52  vtkNew<TDataArray> new_vtk_array;
53  new_vtk_array->SetName(dm_name.c_str());
54  vtk_array = new_vtk_array.GetPointer();
55  vtk_array->SetNumberOfComponents(components);
56  auto* point_data = so_grid->data_->GetPointData();
57  point_data->AddArray(vtk_array);
58  data_arrays.insert({dm_name, VtkDataArrayWrapper(vtk_array)});
59  }
60 
61  return vtk_array;
62 }

113 void ParaviewSoVisitor::Visit(const std::string& dm_name, size_t type_hash_code,
114  const void* data) {
115  if (type_hash_code == typeid(double).hash_code()) {
116  Double(dm_name, data);
117  } else if (type_hash_code == typeid(int).hash_code()) {
118  Int(dm_name, data);
119  } else if (type_hash_code == typeid(uint64_t).hash_code()) {
120  Uint64T(dm_name, data);
121  } else if (type_hash_code == typeid(SoUid).hash_code()) {
122  uint64_t uid = *static_cast<const SoUid*>(data);
123  Uint64T(dm_name, &uid);
124  } else if (type_hash_code == typeid(Double3).hash_code()) {
125  MathArray3(dm_name, data);
126  } else if (type_hash_code == typeid(std::array<int, 3>).hash_code()) {
127  Int3(dm_name, data);
128  } else {
129  Log::Fatal("ParaviewSoVisitor::Visit",
130  "This data member is not supported for visualization");
131  }
132 }
