164 void Scheduler::Execute() {
165  auto* sim = Simulation::GetActive();
166  auto* rm = sim->GetResourceManager();
167  auto* grid = sim->GetGrid();
168  auto* param = sim->GetParam();
169 
170  Timing::Time("Set up exec context", [&]() {
171  const auto& all_exec_ctxts = sim->GetAllExecCtxts();
172  all_exec_ctxts[0]->SetupIterationAll(all_exec_ctxts);
173  });
174 
175  Timing::Time("visualize", [&]() {
176  if (visualization_ != nullptr) {
177  visualization_->Visualize();
178  }
179  });
180  Timing::Time("neighbors", [&]() { grid->UpdateGrid(); });
181 
182  // update all sim objects: run all CPU operations
183  auto scheduled_ops = GetScheduleOps();
184  RunAllScheduldedOps functor(scheduled_ops);
185  rm->ApplyOnAllElementsParallelDynamic(param->scheduling_batch_size_, functor);
186 
187  // update all sim objects: hardware accelerated operations
188  if (param->run_mechanical_interactions_ && !displacement_->UseCpu()) {
189  Timing::Time("displacement (GPU/FPGA)", *displacement_);
190  }
191 
192  // finish updating sim objects
193  Timing::Time("Tear down exec context", [&]() {
194  const auto& all_exec_ctxts = sim->GetAllExecCtxts();
195  all_exec_ctxts[0]->TearDownIterationAll(all_exec_ctxts);
196  });
197 
198  // update all substances (DiffusionGrids)
199  Timing::Time("diffusion", *diffusion_);
200 }

217 bool Scheduler::Restore(uint64_t* steps) {
218  if (backup_->RestoreEnabled() && restore_point_ > total_steps_ + *steps) {
219  total_steps_ += *steps;
220  // restore requested, but not last backup was not done during this call to
221  // Simualte. Therefore, we skip it.
222  return true;
223  } else if (backup_->RestoreEnabled() && restore_point_ > total_steps_ &&
224  restore_point_ < total_steps_ + *steps) {
225  // Restore
226  backup_->Restore();
227  *steps = total_steps_ + *steps - restore_point_;
228  total_steps_ = restore_point_;
229  }
230  return false;
231 }

260 std::vector<Operation*> Scheduler::GetScheduleOps() {
261  std::vector<Operation*> scheduled_ops;
262  scheduled_ops.reserve(operations_.size());
263  auto* param = Simulation::GetActive()->GetParam();
264  for (auto* op : operations_) {
265  // special condition for displacement
266  if (op->name_ == "displacement" &&
267  (!param->run_mechanical_interactions_ || !displacement_->UseCpu())) {
268  continue;
269  }
270  if (total_steps_ % op->frequency_ == 0) {
271  scheduled_ops.push_back(op);
272  }
273  }
274  return scheduled_ops;
275 }
