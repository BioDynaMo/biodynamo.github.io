83 void Scheduler::Simulate(uint64_t steps) {

134 void Scheduler::Execute(bool last_iteration) {
135  auto* sim = Simulation::GetActive();
136  auto* rm = sim->GetResourceManager();
137  auto* grid = sim->GetGrid();
138  auto* param = sim->GetParam();
139 
140  Timing::Time("Set up exec context", [&]() {
141  const auto& all_exec_ctxts = sim->GetAllExecCtxts();
142  all_exec_ctxts[0]->SetupIterationAll(all_exec_ctxts);
143  });
144 
145  Timing::Time("visualize", [&]() {
146  visualization_->Visualize(total_steps_, last_iteration);
147  });
148  Timing::Time("neighbors", [&]() { grid->UpdateGrid(); });
149 
150  // update all sim objects: run all CPU operations
151  const auto& scheduled_ops = GetScheduleOps();
152  rm->ApplyOnAllElementsParallelDynamic(
153  param->scheduling_batch_size_, [&](SimObject* so, SoHandle) {
154  sim->GetExecutionContext()->Execute(so, scheduled_ops);
155  });
156 
157  // update all sim objects: hardware accelerated operations
158  if (param->run_mechanical_interactions_ && !displacement_->UseCpu()) {
159  Timing::Time("displacement (GPU/FPGA)", *displacement_);
160  }
161 
162  // finish updating sim objects
163  Timing::Time("Tear down exec context", [&]() {
164  const auto& all_exec_ctxts = sim->GetAllExecCtxts();
165  all_exec_ctxts[0]->TearDownIterationAll(all_exec_ctxts);
166  });
167 
168  // update all substances (DiffusionGrids)
169  Timing::Time("diffusion", *diffusion_);
170 }

187 bool Scheduler::Restore(uint64_t* steps) {
188  if (backup_->RestoreEnabled() && restore_point_ > total_steps_ + *steps) {
189  total_steps_ += *steps;
190  // restore requested, but not last backup was not done during this call to
191  // Simualte. Therefore, we skip it.
192  return true;
193  } else if (backup_->RestoreEnabled() && restore_point_ > total_steps_ &&
194  restore_point_ < total_steps_ + *steps) {
195  // Restore
196  backup_->Restore();
197  *steps = total_steps_ + *steps - restore_point_;
198  total_steps_ = restore_point_;
199  }
200  return false;
201 }

237 std::vector<Operation> Scheduler::GetScheduleOps() {
238  std::vector<Operation> scheduled_ops;
239  scheduled_ops.reserve(operations_.size());
240  auto* param = Simulation::GetActive()->GetParam();
241  for (auto& op : operations_) {
242  // special condition for displacement
243  if (op.name_ == "displacement" &&
244  (!param->run_mechanical_interactions_ || !displacement_->UseCpu())) {
245  continue;
246  }
247  if (total_steps_ % op.frequency_ == 0) {
248  scheduled_ops.push_back(op);
249  }
250  }
251  return scheduled_ops;
252 }
