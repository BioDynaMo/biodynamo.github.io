46  void operator()() {
47  auto* sim = Simulation::GetActive();
48  auto* grid = sim->GetGrid();
49  auto* param = sim->GetParam();
50  auto* rm = sim->GetResourceManager();
51 
52  // Check the number of NUMA domains on the system. Currently only 1 is
53  // supported for GPU execution.
54  if (ThreadInfo::GetInstance()->GetNumaNodes() > 1) {
55  Log::Fatal(
56  "DisplacementOpCuda",
57  "\nThe GPU execution only supports systems with 1 NUMA domain.");
58  return;
59  }
60 
61  uint32_t num_objects = rm->GetNumSimObjects();
62 
63  // Cannot use Double3 here, because the `data()` function returns a const
64  // pointer to the underlying array, whereas the CUDA kernal will cast it to
65  // a void pointer. The conversion of `const double *` to `void *` is
66  // illegal.
67  std::vector<std::array<double, 3>> cell_movements(num_objects);
68  std::vector<Double3> cell_positions(num_objects);
69  std::vector<double> cell_diameters(num_objects);
70  std::vector<double> cell_adherence(num_objects);
71  std::vector<Double3> cell_tractor_force(num_objects);
72  std::vector<uint32_t> cell_boxid(num_objects);
73  std::vector<double> mass(num_objects);
74  std::vector<uint32_t> starts;
75  std::vector<uint16_t> lengths;
76  std::vector<uint32_t> successors(num_objects);
77  uint32_t box_length;
78  std::array<uint32_t, 3> num_boxes_axis;
79  std::array<int32_t, 3> grid_dimensions;
80  double squared_radius =
81  grid->GetLargestObjectSize() * grid->GetLargestObjectSize();
82 
83  bool is_non_spherical_object = false;
84 
85  rm->ApplyOnAllElements([&](SimObject* so, SoHandle soh) {
86  // Check if there are any non-spherical objects in our simulation, because
87  // GPU accelerations currently supports only sphere-sphere interactions
88  IsNonSphericalObjectPresent(so, &is_non_spherical_object);
89  if (is_non_spherical_object) {
90  Log::Fatal("DisplacementOpCuda",
91  "\nWe detected a non-spherical object during the GPU "
92  "execution. This is currently not supported.");
93  return;
94  }
95  auto* cell = bdm_static_cast<Cell*>(so);
96  auto idx = soh.GetElementIdx();
97  mass[idx] = cell->GetMass();
98  cell_diameters[idx] = cell->GetDiameter();
99  cell_adherence[idx] = cell->GetAdherence();
100  cell_tractor_force[idx] = cell->GetTractorForce();
101  cell_positions[idx] = cell->GetPosition();
102  cell_boxid[idx] = cell->GetBoxIdx();
103  });
104 
105  uint16_t numa_node = 0; // GPU code only supports 1 NUMA domain currently
106  for (size_t i = 0; i < grid->successors_.size(numa_node); i++) {
107  auto sh = SoHandle(numa_node, i);
108  successors[i] = grid->successors_[sh].GetElementIdx();
109  }
110 
111  starts.resize(grid->boxes_.size());
112  lengths.resize(grid->boxes_.size());
113  size_t i = 0;
114  for (auto& box : grid->boxes_) {
115  starts[i] = box.start_.GetElementIdx();
116  lengths[i] = box.length_;
117  i++;
118  }
119  grid->GetGridInfo(&box_length, &num_boxes_axis, &grid_dimensions);
120 
121  // If this is the first time we perform physics on GPU using CUDA
122  if (cdo_ == nullptr) {
123  // Allocate 25% more memory so we don't need to reallocate GPU memory
124  // for every (small) change
125  uint32_t new_num_objects = static_cast<uint32_t>(1.25 * num_objects);
126  uint32_t new_num_boxes = static_cast<uint32_t>(1.25 * starts.size());
127 
128  // Store these extended buffer sizes for future reference
129  num_objects_ = new_num_objects;
130  num_boxes_ = new_num_boxes;
131 
132  // Allocate required GPU memory
133  cdo_ = new DisplacementOpCudaKernel(new_num_objects, new_num_boxes);
134  } else {
135  // If the number of simulation objects increased
136  if (num_objects >= num_objects_) {
137  Log::Info("DisplacementOpCuda",
138  "\nThe number of cells increased signficantly (from ",
139  num_objects_, " to ", num_objects,
140  "), so we allocate bigger GPU buffers\n");
141  uint32_t new_num_objects = static_cast<uint32_t>(1.25 * num_objects);
142  num_objects_ = new_num_objects;
143  cdo_->ResizeCellBuffers(new_num_objects);
144  }
145 
146  // If the neighbor grid size increased
147  if (starts.size() >= num_boxes_) {
148  Log::Info("DisplacementOpCuda",
149  "\nThe number of boxes increased signficantly (from ",
150  num_boxes_, " to ", "), so we allocate bigger GPU buffers\n");
151  uint32_t new_num_boxes = static_cast<uint32_t>(1.25 * starts.size());
152  num_boxes_ = new_num_boxes;
153  cdo_->ResizeGridBuffers(new_num_boxes);
154  }
155  }
156 
157  cdo_->LaunchDisplacementKernel(
158  cell_positions.data()->data(), cell_diameters.data(),
159  cell_tractor_force.data()->data(), cell_adherence.data(),
160  cell_boxid.data(), mass.data(), &(param->simulation_time_step_),
161  &(param->simulation_max_displacement_), &squared_radius, &num_objects,
162  starts.data(), lengths.data(), successors.data(), &box_length,
163  num_boxes_axis.data(), grid_dimensions.data(),
164  cell_movements.data()->data());
165 
166  // set new positions after all updates have been calculated
167  // otherwise some cells would see neighbors with already updated positions
168  // which would lead to inconsistencies
169  rm->ApplyOnAllElements([&](SimObject* so, SoHandle soh) {
170  auto* cell = dynamic_cast<Cell*>(so);
171  auto idx = soh.GetElementIdx();
172  Double3 new_pos;
173  new_pos[0] = cell_movements[idx][0];
174  new_pos[1] = cell_movements[idx][1];
175  new_pos[2] = cell_movements[idx][2];
176  cell->UpdatePosition(new_pos);
177  if (param->bound_space_) {
178  ApplyBoundingBox(so, param->min_bound_, param->max_bound_);
179  }
180  });
181  }
182 
183  private:
184  DisplacementOpCudaKernel* cdo_ = nullptr;
185  uint32_t num_boxes_ = 0;
186  uint32_t num_objects_ = 0;
187 };
