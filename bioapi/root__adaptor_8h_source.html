124  void AddBranch(const SimObject *so, TGeoVolume *container) {
125  switch (so->GetShape()) {

142  void AddSphere(const SimObject *so, TGeoVolume *container) {
143  std::string name = so->GetTypeName() + std::to_string(so->GetUid());
144  auto radius = so->GetDiameter() / 2;
145  auto massLocation = so->GetPosition();
146  auto x = massLocation[0];
147  auto y = massLocation[1];
148  auto z = massLocation[2];

150  auto volume = gGeoManager->MakeSphere(name.c_str(), med_solid_, 0, radius);
151  volume->SetLineColor(kBlue);

156  void AddCylinder(const SimObject *so, TGeoVolume *container) {
157  if (auto neurite = dynamic_cast<const NeuriteElement *>(so)) {
158  std::string name = so->GetTypeName() + std::to_string(so->GetUid());
159  auto radius = neurite->GetDiameter() / 2;
160  auto half_length = neurite->GetLength() / 2;
161  auto massLocation = neurite->GetPosition();
162  auto x = massLocation[0];
163  auto y = massLocation[1];
164  auto z = massLocation[2];
165  auto trans = new TGeoTranslation(x, y, z);
166 
167  // The initial orientation of the cylinder
168  TVector3 orig(0, 0, 1);
169 
170  // The vector that we want to orient the cylinder to (symmetry axis
171  // aligns with this vector)
172  Double3 d = neurite->GetSpringAxis();
173  TVector3 dir(d[0], d[1], d[2]);
174  dir = dir.Unit();
175 
176  // Compute the Axis-Angle rotation representation
177  auto dot_product = dir.Dot(orig);
178  auto angle = std::acos(dot_product);
179  // TODO: make sure it is `z x dir, and not `dir x z`
180  TVector3 n = dir.Cross(orig);
181  n = n.Unit();
182  auto axis = AxisAngle::AxisVector(n[0], n[1], n[2]);
183  AxisAngle aa(axis, angle);
184  EulerAngles ea(aa);
185  TGeoRotation *rot = new TGeoRotation("rot", Math::ToDegree(ea.Phi()),
186  Math::ToDegree(ea.Theta()),
187  Math::ToDegree(ea.Psi()));
188  TGeoCombiTrans *transrot = new TGeoCombiTrans(*trans, *rot);
189  auto volume = gGeoManager->MakeTube(name.c_str(), med_solid_, 0, radius,
190  half_length);
191  volume->SetLineColor(kBlue);
192  container->AddNode(volume, container->GetNdaughters(), transrot);
193  } else {
194  Log::Error("RootAdaptor", "This is not a cylindrical shaped object!");
195  }
196  }
