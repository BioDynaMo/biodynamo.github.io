150 inline void GenerateSimulationInfoJson(
151  const std::unordered_map<std::string, VtkSoGrid*>& vtk_so_grids,
152  const std::unordered_map<std::string, VtkDiffusionGrid*>& vtk_dgrids) {
153  auto* sim = Simulation::GetActive();
154  auto* param = sim->GetParam();
155  // simulation objects
156  std::stringstream sim_objects;
157  uint64_t num_sim_objects = param->visualize_sim_objects_.size();
158  uint64_t counter = 0;
159  for (const auto& entry : param->visualize_sim_objects_) {
160  std::string so_name = entry.first;
161 
162  auto search = vtk_so_grids.find(so_name);
163  if (search == vtk_so_grids.end()) {
164  continue;
165  }
166  auto* so_grid = search->second;
167  // user wanted to export this simulation object type, but not a single
168  // instance has been created during the entire simulation
169  if (!so_grid->initialized_) {
170  Log::Warning("Visualize Simulation Objects",
171  "You are trying to visualize simulation object type ",
172  so_name,
173  ", but not a single instance has been created during the "
174  "entire simulation. Please make sure the names in the "
175  "configuration file match the ones in the simulation.");
176  num_sim_objects--;
177  continue;
178  }
179 
180  sim_objects << " {" << std::endl
181  << " \"name\":\"" << so_name << "\"," << std::endl;
182  if (so_grid->shape_ == Shape::kSphere) {
183  sim_objects << " \"glyph\":\"Glyph\"," << std::endl
184  << " \"shape\":\"Sphere\"," << std::endl
185  << " \"scaling_attribute\":\"diameter_\"" << std::endl;
186  } else if (so_grid->shape_ == kCylinder) {
187  sim_objects << " \"glyph\":\"BDMGlyph\"," << std::endl
188  << " \"shape\":\"Cylinder\"," << std::endl
189  << " \"x_scaling_attribute\":\"diameter_\"," << std::endl
190  << " \"y_scaling_attribute\":\"actual_length_\","
191  << std::endl
192  << " \"z_scaling_attribute\":\"diameter_\"," << std::endl
193  << " \"Vectors\":\"spring_axis_\"," << std::endl
194  << " \"MassLocation\":\"mass_location_\"" << std::endl;
195  }
196  sim_objects << " }";
197  if (counter != num_sim_objects - 1) {
198  sim_objects << ",";
199  }
200  sim_objects << std::endl;
201  counter++;
202  }
203 
204  // extracellular substances
205  std::stringstream substances;
206  uint64_t num_substances = param->visualize_diffusion_.size();
207  for (uint64_t i = 0; i < num_substances; i++) {
208  auto& name = param->visualize_diffusion_[i].name_;
209 
210  auto search = vtk_dgrids.find(name);
211  if (search == vtk_dgrids.end()) {
212  continue;
213  }
214  auto* dgrid = search->second;
215  // user wanted to export this substance, but it did not exist during
216  // the entire simulation
217  if (!dgrid->used_) {
218  Log::Warning(
219  "Visualize Diffusion Grids",
220  "You are trying to visualize diffusion grid ", name,
221  ", but it has not been created during the entire simulation. "
222  "Please make sure the names in the "
223  "configuration file match the ones in the simulation.");
224  num_substances--;
225  continue;
226  }
227  substances << " { \"name\":\"" << name << "\", ";
228  std::string has_gradient =
229  param->visualize_diffusion_[i].gradient_ ? "true" : "false";
230  substances << "\"has_gradient\":\"" << has_gradient << "\" }";
231 
232  if (i != num_substances - 1) {
233  substances << "," << std::endl;
234  }
235  }
236 
237  // write to file
238  std::ofstream ofstr;
239  ofstr.open(Concat(sim->GetOutputDir(), "/", kSimulationInfoJson));
240  ofstr << "{" << std::endl
241  << " \"simulation\": {" << std::endl
242  << " \"name\":\"" << sim->GetUniqueName() << "\"," << std::endl
243  << " \"result_dir\":\"" << sim->GetOutputDir() << "\"" << std::endl
244  << " }," << std::endl
245  << " \"sim_objects\": [" << std::endl
246  << sim_objects.str() << " ]," << std::endl
247  << " \"extracellular_substances\": [" << std::endl
248  << substances.str() << std::endl
249  << " ]" << std::endl
250  << "}" << std::endl;
251  ofstr.close();
252 }
