91 void ParaviewAdaptor::Visualize() {
92  if (!initialized_) {
93  Initialize();

97  auto* sim = Simulation::GetActive();

99  uint64_t total_steps = sim->GetScheduler()->GetSimulatedSteps();
100  if (total_steps % param->visualization_export_interval_ != 0) {
101  return;
102  }
103 
104  if (param->live_visualization_) {
105  double time = param->simulation_time_step_ * total_steps;
106  LiveVisualization(time, total_steps);
107  }
108  if (param->export_visualization_) {
109  double time = param->simulation_time_step_ * total_steps;
110  ExportVisualization(time, total_steps);
111  }

114 void ParaviewAdaptor::Initialize() {
115  auto* sim = Simulation::GetActive();
116  auto* param = sim->GetParam();
117 
118  exclusive_export_viz_ =
119  param->export_visualization_ && !param->live_visualization_;
120  if (param->live_visualization_ || param->export_visualization_) {
121  if (impl_->g_processor_ == nullptr) {
122  impl_->g_processor_ = vtkCPProcessor::New();
123  impl_->g_processor_->Initialize();
124  }
125 
126  if (param->live_visualization_ &&
127  impl_->g_processor_->GetNumberOfPipelines() != 0) {
128  Log::Fatal("ParaviewAdaptor",
129  "Live visualization does not support multiple "
130  "simulations. Turning off live visualization for ",
131  sim->GetUniqueName());
132  } else if (param->python_paraview_pipeline_) {
133  vtkNew<vtkCPPythonScriptPipeline> pipeline;
134  std::string python_script =
135  std::string(std::getenv("BDM_SRC_DIR")) +
136  std::string("/visualization/paraview/simple_pipeline.py");
137  pipeline->Initialize(python_script.c_str());
138  impl_->g_processor_->AddPipeline(pipeline.GetPointer());
139  } else if (!exclusive_export_viz_) {
140  impl_->pipeline_ = new InSituPipeline();
141  impl_->g_processor_->AddPipeline(impl_->pipeline_);
142  }
143 
144  if (impl_->data_description_ == nullptr) {
145  impl_->data_description_ = vtkCPDataDescription::New();
146  } else {
147  impl_->data_description_->Delete();
148  impl_->data_description_ = vtkCPDataDescription::New();
149  }
150  impl_->data_description_->SetTimeData(0, 0);
151 
152  auto* param = Simulation::GetActive()->GetParam();
153  for (auto& pair : param->visualize_sim_objects_) {
154  impl_->vtk_so_grids_[pair.first.c_str()] =
155  new VtkSoGrid(pair.first.c_str(), impl_->data_description_);
156  }
157  for (auto& entry : param->visualize_diffusion_) {
158  impl_->vtk_dgrids_[entry.name_] =
159  new VtkDiffusionGrid(entry.name_, impl_->data_description_);
160  }
161  }
162 }

209 void ParaviewAdaptor::ProcessSimObject(const SimObject* so) {
210  auto* param = Simulation::GetActive()->GetParam();
211  auto so_name = so->GetTypeName();
212 
213  if (param->visualize_sim_objects_.find(so_name) !=
214  param->visualize_sim_objects_.end()) {
215  // assert(impl_->vtk_so_grids_.find(so->GetTypeName()) !=
216  // impl_->vtk_so_grids_.end() &&
217  // Concat("VtkSoGrid for ", so->GetTypeName(), " not found!");
218  auto* vsg = impl_->vtk_so_grids_[so->GetTypeName()];
219  if (!vsg->initialized_) {
220  vsg->Init(so);
221  }
222 
223  // If we segfault at here it probably means that there is a problem
224  // with the pipeline (either the C++ pipeline or Python pipeline)
225  // We do not need to RequestDataDescription in Export Mode, because
226  // we do not make use of ParaView Catalyst CoProcessing capabilities
227  if (exclusive_export_viz_ ||
228  (impl_->g_processor_->RequestDataDescription(
229  impl_->data_description_)) != 0) {
230  ParaviewSoVisitor visitor(vsg);
231  so->ForEachDataMemberIn(vsg->vis_data_members_, &visitor);
232  }
233  }

254 void ParaviewAdaptor::ProcessDiffusionGrid(const DiffusionGrid* grid) {
255  auto* param = Simulation::GetActive()->GetParam();
256  auto name = grid->GetSubstanceName();
257 
258  // get visualization config
259  const Param::VisualizeDiffusion* vd = nullptr;
260  for (auto& entry : param->visualize_diffusion_) {
261  if (entry.name_ == name) {
262  vd = &entry;
263  }
264  }
265 
266  if (vd != nullptr) {
267  auto* vdg = impl_->vtk_dgrids_[grid->GetSubstanceName()];
268  if (!vdg->used_) {
269  vdg->Init();
270  }
271 
272  // If we segfault at here it probably means that there is a problem
273  // with the pipeline (either the C++ pipeline or Python pipeline)
274  // We do not need to RequestDataDescription in Export Mode, because
275  // we do not make use of Catalyst CoProcessing capabilities
276  if (exclusive_export_viz_ ||
277  (impl_->g_processor_->RequestDataDescription(
278  impl_->data_description_)) != 0) {
279  auto num_boxes = grid->GetNumBoxesArray();
280  auto grid_dimensions = grid->GetDimensions();
281  auto box_length = grid->GetBoxLength();
282  auto total_boxes = grid->GetNumBoxes();
283 
284  double origin_x = grid_dimensions[0];
285  double origin_y = grid_dimensions[2];
286  double origin_z = grid_dimensions[4];
287  vdg->data_->SetOrigin(origin_x, origin_y, origin_z);
288  vdg->data_->SetDimensions(num_boxes[0], num_boxes[1], num_boxes[2]);
289  vdg->data_->SetSpacing(box_length, box_length, box_length);
290 
291  if (vdg->concentration_) {
292  auto* co_ptr = const_cast<double*>(grid->GetAllConcentrations());
293  vdg->concentration_->SetArray(co_ptr,
294  static_cast<vtkIdType>(total_boxes), 1);
295  }
296  if (vdg->gradient_) {
297  auto gr_ptr = const_cast<double*>(grid->GetAllGradients());
298  vdg->gradient_->SetArray(gr_ptr,
299  static_cast<vtkIdType>(total_boxes * 3), 1);
300  }
301  }
302  }
