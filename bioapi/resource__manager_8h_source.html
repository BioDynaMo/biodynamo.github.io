375  void Remove(SoUid uid) {
376  // remove from map
377  auto it = uid_soh_map_.find(uid);
378  if (it != uid_soh_map_.end()) {
379  SoHandle soh = it->second;
380  uid_soh_map_.unsafe_erase(it);
381  // remove from vector
382  auto& numa_sos = sim_objects_[soh.GetNumaNode()];
383  if (soh.GetElementIdx() == numa_sos.size() - 1) {
384  delete numa_sos.back();
385  numa_sos.pop_back();
386  } else {
387  // swap
388  delete numa_sos[soh.GetElementIdx()];
389  auto* reordered = numa_sos.back();
390  numa_sos[soh.GetElementIdx()] = reordered;
391  numa_sos.pop_back();
392  uid_soh_map_[reordered->GetUid()] = soh;
393  }
394  }
395  }
