20 void ResourceManager::ApplyOnAllElementsParallel(
21  Functor<void, SimObject*, SoHandle>& function) {
22 #pragma omp parallel
23  {
24  auto tid = omp_get_thread_num();
25  auto nid = thread_info_->GetNumaNode(tid);
26  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(nid);
27  auto& numa_sos = sim_objects_[nid];
28  assert(thread_info_->GetNumaNode(tid) == numa_node_of_cpu(sched_getcpu()));
29 
30  // use static scheduling for now
31  auto correction = numa_sos.size() % threads_in_numa == 0 ? 0 : 1;
32  auto chunk = numa_sos.size() / threads_in_numa + correction;
33  auto start = thread_info_->GetNumaThreadId(tid) * chunk;
34  auto end = std::min(numa_sos.size(), start + chunk);
35 
36  for (uint64_t i = start; i < end; ++i) {
37  function(numa_sos[i], SoHandle(nid, i));
38  }
39  }
40 }

56 void ResourceManager::ApplyOnAllElementsParallelDynamic(
57  uint64_t chunk, Functor<void, SimObject*, SoHandle>& function) {
58  // adapt chunk size
59  auto num_so = GetNumSimObjects();
60  uint64_t factor = (num_so / thread_info_->GetMaxThreads()) / chunk;
61  chunk = (num_so / thread_info_->GetMaxThreads()) / (factor + 1);
62  chunk = chunk >= 1 ? chunk : 1;
63 
64  // use dynamic scheduling
65  // Unfortunately openmp's built in functionality can't be used, since
66  // threads belong to different numa domains and thus operate on
67  // different containers
68  auto numa_nodes = thread_info_->GetNumaNodes();
69  auto max_threads = omp_get_max_threads();
70  std::vector<uint64_t> num_chunks_per_numa(numa_nodes);
71  for (int n = 0; n < numa_nodes; n++) {
72  auto correction = sim_objects_[n].size() % chunk == 0 ? 0 : 1;
73  num_chunks_per_numa[n] = sim_objects_[n].size() / chunk + correction;
74  }
75 
76  std::vector<std::atomic<uint64_t>*> counters(max_threads, nullptr);
77  std::vector<uint64_t> max_counters(max_threads);
78  for (int thread_cnt = 0; thread_cnt < max_threads; thread_cnt++) {
79  uint64_t current_nid = thread_info_->GetNumaNode(thread_cnt);
80 
81  auto correction =
82  num_chunks_per_numa[current_nid] %
83  thread_info_->GetThreadsInNumaNode(current_nid) ==
84  0
85  ? 0
86  : 1;
87  uint64_t num_chunks_per_thread =
88  num_chunks_per_numa[current_nid] /
89  thread_info_->GetThreadsInNumaNode(current_nid) +
90  correction;
91  auto start =
92  num_chunks_per_thread * thread_info_->GetNumaThreadId(thread_cnt);
93  auto end = std::min(num_chunks_per_numa[current_nid],
94  start + num_chunks_per_thread);
95 
96  counters[thread_cnt] = new std::atomic<uint64_t>(start);
97  max_counters[thread_cnt] = end;
98  }
99 
100 #pragma omp parallel
101  {
102  auto tid = omp_get_thread_num();
103  auto nid = thread_info_->GetNumaNode(tid);
104 
105  // thread private variables (compilation error with
106  // firstprivate(chunk, numa_node_) with some openmp versions clause)
107  auto p_numa_nodes = thread_info_->GetNumaNodes();
108  auto p_max_threads = omp_get_max_threads();
109  auto p_chunk = chunk;
110  assert(thread_info_->GetNumaNode(tid) == numa_node_of_cpu(sched_getcpu()));
111 
112  // dynamic scheduling
113  uint64_t start = 0;
114  uint64_t end = 0;
115 
116  // this loop implements work stealing from other NUMA nodes if there
117  // are imbalances. Each thread starts with its NUMA domain. Once, it
118  // is finished the thread looks for tasks on other domains
119  for (int n = 0; n < p_numa_nodes; n++) {
120  int current_nid = (nid + n) % p_numa_nodes;
121  for (int thread_cnt = 0; thread_cnt < p_max_threads; thread_cnt++) {
122  uint64_t current_tid = (tid + thread_cnt) % p_max_threads;
123  if (current_nid != thread_info_->GetNumaNode(current_tid)) {
124  continue;
125  }
126 
127  auto& numa_sos = sim_objects_[current_nid];
128  uint64_t old_count = (*(counters[current_tid]))++;
129  while (old_count < max_counters[current_tid]) {
130  start = old_count * p_chunk;
131  end =
132  std::min(static_cast<uint64_t>(numa_sos.size()), start + p_chunk);
133 
134  for (uint64_t i = start; i < end; ++i) {
135  function(numa_sos[i], SoHandle(current_nid, i));
136  }
137 
138  old_count = (*(counters[current_tid]))++;
139  }
140  } // work stealing loop numa_nodes_
141  } // work stealing loop threads
142  }
143 
144  for (auto* counter : counters) {
145  delete counter;
146  }
147 }

166 void ResourceManager::SortAndBalanceNumaNodes() {
167  // balance simulation objects per numa node according to the number of
168  // threads associated with each numa domain
169  auto numa_nodes = thread_info_->GetNumaNodes();
170  std::vector<uint64_t> so_per_numa(numa_nodes);
171  uint64_t cummulative = 0;
172  auto max_threads = thread_info_->GetMaxThreads();
173  for (int n = 1; n < numa_nodes; ++n) {
174  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(n);
175  uint64_t num_so = GetNumSimObjects() * threads_in_numa / max_threads;
176  so_per_numa[n] = num_so;
177  cummulative += num_so;
178  }
179  so_per_numa[0] = GetNumSimObjects() - cummulative;
180 
181  // using first touch policy - page will be allocated to the numa domain of
182  // the thread that accesses it first.
183  // alternative, use numa_alloc_onnode.
184  int ret = numa_run_on_node(0);
185  if (ret != 0) {
186  Log::Fatal("ResourceManager", "Run on numa node failed. Return code: ",
187  ret);
188  }
189 
190  // new data structure for rearranged SimObject*
191  decltype(sim_objects_) so_rearranged;
192  so_rearranged.resize(numa_nodes);
193 
194  // numa node -> vector of SoHandles
195  std::vector<std::vector<SoHandle>> sorted_so_handles;
196  sorted_so_handles.resize(numa_nodes);
197 #pragma omp parallel for
198  for (int n = 0; n < numa_nodes; ++n) {
199  if (thread_info_->GetMyNumaNode() == n &&
200  thread_info_->GetMyNumaThreadId() == 0) {
201  sorted_so_handles[n].reserve(so_per_numa[n]);
202  }
203  }
204 
205  uint64_t cnt = 0;
206  uint64_t current_numa = 0;
207 
208  auto rearrange = [&](const SoHandle& handle) {
209  if (cnt == so_per_numa[current_numa]) {
210  cnt = 0;
211  current_numa++;
212  }
213 
214  sorted_so_handles[current_numa].push_back(handle);
215  cnt++;
216  };
217 
218  auto* grid = Simulation::GetActive()->GetGrid();
219  grid->IterateZOrder(rearrange);
220 
221  auto* param = Simulation::GetActive()->GetParam();
222  const bool minimize_memory = param->minimize_memory_while_rebalancing_;
223 
224 // create new objects
225 #pragma omp parallel
226  {
227  auto tid = thread_info_->GetMyThreadId();
228  auto nid = thread_info_->GetNumaNode(tid);
229 
230  for (int n = 0; n < numa_nodes; n++) {
231  if (nid != n) {
232  continue;
233  }
234  auto& dest = so_rearranged[n];
235 
236  if (thread_info_->GetNumaThreadId(tid) == 0) {
237  dest.resize(sorted_so_handles[n].size());
238  }
239 
240 #pragma omp barrier
241 
242  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(nid);
243  auto& sohandles = sorted_so_handles[n];
244  assert(thread_info_->GetNumaNode(tid) ==
245  numa_node_of_cpu(sched_getcpu()));
246 
247  // use static scheduling
248  auto correction = sohandles.size() % threads_in_numa == 0 ? 0 : 1;
249  auto chunk = sohandles.size() / threads_in_numa + correction;
250  auto start = thread_info_->GetNumaThreadId(tid) * chunk;
251  auto end = std::min(sohandles.size(), start + chunk);
252 
253  for (uint64_t e = start; e < end; e++) {
254  auto& handle = sohandles[e];
255  auto* so = sim_objects_[handle.GetNumaNode()][handle.GetElementIdx()];
256  dest[e] = so->GetCopy();
257  if (minimize_memory) {
258  delete so;
259  }
260  }
261  }
262  }
263 
264  // delete old objects. This approach has a high chance that a thread
265  // in the right numa node will delete the object, thus minimizing thread
266  // synchronization overheads. The bdm memory allocator does not have this
267  // issue.
268  if (!minimize_memory) {
269  DeleteSimObjectsFunctor delete_functor;
270  ApplyOnAllElementsParallel(delete_functor);
271  }
272 
273  for (int n = 0; n < numa_nodes; n++) {
274  sim_objects_[n].swap(so_rearranged[n]);
275  }
276 
277  // update uid_soh_map_
278  UpdateUidSoHMapFunctor functor(uid_soh_map_);
279  ApplyOnAllElementsParallel(functor);
280 
281  if (Simulation::GetActive()->GetParam()->debug_numa_) {
282  DebugNuma();
283  }
284 }
