20 void ResourceManager::ApplyOnAllElementsParallel(
21  const std::function<void(SimObject*)>& function) {
22 #pragma omp parallel
23  {
24  auto tid = omp_get_thread_num();
25  auto nid = thread_info_->GetNumaNode(tid);
26  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(nid);
27  auto& numa_sos = sim_objects_[nid];
28  assert(thread_info_->GetNumaNode(tid) == numa_node_of_cpu(sched_getcpu()));
29 
30  // use static scheduling for now
31  auto correction = numa_sos.size() % threads_in_numa == 0 ? 0 : 1;
32  auto chunk = numa_sos.size() / threads_in_numa + correction;
33  auto start = thread_info_->GetNumaThreadId(tid) * chunk;
34  auto end = std::min(numa_sos.size(), start + chunk);
35 
36  for (uint64_t i = start; i < end; ++i) {
37  function(numa_sos[i]);
38  }
39  }
40 }

42 void ResourceManager::ApplyOnAllElementsParallelDynamic(
43  uint64_t chunk, const std::function<void(SimObject*, SoHandle)>& function) {
44  // adapt chunk size
45  auto num_so = GetNumSimObjects();
46  uint64_t factor = (num_so / thread_info_->GetMaxThreads()) / chunk;
47  chunk = (num_so / thread_info_->GetMaxThreads()) / (factor + 1);
48  chunk = chunk >= 1 ? chunk : 1;
49 
50  // use dynamic scheduling
51  // Unfortunately openmp's built in functionality can't be used, since
52  // threads belong to different numa domains and thus operate on
53  // different containers
54  auto numa_nodes = thread_info_->GetNumaNodes();
55  auto max_threads = omp_get_max_threads();
56  std::vector<uint64_t> num_chunks_per_numa(numa_nodes);
57  for (int n = 0; n < numa_nodes; n++) {
58  auto correction = sim_objects_[n].size() % chunk == 0 ? 0 : 1;
59  num_chunks_per_numa[n] = sim_objects_[n].size() / chunk + correction;
60  }
61 
62  std::vector<std::atomic<uint64_t>*> counters(max_threads, nullptr);
63  std::vector<uint64_t> max_counters(max_threads);
64  for (int thread_cnt = 0; thread_cnt < max_threads; thread_cnt++) {
65  uint64_t current_nid = thread_info_->GetNumaNode(thread_cnt);
66 
67  auto correction =
68  num_chunks_per_numa[current_nid] %
69  thread_info_->GetThreadsInNumaNode(current_nid) ==
70  0
71  ? 0
72  : 1;
73  uint64_t num_chunks_per_thread =
74  num_chunks_per_numa[current_nid] /
75  thread_info_->GetThreadsInNumaNode(current_nid) +
76  correction;
77  auto start =
78  num_chunks_per_thread * thread_info_->GetNumaThreadId(thread_cnt);
79  auto end = std::min(num_chunks_per_numa[current_nid],
80  start + num_chunks_per_thread);
81 
82  counters[thread_cnt] = new std::atomic<uint64_t>(start);
83  max_counters[thread_cnt] = end;
84  }
85 
86 #pragma omp parallel
87  {
88  auto tid = omp_get_thread_num();
89  auto nid = thread_info_->GetNumaNode(tid);
90 
91  // thread private variables (compilation error with
92  // firstprivate(chunk, numa_node_) with some openmp versions clause)
93  auto p_numa_nodes = thread_info_->GetNumaNodes();
94  auto p_max_threads = omp_get_max_threads();
95  auto p_chunk = chunk;
96  assert(thread_info_->GetNumaNode(tid) == numa_node_of_cpu(sched_getcpu()));
97 
98  // dynamic scheduling
99  uint64_t start = 0;
100  uint64_t end = 0;
101 
102  // this loop implements work stealing from other NUMA nodes if there
103  // are imbalances. Each thread starts with its NUMA domain. Once, it
104  // is finished the thread looks for tasks on other domains
105  for (int n = 0; n < p_numa_nodes; n++) {
106  int current_nid = (nid + n) % p_numa_nodes;
107  for (int thread_cnt = 0; thread_cnt < p_max_threads; thread_cnt++) {
108  uint64_t current_tid = (tid + thread_cnt) % p_max_threads;
109  if (current_nid != thread_info_->GetNumaNode(current_tid)) {
110  continue;
111  }
112 
113  auto& numa_sos = sim_objects_[current_nid];
114  uint64_t old_count = (*(counters[current_tid]))++;
115  while (old_count < max_counters[current_tid]) {
116  start = old_count * p_chunk;
117  end =
118  std::min(static_cast<uint64_t>(numa_sos.size()), start + p_chunk);
119 
120  for (uint64_t i = start; i < end; ++i) {
121  function(numa_sos[i], SoHandle(current_nid, i));
122  }
123 
124  old_count = (*(counters[current_tid]))++;
125  }
126  } // work stealing loop numa_nodes_
127  } // work stealing loop threads
128  }
129 
130  for (auto* counter : counters) {
131  delete counter;
132  }
133 }

135 void ResourceManager::SortAndBalanceNumaNodes() {
136  // balance simulation objects per numa node according to the number of
137  // threads associated with each numa domain
138  auto numa_nodes = thread_info_->GetNumaNodes();
139  std::vector<uint64_t> so_per_numa(numa_nodes);
140  uint64_t cummulative = 0;
141  auto max_threads = thread_info_->GetMaxThreads();
142  for (int n = 1; n < numa_nodes; ++n) {
143  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(n);
144  uint64_t num_so = GetNumSimObjects() * threads_in_numa / max_threads;
145  so_per_numa[n] = num_so;
146  cummulative += num_so;
147  }
148  so_per_numa[0] = GetNumSimObjects() - cummulative;
149 
150  // using first touch policy - page will be allocated to the numa domain of
151  // the thread that accesses it first.
152  // alternative, use numa_alloc_onnode.
153  int ret = numa_run_on_node(0);
154  if (ret != 0) {
155  Log::Fatal("ResourceManager", "Run on numa node failed. Return code: ",
156  ret);
157  }
158 
159  // new data structure for rearranged SimObject*
160  decltype(sim_objects_) so_rearranged;
161  so_rearranged.resize(numa_nodes);
162 
163  // numa node -> vector of SoHandles
164  std::vector<std::vector<SoHandle>> sorted_so_handles;
165  sorted_so_handles.resize(numa_nodes);
166 
167  uint64_t cnt = 0;
168  uint64_t current_numa = 0;
169 
170  auto rearrange = [&](const SoHandle& handle) {
171  if (cnt == so_per_numa[current_numa]) {
172  cnt = 0;
173  current_numa++;
174  }
175 
176  sorted_so_handles[current_numa].push_back(handle);
177  cnt++;
178  };
179 
180  auto* grid = Simulation::GetActive()->GetGrid();
181  grid->IterateZOrder(rearrange);
182 
183  for (int n = 0; n < numa_nodes; n++) {
184  auto& dest = so_rearranged[n];
185  std::atomic<bool> resized(false);
186 #pragma omp parallel
187  {
188  auto tid = omp_get_thread_num();
189  auto nid = thread_info_->GetNumaNode(tid);
190  if (nid == n) {
191  auto old = std::atomic_exchange(&resized, true);
192  if (!old) {
193  dest.resize(sorted_so_handles[n].size());
194  }
195  }
196  }
197 #pragma omp parallel
198  {
199  auto tid = omp_get_thread_num();
200  auto nid = thread_info_->GetNumaNode(tid);
201 
202  if (nid == n) {
203  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(nid);
204  auto& sohandles = sorted_so_handles[n];
205  assert(thread_info_->GetNumaNode(tid) ==
206  numa_node_of_cpu(sched_getcpu()));
207 
208  // use static scheduling
209  auto correction = sohandles.size() % threads_in_numa == 0 ? 0 : 1;
210  auto chunk = sohandles.size() / threads_in_numa + correction;
211  auto start = thread_info_->GetNumaThreadId(tid) * chunk;
212  auto end = std::min(sohandles.size(), start + chunk);
213 
214  for (uint64_t e = start; e < end; e++) {
215  auto& handle = sohandles[e];
216  auto* so = sim_objects_[handle.GetNumaNode()][handle.GetElementIdx()];
217  dest[e] = so->GetCopy();
218  delete so;
219  }
220  }
221  }
222  }
223 
224  for (int n = 0; n < numa_nodes; n++) {
225  sim_objects_[n].swap(so_rearranged[n]);
226  }
227 
228  // update uid_soh_map_
229  ApplyOnAllElements([this](SimObject* so, SoHandle soh) {
230  this->uid_soh_map_[so->GetUid()] = soh;
231  });
232 
233  if (Simulation::GetActive()->GetParam()->debug_numa_) {
234  DebugNuma();
235  }
236 }
