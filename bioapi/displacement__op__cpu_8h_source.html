40  auto* scheduler = sim->GetScheduler();

42 
43  if (!sim_object->RunDisplacement()) {

45  }
46 
47  // update search radius and delta_time_ at beginning of each iteration
48  auto current_iteration = scheduler->GetSimulatedSteps();
49  if (last_iteration_ != current_iteration) {
50  last_iteration_ = current_iteration;
51 
52  auto* grid = sim->GetGrid();
53  auto search_radius = grid->GetLargestObjectSize();
54  squared_radius_ = search_radius * search_radius;
55  auto current_time =
56  (current_iteration + 1) * param->simulation_time_step_;
57  delta_time_ = current_time - last_time_run_;
58  last_time_run_ = current_time;
59  }
60 
61  const auto& displacement =
62  sim_object->CalculateDisplacement(squared_radius_, delta_time_);
63  sim_object->ApplyDisplacement(displacement);
64  if (param->bound_space_) {
65  ApplyBoundingBox(sim_object, param->min_bound_, param->max_bound_);
66  }
67  }
