45  void operator()(SimObject* sim_object) {
46  auto* sim = Simulation::GetActive();
47  auto* scheduler = sim->GetScheduler();
48  auto* param = sim->GetParam();
49 
50  if (!sim_object->RunDisplacement()) {
51  return;
52  }
53 
54  // update search radius and delta_time_ at beginning of each iteration
55  auto current_iteration = scheduler->GetSimulatedSteps();
56  auto tid = omp_get_thread_num();
57  if (last_iteration_[tid] != current_iteration) {
58  last_iteration_[tid] = current_iteration;
59 
60  auto* grid = sim->GetGrid();
61  auto search_radius = grid->GetLargestObjectSize();
62  squared_radius_ = search_radius * search_radius;
63  auto current_time =
64  (current_iteration + 1) * param->simulation_time_step_;
65  delta_time_[tid] = current_time - last_time_run_[tid];
66  last_time_run_[tid] = current_time;
67  }
68 
69  const auto& displacement =
70  sim_object->CalculateDisplacement(squared_radius_, delta_time_[tid]);
71  sim_object->ApplyDisplacement(displacement);
72  if (param->bound_space_) {
73  ApplyBoundingBox(sim_object, param->min_bound_, param->max_bound_);
74  }
75  }
