42 void InPlaceExecutionContext::ThreadSafeSoUidMap::Insert(
43  const SoUid& uid,
44  const typename InPlaceExecutionContext::ThreadSafeSoUidMap::value_type&
45  value) {
46  auto index = uid.GetIndex();
47  if (map_->size() > index + ThreadInfo::GetInstance()->GetMaxThreads()) {
48  // enough free slots -> no locking required
49  map_->Insert(uid, value);
50  } else {
51  std::lock_guard<Spinlock> guard(lock_);
52  // check again
53  if (map_->size() <= index) {
54  // map is too small -> grow
55  auto* new_map = new Map(*map_);
56  new_map->resize(std::max(static_cast<uint64_t>(1000u),
57  static_cast<uint64_t>(map_->size() * 1.5)));
58  previous_maps_.emplace_back(map_);
59  map_ = new_map;
60  }
61  map_->Insert(uid, value);
62  }
63 }

108 void InPlaceExecutionContext::TearDownIterationAll(
109  const std::vector<InPlaceExecutionContext*>& all_exec_ctxts) const {
110  // group execution contexts by numa domain
111  std::vector<uint64_t> new_so_per_numa(tinfo_->GetNumaNodes());
112  std::vector<uint64_t> thread_offsets(tinfo_->GetMaxThreads());
113 
114  for (int tid = 0; tid < tinfo_->GetMaxThreads(); ++tid) {
115  auto* ctxt = all_exec_ctxts[tid];
116  int nid = tinfo_->GetNumaNode(tid);
117  thread_offsets[tid] = new_so_per_numa[nid];
118  new_so_per_numa[nid] += ctxt->new_sim_objects_.size();
119  }
120 
121  // reserve enough memory in ResourceManager
122  std::vector<uint64_t> numa_offsets(tinfo_->GetNumaNodes());
123  auto* rm = Simulation::GetActive()->GetResourceManager();
124  rm->ResizeUidSohMap();
125  for (unsigned n = 0; n < new_so_per_numa.size(); n++) {
126  numa_offsets[n] = rm->GrowSoContainer(new_so_per_numa[n], n);
127  }
128 
129 // add new_sim_objects_ to the ResourceManager in parallel
130 #pragma omp parallel for schedule(static, 1)
131  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
132  auto* ctxt = all_exec_ctxts[i];
133  int nid = tinfo_->GetNumaNode(i);
134  uint64_t offset = thread_offsets[i] + numa_offsets[nid];
135  rm->AddNewSimObjects(nid, offset, ctxt->new_sim_objects_);
136  ctxt->new_sim_objects_.clear();
137  }
138 
139 // remove
140 #pragma omp parallel for schedule(static, 1)
141  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
142  auto* ctxt = all_exec_ctxts[i];
143  // removed sim objects
144  // remove them after adding new ones (maybe one has been removed
145  // that was in new_sim_objects_)
146  for (auto& uid : ctxt->remove_) {
147  rm->Remove(uid);
148  }
149  ctxt->remove_.clear();
150  }
151 
152  rm->EndOfIteration();
153 
154  // FIXME
155  // new_so_map_.SetOffset(SoUidGenerator::Get()->GetLastId());
156  new_so_map_->RemoveOldCopies();
157  if (rm->GetNumSimObjects() > new_so_map_->Size()) {
158  new_so_map_->Resize(rm->GetNumSimObjects() * 1.5);
159  }
160 }

162 void InPlaceExecutionContext::Execute(
163  SimObject* so, const std::vector<Operation>& operations) {
164  auto* grid = Simulation::GetActive()->GetGrid();
165  auto nb_mutex_builder = grid->GetNeighborMutexBuilder();
166  if (nb_mutex_builder != nullptr) {
167  auto mutex = nb_mutex_builder->GetMutex(so->GetBoxIdx());
168  std::lock_guard<decltype(mutex)> guard(mutex);
169  neighbor_cache_.clear();
170  for (auto& op : operations) {
171  op(so);
172  }
173  } else {
174  neighbor_cache_.clear();
175  for (auto& op : operations) {
176  op(so);
177  }
178  }

202 void InPlaceExecutionContext::ForEachNeighbor(
203  const std::function<void(const SimObject*, double)>& lambda,
204  const SimObject& query) {
205  // use values in cache
206  if (neighbor_cache_.size() != 0) {
207  for (auto& pair : neighbor_cache_) {
208  lambda(pair.first, pair.second);
209  }
210  return;
211  }
212 
213  // forward call to grid and populate cache
214  auto* grid = Simulation::GetActive()->GetGrid();
215  auto* param = Simulation::GetActive()->GetParam();
216  auto for_each = [&, this](const SimObject* so, double squared_distance) {
217  if (param->cache_neighbors_) {
218  this->neighbor_cache_.push_back(make_pair(so, squared_distance));
219  }
220  lambda(so, squared_distance);
221  };
222  grid->ForEachNeighbor(for_each, query);
223 }

276 SimObject* InPlaceExecutionContext::GetCachedSimObject(const SoUid& uid) {
277  if (!new_so_map_->Contains(uid)) {
278  return nullptr;
279  }
280  auto& pair = (*new_so_map_)[uid];
281  auto timesteps = Simulation::GetActive()->GetScheduler()->GetSimulatedSteps();
282  if (pair.second == timesteps) {
283  return pair.first;
284  }
285  return nullptr;
