42 // NB: There is small risk of a race condition here if another thread grows
43 // the container. In this situation, the element will be inserted in a map
44 // that will be moved to previous_maps_.
45 // This race-condition can be mitigated in `ThreadSafeSoUidMap::operator[]`.
46 void InPlaceExecutionContext::ThreadSafeSoUidMap::Insert(
47  const SoUid& uid,
48  const typename InPlaceExecutionContext::ThreadSafeSoUidMap::value_type&
49  value) {
50  auto index = uid.GetIndex();
51  if (map_->size() > index + ThreadInfo::GetInstance()->GetMaxThreads()) {
52  map_->Insert(uid, value);
53  } else {
54  std::lock_guard<Spinlock> guard(lock_);
55  // check again
56  if (map_->size() <= index) {
57  // map is too small -> grow
58  auto* new_map = new Map(*map_);
59  new_map->resize(std::max(static_cast<uint64_t>(1000u),
60  static_cast<uint64_t>(map_->size() * 1.5)));
61  previous_maps_.emplace_back(map_);
62  map_ = new_map;
63  }
64  map_->Insert(uid, value);
65  }
66 }

75 const typename InPlaceExecutionContext::ThreadSafeSoUidMap::value_type&
76  InPlaceExecutionContext::ThreadSafeSoUidMap::operator[](const SoUid& uid) {
77  static InPlaceExecutionContext::ThreadSafeSoUidMap::value_type kDefault;
78  auto& pair = (*map_)[uid];
79  auto timesteps = Simulation::GetActive()->GetScheduler()->GetSimulatedSteps();
80 
81  if (pair.second == timesteps && pair.first != nullptr) {
82  return pair;
83  }
84 
85  std::lock_guard<Spinlock> guard(lock_);
86  for (auto* m : previous_maps_) {
87  if (m->size() <= uid.GetIndex()) {
88  continue;
89  }
90  auto& pair = (*m)[uid];
91  if (pair.second == timesteps && pair.first != nullptr) {
92  map_->Insert(uid, pair);
93  return pair;
94  }
95  }
96 
97  return kDefault;
98 }

132 void InPlaceExecutionContext::TearDownIterationAll(
133  const std::vector<InPlaceExecutionContext*>& all_exec_ctxts) const {
134  // group execution contexts by numa domain
135  std::vector<uint64_t> new_so_per_numa(tinfo_->GetNumaNodes());
136  std::vector<uint64_t> thread_offsets(tinfo_->GetMaxThreads());
137 
138  for (int tid = 0; tid < tinfo_->GetMaxThreads(); ++tid) {
139  auto* ctxt = all_exec_ctxts[tid];
140  int nid = tinfo_->GetNumaNode(tid);
141  thread_offsets[tid] = new_so_per_numa[nid];
142  new_so_per_numa[nid] += ctxt->new_sim_objects_.size();
143  }
144 
145  // reserve enough memory in ResourceManager
146  std::vector<uint64_t> numa_offsets(tinfo_->GetNumaNodes());
147  auto* rm = Simulation::GetActive()->GetResourceManager();
148  rm->ResizeUidSohMap();
149  for (unsigned n = 0; n < new_so_per_numa.size(); n++) {
150  numa_offsets[n] = rm->GrowSoContainer(new_so_per_numa[n], n);
151  }
152 
153 // add new_sim_objects_ to the ResourceManager in parallel
154 #pragma omp parallel for schedule(static, 1)
155  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
156  auto* ctxt = all_exec_ctxts[i];
157  int nid = tinfo_->GetNumaNode(i);
158  uint64_t offset = thread_offsets[i] + numa_offsets[nid];
159  rm->AddNewSimObjects(nid, offset, ctxt->new_sim_objects_);
160  ctxt->new_sim_objects_.clear();
161  }
162 
163  // remove
164  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
165  auto* ctxt = all_exec_ctxts[i];
166  // removed sim objects
167  // remove them after adding new ones (maybe one has been removed
168  // that was in new_sim_objects_)
169  for (auto& uid : ctxt->remove_) {
170  rm->Remove(uid);
171  }
172  ctxt->remove_.clear();
173  }
174 
175  rm->EndOfIteration();
176 
177  // FIXME
178  // new_so_map_.SetOffset(SoUidGenerator::Get()->GetLastId());
179  new_so_map_->RemoveOldCopies();
180  if (rm->GetNumSimObjects() > new_so_map_->Size()) {
181  new_so_map_->Resize(rm->GetNumSimObjects() * 1.5);
182  }
183 }

185 void InPlaceExecutionContext::Execute(
186  SimObject* so, const std::vector<Operation*>& operations) {
187  auto* grid = Simulation::GetActive()->GetGrid();
188  auto* param = Simulation::GetActive()->GetParam();
189 
190  if (param->thread_safety_mechanism_ ==
191  Param::ThreadSafetyMechanism::kUserSpecified) {
192  so->CriticalRegion(&locks);
193  std::sort(locks.begin(), locks.end());
194  for (auto* l : locks) {
195  l->lock();
196  }
197  neighbor_cache_.clear();
198  for (auto& op : operations) {
199  (*op)(so);
200  }
201  for (auto* l : locks) {
202  l->unlock();
203  }
204  locks.clear();
205  } else if (param->thread_safety_mechanism_ ==
206  Param::ThreadSafetyMechanism::kAutomatic) {
207  auto* nb_mutex_builder = grid->GetNeighborMutexBuilder();
208  auto mutex = nb_mutex_builder->GetMutex(so->GetBoxIdx());
209  std::lock_guard<decltype(mutex)> guard(mutex);
210  neighbor_cache_.clear();
211  for (auto* op : operations) {
212  (*op)(so);
213  }
214  } else if (param->thread_safety_mechanism_ ==
215  Param::ThreadSafetyMechanism::kNone) {
216  neighbor_cache_.clear();
217  for (auto* op : operations) {
218  (*op)(so);
219  }
220  } else {
221  Log::Fatal("InPlaceExecutionContext::Execute",
222  "Invalid value for parameter thread_safety_mechanism_: ",
223  param->thread_safety_mechanism_);
224  }
225 }
