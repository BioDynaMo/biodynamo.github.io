58  void Initialize(const std::array<int32_t, 6>& grid_dimensions) {
59  // Get grid properties from neighbor grid
60  grid_dimensions_ = grid_dimensions;
61  assert(resolution_ > 0 && "The resolution cannot be zero!");
62 
63  num_boxes_axis_[0] = resolution_;
64  num_boxes_axis_[1] = resolution_;
65  num_boxes_axis_[2] = resolution_;
66 
67  // Example: diffusion grid dimensions from 0-40 and resolution
68  // of 4. Resolution must be adjusted otherwise one data pointer will be
69  // missing.
70  // Without adjustment:
71  // box_length_: 10
72  // data points {0, 10, 20, 30} - 40 will be misssing!
73  // With adjustment
74  // box_length_: 13.3
75  // data points: {0, 13.3, 26.6, 39.9}
76  box_length_ = (grid_dimensions_[1] - grid_dimensions_[0]) /
77  static_cast<double>(resolution_ - 1);
78  ParametersCheck();
79 
80  box_volume_ = box_length_ * box_length_ * box_length_;
81 
82  assert(box_length_ > 0 &&
83  "Box length of diffusion grid must be greater than zero!");
84 
85  // Set the parity of the number of boxes along the dimensions (since all
86  // dimensions are the same, we just take the x-axis here)
87  parity_ = num_boxes_axis_[0] % 2;
88 
89  total_num_boxes_ =
90  num_boxes_axis_[0] * num_boxes_axis_[1] * num_boxes_axis_[2];

92  // Allocate memory for the concentration and gradient arrays
93  c1_.resize(total_num_boxes_);

96 
97  initialized_ = true;

114  void RunInitializers() {
115  assert(num_boxes_axis_[0] > 0 &&
116  "The number of boxes along an axis was found to be zero!");
117  if (initializers_.empty()) {
118  return;
119  }
120

123  auto nz = num_boxes_axis_[2];
124 
125  // Apply all functions that initialize this diffusion grid
126  for (size_t f = 0; f < initializers_.size(); f++) {
127  for (size_t x = 0; x < nx; x++) {
128  double real_x = grid_dimensions_[0] + x * box_length_;

157  void Update(const std::array<int32_t, 2>& threshold_dimensions) {
158  // Update the grid dimensions such that each dimension ranges from
159  // {treshold_dimensions[0] - treshold_dimensions[1]}
160  auto min_gd = threshold_dimensions[0];
161  auto max_gd = threshold_dimensions[1];
162  grid_dimensions_ = {min_gd, max_gd, min_gd, max_gd, min_gd, max_gd};
163 
164  // If the grid is not perfectly divisible along each dimension by the
165  // box length, extend the grid so that it is
166  int dimension_length = max_gd - min_gd;
167  for (int i = 0; i < 3; i++) {
168  int r = fmod(dimension_length, box_length_);
169  if (r > 1e-9) {
170  // std::abs for the case that box_length_ > dimension_length
171  grid_dimensions_[2 * i + 1] += (box_length_ - r);
172  }
173  }
174 
175  // Calculate by how many boxes each dimension has grown
176  int new_dimension_length = grid_dimensions_[1] - grid_dimensions_[0];
177  int new_num_boxes = std::ceil(new_dimension_length / box_length_);
178  int growth = new_num_boxes - num_boxes_axis_[0];
179 
180  if (growth > 0) {
181  // Store the old number of boxes along each axis for comparison
182  std::array<size_t, 3> tmp_num_boxes_axis = num_boxes_axis_;
183 
184  // Increase number of boxes along axis accordingly
185  num_boxes_axis_[0] += growth;
186  num_boxes_axis_[1] += growth;
187  num_boxes_axis_[2] += growth;
188 
189  // We need to maintain the parity of the number of boxes along each
190  // dimension, otherwise copying of the substances to the increases grid
191  // will not be symmetrically done; resulting in shifting of boxes
192  // We add a box in the negative direction, because the only way the parity
193  // could have changed is because of adding a box in the positive direction
194  // (due to the grid not being perfectly divisible; see above)
195  if (num_boxes_axis_[0] % 2 != parity_) {
196  for (int i = 0; i < 3; i++) {
197  grid_dimensions_[2 * i] -= box_length_;
198  num_boxes_axis_[i]++;
199  }
200  }
201 
202  // Temporarily save previous grid data
203  auto tmp_c1 = c1_;
204  auto tmp_gradients = gradients_;
205 
206  c1_.clear();
207  c2_.clear();
208  gradients_.clear();
209 
210  total_num_boxes_ =
211  num_boxes_axis_[0] * num_boxes_axis_[1] * num_boxes_axis_[2];
212 
213  CopyOldData(tmp_c1, tmp_gradients, tmp_num_boxes_axis);
214 
215  assert(total_num_boxes_ >= tmp_num_boxes_axis[0] * tmp_num_boxes_axis[1] *
216  tmp_num_boxes_axis[2] &&
217  "The diffusion grid tried to shrink! It can only become larger");
218  }
219  }

253  for (size_t k = 0; k < old_num_boxes_axis[2]; k++) {
254  int offset = new_origin + k * num_box_xy;
255  for (size_t j = 0; j < old_num_boxes_axis[1]; j++) {
256  if (j != 0) {
257  offset += num_boxes_axis_[0];
258  }
259  for (size_t i = 0; i < old_num_boxes_axis[0]; i++) {
260  auto idx = k * old_box_xy + j * old_num_boxes_axis[0] + i;
261  c1_[offset + i] = old_c1[idx];
262  gradients_[3 * (offset + i)] = old_gradients[3 * idx];
263  gradients_[3 * (offset + i) + 1] = old_gradients[3 * idx + 1];
264  gradients_[3 * (offset + i) + 2] = old_gradients[3 * idx + 2];
265  }
266  }
267  }

274  void DiffuseWithLeakingEdge() {
275  int nx = num_boxes_axis_[0];
276  int ny = num_boxes_axis_[1];
277  int nz = num_boxes_axis_[2];
278 
279 #define YBF 16
280 #pragma omp parallel for collapse(2)
281  for (int yy = 0; yy < ny; yy += YBF) {
282  for (int z = 0; z < nz; z++) {
283  // To let the edges bleed we set some diffusion coefficients
284  // to zero. This prevents substance building up at the edges
285  auto dc_2_ = dc_;
286  int ymax = yy + YBF;
287  if (ymax >= ny) {
288  ymax = ny;
289  }
290  for (int y = yy; y < ymax; y++) {
291  dc_2_ = dc_;
292  int x;
293  int c, n, s, b, t;
294  x = 0;
295  c = x + y * nx + z * nx * ny;
296  if (y == 0) {
297  n = c;
298  dc_2_[4] = 0;
299  } else {
300  n = c - nx;
301  }
302  if (y == (ny - 1)) {
303  s = c;
304  dc_2_[3] = 0;
305  } else {
306  s = c + nx;
307  }
308  if (z == 0) {
309  b = c;
310  dc_2_[5] = 0;
311  } else {
312  b = c - nx * ny;
313  }
314  if (z == (nz - 1)) {
315  t = c;
316  dc_2_[6] = 0;
317  } else {
318  t = c + nx * ny;
319  }
320  // x = 0; we leak out substances past this edge (so multiply by 0)
321  c2_[c] = (dc_2_[0] * c1_[c] + 0 * c1_[c] + dc_2_[2] * c1_[c + 1] +
322  dc_2_[3] * c1_[s] + dc_2_[4] * c1_[n] + dc_2_[5] * c1_[b] +
323  dc_2_[6] * c1_[t]) *
324  (1 - mu_);
325 #pragma omp simd
326  for (x = 1; x < nx - 1; x++) {
327  ++c;
328  ++n;
329  ++s;
330  ++b;
331  ++t;
332  c2_[c] =
333  (dc_2_[0] * c1_[c] + dc_2_[1] * c1_[c - 1] +
334  dc_2_[2] * c1_[c + 1] + dc_2_[3] * c1_[s] + dc_2_[4] * c1_[n] +
335  dc_2_[5] * c1_[b] + dc_2_[6] * c1_[t]) *
336  (1 - mu_);
337  }
338  ++c;
339  ++n;
340  ++s;
341  ++b;
342  ++t;
343  // x = nx-1; we leak out substances past this edge (so multiply by 0)
344  c2_[c] = (dc_2_[0] * c1_[c] + dc_2_[1] * c1_[c - 1] + 0 * c1_[c] +
345  dc_2_[3] * c1_[s] + dc_2_[4] * c1_[n] + dc_2_[5] * c1_[b] +
346  dc_2_[6] * c1_[t]) *
347  (1 - mu_);
348  } // tile ny
349  } // tile nz
350  } // block ny
351  c1_.swap(c2_);
352  }

423  const double d = 1 - dc_[0];
424 
425 #define YBF 16
426 #pragma omp parallel for collapse(2)
427  for (size_t yy = 0; yy < ny; yy += YBF) {
428  for (size_t z = 0; z < nz; z++) {
429  size_t ymax = yy + YBF;
430  if (ymax >= ny) {
431  ymax = ny;
432  }
433  for (size_t y = yy; y < ymax; y++) {
434  size_t x = 0;
435  int c, n, s, b, t;
436  c = x + y * nx + z * nx * ny;
437 #pragma omp simd
438  for (x = 1; x < nx - 1; x++) {
439  ++c;
440  ++n;
441  ++s;
442  ++b;
443  ++t;
444 
445  if (y == 0 || y == (ny - 1) || z == 0 || z == (nz - 1)) {
446  continue;
447  }
448 
449  n = c - nx;
450  s = c + nx;
451  b = c - nx * ny;
452  t = c + nx * ny;

570  void CalculateGradient() {
571  // check if gradient has been calculated once
572  // and if diffusion coefficient and decay constant are 0
573  // i.e. if we don't need to calculate gradient update
574  if (init_gradient_ && IsFixedSubstance()) {
575  return;
576  }
577 
578  double gd = 1 / (box_length_ * 2);
579 
580  auto nx = num_boxes_axis_[0];
581  auto ny = num_boxes_axis_[1];
582  auto nz = num_boxes_axis_[2];
583 
584 #pragma omp parallel for collapse(2)
585  for (size_t z = 0; z < nz; z++) {
586  for (size_t y = 0; y < ny; y++) {
587  for (size_t x = 0; x < nx; x++) {
588  int c, e, w, n, s, b, t;
589  c = x + y * nx + z * nx * ny;
590 
591  if (x == 0) {
592  e = c;
593  w = c + 2;
594  } else if (x == nx - 1) {
595  e = c - 2;
596  w = c;
597  } else {
598  e = c - 1;
599  w = c + 1;
600  }
601 
602  if (y == 0) {
603  n = c + 2 * nx;
604  s = c;
605  } else if (y == ny - 1) {
606  n = c;
607  s = c - 2 * nx;
608  } else {
609  n = c + nx;
610  s = c - nx;
611  }
612 
613  if (z == 0) {
614  t = c + 2 * nx * ny;
615  b = c;
616  } else if (z == nz - 1) {
617  t = c;
618  b = c - 2 * nx * ny;
619  } else {
620  t = c + nx * ny;
621  b = c - nx * ny;
622  }
623 
624  // Let the gradient point from low to high concentration
625  gradients_[3 * c + 0] = (c1_[w] - c1_[e]) * gd;
626  gradients_[3 * c + 1] = (c1_[n] - c1_[s]) * gd;
627  gradients_[3 * c + 2] = (c1_[t] - c1_[b]) * gd;
628  }
629  }
630  }

632  init_gradient_ = true;
633  }
