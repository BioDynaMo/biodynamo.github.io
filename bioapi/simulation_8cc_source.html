91 void Simulation::Restore(Simulation&& restored) {
92  // random_
93  if (random_.size() != restored.random_.size()) {
94  Log::Warning("Simulation", "The restore file (", param_->restore_file_,
95  ") was run with a different number of threads. Can't restore "
96  "complete random number generator state.");
97  uint64_t min = std::min(random_.size(), restored.random_.size());
98  for (uint64_t i = 0; i < min; i++) {
99  *(random_[i]) = *(restored.random_[i]);
100  }
101  } else {
102  for (uint64_t i = 0; i < random_.size(); i++) {
103  *(random_[i]) = *(restored.random_[i]);
104  }
105  }
106 
107  // param and rm
108  param_->Restore(std::move(*restored.param_));
109  restored.param_ = nullptr;
110  *rm_ = std::move(*restored.rm_);
111  restored.rm_ = nullptr;
112 
113  // name_ and unique_name_
114  InitializeUniqueName(restored.name_);
115  InitializeOutputDir();
116 }

211 void Simulation::InitializeMembers() {
212  if (param_->use_bdm_mem_mgr_) {
213  mem_mgr_ = new MemoryManager(param_->mem_mgr_aligned_pages_shift_,
214  param_->mem_mgr_growth_rate_,
215  param_->mem_mgr_max_mem_per_thread_);
216  }
217  so_uid_generator_ = new SoUidGenerator();
218  if (param_->debug_numa_) {
219  std::cout << "ThreadInfo:\n" << *ThreadInfo::GetInstance() << std::endl;
220  }
221 
222  random_.resize(omp_get_max_threads());
223 #pragma omp parallel for schedule(static, 1)
224  for (uint64_t i = 0; i < random_.size(); i++) {
225  random_[i] = new Random();
226  }
227  exec_ctxt_.resize(omp_get_max_threads());
228  auto map =
229  std::make_shared<typename InPlaceExecutionContext::ThreadSafeSoUidMap>();
230 #pragma omp parallel for schedule(static, 1)
231  for (uint64_t i = 0; i < exec_ctxt_.size(); i++) {
232  exec_ctxt_[i] = new InPlaceExecutionContext(map);
233  }
234  rm_ = new ResourceManager();
235  grid_ = new Grid();
236  scheduler_ = new Scheduler();
237 }

239 void Simulation::InitializeRuntimeParams(
240  CommandLineOptions* clo, const std::function<void(Param*)>& set_param,
241  const std::string& ctor_config) {
242  // Renew thread info just in case it has been initialised as static and a
243  // simulation calls e.g. `omp_set_num_threads()` within main.
244  ThreadInfo::GetInstance()->Renew();
245 
246  param_ = new Param();
247 
248  // detect if the biodynamo environment has been sourced
249  if (std::getenv("BDMSYS") == nullptr) {
250  Log::Fatal("Simulation::InitializeRuntimeParams",
251  "The BioDynaMo environment is not set up correctly. Please execute "
252  "'source <path-to-bdm-installation>/bin/thisbdm.sh' and retry this command.");
253  }
254 
255  static bool read_env = false;
256  if (!read_env) {
257  // Read, only once, bdm.rootrc to set BioDynaMo-related settings for ROOT
258  std::stringstream ss;
259  ss << std::getenv("BDMSYS") << "/etc/bdm.rootrc";
260  gEnv->ReadFile(ss.str().c_str(), kEnvUser);
261  read_env = true;
262  }
263 
264  LoadConfigFile(ctor_config, clo->Get<std::string>("config"));
265  if (clo->Get<std::string>("backup") != "") {
266  param_->backup_file_ = clo->Get<std::string>("backup");
267  }
268  if (clo->Get<std::string>("restore") != "") {
269  param_->restore_file_ = clo->Get<std::string>("restore");
270  }
271 
272  // Handle "cuda" and "opencl" arguments
273  if (clo->Get<bool>("cuda")) {
274  param_->use_gpu_ = true;
275  }
276 
277  if (clo->Get<bool>("opencl")) {
278  param_->use_gpu_ = true;
279  param_->use_opencl_ = true;
280  }
281 
282  ocl_state_ = new OpenCLState();
283 
284  set_param(param_);
285 
286  if (!is_gpu_environment_initialized_ && param_->use_gpu_) {
287  GpuHelper::GetInstance()->InitializeGPUEnvironment();
288  is_gpu_environment_initialized_ = true;
289  }
290 
291  // Removing this line causes an unexplainable segfault due to setting the
292  // gErrorIngoreLevel global parameter of ROOT. We need to log at least one
293  // thing before setting that parameter.
294  Log::Info("", "Initialize new simulation using BioDynaMo ",
295  Version::String());
296 }

298 void Simulation::LoadConfigFile(const std::string& ctor_config,
299  const std::string& cli_config) {
300  constexpr auto kConfigFile = "bdm.toml";
301  constexpr auto kConfigFileParentDir = "../bdm.toml";
302  if (ctor_config != "") {
303  if (FileExists(ctor_config)) {
304  auto config = cpptoml::parse_file(ctor_config);
305  param_->AssignFromConfig(config);
306  } else {
307  Log::Fatal("Simulation::LoadConfigFile", "The config file ", ctor_config,
308  " specified in the constructor of bdm::Simulation "
309  "could not be found.");
310  }
311  } else if (cli_config != "") {
312  if (FileExists(cli_config)) {
313  auto config = cpptoml::parse_file(cli_config);
314  param_->AssignFromConfig(config);
315  } else {
316  Log::Fatal("Simulation::LoadConfigFile", "The config file ", cli_config,
317  " specified as command line argument "
318  "could not be found.");
319  }
320  } else if (FileExists(kConfigFile)) {
321  auto config = cpptoml::parse_file(kConfigFile);
322  param_->AssignFromConfig(config);
323  } else if (FileExists(kConfigFileParentDir)) {
324  auto config = cpptoml::parse_file(kConfigFileParentDir);
325  param_->AssignFromConfig(config);
326  } else {
327  Log::Warning("Simulation::LoadConfigFile",
328  "Config file ", kConfigFile, " not found in `.` or `..` directory.");
329  }
330 }
