62  : Base(event, mother, new_oid) {
63  const CellDivisionEvent* cdevent =
64  dynamic_cast<const CellDivisionEvent*>(&event);
65  Cell* mother_cell = dynamic_cast<Cell*>(mother);
66  if (cdevent && mother_cell) {
67  auto* daughter = this; // FIXME
68  // A) Defining some values
69  // ..................................................................
70  // defining the two radii s.t total volume is conserved
71  // * radius^3 = r1^3 + r2^3 ;
72  // * volume_ratio = r2^3 / r1^3
73  double radius = mother_cell->GetDiameter() * 0.5;
74 
75  // define an axis for division (along which the nuclei will move)
76  double x_coord = std::cos(cdevent->theta_) * std::sin(cdevent->phi_);
77  double y_coord = std::sin(cdevent->theta_) * std::sin(cdevent->phi_);
78  double z_coord = std::cos(cdevent->phi_);
79  Double3 coords = {x_coord, y_coord, z_coord};
80  double total_length_of_displacement = radius / 4.0;
81 
82  const auto x_axis = mother_cell->kXAxis;
83  const auto y_axis = mother_cell->kYAxis;
84  const auto z_axis = mother_cell->kZAxis;
85 
86  Double3 axis_of_division =
87  (coords.EntryWiseProduct(x_axis) + coords.EntryWiseProduct(y_axis) +
88  coords.EntryWiseProduct(z_axis)) *
89  total_length_of_displacement;
90 
91  // two equations for the center displacement :
92  // 1) d2/d1= v2/v1 = volume_ratio (each sphere is shifted inver.
93  // proportionally to its volume)
94  // 2) d1 + d2 = TOTAL_LENGTH_OF_DISPLACEMENT
95  double d_2 = total_length_of_displacement / (cdevent->volume_ratio_ + 1);
96  double d_1 = total_length_of_displacement - d_2;
97 
98  double mother_volume = mother_cell->GetVolume();
99  double new_volume = mother_volume / (cdevent->volume_ratio_ + 1);
100  daughter->SetVolume(mother_volume - new_volume);
101 
102  // position
103  auto mother_pos = mother_cell->GetPosition();
104  auto new_position = mother_pos + (axis_of_division * d_2);
105  daughter->SetPosition(new_position);
106 
107  // E) This sphere becomes the 1st daughter
108  // move these cells on opposite direction
109  mother_pos -= axis_of_division * d_1;
110  // update mother here and not in EventHandler to avoid recomputation
111  mother_cell->SetPosition(mother_pos);
112  mother_cell->SetVolume(new_volume);
113 
114  daughter->SetAdherence(mother_cell->GetAdherence());
115  daughter->SetDensity(mother_cell->GetDensity());

261  Double3 CalculateDisplacement(double squared_radius, double dt) override {
262  // Basically, the idea is to make the sum of all the forces acting
263  // on the Point mass. It is stored in translationForceOnPointMass.
264  // There is also a computation of the torque (only applied
265  // by the daughter neurites), stored in rotationForce.
266 
267  // TODO(roman) : There might be a problem, in the sense that the biology
268  // is not applied if the total Force is smaller than adherence.
269  // Once, I should look at this more carefully.
270 
271  // fixme why? copying
272  const auto& tf = GetTractorForce();
273 
274  // the 3 types of movement that can occur
275  // bool biological_translation = false;
276  bool physical_translation = false;
277  // bool physical_rotation = false;
278 
279  double h = dt;
280  Double3 movement_at_next_step{0, 0, 0};
281 
282  // BIOLOGY :
283  // 0) Start with tractor force : What the biology defined as active
284  // movement------------
285  movement_at_next_step += tf * h;
286 
287  // PHYSICS
288  // the physics force to move the point mass
289  Double3 translation_force_on_point_mass{0, 0, 0};
290  // the physics force to rotate the cell
291  // Double3 rotation_force { 0, 0, 0 };
292 
293  // 1) "artificial force" to maintain the sphere in the ecm simulation
294  // boundaries--------
295  // 2) Spring force from my neurites (translation and
296  // rotation)--------------------------
297  // 3) Object avoidance force
298  // -----------------------------------------------------------
299  // (We check for every neighbor object if they touch us, i.e. push us
300  // away)
301 
302  auto calculate_neighbor_forces = [&, this](const auto* neighbor) {
303  DefaultForce default_force;
304  auto neighbor_force = default_force.GetForce(this, neighbor);
305  translation_force_on_point_mass[0] += neighbor_force[0];
306  translation_force_on_point_mass[1] += neighbor_force[1];
307  translation_force_on_point_mass[2] += neighbor_force[2];
308  };
309 
310  auto* ctxt = Simulation::GetActive()->GetExecutionContext();
311  ctxt->ForEachNeighborWithinRadius(calculate_neighbor_forces, *this,
312  squared_radius);
313 
314  // 4) PhysicalBonds
315  // How the physics influences the next displacement
316  double norm_of_force = std::sqrt(translation_force_on_point_mass *
317  translation_force_on_point_mass);
318 
319  // is there enough force to :
320  // - make us biologically move (Tractor) :
321  // - break adherence and make us translate ?
322  physical_translation = norm_of_force > GetAdherence();
323 
324  assert(GetMass() != 0 && "The mass of a cell was found to be zero!");
325  double mh = h / GetMass();
326  // adding the physics translation (scale by weight) if important enough
327  if (physical_translation) {
328  // We scale the move with mass and time step
329  movement_at_next_step += translation_force_on_point_mass * mh;
330 
331  // Performing the translation itself :
332  // but we want to avoid huge jumps in the simulation, so there are
333  // maximum distances possible
334  auto* param = Simulation::GetActive()->GetParam();
335  if (norm_of_force * mh > param->simulation_max_displacement_) {
336  movement_at_next_step.Normalize();
337  movement_at_next_step *= param->simulation_max_displacement_;
338  }
339  }
340  return movement_at_next_step;
341  }
