63  : Base(event, mother, new_oid) {
64  const CellDivisionEvent* cdevent =
65  dynamic_cast<const CellDivisionEvent*>(&event);
66  Cell* mother_cell = dynamic_cast<Cell*>(mother);
67  if (cdevent && mother_cell) {
68  auto* daughter = this; // FIXME
69  // A) Defining some values
70  // ..................................................................
71  // defining the two radii s.t total volume is conserved
72  // * radius^3 = r1^3 + r2^3 ;
73  // * volume_ratio = r2^3 / r1^3
74  double radius = mother_cell->GetDiameter() * 0.5;
75 
76  // define an axis for division (along which the nuclei will move)
77  double x_coord = std::cos(cdevent->theta_) * std::sin(cdevent->phi_);
78  double y_coord = std::sin(cdevent->theta_) * std::sin(cdevent->phi_);
79  double z_coord = std::cos(cdevent->phi_);
80  Double3 coords = {x_coord, y_coord, z_coord};
81  double total_length_of_displacement = radius / 4.0;
82 
83  const auto x_axis = mother_cell->kXAxis;
84  const auto y_axis = mother_cell->kYAxis;
85  const auto z_axis = mother_cell->kZAxis;
86 
87  Double3 axis_of_division =
88  (coords.EntryWiseProduct(x_axis) + coords.EntryWiseProduct(y_axis) +
89  coords.EntryWiseProduct(z_axis)) *
90  total_length_of_displacement;
91 
92  // two equations for the center displacement :
93  // 1) d2/d1= v2/v1 = volume_ratio (each sphere is shifted inver.
94  // proportionally to its volume)
95  // 2) d1 + d2 = TOTAL_LENGTH_OF_DISPLACEMENT
96  double d_2 = total_length_of_displacement / (cdevent->volume_ratio_ + 1);
97  double d_1 = total_length_of_displacement - d_2;
98 
99  double mother_volume = mother_cell->GetVolume();
100  double new_volume = mother_volume / (cdevent->volume_ratio_ + 1);
101  daughter->SetVolume(mother_volume - new_volume);
102 
103  // position
104  auto mother_pos = mother_cell->GetPosition();
105  auto new_position = mother_pos + (axis_of_division * d_2);
106  daughter->SetPosition(new_position);
107 
108  // E) This sphere becomes the 1st daughter
109  // move these cells on opposite direction
110  mother_pos -= axis_of_division * d_1;
111  // update mother here and not in EventHandler to avoid recomputation
112  mother_cell->SetPosition(mother_pos);
113  mother_cell->SetVolume(new_volume);
114 
115  daughter->SetAdherence(mother_cell->GetAdherence());
116  daughter->SetDensity(mother_cell->GetDensity());

278  Double3 CalculateDisplacement(double squared_radius, double dt) override {
279  // Basically, the idea is to make the sum of all the forces acting
280  // on the Point mass. It is stored in translationForceOnPointMass.
281  // There is also a computation of the torque (only applied
282  // by the daughter neurites), stored in rotationForce.
283 
284  // TODO(roman) : There might be a problem, in the sense that the biology
285  // is not applied if the total Force is smaller than adherence.
286  // Once, I should look at this more carefully.
287 
288  // fixme why? copying
289  const auto& tf = GetTractorForce();
290 
291  // the 3 types of movement that can occur
292  // bool biological_translation = false;
293  bool physical_translation = false;
294  // bool physical_rotation = false;
295 
296  double h = dt;
297  Double3 movement_at_next_step{0, 0, 0};
298 
299  // BIOLOGY :
300  // 0) Start with tractor force : What the biology defined as active
301  // movement------------
302  movement_at_next_step += tf * h;
303 
304  // PHYSICS
305  // the physics force to move the point mass
306 
307  // the physics force to rotate the cell
308  // Double3 rotation_force { 0, 0, 0 };
309 
310  // 1) "artificial force" to maintain the sphere in the ecm simulation
311  // boundaries--------
312  // 2) Spring force from my neurites (translation and
313  // rotation)--------------------------
314  // 3) Object avoidance force
315  // -----------------------------------------------------------
316  // (We check for every neighbor object if they touch us, i.e. push us
317  // away)
318 
319  DisplacementFunctor calculate_neighbor_forces(this);
320  auto* ctxt = Simulation::GetActive()->GetExecutionContext();
321  ctxt->ForEachNeighborWithinRadius(calculate_neighbor_forces, *this,
322  squared_radius);
323 
324  // 4) PhysicalBonds
325  // How the physics influences the next displacement
326  double norm_of_force =
327  std::sqrt(calculate_neighbor_forces.translation_force_on_point_mass *
328  calculate_neighbor_forces.translation_force_on_point_mass);
329 
330  // is there enough force to :
331  // - make us biologically move (Tractor) :
332  // - break adherence and make us translate ?
333  physical_translation = norm_of_force > GetAdherence();
334 
335  assert(GetMass() != 0 && "The mass of a cell was found to be zero!");
336  double mh = h / GetMass();
337  // adding the physics translation (scale by weight) if important enough
338  if (physical_translation) {
339  // We scale the move with mass and time step
340  movement_at_next_step +=
341  calculate_neighbor_forces.translation_force_on_point_mass * mh;
342 
343  // Performing the translation itself :
344  // but we want to avoid huge jumps in the simulation, so there are
345  // maximum distances possible
346  auto* param = Simulation::GetActive()->GetParam();
347  if (norm_of_force * mh > param->simulation_max_displacement_) {
348  movement_at_next_step.Normalize();
349  movement_at_next_step *= param->simulation_max_displacement_;
350  }
351  }
352  return movement_at_next_step;
353  }
