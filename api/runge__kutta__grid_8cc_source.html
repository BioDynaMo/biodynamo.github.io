20 void RungeKuttaGrid::DiffuseWithClosedEdge(real_t dt) {
21  const auto nx = resolution_;
22  const auto ny = resolution_;
23  const auto nz = resolution_;
24 
25  const real_t ibl2 = 1 / (box_length_ * box_length_);
26  const real_t d = 1 - dc_[0];
27  real_t step = diffusion_step_;
28  real_t h = dt / step;
29 #define YBF 16
30  for (size_t i = 0; i < step; i++) {
31  for (size_t order = 0; order < 2; order++) {
32 #pragma omp parallel for collapse(2)
33  for (size_t yy = 0; yy < ny; yy += YBF) {
34  for (size_t z = 0; z < nz; z++) {
35  size_t ymax = yy + YBF;
36  if (ymax >= ny) {
37  ymax = ny;
38  }
39  for (size_t y = yy; y < ymax; y++) {
40  size_t x = 0;
41  int c, n, s, b, t;
42  c = x + y * nx + z * nx * ny;
43 #pragma omp simd
44  for (x = 1; x < nx - 1; x++) {
45  ++c;
46 
47  if (y == 0 || y == (ny - 1) || z == 0 || z == (nz - 1)) {
48  continue;
49  }
50 
51  n = c - nx;
52  s = c + nx;
53  b = c - nx * ny;
54  t = c + nx * ny;
55 
56  real_t h2 = h / 2.0;
57 
58  if (order == 0) {
59  k_[0] = (d * (c1_[c - 1] - 2 * c1_[c] + c1_[c + 1]) * ibl2 +
60  d * (c1_[s] - 2 * c1_[c] + c1_[n]) * ibl2 +
61  d * (c1_[b] - 2 * c1_[c] + c1_[t]) * ibl2);
62  r1_[c] = c1_[c] + (k_[0] * h2);
63  } else if (order == 1) {
64  k_[1] = (d * (c1_[c - 1] - 2 * c1_[c] + c1_[c + 1]) * ibl2 +
65  d * (c1_[s] - 2 * c1_[c] + c1_[n]) * ibl2 +
66  d * (c1_[b] - 2 * c1_[c] + c1_[t]) * ibl2);
67 
68  c2_[c] = c1_[c] + (k_[1] * h);
69  }
70  }
71  } // tile ny
72  } // tile nz
73  } // block ny
74  }
75  c1_.swap(c2_);
76  }
77 }
