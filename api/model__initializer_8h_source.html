45  template <typename Function, typename TResourceManager = ResourceManager<>>
46  static void Grid3D(size_t cells_per_dim, double space,
47  Function cell_builder) {
48  auto rm = TResourceManager::Get();
49  // Determine simulation object type which is returned by the cell_builder
50  using FunctionReturnType = decltype(cell_builder({0, 0, 0}));
51 
52  auto container = rm->template Get<FunctionReturnType>();
53  container->reserve(cells_per_dim * cells_per_dim * cells_per_dim);
54  for (size_t x = 0; x < cells_per_dim; x++) {
55  auto x_pos = x * space;
56  for (size_t y = 0; y < cells_per_dim; y++) {
57  auto y_pos = y * space;
58  for (size_t z = 0; z < cells_per_dim; z++) {
59  auto new_simulation_object = cell_builder({x_pos, y_pos, z * space});
60  container->push_back(new_simulation_object);
61  }
62  }
63  }
64  container->Commit();
65  }
