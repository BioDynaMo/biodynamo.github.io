160 #pragma omp for
161  for (uint64_t i = 0; i < num_agents; i++) {
162  if (rng != nullptr) {
163  Double3 pos;
164  bool in_range = false;
165  do {
166  pos = rng->Sample3();
167  in_range = (pos[0] >= min) && (pos[0] <= max) && (pos[1] >= min) &&
168  (pos[1] <= max) && (pos[2] >= min) && (pos[2] <= max);
169  } while (!in_range);
170  auto* new_agent = agent_builder(pos);
171  ctxt->AddAgent(new_agent);
172  } else {
173  auto* new_agent = agent_builder(random->UniformArray<3>(min, max));
174  ctxt->AddAgent(new_agent);
175  }
176  }
177  }

335  template <typename Function>
336  static void CreateAgentsInSphereRndm(const Double3& center, double radius,
337  uint64_t num_agents,
338  Function agent_builder) {
339  // We use a probability density function (PDF) to model the probability of
340  // an agent to occur at a distance `r>=0` of the center. As the surface of
341  // a sphere scales as `r^2`, the PDF does as well. Thus
342  // `p(r)=a*r^2*\Theta(R-r)`, where `\Theta` is a heavyside function and R is
343  // largest allowed radius (interpretation: no agents outside the sphere). We
344  // can fix `a` by requiring `\int_0^\inf p(r') dr' = 1` and obtain
345  // `a=3/R^3`.
346  auto radial_pdf_sphere = [](const double* x, const double* params) {
347  double R{params[0]};
348  double r{x[0]};
349  if (r > 0.0 && r <= R) {
350  return 3.0 * std::pow(r, 2.0) / std::pow(R, 3.0);
351  } else {
352  return 0.0;
353  }
354  };
355 
356  // Get a random number generator to sample from our PDF.
357  auto* random = Simulation::GetActive()->GetRandom();
358  auto rng =
359  random->GetUserDefinedDistRng1D(radial_pdf_sphere, {radius}, 0, radius);
360 
361  // Create a random radius for each of the agents. Note: this is done
362  // serially because we GetUserDefinedDistRng1D does not work in parallel
363  // regions at the moment.
364  std::vector<double> random_radius;
365  random_radius.resize(num_agents);
366  for (size_t i = 0; i < num_agents; i++) {
367  random_radius[i] = rng.Sample();
368  }
369 #pragma omp parallel shared(random_radius)
370  {
371  auto* ctxt_tl = Simulation::GetActive()->GetExecutionContext();
372 #pragma omp for schedule(static)
373  for (uint64_t i = 0; i < num_agents; i++) {
374  auto pos = random->Sphere(random_radius[i]) + center;
375  auto* new_agent = agent_builder(pos);
376  ctxt_tl->AddAgent(new_agent);
377  }
378  }
379  }
