370  void RetractTerminalEnd(double speed) {
371  // check if is a terminal branch
372  if (daughter_left_[kIdx] != nullptr) {
373  return;
374  }
375  // scaling for integration step
376  auto* param = Simulation_t::GetActive()->GetParam();
377  speed *= param->simulation_time_step_;
378 
379  if (actual_length_[kIdx] > speed + 0.1) {
380  // if actual_length_ > length : retraction keeping the same tension
381  // (putting a limit on how short a branch can be is absolutely necessary
382  // otherwise the tension might explode)
383 
384  double new_actual_length = actual_length_[kIdx] - speed;
385  double factor = new_actual_length / actual_length_[kIdx];
386  actual_length_[kIdx] = new_actual_length;
387  // cf removeproximalCylinder()
388  resting_length_[kIdx] = spring_constant_[kIdx] * actual_length_[kIdx] /
389  (tension_[kIdx] + spring_constant_[kIdx]);
390  spring_axis_[kIdx] = Math::ScalarMult(factor, spring_axis_[kIdx]);
391 
392  mass_location_[kIdx] = Math::Add(
393  mother_[kIdx].OriginOf(Base::GetElementIdx()), spring_axis_[kIdx]);
394  UpdateVolume(); // and update concentration of internal stuff.
395  } else if (mother_[kIdx].IsNeuronSoma()) {
396  mother_[kIdx].RemoveDaughter(GetSoPtr());
397  RemoveFromSimulation();
398  } else if (mother_[kIdx].IsNeuriteElement() &&
399  mother_[kIdx].GetDaughterRight() == nullptr) {
400  // if actual_length_ < length and mother is a neurite element with no
401  // other daughter : merge with mother
402  RemoveProximalNeuriteElement(); // also updates volume_...
403  RetractTerminalEnd(speed / param->simulation_time_step_);
404  } else {
405  // if mother is neurite element with other daughter or is not a neurite
406  // segment: disappear.
407  mother_[kIdx].RemoveDaughter(GetSoPtr());
408  RemoveFromSimulation();
409 
410  mother_[kIdx].UpdateDependentPhysicalVariables();
411  }

524  std::array<MostDerivedSoPtr, 2> Bifurcate(
525  double length, double diameter_1, double diameter_2,
526  const std::array<double, 3>& direction_1,
527  const std::array<double, 3>& direction_2) {
528  // 1) physical bifurcation
529  // check it is a terminal branch
530  if (daughter_left_[kIdx] != nullptr) {
531  Fatal("NeuriteElements",
532  "Bifurcation only allowed on a terminal neurite element");
533  }
534  auto* rm = Simulation_t::GetActive()->GetResourceManager();
535  auto new_branch_l = rm->template New<MostDerivedScalar>();
536  auto new_branch_r = rm->template New<MostDerivedScalar>();
537  new_branch_l.Copy(*static_cast<MostDerived<Backend>*>(this));
538  new_branch_r.Copy(*static_cast<MostDerived<Backend>*>(this));
539  // set family relations
540  daughter_left_[kIdx] = new_branch_l.GetSoPtr();
541  new_branch_l.SetMother(GetSoPtr());
542  daughter_right_[kIdx] = new_branch_r.GetSoPtr();
543  new_branch_r.SetMother(GetSoPtr());
544 
545  // check that the directions are not pointing backwards
546  auto dir_1 = direction_1; // todo avoid cpy
547  auto dir_2 = direction_2;
548  if (Math::AngleRadian(spring_axis_[kIdx], direction_1) > Math::kPi / 2.0) {
549  auto proj = Math::ProjectionOnto(direction_1, spring_axis_[kIdx]);
550  proj = Math::ScalarMult(-1, proj);
551  dir_1 = Math::Add(direction_1, proj);
552  }
553  if (Math::AngleRadian(spring_axis_[kIdx], direction_2) > Math::kPi / 2.0) {
554  auto proj = Math::ProjectionOnto(direction_2, spring_axis_[kIdx]);
555  proj = Math::ScalarMult(-1, proj);
556  dir_2 = Math::Add(direction_2, proj);
557  }
558 
559  // mass location and spring axis
560  new_branch_l.SetSpringAxis(
561  Math::ScalarMult(length, Math::Normalize(dir_1)));
562  new_branch_l.SetMassLocation(
563  Math::Add(mass_location_[kIdx], new_branch_l.GetSpringAxis()));
564  new_branch_l
565  .UpdateLocalCoordinateAxis(); // (important so that x_axis_ is correct)
566 
567  new_branch_r.SetSpringAxis(
568  Math::ScalarMult(length, Math::Normalize(dir_2)));
569  new_branch_r.SetMassLocation(
570  Math::Add(mass_location_[kIdx], new_branch_r.GetSpringAxis()));
571  new_branch_r.UpdateLocalCoordinateAxis();
572 
573  // physics of tension :
574  new_branch_l.SetActualLength(length);
575  new_branch_r.SetActualLength(length);
576  new_branch_r.SetRestingLengthForDesiredTension(
577  Param::kNeuriteDefaultTension);
578  new_branch_l.SetRestingLengthForDesiredTension(
579  Param::kNeuriteDefaultTension);
580 
581  // set local coordinate axis in the new branches
582  // TODO(neurites) again?? alreay done a few lines up
583  new_branch_l.UpdateLocalCoordinateAxis();
584  new_branch_r.UpdateLocalCoordinateAxis();
585 
586  // 2) creating the first daughter branch
587  new_branch_l.SetDiameter(diameter_1);
588  new_branch_l.SetBranchOrder(branch_order_[kIdx] + 1);
589 
590  // 3) the second one
591  new_branch_r.SetDiameter(diameter_2);
592  new_branch_r.SetBranchOrder(branch_order_[kIdx] + 1);
593 
594  // 4) the biological modules :
595  std::vector<BiologyModules> biology_modules_l;
596  BiologyModuleEventHandler(gNeuriteBifurcation, &biology_modules_l);
597  new_branch_l.SetBiologyModules(std::move(biology_modules_l));
598 
599  std::vector<BiologyModules> biology_modules_r;
600  BiologyModuleEventHandler(gNeuriteBifurcation, &biology_modules_r, true);
601  new_branch_r.SetBiologyModules(std::move(biology_modules_r));
602 
603  new_branch_l.UpdateDependentPhysicalVariables();
604  new_branch_r.UpdateDependentPhysicalVariables();
605 
606  return {new_branch_l.GetSoPtr(), new_branch_r.GetSoPtr()};
607  }

806  // TODO(neurites) documentation
807  template <typename TGrid>
808  std::array<double, 3> CalculateDisplacement(TGrid * grid,
809  double squared_radius) {
810  // decide first if we have to split or fuse this cylinder. Usually only
811  // terminal branches (growth cone) do
812  if (daughter_left_[kIdx] == nullptr) {
813  RunDiscretization();
814  }
815 
816  std::array<double, 3> force_on_my_point_mass{0, 0, 0};
817  std::array<double, 3> force_on_my_mothers_point_mass{0, 0, 0};
818 
819  // 1) Spring force
820  // Only the spring of this cylinder. The daughters spring also act on this
821  // mass, but they are treated in point (2)
822  double factor =
823  -tension_[kIdx] / actual_length_[kIdx]; // the minus sign is important
824  // because the spring axis goes
825  // in the opposite direction
826  force_on_my_point_mass = Math::Add(
827  force_on_my_point_mass, Math::ScalarMult(factor, spring_axis_[kIdx]));
828 
829  // 2) Force transmitted by daugthers (if they exist)
830  if (daughter_left_[kIdx] != nullptr) {
831  auto force_from_daughter =
832  daughter_left_[kIdx]->ForceTransmittedFromDaugtherToMother(
833  GetSoPtr());
834  force_on_my_point_mass =
835  Math::Add(force_on_my_point_mass, force_from_daughter);
836  }
837  if (daughter_right_[kIdx] != nullptr) {
838  auto force_from_daughter =
839  daughter_right_[kIdx]->ForceTransmittedFromDaugtherToMother(
840  GetSoPtr());
841  force_on_my_point_mass =
842  Math::Add(force_on_my_point_mass, force_from_daughter);
843  }
844 
845  // 3) Object avoidance force
846  // (We check for every neighbor object if they touch us, i.e. push us away)
847  auto calculate_neighbor_forces = [this, &force_on_my_point_mass,
848  &force_on_my_mothers_point_mass](
849  auto&& neighbor, SoHandle neighbor_handle) {
850  // TODO(lukas) once we switch to C++17 use if constexpr.
851  // As a consequence the reinterpret_cast won't be needed anymore.
852  // if neighbor is a NeuriteElement
853  if (neighbor.IsSoType(this)) {
854  auto&& neighbor_rc =
855  neighbor.template ReinterpretCast<MostDerivedScalar>();
856  auto n_soptr = neighbor_rc.GetSoPtr();
857  // if it is a direct relative, or sister branch, we don't take it into
858  // account
859  if (n_soptr == this->GetDaughterLeft() ||
860  n_soptr == this->GetDaughterRight() ||
861  (this->GetMother().IsNeuriteElement() &&
862  this->GetMother().GetNeuriteElementSoPtr() == n_soptr) ||
863  n_soptr->GetMother() == this->GetMother()) {
864  return;
865  }
866  } else if (neighbor.template IsSoType<NeuronSoma>()) {
867  // if neighbor is NeuronSoma
868  // if it is a direct relative, we don't take it into account
869  auto&& neighbor_rc = neighbor.template ReinterpretCast<NeuronSoma>();
870  auto n_soptr = neighbor_rc.GetSoPtr();
871  if (this->GetMother().IsNeuronSoma() &&
872  this->GetMother().GetNeuronSomaSoPtr() == n_soptr) {
873  return;
874  }
875  }
876 
877  DefaultForce force;
878  std::array<double, 4> force_from_neighbor =
879  force.GetForce(this, &neighbor);
880 
881  if (std::abs(force_from_neighbor[3]) <
882  1E-10) { // TODO(neurites) hard coded value
883  // (if all the force is transmitted to the (distal end) point mass)
884  force_on_my_point_mass[0] += force_from_neighbor[0];
885  force_on_my_point_mass[1] += force_from_neighbor[1];
886  force_on_my_point_mass[2] += force_from_neighbor[2];
887  } else {
888  // (if there is a part transmitted to the proximal end)
889  double part_for_point_mass = 1.0 - force_from_neighbor[3];
890  force_on_my_point_mass[0] +=
891  force_from_neighbor[0] * part_for_point_mass;
892  force_on_my_point_mass[1] +=
893  force_from_neighbor[1] * part_for_point_mass;
894  force_on_my_point_mass[2] +=
895  force_from_neighbor[2] * part_for_point_mass;
896  force_on_my_mothers_point_mass[0] +=
897  force_from_neighbor[0] * force_from_neighbor[3];
898  force_on_my_mothers_point_mass[1] +=
899  force_from_neighbor[1] * force_from_neighbor[3];
900  force_on_my_mothers_point_mass[2] +=
901  force_from_neighbor[2] * force_from_neighbor[3];
902  }
903  };
904 
905  grid->ForEachNeighborWithinRadius(calculate_neighbor_forces, *this,
906  GetSoHandle(), squared_radius);
907 
908  bool anti_kink = false;
909  // TEST : anti-kink
910  if (anti_kink) {
911  double kk = 5;
912  if (daughter_left_[kIdx] != nullptr && daughter_right_[kIdx] == nullptr) {
913  if (daughter_left_[kIdx]->GetDaughterLeft() != nullptr) {
914  auto downstream = daughter_left_[kIdx]->GetDaughterLeft();
915  double rresting = daughter_left_[kIdx]->GetRestingLength() +
916  downstream->GetRestingLength();
917  auto down_to_me =
918  Math::Subtract(GetMassLocation(), downstream->GetMassLocation());
919  double aactual = Math::Norm(down_to_me);
920 
921  force_on_my_point_mass =
922  Math::Add(force_on_my_point_mass,
923  Math::ScalarMult(kk * (rresting - aactual),
924  Math::Normalize(down_to_me)));
925  }
926  }
927 
928  if (daughter_left_[kIdx] != nullptr && mother_[kIdx].IsNeuriteElement()) {
929  auto mother = mother_[kIdx].GetNeuriteElementSoPtr();
930  double rresting = GetRestingLength() + mother->GetRestingLength();
931  auto down_to_me =
932  Math::Subtract(GetMassLocation(), mother->ProximalEnd());
933  double aactual = Math::Norm(down_to_me);
934 
935  force_on_my_point_mass =
936  Math::Add(force_on_my_point_mass,
937  Math::ScalarMult(kk * (rresting - aactual),
938  Math::Normalize(down_to_me)));
939  }
940  }
941 
942  // 5) define the force that will be transmitted to the mother
943  force_to_transmit_to_proximal_mass_[kIdx] = force_on_my_mothers_point_mass;
944  // 6.1) Define movement scale
945  double h_over_m = 1.0;
946  double force_norm = Math::Norm(force_on_my_point_mass);
947  // 6.2) If is F not strong enough -> no movements
948  if (force_norm < adherence_[kIdx]) {
949  return {0, 0, 0};
950  }
951  // if this or its mother is a branching point, displacement have to be
952  // reduced to avoid kink behaviour
953  auto* param = Simulation_t::GetActive()->GetParam();
954  if (mother_[kIdx].IsNeuriteElement()) {
955  auto mother = mother_[kIdx].GetNeuriteElementSoPtr();
956  if (mother->GetDaughterLeft() != nullptr &&
957  mother->GetDaughterRight() != nullptr) {
958  double h = param->simulation_time_step_;
959  h_over_m = h / GetMass();
960  }
961  }
962  if (daughter_left_[kIdx] != nullptr && daughter_right_[kIdx] != nullptr) {
963  double h = param->simulation_time_step_;
964  h_over_m = h / GetMass();
965  }
966 
967  // So, what follows is only executed if we do actually move :
968 
969  // 6.3) Since there's going be a move, we calculate it
970  auto displacement = Math::ScalarMult(h_over_m, force_on_my_point_mass);
971  double displacement_norm = force_norm * h_over_m;
972 
973  // 6.4) There is an upper bound for the movement.
974  if (displacement_norm > param->simulation_max_displacement_) {
975  displacement = Math::ScalarMult(
976  param->simulation_max_displacement_ / displacement_norm,
977  displacement);
978  }
979 
980  return displacement;
981  }
