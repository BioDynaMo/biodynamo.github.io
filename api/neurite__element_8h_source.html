256  void RetractTerminalEnd(double speed) {
257  // check if is a terminal branch
258  if (daughter_left_ != nullptr) {
259  return;
260  }
261  // scaling for integration step
262  auto* core_param = Simulation::GetActive()->GetParam();
263  speed *= core_param->simulation_time_step;
264 
265  auto* mother_agentma = dynamic_cast<NeuronSoma*>(mother_.Get());
266  auto* mother_neurite = dynamic_cast<NeuriteElement*>(mother_.Get());
267 
268  if (actual_length_ > speed + 0.1) {
269  // if actual_length_ > length : retraction keeping the same tension
270  // (putting a limit on how short a branch can be is absolutely necessary
271  // otherwise the tension might explode)
272 
273  double new_actual_length = actual_length_ - speed;
274  double factor = new_actual_length / actual_length_;
275  SetActualLength(new_actual_length);
276  // cf removeproximalCylinder()
277  resting_length_ =
278  spring_constant_ * actual_length_ / (tension_ + spring_constant_);
279  SetSpringAxis(spring_axis_ * factor);
280 
281  SetMassLocation(mother_->OriginOf(Base::GetUid()) + spring_axis_);
282  UpdatePosition();
283  UpdateVolume(); // and update concentration of internal stuff.
284  } else if (mother_agentma) {
285  mother_->RemoveDaughter(Base::GetAgentPtr<NeuriteElement>());
286  this->RemoveFromSimulation();
287  } else if (mother_neurite &&
288  mother_neurite->GetDaughterRight() == nullptr) {
289  // if actual_length_ < length and mother is a neurite element with no
290  // other daughter : merge with mother
291  RemoveProximalNeuriteElement(); // also updates volume_...
292  RetractTerminalEnd(speed / core_param->simulation_time_step);
293  } else {
294  // if mother is neurite element with other daughter or is not a neurite
295  // segment: disappear.
296  mother_->RemoveDaughter(Base::GetAgentPtr<NeuriteElement>());
297  this->RemoveFromSimulation();
298 
299  mother_->UpdateDependentPhysicalVariables();
300  }

553  void MovePointMass(double speed, const Double3& direction) {
554  // check if is a terminal branch
555  if (daughter_left_ != nullptr) {
556  return;
557  }
558 
559  // scaling for integration step
560  auto* core_param = Simulation::GetActive()->GetParam();
561  double length = speed * core_param->simulation_time_step;
562  auto dir = direction;
563  auto displacement = dir.Normalize() * length;
564  auto new_mass_location = displacement + mass_location_;
565  // here I have to define the actual length ..........
566  auto relative_ml = mother_->OriginOf(Base::GetUid()); // change to auto&&
567  SetSpringAxis(new_mass_location - relative_ml);
568  SetMassLocation(new_mass_location);
569  UpdatePosition();
570  SetActualLength(std::sqrt(spring_axis_ * spring_axis_));
571  // process of elongation : setting tension to 0 increases the resting length
572  SetRestingLengthForDesiredTension(0.0);
573 
574  // some physics and computation obligations....

576  UpdateLocalCoordinateAxis();

635  void operator()(Agent* neighbor, double squared_distance) override {
636  // if neighbor is a NeuriteElement
637  // use shape to determine if neighbor is a NeuriteElement
638  // this is much faster than using a dynamic_cast
639  if (neighbor->GetShape() == Shape::kCylinder) {
640  // if it is a direct relative, or sister branch, we don't take it into
641  // account
642  if (ne->GetDaughterLeft() == neighbor ||
643  ne->GetDaughterRight() == neighbor ||
644  ne->GetMother() ==
645  bdm_static_cast<const NeuriteElement*>(neighbor)->GetMother() ||
646  (ne->GetMother() == neighbor)) {
647  return;
648  }
649  } else if (auto* neighbor_soma =
650  dynamic_cast<const NeuronSoma*>(neighbor)) {
651  // if neighbor is NeuronSoma
652  // if it is a direct relative, we don't take it into account
653  if (ne->GetMother() == neighbor_soma) {
654  return;
655  }
656  }
657 
658  Double4 force_from_neighbor = force->Calculate(ne, neighbor);
659 
660  // hack: if the neighbour is a neurite, we need to reduce the force from
661  // that neighbour in order to avoid kink behaviour
662  if (neighbor->GetShape() == Shape::kCylinder) {
663  force_from_neighbor = force_from_neighbor * h_over_m;
664  has_neurite_neighbor = true;
665  }
666 
667  if (std::abs(force_from_neighbor[3]) <
668  1E-10) { // TODO(neurites) hard coded value
669  // (if all the force is transmitted to the (distal end) point mass)
670  force_from_neighbors[0] += force_from_neighbor[0];
671  force_from_neighbors[1] += force_from_neighbor[1];
672  force_from_neighbors[2] += force_from_neighbor[2];
673  } else {
674  // (if there is a part transmitted to the proximal end)
675  double part_for_point_mass = 1.0 - force_from_neighbor[3];
676  force_from_neighbors[0] += force_from_neighbor[0] * part_for_point_mass;
677  force_from_neighbors[1] += force_from_neighbor[1] * part_for_point_mass;
678  force_from_neighbors[2] += force_from_neighbor[2] * part_for_point_mass;
679  force_on_my_mothers_point_mass[0] +=
680  force_from_neighbor[0] * force_from_neighbor[3];
681  force_on_my_mothers_point_mass[1] +=
682  force_from_neighbor[1] * force_from_neighbor[3];
683  force_on_my_mothers_point_mass[2] +=
684  force_from_neighbor[2] * force_from_neighbor[3];
685  }
686  }
