512  void RetractTerminalEnd(double speed) {
513  // check if is a terminal branch
514  if (daughter_left_[kIdx] != nullptr) {
515  return;
516  }
517  // scaling for integration step
518  auto* param = Simulation_t::GetActive()->GetParam();
519  speed *= param->simulation_time_step_;
520 
521  if (actual_length_[kIdx] > speed + 0.1) {
522  // if actual_length_ > length : retraction keeping the same tension
523  // (putting a limit on how short a branch can be is absolutely necessary
524  // otherwise the tension might explode)
525 
526  double new_actual_length = actual_length_[kIdx] - speed;
527  double factor = new_actual_length / actual_length_[kIdx];
528  actual_length_[kIdx] = new_actual_length;
529  // cf removeproximalCylinder()
530  resting_length_[kIdx] = spring_constant_[kIdx] * actual_length_[kIdx] /
531  (tension_[kIdx] + spring_constant_[kIdx]);
532  spring_axis_[kIdx] = Math::ScalarMult(factor, spring_axis_[kIdx]);
533 
534  mass_location_[kIdx] = Math::Add(
535  mother_[kIdx].OriginOf(Base::GetElementIdx()), spring_axis_[kIdx]);
536  UpdateVolume(); // and update concentration of internal stuff.
537  } else if (mother_[kIdx].IsNeuronSoma()) {
538  mother_[kIdx].RemoveDaughter(GetSoPtr());
539  RemoveFromSimulation();
540  } else if (mother_[kIdx].IsNeuriteElement() &&
541  mother_[kIdx].GetDaughterRight() == nullptr) {
542  // if actual_length_ < length and mother is a neurite element with no
543  // other daughter : merge with mother
544  RemoveProximalNeuriteElement(); // also updates volume_...
545  RetractTerminalEnd(speed / param->simulation_time_step_);
546  } else {
547  // if mother is neurite element with other daughter or is not a neurite
548  // segment: disappear.
549  mother_[kIdx].RemoveDaughter(GetSoPtr());
550  RemoveFromSimulation();
551 
552  mother_[kIdx].UpdateDependentPhysicalVariables();
553  }
554  }

884  template <typename TGrid>
885  std::array<double, 3> CalculateDisplacement(TGrid * grid,
886  double squared_radius) {
887  std::array<double, 3> force_on_my_point_mass{0, 0, 0};
888  std::array<double, 3> force_on_my_mothers_point_mass{0, 0, 0};
889 
890  // 1) Spring force
891  // Only the spring of this cylinder. The daughters spring also act on this
892  // mass, but they are treated in point (2)
893  double factor =
894  -tension_[kIdx] / actual_length_[kIdx]; // the minus sign is important
895  // because the spring axis goes
896  // in the opposite direction
897  force_on_my_point_mass = Math::Add(
898  force_on_my_point_mass, Math::ScalarMult(factor, spring_axis_[kIdx]));
899 
900  // 2) Force transmitted by daugthers (if they exist)
901  if (daughter_left_[kIdx] != nullptr) {
902  auto force_from_daughter =
903  daughter_left_[kIdx]->ForceTransmittedFromDaugtherToMother(
904  GetSoPtr());
905  force_on_my_point_mass =
906  Math::Add(force_on_my_point_mass, force_from_daughter);
907  }
908  if (daughter_right_[kIdx] != nullptr) {
909  auto force_from_daughter =
910  daughter_right_[kIdx]->ForceTransmittedFromDaugtherToMother(
911  GetSoPtr());
912  force_on_my_point_mass =
913  Math::Add(force_on_my_point_mass, force_from_daughter);
914  }
915 
916  std::array<double, 3> force_from_neighbors = {0, 0, 0};
917 
918  auto* param = Simulation_t::GetActive()->GetParam();
919  // this value will be used to reduce force for neurite/neurite interactions
920  double h_over_m = 0.01;
921 
922  // 3) Object avoidance force
923  bool has_neurite_neighbor = false;
924  // (We check for every neighbor object if they touch us, i.e. push us away)
925  auto calculate_neighbor_forces = [this, &force_from_neighbors,
926  &force_on_my_mothers_point_mass,
927  &h_over_m, &has_neurite_neighbor](
928  auto&& neighbor, SoHandle neighbor_handle) {
929  // TODO(lukas) once we switch to C++17 use if constexpr.
930  // As a consequence the reinterpret_cast won't be needed anymore.
931  // if neighbor is a NeuriteElement
932  if (neighbor.template IsSoType<MostDerivedScalar>()) {
933  auto&& neighbor_rc =
934  neighbor.template ReinterpretCast<MostDerivedScalar>();
935  auto n_soptr = neighbor_rc.GetSoPtr();
936  // if it is a direct relative, or sister branch, we don't take it into
937  // account
938  if (n_soptr == this->GetDaughterLeft() ||
939  n_soptr == this->GetDaughterRight() ||
940  (this->GetMother().IsNeuriteElement() &&
941  this->GetMother().GetNeuriteElementSoPtr() == n_soptr) ||
942  n_soptr->GetMother() == this->GetMother()) {
943  return;
944  }
945  } else if (neighbor.template IsSoType<NeuronSoma>()) {
946  // if neighbor is NeuronSoma
947  // if it is a direct relative, we don't take it into account
948  auto&& neighbor_rc = neighbor.template ReinterpretCast<NeuronSoma>();
949  auto n_soptr = neighbor_rc.GetSoPtr();
950  if (this->GetMother().IsNeuronSoma() &&
951  this->GetMother().GetNeuronSomaSoPtr() == n_soptr) {
952  return;
953  }
954  }
955 
956  DefaultForce force;
957  std::array<double, 4> force_from_neighbor =
958  force.GetForce(this, &neighbor);
959 
960  // hack: if the neighbour is a neurite, we need to reduce the force from
961  // that neighbour in order to avoid kink behaviour
962  if (neighbor.template IsSoType<MostDerivedScalar>()) {
963  force_from_neighbor = Math::ScalarMult(h_over_m, force_from_neighbor);
964  has_neurite_neighbor = true;
965  }
966 
967  if (std::abs(force_from_neighbor[3]) <
968  1E-10) { // TODO(neurites) hard coded value
969  // (if all the force is transmitted to the (distal end) point mass)
970  force_from_neighbors[0] += force_from_neighbor[0];
971  force_from_neighbors[1] += force_from_neighbor[1];
972  force_from_neighbors[2] += force_from_neighbor[2];
973  } else {
974  // (if there is a part transmitted to the proximal end)
975  double part_for_point_mass = 1.0 - force_from_neighbor[3];
976  force_from_neighbors[0] += force_from_neighbor[0] * part_for_point_mass;
977  force_from_neighbors[1] += force_from_neighbor[1] * part_for_point_mass;
978  force_from_neighbors[2] += force_from_neighbor[2] * part_for_point_mass;
979  force_on_my_mothers_point_mass[0] +=
980  force_from_neighbor[0] * force_from_neighbor[3];
981  force_on_my_mothers_point_mass[1] +=
982  force_from_neighbor[1] * force_from_neighbor[3];
983  force_on_my_mothers_point_mass[2] +=
984  force_from_neighbor[2] * force_from_neighbor[3];
985  }
986  };
987 
988  grid->ForEachNeighborWithinRadius(calculate_neighbor_forces, *this,
989  GetSoHandle(), squared_radius);
990 
991  // hack: if the neighbour is a neurite, and as we reduced the force from
992  // that neighbour, we also need to reduce my internal force (from internal
993  // tension and daughters)
994  if (has_neurite_neighbor) {
995  force_on_my_point_mass =
996  Math::ScalarMult(h_over_m, force_on_my_point_mass);
997  }
998 
999  force_on_my_point_mass =
1000  Math::Add(force_on_my_point_mass, force_from_neighbors);
1001 
1002  // 5) define the force that will be transmitted to the mother
1003  force_to_transmit_to_proximal_mass_[kIdx] = force_on_my_mothers_point_mass;
1004  // 6.1) Define movement scale
1005  double force_norm = Math::Norm(force_on_my_point_mass);
1006  // 6.2) If is F not strong enough -> no movements
1007  if (force_norm < adherence_[kIdx]) {
1008  return {0, 0, 0};
1009  }
1010 
1011  // So, what follows is only executed if we do actually move :
1012 
1013  // 6.3) Since there's going be a move, we calculate it
1014  auto& displacement = force_on_my_point_mass;
1015  double& displacement_norm = force_norm;
1016 
1017  // 6.4) There is an upper bound for the movement.
1018  if (displacement_norm > param->simulation_max_displacement_) {
1019  displacement = Math::ScalarMult(
1020  param->simulation_max_displacement_ / displacement_norm,
1021  displacement);
1022  }
1023 
1024  return displacement;
1025  }
