259  void RetractTerminalEnd(double speed) {
260  // check if is a terminal branch
261  if (daughter_left_ != nullptr) {
262  return;
263  }
264  // scaling for integration step
265  auto* core_param = Simulation::GetActive()->GetParam();
266  speed *= core_param->simulation_time_step_;
267 
268  auto* mother_soma = dynamic_cast<NeuronSoma*>(mother_.Get());
269  auto* mother_neurite = dynamic_cast<NeuriteElement*>(mother_.Get());
270 
271  if (actual_length_ > speed + 0.1) {
272  // if actual_length_ > length : retraction keeping the same tension
273  // (putting a limit on how short a branch can be is absolutely necessary
274  // otherwise the tension might explode)
275 
276  double new_actual_length = actual_length_ - speed;
277  double factor = new_actual_length / actual_length_;
278  SetActualLength(new_actual_length);
279  // cf removeproximalCylinder()
280  resting_length_ =
281  spring_constant_ * actual_length_ / (tension_ + spring_constant_);
282  SetSpringAxis(spring_axis_ * factor);
283 
284  SetMassLocation(mother_->OriginOf(Base::GetUid()) + spring_axis_);
285  UpdatePosition();
286  UpdateVolume(); // and update concentration of internal stuff.
287  } else if (mother_soma) {
288  mother_->RemoveDaughter(Base::GetSoPtr<NeuriteElement>());
289  this->RemoveFromSimulation();
290  } else if (mother_neurite &&
291  mother_neurite->GetDaughterRight() == nullptr) {
292  // if actual_length_ < length and mother is a neurite element with no
293  // other daughter : merge with mother
294  RemoveProximalNeuriteElement(); // also updates volume_...
295  RetractTerminalEnd(speed / core_param->simulation_time_step_);
296  } else {
297  // if mother is neurite element with other daughter or is not a neurite
298  // segment: disappear.
299  mother_->RemoveDaughter(Base::GetSoPtr<NeuriteElement>());
300  this->RemoveFromSimulation();
301 
302  mother_->UpdateDependentPhysicalVariables();
303  }

568  // check if is a terminal branch
569  if (daughter_left_ != nullptr) {
570  return;
571  }

574  auto* core_param = Simulation::GetActive()->GetParam();

576  auto dir = direction;
577  auto displacement = dir.Normalize() * length;
578  auto new_mass_location = displacement + mass_location_;
579  // here I have to define the actual length ..........
580  auto relative_ml = mother_->OriginOf(Base::GetUid()); // change to auto&&
581  SetSpringAxis(new_mass_location - relative_ml);
582  SetMassLocation(new_mass_location);
583  UpdatePosition();

646  void operator()(const SimObject* neighbor,
647  double squared_distance) override {
648  // if neighbor is a NeuriteElement
649  // use shape to determine if neighbor is a NeuriteElement
650  // this is much faster than using a dynamic_cast
651  if (neighbor->GetShape() == Shape::kCylinder) {
652  // if it is a direct relative, or sister branch, we don't take it into
653  // account
654  if (ne->GetDaughterLeft() == neighbor ||
655  ne->GetDaughterRight() == neighbor ||
656  ne->GetMother() ==
657  bdm_static_cast<const NeuriteElement*>(neighbor)->GetMother() ||
658  (ne->GetMother() == neighbor)) {
659  return;
660  }
661  } else if (auto* neighbor_soma =
662  dynamic_cast<const NeuronSoma*>(neighbor)) {
663  // if neighbor is NeuronSoma
664  // if it is a direct relative, we don't take it into account
665  if (ne->GetMother() == neighbor_soma) {
666  return;
667  }
668  }
669 
670  Double4 force_from_neighbor = force.GetForce(ne, neighbor);
671 
672  // hack: if the neighbour is a neurite, we need to reduce the force from
673  // that neighbour in order to avoid kink behaviour
674  if (neighbor->GetShape() == Shape::kCylinder) {
675  force_from_neighbor = force_from_neighbor * h_over_m;
676  has_neurite_neighbor = true;
677  }
678 
679  if (std::abs(force_from_neighbor[3]) <
680  1E-10) { // TODO(neurites) hard coded value
681  // (if all the force is transmitted to the (distal end) point mass)
682  force_from_neighbors[0] += force_from_neighbor[0];
683  force_from_neighbors[1] += force_from_neighbor[1];
684  force_from_neighbors[2] += force_from_neighbor[2];
685  } else {
686  // (if there is a part transmitted to the proximal end)
687  double part_for_point_mass = 1.0 - force_from_neighbor[3];
688  force_from_neighbors[0] += force_from_neighbor[0] * part_for_point_mass;
689  force_from_neighbors[1] += force_from_neighbor[1] * part_for_point_mass;
690  force_from_neighbors[2] += force_from_neighbor[2] * part_for_point_mass;
691  force_on_my_mothers_point_mass[0] +=
692  force_from_neighbor[0] * force_from_neighbor[3];
693  force_on_my_mothers_point_mass[1] +=
694  force_from_neighbor[1] * force_from_neighbor[3];

705  Double3 CalculateDisplacement(double squared_radius, double dt) override {
706  Double3 force_on_my_point_mass{0, 0, 0};
707  Double3 force_on_my_mothers_point_mass{0, 0, 0};
708 
709  // 1) Spring force
710  // Only the spring of this cylinder. The daughters spring also act on this
711  // mass, but they are treated in point (2)
712  double factor = -tension_ / actual_length_; // the minus sign is important
713  // because the spring axis goes
714  // in the opposite direction
715  force_on_my_point_mass = force_on_my_point_mass + (spring_axis_ * factor);
716 
717  // 2) Force transmitted by daugthers (if they exist)
718  if (daughter_left_ != nullptr) {
719  auto force_from_daughter =
720  daughter_left_->ForceTransmittedFromDaugtherToMother(*this);
721  force_on_my_point_mass = force_on_my_point_mass + force_from_daughter;
722  }
723  if (daughter_right_ != nullptr) {
724  auto force_from_daughter =
725  daughter_right_->ForceTransmittedFromDaugtherToMother(*this);
726  force_on_my_point_mass = force_on_my_point_mass + force_from_daughter;
727  }
728 
729  Double3 force_from_neighbors = {0, 0, 0};
730 
731  auto* core_param = Simulation::GetActive()->GetParam();
732  // this value will be used to reduce force for neurite/neurite interactions
733  double h_over_m = 0.01;
734 
735  // 3) Object avoidance force
736  bool has_neurite_neighbor = false;
737  // (We check for every neighbor object if they touch us, i.e. push us away)
738  auto* ctxt = Simulation::GetActive()->GetExecutionContext();
739  DisplacementFunctor calculate_neighbor_forces(
740  this, force_from_neighbors, force_on_my_mothers_point_mass, h_over_m,
741  has_neurite_neighbor);
742  ctxt->ForEachNeighborWithinRadius(calculate_neighbor_forces, *this,
743  squared_radius);
744  // hack: if the neighbour is a neurite, and as we reduced the force from
745  // that neighbour, we also need to reduce my internal force (from internal
746  // tension and daughters)
747  if (has_neurite_neighbor) {
748  force_on_my_point_mass = force_on_my_point_mass * h_over_m;
749  }
750 
751  force_on_my_point_mass = force_on_my_point_mass + force_from_neighbors;
752 
753  // 5) define the force that will be transmitted to the mother
754  force_to_transmit_to_proximal_mass_ = force_on_my_mothers_point_mass;
755  // 6.1) Define movement scale
756  double force_norm = force_on_my_point_mass.Norm();
757  // 6.2) If is F not strong enough -> no movements
758  if (force_norm < adherence_) {
759  return {0, 0, 0};
760  }
761 
762  // So, what follows is only executed if we do actually move :
763 
764  // 6.3) Since there's going be a move, we calculate it
765  auto& displacement = force_on_my_point_mass;
766  double& displacement_norm = force_norm;
767 
768  // 6.4) There is an upper bound for the movement.
769  if (displacement_norm > core_param->simulation_max_displacement_) {
770  displacement = displacement * (core_param->simulation_max_displacement_ /
771  displacement_norm);
772  }
773  return displacement;
774  }
