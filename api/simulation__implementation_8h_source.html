75 template <typename T>
76 void Simulation<T>::Restore(Simulation<T>&& restored) {
77  // random_
78  if (random_.size() != restored.random_.size()) {
79  Log::Warning("Simulation", "The restore file (", param_->restore_file_,
80  ") was run with a different number of threads. Can't restore "
81  "complete random number generator state.");
82  uint64_t min = std::min(random_.size(), restored.random_.size());
83  for (uint64_t i = 0; i < min; i++) {
84  *(random_[i]) = *(restored.random_[i]);
85  }
86  } else {
87  for (uint64_t i = 0; i < random_.size(); i++) {
88  *(random_[i]) = *(restored.random_[i]);
89  }
90  }
91 
92  // param and rm
93  *param_ = *restored.param_;
94  *rm_ = std::move(*restored.rm_);
95 
96  // name_ and unique_name_
97  InitializeUniqueName(restored.name_);
98  InitializeOutputDir();
99 }

189 template <typename T>
190 template <typename TSetParamLambda>
191 void Simulation<T>::InitializeRuntimeParams(int argc, const char** argv,
192  const TSetParamLambda& set_param) {
193  param_ = new Param_t();
194 
195  // Removing this line causes an unexplainable segfault due to setting the
196  // gErrorIngoreLevel global parameter of ROOT. We need to log at least one
197  // thing before setting that parameter.
198  Log::Info("", "Initialize new simulation using BioDynaMo ",
199  Version::String());
200 
201  // detect if the biodynamo environment has been sourced
202  if (std::getenv("BDM_CMAKE_DIR") == nullptr) {
203  Log::Fatal("Simulation::InitializeRuntimeParams",
204  "The BioDynaMo environment is not set up correctly. Please call "
205  "$use_biodynamo "
206  "and retry this command.");
207  }
208 
209  auto options = bdm::DefaultSimulationOptionParser(argc, argv);
210  constexpr auto kConfigFile = "bdm.toml";
211  constexpr auto kConfigFileParentDir = "../bdm.toml";
212  if (FileExists(kConfigFile)) {
213  auto config = cpptoml::parse_file(kConfigFile);
214  param_->AssignFromConfig(config);
215  } else if (FileExists(kConfigFileParentDir)) {
216  auto config = cpptoml::parse_file(kConfigFileParentDir);
217  param_->AssignFromConfig(config);
218  } else {
219  Log::Warning("Simulation::InitializeRuntimeParams",
220  "Config file %s not found in `.` or `../` directory.",
221  kConfigFile);
222  }
223  if (options.backup_file_ != "") {
224  param_->backup_file_ = options.backup_file_;
225  param_->restore_file_ = options.restore_file_;
226  }
227  set_param(param_);
228 }

243  std::string s(path);
