63 template <typename T>
64 void Simulation<T>::Restore(Simulation<T>&& restored) {
65  // random_
66  if (random_.size() != restored.random_.size()) {
67  Log::Warning("Simulation", "The restore file (", param_->restore_file_,
68  ") was run with a different number of threads. Can't restore "
69  "complete random number generator state.");
70  uint64_t min = std::min(random_.size(), restored.random_.size());
71  for (uint64_t i = 0; i < min; i++) {
72  *(random_[i]) = *(restored.random_[i]);
73  }
74  } else {
75  for (uint64_t i = 0; i < random_.size(); i++) {
76  *(random_[i]) = *(restored.random_[i]);
77  }
78  }
79 
80  // param and rm
81  *param_ = *restored.param_;
82  *rm_ = std::move(*restored.rm_);
83 
84  // name_ and unique_name_
85  InitializeUniqueName(restored.name_);
86  InitializeOutputDir();
87 }

175 template <typename T>
176 void Simulation<T>::InitializeRuntimeParams(int argc, const char** argv) {
177  param_ = new Param();
178 
179  // Removing this line causes an unexplainable segfault due to setting the
180  // gErrorIngoreLevel global parameter of ROOT. We need to log at least one
181  // thing before setting that parameter.
182  Log::Info("", "Initialize new simulation using BioDynaMo ",
183  Version::String());
184 
185  // detect if the biodynamo environment has been sourced
186  if (std::getenv("BDM_CMAKE_DIR") == nullptr) {
187  Log::Fatal("Simulation::InitializeRuntimeParams",
188  "The BioDynaMo environment is not set up correctly. Please call "
189  "$use_biodynamo "
190  "and retry this command.");
191  }
192 
193  auto options = bdm::DefaultSimulationOptionParser(argc, argv);
194  constexpr auto kConfigFile = "bdm.toml";
195  constexpr auto kConfigFileParentDir = "../bdm.toml";
196  if (FileExists(kConfigFile)) {
197  auto config = cpptoml::parse_file(kConfigFile);
198  param_->AssignFromConfig(config);
199  } else if (FileExists(kConfigFileParentDir)) {
200  auto config = cpptoml::parse_file(kConfigFileParentDir);
201  param_->AssignFromConfig(config);
202  } else {
203  Log::Warning("Simulation::InitializeRuntimeParams",
204  "Config file %s not found in `.` or `../` directory.",
205  kConfigFile);
206  }
207  if (options.backup_file_ != "") {
208  param_->backup_file_ = options.backup_file_;
209  param_->restore_file_ = options.restore_file_;
210  }
211 }
