89  void CompileOpenCLKernels() {
90  auto* sim = Simulation::GetActive();
91  auto* param = sim->GetParam();
92  auto* ocl_state = sim->GetOpenCLState();
93 
94  std::vector<cl::Program>* all_programs = ocl_state->GetOpenCLProgramList();
95  cl::Context* context = ocl_state->GetOpenCLContext();
96  std::vector<cl::Device>* devices = ocl_state->GetOpenCLDeviceList();
97  // Compile OpenCL program for found device
98  // TODO(ahmad): create more convenient way to compile all OpenCL kernels, by
99  // going through a list of header files. Also, create a stringifier that
100  // goes from .cl --> .h, since OpenCL kernels must be input as a string here
101  std::string bdmsys = std::getenv("BDMSYS");
102  std::ifstream cl_file(bdmsys +
103  "/include/core/gpu/displacement_op_opencl_kernel.cl");
104  if (cl_file.fail()) {
105  Log::Error("CompileOpenCLKernels", "Kernel file does not exists!");
106  }
107  std::stringstream buffer;
108  buffer << cl_file.rdbuf();
109 
110  cl::Program displacement_op_program(*context, buffer.str());
111 
112  all_programs->push_back(displacement_op_program);
113 
114  Log::Info("", "Compiling OpenCL kernels...");
115 
116  std::string options;
117  if (param->opencl_debug_) {
118  Log::Info("", "Building OpenCL kernels with debugging symbols");
119  options = "-g -O0";
120  } else {
121  Log::Info("", "Building OpenCL kernels without debugging symbols");
122  }
123 
124  for (auto& prog : *all_programs) {
125  try {
126  prog.build(*devices, options.c_str());
127  } catch (const cl::Error&) {
128  Log::Error("CompileOpenCLKernels", "OpenCL compilation error: ",
129  prog.getBuildInfo<CL_PROGRAM_BUILD_LOG>((*devices)[0]));
130  }
131  }
132  }

134  void FindGpuDevicesOpenCL() {
135  try {
136  // We keep the context and device list in the resource manager to be
137  // accessible elsewhere to create command queues and buffers from
138  auto* sim = Simulation::GetActive();
139  auto* param = sim->GetParam();
140  auto* ocl_state = sim->GetOpenCLState();
141 
142  cl::Context* context = ocl_state->GetOpenCLContext();
143  cl::CommandQueue* queue = ocl_state->GetOpenCLCommandQueue();
144  std::vector<cl::Device>* devices = ocl_state->GetOpenCLDeviceList();
145 
146  // Get list of OpenCL platforms.
147  std::vector<cl::Platform> platform;
148  // If we get stuck here, it might be because the DISPLAY envar is not set.
149  // Set it to 0 to avoid getting stuck. It's an AMD specific issue
150  cl::Platform::get(&platform);
151 
152  if (platform.empty()) {
153  Log::Error("FindGpuDevicesOpenCL", "No OpenCL platforms found");
154  }
155 
156  // Go over all available platforms and devices until all compatible
157  // devices are found
158  for (auto p = platform.begin(); p != platform.end(); p++) {
159  std::vector<cl::Device> pldev;
160 
161  try {
162  // Only select GPU's
163  p->getDevices(CL_DEVICE_TYPE_GPU, &pldev);
164 
165  for (auto d = pldev.begin(); d != pldev.end(); d++) {
166  if (!d->getInfo<CL_DEVICE_AVAILABLE>()) // NOLINT
167  continue;
168 
169  // Only select GPU's with double support
170  if (!d->getInfo<
171  CL_DEVICE_PREFERRED_VECTOR_WIDTH_DOUBLE>()) // NOLINT
172  continue;
173 
174  // The OpenCL extension available on this device
175  std::string ext = d->getInfo<CL_DEVICE_EXTENSIONS>();
176 
177  devices->push_back(*d);
178  }
179  } catch (...) {
180  Log::Error("FindGpuDevicesOpenCL",
181  "Found bad OpenCL platform! Continuing to next one");
182  devices->clear();
183  continue;
184  }
185  }
186 
187  if (devices->empty()) {
188  Log::Fatal("FindGpuDevicesOpenCL",
189  "No OpenCL compatible GPU's found on this machine!");
190  return;
191  }
192 
193  *context = cl::Context(*devices);
194 
195  Log::Info("", "Found ", devices->size(),
196  " OpenCL-compatible GPU device(s): ");
197 
198  for (size_t i = 0; i < devices->size(); i++) {
199  Log::Info("", " [", i, "] ", (*devices)[i].getInfo<CL_DEVICE_NAME>());
200  }
201 
202  int selected_gpu = param->preferred_gpu_;
203  Log::Info("", "Selected GPU [", selected_gpu,
204  "]: ", (*devices)[selected_gpu].getInfo<CL_DEVICE_NAME>());
205 
206  // Create command queue for that GPU
207  cl_int queue_err;
208  *queue = cl::CommandQueue(*context, (*devices)[selected_gpu],
209  CL_QUEUE_PROFILING_ENABLE, &queue_err);
210  ClOk(queue_err);
211 
212  // Compile the OpenCL kernels
213  CompileOpenCLKernels();
214  } catch (const cl::Error& err) {
215  Log::Error("FindGpuDevicesOpenCL", "OpenCL error: ", err.what(), "(",
216  err.err(), ")");
217  }
218  }
