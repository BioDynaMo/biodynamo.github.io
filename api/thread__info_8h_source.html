72  void Renew() {
73 // (openmp thread id -> numa node)
74 #pragma omp parallel
75  {
76  int tid = omp_get_thread_num();
77  thread_numa_mapping_[tid] = numa_node_of_cpu(sched_getcpu());
78  }
79 
80  // (numa -> number of associated threads), and
81  // (omp_thread_id -> thread id in numa)
82  for (uint16_t n = 0; n < numa_nodes_; n++) {
83  uint64_t cnt = 0;
84  for (uint64_t t = 0; t < thread_numa_mapping_.size(); t++) {
85  int numa = thread_numa_mapping_[t];
86  if (n == numa) {
87  numa_thread_id_[t] = cnt;
88  cnt++;
89  }
90  }
91  threads_in_numa_[n] = cnt;
92  }
93  }
