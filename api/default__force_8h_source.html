84  template <typename TSphereLhs, typename TSphereRhs,
85  typename TSimulation = Simulation<>>
86  void ForceBetweenSpheres(const TSphereLhs* sphere_lhs,
87  const TSphereRhs* sphere_rhs,
88  std::array<double, 3>* result) const {
89  const std::array<double, 3>& ref_mass_location = sphere_lhs->GetPosition();
90  double ref_diameter = sphere_lhs->GetDiameter();
91  double ref_iof_coefficient = 0.15;
92  const std::array<double, 3>& nb_mass_location = sphere_rhs->GetPosition();
93  double nb_diameter = sphere_rhs->GetDiameter();
94  double nb_iof_coefficient = 0.15;
95 
96  auto c1 = ref_mass_location;
97  double r1 = 0.5 * ref_diameter;
98  auto c2 = nb_mass_location;
99  double r2 = 0.5 * nb_diameter;
100  // We take virtual bigger radii to have a distant interaction, to get a
101  // desired density.
102  double additional_radius =
103  10.0 * std::min(ref_iof_coefficient, nb_iof_coefficient);
104  r1 += additional_radius;
105  r2 += additional_radius;
106  // the 3 components of the vector c2 -> c1
107  double comp1 = c1[0] - c2[0];
108  double comp2 = c1[1] - c2[1];
109  double comp3 = c1[2] - c2[2];
110  double center_distance =
111  std::sqrt(comp1 * comp1 + comp2 * comp2 + comp3 * comp3);
112  // the overlap distance (how much one penetrates in the other)
113  double delta = r1 + r2 - center_distance;
114  // if no overlap : no force
115  if (delta < 0) {
116  *result = {0.0, 0.0, 0.0};
117  return;
118  }
119  // to avoid a division by 0 if the centers are (almost) at the same
120  // location
121  if (center_distance < 0.00000001) {
122  auto* random = TSimulation::GetActive()->GetRandom();
123  auto force2on1 = random->template UniformArray<3>(-3.0, 3.0);
124  *result = force2on1;
125  return;
126  }
127  // the force itself
128  double r = (r1 * r2) / (r1 + r2);
129  double gamma = 1; // attraction coeff
130  double k = 2; // repulsion coeff
131  double f = k * delta - gamma * std::sqrt(r * delta);
132 
133  double module = f / center_distance;
134  std::array<double, 3> force2on1(
135  {module * comp1, module * comp2, module * comp3});
136  *result = force2on1;
137  }

139  template <typename TCylinder, typename TSphere>
140  void ForceOnACylinderFromASphere(const TCylinder* cylinder,
141  const TSphere* sphere,
142  std::array<double, 4>* result) const {
143  auto proximal_end = cylinder->ProximalEnd();
144  auto distal_end = cylinder->DistalEnd();
145  auto axis = cylinder->GetSpringAxis();
146  // TODO(neurites) use cylinder.GetActualLength() ??
147  double actual_length = Math::Norm(axis);
148  double d = cylinder->GetDiameter();
149  auto c = sphere->GetPosition();
150  double r = 0.5 * sphere->GetDiameter();
151 
152  // I. If the cylinder is small with respect to the sphere:
153  // we only consider the interaction between the sphere and the point mass
154  // (i.e. distal point) of the cylinder - that we treat as a sphere.
155  if (actual_length < r) {
156  // move back sphere center by 1 cylinder radius from distal_end
157  // vector_x = rc * (axis[0]/actual_length)
158  // vector_y = rc * (axis[1]/actual_length)
159  // vector_z = rc * (axis[2]/actual_length)
160  double rc = 0.5 * d;
161  std::array<double, 3> dvec = {
162  rc * (axis[0] / actual_length), rc * (axis[1] / actual_length),
163  rc * (axis[2] / actual_length)}; // displacement vector
164  std::array<double, 3> npd = {
165  distal_end[0] - dvec[0], distal_end[1] - dvec[1],
166  distal_end[2] - dvec[2]}; // new sphere center
167  *result = ComputeForceOfASphereOnASphere(npd, rc, c, r);
168  return;
169  }
170 
171  // II. If the cylinder is of the same scale or bigger than the sphere,
172  // we look at the interaction between the sphere and the closest point
173  // (to the sphere center) on the cylinder. This interaction is distributed
174  // to
175  // the two ends of the cylinder: the distal (point mass of the segment) and
176  // the proximal (point mass of the mother of the segment).
177 
178  // 1) Finding cc : the closest point to c on the line proximal_end
179  // proximal_end
180  // ("line" and not "segment")
181  // It is the projection of the vector proximal_end->c onto the vector
182  // proximal_end->distal_end
183  // (=axis)
184  auto proximal_end_closest = Math::Subtract(c, proximal_end);
185 
186  // projection of proximal_end_closest onto axis =
187  // (proximal_end_closest.axis)/norm(axis)^2 * axis
188  // length of the projection = (proximal_end_closest.axis)/norm(axis)
189 
190  double proximal_end_closest_axis = proximal_end_closest[0] * axis[0] +
191  proximal_end_closest[1] * axis[1] +
192  proximal_end_closest[2] * axis[2];
193  double k = proximal_end_closest_axis / (actual_length * actual_length);
194  // cc = proximal_end + k* axis
195  std::array<double, 3> cc{proximal_end[0] + k * axis[0],
196  proximal_end[1] + k * axis[1],
197  proximal_end[2] + k * axis[2]};
198 
199  // 2) Look if c -and hence cc- is (a) between proximal_end and distal_end,
200  // (b) before proximal_end or
201  // (c) after distal_end
202  double proportion_to_proximal_end;
203  if (k <= 1.0 && k >= 0.0) {
204  // a) if cc (the closest point to c on the line pPpD) is between
205  // proximal_end
206  // and distal_end
207  // the force is distributed to the two nodes
208  proportion_to_proximal_end = 1.0 - k;
209  } else if (k < 0) {
210  // b) if the closest point to c on the line pPpD is before
211  // proximal_end
212  // the force is only on the proximal end (the mother point mass)
213  proportion_to_proximal_end = 1.0;
214  cc = proximal_end;
215  } else { // if(k>1)
216  // c) if cc is after distal_end, the force is only on the distal end
217  // (the
218  // segment's point mass).
219  proportion_to_proximal_end = 0.0;
220  cc = distal_end;
221  }
222 
223  // 3) If the smallest distance between the cylinder and the center of the
224  // sphere
225  // is larger than the radius of the two objects , there is no
226  // interaction:
227  double penetration = d / 2 + r - Math::GetL2Distance(c, cc);
228  if (penetration <= 0) {
229  *result = std::array<double, 4>{0.0, 0.0, 0.0, 0.0};
230  return;
231  }
232  auto force = ComputeForceOfASphereOnASphere(cc, d * 0.5, c, r);
233  *result = {force[0], force[1], force[2], proportion_to_proximal_end};
234  return;
235  }

248  template <typename TCylinderLhs, typename TCylinderRhs>
249  void ForceBetweenCylinders(const TCylinderLhs* cylinder1,
250  const TCylinderRhs* cylinder2,
251  std::array<double, 4>* result) const {
252  auto a = cylinder1->ProximalEnd();
253  auto b = cylinder1->GetMassLocation();
254  double d1 = cylinder1->GetDiameter();
255  auto c = cylinder2->ProximalEnd();
256  auto d = cylinder2->GetMassLocation();
257  double d2 = cylinder2->GetDiameter();
258 
259  double k = 0.5; // part devoted to the distal node
260 
261  // looking for closest point on them
262  // (based on http://local.wasp.uwa.edu.au/~pbourke/geometry/lineline3d/)
263  double p13x = a[0] - c[0];
264  double p13y = a[1] - c[1];
265  double p13z = a[2] - c[2];
266  double p43x = d[0] - c[0];
267  double p43y = d[1] - c[1];
268  double p43z = d[2] - c[2];
269  double p21x = b[0] - a[0];
270  double p21y = b[1] - a[1];
271  double p21z = b[2] - a[2];
272 
273  double d1343 = p13x * p43x + p13y * p43y + p13z * p43z;
274  double d4321 = p21x * p43x + p21y * p43y + p21z * p43z;
275  double d1321 = p21x * p13x + p21y * p13y + p21z * p13z;
276  double d4343 = p43x * p43x + p43y * p43y + p43z * p43z;
277  double d2121 = p21x * p21x + p21y * p21y + p21z * p21z;
278 
279  std::array<double, 3> p1, p2;
280 
281  double denom = d2121 * d4343 - d4321 * d4321;
282 
283  // if the two segments are not ABSOLUTLY parallel
284  if (denom > 0.000000000001) {
285  double numer = d1343 * d4321 - d1321 * d4343;
286 
287  double mua = numer / denom;
288  double mub = (d1343 + mua * d4321) / d4343;
289 
290  if (mua < 0) {
291  p1 = a;
292  k = 1;
293  } else if (mua > 1) {
294  p1 = b;
295  k = 0;
296  } else {
297  p1 = std::array<double, 3>{a[0] + mua * p21x, a[1] + mua * p21y,
298  a[2] + mua * p21z};
299  k = 1 - mua;
300  }
301 
302  if (mub < 0) {
303  p2 = c;
304  } else if (mub > 1) {
305  p2 = d;
306  } else {
307  p2 = std::array<double, 3>{c[0] + mub * p43x, c[1] + mub * p43y,
308  c[2] + mub * p43z};
309  }
310 
311  } else {
312  p1 = Math::Add(a, Math::ScalarMult(0.5, Math::Subtract(b, a)));
313  p2 = Math::Add(c, Math::ScalarMult(0.5, Math::Subtract(d, c)));
314  }
315 
316  // W put a virtual sphere on the two cylinders
317  auto force = Math::ScalarMult(
318  10, ComputeForceOfASphereOnASphere(p1, d1 / 2.0, p2, d2 / 2.0));
319 
320  *result = {force[0], force[1], force[2], k};
321  }

323  template <typename TSimulation = Simulation<>>
324  std::array<double, 4> ComputeForceOfASphereOnASphere(
325  const std::array<double, 3>& c1, double r1,
326  const std::array<double, 3>& c2, double r2) const {
327  double comp1 = c1[0] - c2[0];
328  double comp2 = c1[1] - c2[1];
329  double comp3 = c1[2] - c2[2];
330  double distance_between_centers =
331  std::sqrt(comp1 * comp1 + comp2 * comp2 + comp3 * comp3);
332  // the overlap distance (how much one penetrates in the other)
333  double a = r1 + r2 - distance_between_centers;
334  // if no overlap: no force
335  if (a < 0) {
336  return std::array<double, 4>{0.0, 0.0, 0.0, 0.0};
337  }
338  // to avoid a division by 0 if the centers are (almost) at the same location
339  if (distance_between_centers <
340  0.00000001) { // TODO(neurites) hard coded values
341  auto* random = TSimulation::GetActive()->GetRandom();
342  auto force2on1 = random->template UniformArray<3>(-3.0, 3.0);
343  return std::array<double, 4>{force2on1[0], force2on1[1], force2on1[2],
344  0.0};
345  } else {
346  // the force is prop to the square of the interpentration distance and to
347  // the radii.
348  double module = a / distance_between_centers;
349  std::array<double, 4> force2on1(
350  {module * comp1, module * comp2, module * comp3, 0.0});
351  return force2on1;
352  }
353  }
