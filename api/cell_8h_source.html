68  void Initialize(const NewAgentEvent& event) override {
69  Base::Initialize(event);
70 
71  if (event.GetUid() == CellDivisionEvent::kUid) {
72  const auto& cdevent = static_cast<const CellDivisionEvent&>(event);
73  auto* mother_cell = bdm_static_cast<Cell*>(event.existing_agent);
74  auto* daughter = this; // FIXME
75  // A) Defining some values
76  // ..................................................................
77  // defining the two radii s.t total volume is conserved
78  // * radius^3 = r1^3 + r2^3 ;
79  // * volume_ratio = r2^3 / r1^3
80  double radius = mother_cell->GetDiameter() * 0.5;
81 
82  // define an axis for division (along which the nuclei will move)
83  double x_coord = std::cos(cdevent.theta) * std::sin(cdevent.phi);
84  double y_coord = std::sin(cdevent.theta) * std::sin(cdevent.phi);
85  double z_coord = std::cos(cdevent.phi);
86  Double3 coords = {x_coord, y_coord, z_coord};
87  double total_length_of_displacement = radius / 4.0;
88 
89  const auto& x_axis = mother_cell->kXAxis;
90  const auto& y_axis = mother_cell->kYAxis;
91  const auto& z_axis = mother_cell->kZAxis;
92 
93  Double3 axis_of_division =
94  (coords.EntryWiseProduct(x_axis) + coords.EntryWiseProduct(y_axis) +
95  coords.EntryWiseProduct(z_axis)) *
96  total_length_of_displacement;
97 
98  // two equations for the center displacement :
99  // 1) d2/d1= v2/v1 = volume_ratio (each sphere is shifted inver.
100  // proportionally to its volume)
101  // 2) d1 + d2 = TOTAL_LENGTH_OF_DISPLACEMENT
102  double d_2 = total_length_of_displacement / (cdevent.volume_ratio + 1);
103  double d_1 = total_length_of_displacement - d_2;
104 
105  double mother_volume = mother_cell->GetVolume();
106  double new_volume = mother_volume / (cdevent.volume_ratio + 1);
107  daughter->SetVolume(mother_volume - new_volume);
108 
109  // position
110  auto mother_pos = mother_cell->GetPosition();
111  auto new_position = mother_pos + (axis_of_division * d_2);
112  daughter->SetPosition(new_position);
113 
114  // E) This sphere becomes the 1st daughter
115  // move these cells on opposite direction
116  mother_pos -= axis_of_division * d_1;
117  // update mother here and not in Update method to avoid recomputation
118  mother_cell->SetPosition(mother_pos);
119  mother_cell->SetVolume(new_volume);
120 
121  daughter->SetAdherence(mother_cell->GetAdherence());
122  daughter->SetDensity(mother_cell->GetDensity());
123  // G) TODO(lukas) Copy the intracellular and membrane bound Substances
124  }
125  }

267  Double3 CalculateDisplacement(const InteractionForce* force,
268  double squared_radius, double dt) override {
269  // Basically, the idea is to make the sum of all the forces acting
270  // on the Point mass. It is stored in translationForceOnPointMass.
271  // There is also a computation of the torque (only applied
272  // by the daughter neurites), stored in rotationForce.
273 
274  // TODO(roman) : There might be a problem, in the sense that the biology
275  // is not applied if the total Force is smaller than adherence.
276  // Once, I should look at this more carefully.
277 
278  // fixme why? copying
279  const auto& tf = GetTractorForce();
280 
281  // the 3 types of movement that can occur
282  // bool biological_translation = false;
283  bool physical_translation = false;
284  // bool physical_rotation = false;
285 
286  double h = dt;
287  Double3 movement_at_next_step{0, 0, 0};
288 
289  // BIOLOGY :
290  // 0) Start with tractor force : What the biology defined as active
291  // movement------------
292  movement_at_next_step += tf * h;
293 
294  // PHYSICS
295  // the physics force to move the point mass
296 
297  // the physics force to rotate the cell
298  // Double3 rotation_force { 0, 0, 0 };
299 
300  // 1) "artificial force" to maintain the sphere in the ecm simulation
301  // boundaries--------
302  // 2) Spring force from my neurites (translation and
303  // rotation)--------------------------
304  // 3) Object avoidance force
305  // -----------------------------------------------------------
306  // (We check for every neighbor object if they touch us, i.e. push us
307  // away)
308 
309  MechanicalForcesFunctor calculate_neighbor_forces(force, this);
310  auto* ctxt = Simulation::GetActive()->GetExecutionContext();
311  ctxt->ForEachNeighbor(calculate_neighbor_forces, *this, squared_radius);
312 
313  // 4) PhysicalBonds
314  // How the physics influences the next displacement
315  double norm_of_force =
316  std::sqrt(calculate_neighbor_forces.translation_force_on_point_mass *
317  calculate_neighbor_forces.translation_force_on_point_mass);
318 
319  // is there enough force to :
320  // - make us biologically move (Tractor) :
321  // - break adherence and make us translate ?
322  physical_translation = norm_of_force > GetAdherence();
323 
324  assert(GetMass() != 0 && "The mass of a cell was found to be zero!");
325  double mh = h / GetMass();
326  // adding the physics translation (scale by weight) if important enough
327  if (physical_translation) {
328  // We scale the move with mass and time step
329  movement_at_next_step +=
330  calculate_neighbor_forces.translation_force_on_point_mass * mh;
331 
332  // Performing the translation itself :
333  // but we want to avoid huge jumps in the simulation, so there are
334  // maximum distances possible
335  auto* param = Simulation::GetActive()->GetParam();
336  if (norm_of_force * mh > param->simulation_max_displacement) {
337  movement_at_next_step.Normalize();
338  movement_at_next_step *= param->simulation_max_displacement;
339  }
340  }
341  return movement_at_next_step;
342  }
