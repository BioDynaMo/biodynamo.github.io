68  template <typename TMother>
69  CellExt(const CellDivisionEvent& event, TMother* mother, uint64_t new_oid = 0)
70  : Base(event, mother, new_oid) {
71  auto* daughter = ThisMD(); // FIXME
72  // A) Defining some values
73  // ..................................................................
74  // defining the two radii s.t total volume is conserved
75  // * radius^3 = r1^3 + r2^3 ;
76  // * volume_ratio = r2^3 / r1^3
77  double radius = mother->GetDiameter() * 0.5;
78 
79  // define an axis for division (along which the nuclei will move)
80  double x_coord = std::cos(event.theta_) * std::sin(event.phi_);
81  double y_coord = std::sin(event.theta_) * std::sin(event.phi_);
82  double z_coord = std::cos(event.phi_);
83  double total_length_of_displacement = radius / 4.0;
84 
85  const auto x_axis = mother->kXAxis;
86  const auto y_axis = mother->kYAxis;
87  const auto z_axis = mother->kZAxis;
88  std::array<double, 3> axis_of_division{
89  total_length_of_displacement *
90  (x_coord * x_axis[0] + y_coord * y_axis[0] + z_coord * z_axis[0]),
91  total_length_of_displacement *
92  (x_coord * x_axis[1] + y_coord * y_axis[1] + z_coord * z_axis[1]),
93  total_length_of_displacement *
94  (x_coord * x_axis[2] + y_coord * y_axis[2] + z_coord * z_axis[2])};
95 
96  // two equations for the center displacement :
97  // 1) d2/d1= v2/v1 = volume_ratio (each sphere is shifted inver.
98  // proportionally to its volume)
99  // 2) d1 + d2 = TOTAL_LENGTH_OF_DISPLACEMENT
100  double d_2 = total_length_of_displacement / (event.volume_ratio_ + 1);
101  double d_1 = total_length_of_displacement - d_2;
102 
103  double mother_volume = mother->GetVolume();
104  double new_volume = mother_volume / (event.volume_ratio_ + 1);
105  daughter->SetVolume(mother_volume - new_volume);
106 
107  // position
108  auto mother_pos = mother->GetPosition();
109  std::array<double, 3> new_position{
110  mother_pos[0] + d_2 * axis_of_division[0],
111  mother_pos[1] + d_2 * axis_of_division[1],
112  mother_pos[2] + d_2 * axis_of_division[2]};
113  daughter->SetPosition(new_position);
114 
115  // E) This sphere becomes the 1st daughter
116  // move these cells on opposite direction
117  mother_pos[0] -= d_1 * axis_of_division[0];
118  mother_pos[1] -= d_1 * axis_of_division[1];
119  mother_pos[2] -= d_1 * axis_of_division[2];
120  // update mother here and not in EventHandler to avoid recomputation
121  mother->SetPosition(mother_pos);
122  mother->SetVolume(new_volume);
123 
124  daughter->SetAdherence(mother->GetAdherence());
125  daughter->SetDensity(mother->GetDensity());
126  daughter->SetBoxIdx(mother->GetBoxIdx());
127  // G) TODO(lukas) Copy the intracellular and membrane bound Substances
128  }

317 BDM_SO_DEFINE(template <typename TGrid> inline std::array<double, 3>
318  CellExt)::CalculateDisplacement(TGrid* grid,
319  double squared_radius) {
320  // Basically, the idea is to make the sum of all the forces acting
321  // on the Point mass. It is stored in translationForceOnPointMass.
322  // There is also a computation of the torque (only applied
323  // by the daughter neurites), stored in rotationForce.
324 
325  // TODO(roman) : There might be a problem, in the sense that the biology
326  // is not applied if the total Force is smaller than adherence.
327  // Once, I should look at this more carefully.
328 
329  // If we detect enough forces to make us move, we will re-schedule us
330  // setOnTheSchedulerListForPhysicalObjects(false);
331 
332  // fixme why? copying
333  const auto& tf = GetTractorForce();
334 
335  // the 3 types of movement that can occur
336  // bool biological_translation = false;
337  bool physical_translation = false;
338  // bool physical_rotation = false;
339 
340  auto* param = Simulation_t::GetActive()->GetParam();
341  double h = param->simulation_time_step_;
342  std::array<double, 3> movement_at_next_step{0, 0, 0};
343 
344  // BIOLOGY :
345  // 0) Start with tractor force : What the biology defined as active
346  // movement------------
347  movement_at_next_step[0] += h * tf[0];
348  movement_at_next_step[1] += h * tf[1];
349  movement_at_next_step[2] += h * tf[2];
350 
351  // PHYSICS
352  // the physics force to move the point mass
353  std::array<double, 3> translation_force_on_point_mass{0, 0, 0};
354  // the physics force to rotate the cell
355  // std::array<double, 3> rotation_force { 0, 0, 0 };
356 
357  // 1) "artificial force" to maintain the sphere in the ecm simulation
358  // boundaries--------
359  // 2) Spring force from my neurites (translation and
360  // rotation)--------------------------
361  // 3) Object avoidance force
362  // -----------------------------------------------------------
363  // (We check for every neighbor object if they touch us, i.e. push us
364  // away)
365 
366  auto calculate_neighbor_forces = [&, this](auto&& neighbor,
367  auto&& neighbor_handle) {
368  DefaultForce default_force;
369  auto neighbor_force = default_force.GetForce(this, &neighbor);
370  translation_force_on_point_mass[0] += neighbor_force[0];
371  translation_force_on_point_mass[1] += neighbor_force[1];
372  translation_force_on_point_mass[2] += neighbor_force[2];
373  };
374 
375  grid->ForEachNeighborWithinRadius(calculate_neighbor_forces, *this,
376  GetSoHandle(), squared_radius);
377 
378  // 4) PhysicalBonds
379  // How the physics influences the next displacement
380  double norm_of_force = std::sqrt(
381  translation_force_on_point_mass[0] * translation_force_on_point_mass[0] +
382  translation_force_on_point_mass[1] * translation_force_on_point_mass[1] +
383  translation_force_on_point_mass[2] * translation_force_on_point_mass[2]);
384 
385  // is there enough force to :
386  // - make us biologically move (Tractor) :
387  // - break adherence and make us translate ?
388  physical_translation = norm_of_force > GetAdherence();
389 
390  assert(GetMass() != 0 && "The mass of a cell was found to be zero!");
391  double mh = h / GetMass();
392  // adding the physics translation (scale by weight) if important enough
393  if (physical_translation) {
394  // We scale the move with mass and time step
395  movement_at_next_step[0] += translation_force_on_point_mass[0] * mh;
396  movement_at_next_step[1] += translation_force_on_point_mass[1] * mh;
397  movement_at_next_step[2] += translation_force_on_point_mass[2] * mh;
398 
399  // Performing the translation itself :
400 
401  // but we want to avoid huge jumps in the simulation, so there are
402  // maximum distances possible
403  auto* param = Simulation_t::GetActive()->GetParam();
404  if (norm_of_force * mh > param->simulation_max_displacement_) {
405  const auto& norm = Math::Normalize(movement_at_next_step);
406  movement_at_next_step[0] = norm[0] * param->simulation_max_displacement_;
407  movement_at_next_step[1] = norm[1] * param->simulation_max_displacement_;
408  movement_at_next_step[2] = norm[2] * param->simulation_max_displacement_;
409  }
410  }
411  return movement_at_next_step;
412 }

421 BDM_SO_DEFINE(inline std::array<double, 3> CellExt)::
422  TransformCoordinatesGlobalToPolar(const std::array<double, 3>& pos) const {
423  auto vector_to_point = Math::Subtract(pos, position_[kIdx]);
424  std::array<double, 3> local_cartesian{Math::Dot(kXAxis, vector_to_point),
425  Math::Dot(kYAxis, vector_to_point),
426  Math::Dot(kZAxis, vector_to_point)};
427  double radius = std::sqrt(local_cartesian[0] * local_cartesian[0] +
428  local_cartesian[1] * local_cartesian[1] +
429  local_cartesian[2] * local_cartesian[2]);
430  return {radius, std::acos(local_cartesian[2] / radius),
431  std::atan2(local_cartesian[1], local_cartesian[0])};
432 }
