62  : Base(event, mother, new_oid) {
63  const CellDivisionEvent* cdevent =
64  dynamic_cast<const CellDivisionEvent*>(&event);
65  Cell* mother_cell = dynamic_cast<Cell*>(mother);
66  if (cdevent && mother_cell) {
67  auto* daughter = this; // FIXME
68  // A) Defining some values
69  // ..................................................................
70  // defining the two radii s.t total volume is conserved
71  // * radius^3 = r1^3 + r2^3 ;
72  // * volume_ratio = r2^3 / r1^3
73  double radius = mother_cell->GetDiameter() * 0.5;
74 
75  // define an axis for division (along which the nuclei will move)
76  double x_coord = std::cos(cdevent->theta_) * std::sin(cdevent->phi_);
77  double y_coord = std::sin(cdevent->theta_) * std::sin(cdevent->phi_);
78  double z_coord = std::cos(cdevent->phi_);
79  double total_length_of_displacement = radius / 4.0;
80 
81  const auto x_axis = mother_cell->kXAxis;
82  const auto y_axis = mother_cell->kYAxis;
83  const auto z_axis = mother_cell->kZAxis;
84  std::array<double, 3> axis_of_division{
85  total_length_of_displacement *
86  (x_coord * x_axis[0] + y_coord * y_axis[0] + z_coord * z_axis[0]),
87  total_length_of_displacement *
88  (x_coord * x_axis[1] + y_coord * y_axis[1] + z_coord * z_axis[1]),
89  total_length_of_displacement *
90  (x_coord * x_axis[2] + y_coord * y_axis[2] +
91  z_coord * z_axis[2])};
92 
93  // two equations for the center displacement :
94  // 1) d2/d1= v2/v1 = volume_ratio (each sphere is shifted inver.
95  // proportionally to its volume)
96  // 2) d1 + d2 = TOTAL_LENGTH_OF_DISPLACEMENT
97  double d_2 = total_length_of_displacement / (cdevent->volume_ratio_ + 1);
98  double d_1 = total_length_of_displacement - d_2;
99 
100  double mother_volume = mother_cell->GetVolume();
101  double new_volume = mother_volume / (cdevent->volume_ratio_ + 1);
102  daughter->SetVolume(mother_volume - new_volume);
103 
104  // position
105  auto mother_pos = mother_cell->GetPosition();
106  std::array<double, 3> new_position{
107  mother_pos[0] + d_2 * axis_of_division[0],
108  mother_pos[1] + d_2 * axis_of_division[1],
109  mother_pos[2] + d_2 * axis_of_division[2]};
110  daughter->SetPosition(new_position);
111 
112  // E) This sphere becomes the 1st daughter
113  // move these cells on opposite direction
114  mother_pos[0] -= d_1 * axis_of_division[0];
115  mother_pos[1] -= d_1 * axis_of_division[1];
116  mother_pos[2] -= d_1 * axis_of_division[2];
117  // update mother here and not in EventHandler to avoid recomputation
118  mother_cell->SetPosition(mother_pos);
119  mother_cell->SetVolume(new_volume);

276  std::array<double, 3> CalculateDisplacement(double squared_radius) override {
277  // Basically, the idea is to make the sum of all the forces acting
278  // on the Point mass. It is stored in translationForceOnPointMass.
279  // There is also a computation of the torque (only applied
280  // by the daughter neurites), stored in rotationForce.
281 
282  // TODO(roman) : There might be a problem, in the sense that the biology
283  // is not applied if the total Force is smaller than adherence.
284  // Once, I should look at this more carefully.
285 
286  // fixme why? copying
287  const auto& tf = GetTractorForce();
288 
289  // the 3 types of movement that can occur
290  // bool biological_translation = false;
291  bool physical_translation = false;
292  // bool physical_rotation = false;
293 
294  auto* param = Simulation::GetActive()->GetParam();
295  double h = param->simulation_time_step_;
296  std::array<double, 3> movement_at_next_step{0, 0, 0};
297 
298  // BIOLOGY :
299  // 0) Start with tractor force : What the biology defined as active
300  // movement------------
301  movement_at_next_step[0] += h * tf[0];
302  movement_at_next_step[1] += h * tf[1];
303  movement_at_next_step[2] += h * tf[2];
304 
305  // PHYSICS
306  // the physics force to move the point mass
307  std::array<double, 3> translation_force_on_point_mass{0, 0, 0};
308  // the physics force to rotate the cell
309  // std::array<double, 3> rotation_force { 0, 0, 0 };
310 
311  // 1) "artificial force" to maintain the sphere in the ecm simulation
312  // boundaries--------
313  // 2) Spring force from my neurites (translation and
314  // rotation)--------------------------
315  // 3) Object avoidance force
316  // -----------------------------------------------------------
317  // (We check for every neighbor object if they touch us, i.e. push us
318  // away)
319 
320  auto calculate_neighbor_forces = [&, this](const auto* neighbor) {
321  DefaultForce default_force;
322  auto neighbor_force = default_force.GetForce(this, neighbor);
323  translation_force_on_point_mass[0] += neighbor_force[0];
324  translation_force_on_point_mass[1] += neighbor_force[1];
325  translation_force_on_point_mass[2] += neighbor_force[2];
326  };
327 
328  auto* ctxt = Simulation::GetActive()->GetExecutionContext();
329  ctxt->ForEachNeighborWithinRadius(calculate_neighbor_forces, *this,
330  squared_radius);
331 
332  // 4) PhysicalBonds
333  // How the physics influences the next displacement
334  double norm_of_force = std::sqrt(translation_force_on_point_mass[0] *
335  translation_force_on_point_mass[0] +
336  translation_force_on_point_mass[1] *
337  translation_force_on_point_mass[1] +
338  translation_force_on_point_mass[2] *
339  translation_force_on_point_mass[2]);
340 
341  // is there enough force to :
342  // - make us biologically move (Tractor) :
343  // - break adherence and make us translate ?
344  physical_translation = norm_of_force > GetAdherence();
345 
346  assert(GetMass() != 0 && "The mass of a cell was found to be zero!");
347  double mh = h / GetMass();
348  // adding the physics translation (scale by weight) if important enough
349  if (physical_translation) {
350  // We scale the move with mass and time step
351  movement_at_next_step[0] += translation_force_on_point_mass[0] * mh;
352  movement_at_next_step[1] += translation_force_on_point_mass[1] * mh;
353  movement_at_next_step[2] += translation_force_on_point_mass[2] * mh;
354 
355  // Performing the translation itself :
356 
357  // but we want to avoid huge jumps in the simulation, so there are
358  // maximum distances possible
359  auto* param = Simulation::GetActive()->GetParam();
360  if (norm_of_force * mh > param->simulation_max_displacement_) {
361  const auto& norm = Math::Normalize(movement_at_next_step);
362  movement_at_next_step[0] =
363  norm[0] * param->simulation_max_displacement_;
364  movement_at_next_step[1] =
365  norm[1] * param->simulation_max_displacement_;
366  movement_at_next_step[2] =
367  norm[2] * param->simulation_max_displacement_;
368  }
369  }
370  return movement_at_next_step;
371  }
