284  void DivideImpl(MostDerivedSoPtr daughter, double volume_ratio, double phi,
285  double theta) {
286  // A) Defining some values
287  // ..................................................................
288  // defining the two radii s.t total volume is conserved
289  // * radius^3 = r1^3 + r2^3 ;
290  // * volume_ratio = r2^3 / r1^3
291  double radius = diameter_[kIdx] * 0.5;
292 
293  // define an axis for division (along which the nuclei will move)
294  double x_coord = std::cos(theta) * std::sin(phi);
295  double y_coord = std::sin(theta) * std::sin(phi);
296  double z_coord = std::cos(phi);
297  double total_length_of_displacement = radius / 4.0;
298  std::array<double, 3> axis_of_division{
299  total_length_of_displacement *
300  (x_coord * kXAxis[0] + y_coord * kYAxis[0] + z_coord * kZAxis[0]),
301  total_length_of_displacement *
302  (x_coord * kXAxis[1] + y_coord * kYAxis[1] + z_coord * kZAxis[1]),
303  total_length_of_displacement *
304  (x_coord * kXAxis[2] + y_coord * kYAxis[2] + z_coord * kZAxis[2])};
305 
306  // two equations for the center displacement :
307  // 1) d2/d1= v2/v1 = volume_ratio (each sphere is shifted inver.
308  // proportionally to its volume)
309  // 2) d1 + d2 = TOTAL_LENGTH_OF_DISPLACEMENT
310  double d_2 = total_length_of_displacement / (volume_ratio + 1);
311  double d_1 = total_length_of_displacement - d_2;
312 
313  daughter->SetAdherence(adherence_[kIdx]);
314  daughter->SetDensity(density_[kIdx]);
315 
316  double mother_volume = volume_[kIdx];
317  double new_volume = mother_volume / (volume_ratio + 1);
318  daughter->SetVolume(mother_volume - new_volume);
319  SetVolume(new_volume);
320 
321  // position
322  std::array<double, 3> new_position{
323  position_[kIdx][0] + d_2 * axis_of_division[0],
324  position_[kIdx][1] + d_2 * axis_of_division[1],
325  position_[kIdx][2] + d_2 * axis_of_division[2]};
326  daughter->SetPosition(new_position);
327 
328  std::vector<BiologyModules> branch_biology_modules;
329  BiologyModuleEventHandler(gCellDivision, &branch_biology_modules);
330  daughter->SetBiologyModules(std::move(branch_biology_modules));
331 
332  // E) This sphere becomes the 1st daughter
333  // move these cells on opposite direction
334  position_[kIdx][0] -= d_1 * axis_of_division[0];
335  position_[kIdx][1] -= d_1 * axis_of_division[1];
336  position_[kIdx][2] -= d_1 * axis_of_division[2];
337 
338  daughter->SetBoxIdx(box_idx_[kIdx]);
339 
340  // G) TODO(lukas) Copy the intracellular and membrane bound Substances
341  }

352  void BiologyModuleEventHandler(BmEvent event,
353  std::vector<BiologyModules> * destination,
354  bool skip_removal = false) {
355  CopyVisitor<std::vector<BiologyModules>> visitor(event, destination);
356  for (auto& module : biology_modules_[kIdx]) {
357  visit(visitor, module);
358  }
359 
360  if (skip_removal) {
361  return;
362  }
363 
364  RemoveVisitor remove_visitor(event);
365  for (auto it = biology_modules_[kIdx].begin();
366  it != biology_modules_[kIdx].end();) {
367  visit(remove_visitor, *it);
368  if (remove_visitor.return_value_) {
369  it = biology_modules_[kIdx].erase(it);
370  } else {
371  ++it;
372  }
373  }
374  }

396 BDM_SO_DEFINE(template <typename TGrid> inline std::array<double, 3>
397  CellExt)::CalculateDisplacement(TGrid* grid,
398  double squared_radius) {
399  // Basically, the idea is to make the sum of all the forces acting
400  // on the Point mass. It is stored in translationForceOnPointMass.
401  // There is also a computation of the torque (only applied
402  // by the daughter neurites), stored in rotationForce.
403 
404  // TODO(roman) : There might be a problem, in the sense that the biology
405  // is not applied if the total Force is smaller than adherence.
406  // Once, I should look at this more carefully.
407 
408  // If we detect enough forces to make us move, we will re-schedule us
409  // setOnTheSchedulerListForPhysicalObjects(false);
410 
411  // fixme why? copying
412  const auto& tf = GetTractorForce();
413 
414  // the 3 types of movement that can occur
415  // bool biological_translation = false;
416  bool physical_translation = false;
417  // bool physical_rotation = false;
418 
419  auto* param = Simulation_t::GetActive()->GetParam();
420  double h = param->simulation_time_step_;
421  std::array<double, 3> movement_at_next_step{0, 0, 0};
422 
423  // BIOLOGY :
424  // 0) Start with tractor force : What the biology defined as active
425  // movement------------
426  movement_at_next_step[0] += h * tf[0];
427  movement_at_next_step[1] += h * tf[1];
428  movement_at_next_step[2] += h * tf[2];
429 
430  // PHYSICS
431  // the physics force to move the point mass
432  std::array<double, 3> translation_force_on_point_mass{0, 0, 0};
433  // the physics force to rotate the cell
434  // std::array<double, 3> rotation_force { 0, 0, 0 };
435 
436  // 1) "artificial force" to maintain the sphere in the ecm simulation
437  // boundaries--------
438  // 2) Spring force from my neurites (translation and
439  // rotation)--------------------------
440  // 3) Object avoidance force
441  // -----------------------------------------------------------
442  // (We check for every neighbor object if they touch us, i.e. push us
443  // away)
444 
445  auto calculate_neighbor_forces = [&, this](auto&& neighbor,
446  auto&& neighbor_handle) {
447  DefaultForce default_force;
448  auto neighbor_force = default_force.GetForce(this, &neighbor);
449  translation_force_on_point_mass[0] += neighbor_force[0];
450  translation_force_on_point_mass[1] += neighbor_force[1];
451  translation_force_on_point_mass[2] += neighbor_force[2];
452  };
453 
454  grid->ForEachNeighborWithinRadius(calculate_neighbor_forces, *this,
455  GetSoHandle(), squared_radius);
456 
457  // 4) PhysicalBonds
458  // How the physics influences the next displacement
459  double norm_of_force = std::sqrt(
460  translation_force_on_point_mass[0] * translation_force_on_point_mass[0] +
461  translation_force_on_point_mass[1] * translation_force_on_point_mass[1] +
462  translation_force_on_point_mass[2] * translation_force_on_point_mass[2]);
463 
464  // is there enough force to :
465  // - make us biologically move (Tractor) :
466  // - break adherence and make us translate ?
467  physical_translation = norm_of_force > GetAdherence();
468 
469  assert(GetMass() != 0 && "The mass of a cell was found to be zero!");
470  double mh = h / GetMass();
471  // adding the physics translation (scale by weight) if important enough
472  if (physical_translation) {
473  // We scale the move with mass and time step
474  movement_at_next_step[0] += translation_force_on_point_mass[0] * mh;
475  movement_at_next_step[1] += translation_force_on_point_mass[1] * mh;
476  movement_at_next_step[2] += translation_force_on_point_mass[2] * mh;
477 
478  // Performing the translation itself :
479 
480  // but we want to avoid huge jumps in the simulation, so there are
481  // maximum distances possible
482  auto* param = Simulation_t::GetActive()->GetParam();
483  if (norm_of_force * mh > param->simulation_max_displacement_) {
484  const auto& norm = Math::Normalize(movement_at_next_step);
485  movement_at_next_step[0] = norm[0] * param->simulation_max_displacement_;
486  movement_at_next_step[1] = norm[1] * param->simulation_max_displacement_;
487  movement_at_next_step[2] = norm[2] * param->simulation_max_displacement_;
488  }
489  }
490  return movement_at_next_step;
491 }

500 BDM_SO_DEFINE(inline std::array<double, 3> CellExt)::
501  TransformCoordinatesGlobalToPolar(const std::array<double, 3>& pos) const {
502  auto vector_to_point = Math::Subtract(pos, position_[kIdx]);
503  std::array<double, 3> local_cartesian{Math::Dot(kXAxis, vector_to_point),
504  Math::Dot(kYAxis, vector_to_point),
505  Math::Dot(kZAxis, vector_to_point)};
506  double radius = std::sqrt(local_cartesian[0] * local_cartesian[0] +
507  local_cartesian[1] * local_cartesian[1] +
508  local_cartesian[2] * local_cartesian[2]);
509  return {radius, std::acos(local_cartesian[2] / radius),
510  std::atan2(local_cartesian[1], local_cartesian[0])};
511 }
