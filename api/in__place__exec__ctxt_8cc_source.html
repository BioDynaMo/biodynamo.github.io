44 void InPlaceExecutionContext::ThreadSafeAgentUidMap::Insert(
45  const AgentUid& uid,
46  const typename InPlaceExecutionContext::ThreadSafeAgentUidMap::value_type&
47  value) {
48  auto index = uid.GetIndex();
49  auto bidx = index / kBatchSize;
50  if (bidx >= num_batches_) {
51  auto new_size =
52  std::max(index, static_cast<uint32_t>(num_batches_ * kBatchSize * 1.1));
53  Resize(new_size);
54  }
55 
56  auto el_idx = index % kBatchSize;
57  (*batches_.load()[bidx])[el_idx] = value;
58 }

81 void InPlaceExecutionContext::ThreadSafeAgentUidMap::Resize(uint64_t new_size) {
82  using Batch = InPlaceExecutionContext::ThreadSafeAgentUidMap::Batch;
83  auto new_num_batches = new_size / kBatchSize + 1;
84  std::lock_guard<Spinlock> guard(lock_);
85  if (new_num_batches >= num_batches_) {
86  auto bcopy = new Batch*[new_num_batches];
87  auto** non_atomic_batches = batches_.load();
88  for (uint64_t i = 0; i < new_num_batches; ++i) {
89  if (i < num_batches_) {
90  bcopy[i] = non_atomic_batches[i];
91  } else {
92  bcopy[i] = new Batch();
93  bcopy[i]->reserve(kBatchSize);
94  }
95  }
96  batches_.exchange(bcopy);
97  old_copies_.push_back(non_atomic_batches);
98  num_batches_ = new_num_batches;
99  }
100 }

129 void InPlaceExecutionContext::TearDownIterationAll(
130  const std::vector<InPlaceExecutionContext*>& all_exec_ctxts) const {
131  // group execution contexts by numa domain
132  std::vector<uint64_t> new_agent_per_numa(tinfo_->GetNumaNodes());
133  std::vector<uint64_t> thread_offsets(tinfo_->GetMaxThreads());
134 
135  for (int tid = 0; tid < tinfo_->GetMaxThreads(); ++tid) {
136  auto* ctxt = all_exec_ctxts[tid];
137  int nid = tinfo_->GetNumaNode(tid);
138  thread_offsets[tid] = new_agent_per_numa[nid];
139  new_agent_per_numa[nid] += ctxt->new_agents_.size();
140  }
141 
142  // reserve enough memory in ResourceManager
143  std::vector<uint64_t> numa_offsets(tinfo_->GetNumaNodes());
144  auto* rm = Simulation::GetActive()->GetResourceManager();
145  rm->ResizeAgentUidMap();
146  for (unsigned n = 0; n < new_agent_per_numa.size(); n++) {
147  numa_offsets[n] = rm->GrowAgentContainer(new_agent_per_numa[n], n);
148  }
149 
150 // add new_agents_ to the ResourceManager in parallel
151 #pragma omp parallel for schedule(static, 1)
152  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
153  auto* ctxt = all_exec_ctxts[i];
154  int nid = tinfo_->GetNumaNode(i);
155  uint64_t offset = thread_offsets[i] + numa_offsets[nid];
156  rm->AddAgents(nid, offset, ctxt->new_agents_);
157  ctxt->new_agents_.clear();
158  }
159 
160  // remove
161  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
162  auto* ctxt = all_exec_ctxts[i];
163  // removed agents
164  // remove them after adding new ones (maybe one has been removed
165  // that was in new_agents_)
166  for (auto& uid : ctxt->remove_) {
167  rm->RemoveAgent(uid);
168  }
169  ctxt->remove_.clear();
170  }
171 
172  rm->EndOfIteration();
173 
174  new_agent_map_->DeleteOldCopies();
175  if (rm->GetNumAgents() > new_agent_map_->Size()) {
176  new_agent_map_->Resize(rm->GetNumAgents() * 1.5);
177  }
178 }

180 void InPlaceExecutionContext::Execute(
181  Agent* agent, const std::vector<Operation*>& operations) {
182  auto* env = Simulation::GetActive()->GetEnvironment();
183  auto* param = Simulation::GetActive()->GetParam();
184 
185  if (param->thread_safety_mechanism ==
186  Param::ThreadSafetyMechanism::kUserSpecified) {
187  agent->CriticalRegion(&locks);
188  std::sort(locks.begin(), locks.end());
189  for (auto* l : locks) {
190  l->lock();
191  }
192  neighbor_cache_.clear();
193  for (auto& op : operations) {
194  (*op)(agent);
195  }
196  for (auto* l : locks) {
197  l->unlock();
198  }
199  locks.clear();
200  } else if (param->thread_safety_mechanism ==
201  Param::ThreadSafetyMechanism::kAutomatic) {
202  auto* nb_mutex_builder = env->GetNeighborMutexBuilder();
203  auto* mutex = nb_mutex_builder->GetMutex(agent->GetBoxIdx());
204  std::lock_guard<decltype(*mutex)> guard(*mutex);
205  neighbor_cache_.clear();
206  for (auto* op : operations) {
207  (*op)(agent);
208  }
209  } else if (param->thread_safety_mechanism ==
210  Param::ThreadSafetyMechanism::kNone) {
211  neighbor_cache_.clear();
212  for (auto* op : operations) {
213  (*op)(agent);
214  }
215  } else {
216  Log::Fatal("InPlaceExecutionContext::Execute",
217  "Invalid value for parameter thread_safety_mechanism: ",
218  param->thread_safety_mechanism);
219  }
220 }

246 void InPlaceExecutionContext::ForEachNeighbor(
247  Functor<void, const Agent*, double>& lambda, const Agent& query) {
248  // use values in cache
