33 InPlaceExecutionContext::ThreadSafeSoUidMap::~ThreadSafeSoUidMap() {
34  if (map_) {
35  delete map_;
36  }
37  if (next_) {
38  delete next_;
39  }
40  for (auto* map : previous_maps_) {
41  delete map;
42  }
43  previous_maps_.clear();
44 }

46 // NB: There is small risk of a race condition here if another thread grows
47 // the container. In this situation, the element will be inserted in a map
48 // that will be moved to previous_maps_.
49 // This race-condition can be mitigated in `ThreadSafeSoUidMap::operator[]`.
50 void InPlaceExecutionContext::ThreadSafeSoUidMap::Insert(
51  const SoUid& uid,
52  const typename InPlaceExecutionContext::ThreadSafeSoUidMap::value_type&
53  value) {
54  auto index = uid.GetIndex();
55  if (map_->size() > index + ThreadInfo::GetInstance()->GetMaxThreads()) {
56  map_->Insert(uid, value);
57  } else {
58  bool resized = false;
59  {
60  std::lock_guard<Spinlock> guard(lock_);
61  // check again
62  if (map_->size() <= index + ThreadInfo::GetInstance()->GetMaxThreads()) {
63  resized = true;
64  if (next_ == nullptr) {
65  // another thread is still creating a new map
66  // wait until this operation is finished.
67  std::lock_guard<Spinlock> guard(next_lock_);
68  }
69  previous_maps_.emplace_back(map_);
70  map_ = next_;
71  next_ = nullptr;
72  }
73  }
74  if (resized) {
75  std::lock_guard<Spinlock> guard(next_lock_);
76  auto new_size =
77  std::max(uid.GetIndex(), static_cast<uint32_t>(map_->size() * 2));
78  next_ = new ThreadSafeSoUidMap::Map(new_size);
79  }
80  // new map might still be too small.
81  Insert(uid, value);
82  }
83 }

92 const typename InPlaceExecutionContext::ThreadSafeSoUidMap::value_type&
93 InPlaceExecutionContext::ThreadSafeSoUidMap::operator[](const SoUid& uid) {
94  static InPlaceExecutionContext::ThreadSafeSoUidMap::value_type kDefault;
95  auto& pair = (*map_)[uid];
96  auto timesteps = Simulation::GetActive()->GetScheduler()->GetSimulatedSteps();
97 
98  if (pair.second == timesteps && pair.first != nullptr) {
99  return pair;
100  }
101 
102  std::lock_guard<Spinlock> guard(lock_);
103  for (auto* m : previous_maps_) {
104  if (m->size() <= uid.GetIndex()) {
105  continue;
106  }
107  auto& pair = (*m)[uid];
108  if (pair.second == timesteps && pair.first != nullptr) {
109  map_->Insert(uid, pair);
110  return pair;
111  }
112  }
113 
114  return kDefault;
115 }

149 void InPlaceExecutionContext::TearDownIterationAll(
150  const std::vector<InPlaceExecutionContext*>& all_exec_ctxts) const {
151  // group execution contexts by numa domain
152  std::vector<uint64_t> new_so_per_numa(tinfo_->GetNumaNodes());
153  std::vector<uint64_t> thread_offsets(tinfo_->GetMaxThreads());
154 
155  for (int tid = 0; tid < tinfo_->GetMaxThreads(); ++tid) {
156  auto* ctxt = all_exec_ctxts[tid];
157  int nid = tinfo_->GetNumaNode(tid);
158  thread_offsets[tid] = new_so_per_numa[nid];
159  new_so_per_numa[nid] += ctxt->new_sim_objects_.size();
160  }
161 
162  // reserve enough memory in ResourceManager
163  std::vector<uint64_t> numa_offsets(tinfo_->GetNumaNodes());
164  auto* rm = Simulation::GetActive()->GetResourceManager();
165  rm->ResizeUidSohMap();
166  for (unsigned n = 0; n < new_so_per_numa.size(); n++) {
167  numa_offsets[n] = rm->GrowSoContainer(new_so_per_numa[n], n);
168  }
169 
170 // add new_sim_objects_ to the ResourceManager in parallel
171 #pragma omp parallel for schedule(static, 1)
172  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
173  auto* ctxt = all_exec_ctxts[i];
174  int nid = tinfo_->GetNumaNode(i);
175  uint64_t offset = thread_offsets[i] + numa_offsets[nid];
176  rm->AddNewSimObjects(nid, offset, ctxt->new_sim_objects_);
177  ctxt->new_sim_objects_.clear();
178  }
179 
180  // remove
181  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
182  auto* ctxt = all_exec_ctxts[i];
183  // removed sim objects
184  // remove them after adding new ones (maybe one has been removed
185  // that was in new_sim_objects_)
186  for (auto& uid : ctxt->remove_) {
187  rm->Remove(uid);
188  }
189  ctxt->remove_.clear();
190  }
191 
192  rm->EndOfIteration();
193 
194  // FIXME
195  // new_so_map_.SetOffset(SoUidGenerator::Get()->GetLastId());
196  new_so_map_->RemoveOldCopies();
197  if (rm->GetNumSimObjects() > new_so_map_->Size()) {
198  new_so_map_->Resize(rm->GetNumSimObjects() * 1.5);
199  }
200 }

202 void InPlaceExecutionContext::Execute(
203  SimObject* so, const std::vector<Operation*>& operations) {
204  auto* env = Simulation::GetActive()->GetEnvironment();
205  auto* param = Simulation::GetActive()->GetParam();
206 
207  if (param->thread_safety_mechanism_ ==
208  Param::ThreadSafetyMechanism::kUserSpecified) {
209  so->CriticalRegion(&locks);
210  std::sort(locks.begin(), locks.end());
211  for (auto* l : locks) {
212  l->lock();
213  }
214  neighbor_cache_.clear();
215  for (auto& op : operations) {
216  (*op)(so);
217  }
218  for (auto* l : locks) {
219  l->unlock();
220  }
221  locks.clear();
222  } else if (param->thread_safety_mechanism_ ==
223  Param::ThreadSafetyMechanism::kAutomatic) {
224  auto* nb_mutex_builder = env->GetNeighborMutexBuilder();
225  auto* mutex = nb_mutex_builder->GetMutex(so->GetBoxIdx());
226  std::lock_guard<decltype(*mutex)> guard(*mutex);
227  neighbor_cache_.clear();
228  for (auto* op : operations) {
229  (*op)(so);
230  }
231  } else if (param->thread_safety_mechanism_ ==
232  Param::ThreadSafetyMechanism::kNone) {
233  neighbor_cache_.clear();
234  for (auto* op : operations) {
235  (*op)(so);
236  }
237  } else {
238  Log::Fatal("InPlaceExecutionContext::Execute",
239  "Invalid value for parameter thread_safety_mechanism_: ",
240  param->thread_safety_mechanism_);
241  }
242 }

268 void InPlaceExecutionContext::ForEachNeighbor(
269  Functor<void, const SimObject*, double>& lambda, const SimObject& query) {
270  // use values in cache
271  if (neighbor_cache_.size() != 0) {
272  for (auto& pair : neighbor_cache_) {
273  lambda(pair.first, pair.second);
274  }
275  return;
276  }
277 
278  // forward call to env and populate cache
279  auto* env = Simulation::GetActive()->GetEnvironment();
280  ForEachNeighborFunctor for_each(lambda, neighbor_cache_);
281  env->ForEachNeighbor(for_each, query);
