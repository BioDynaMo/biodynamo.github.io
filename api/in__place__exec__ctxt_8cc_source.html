38 void InPlaceExecutionContext::TearDownIterationAll(
39  const std::vector<InPlaceExecutionContext*>& all_exec_ctxts) const {
40  // group execution contexts by numa domain
41  std::vector<uint64_t> new_so_per_numa(tinfo_->GetNumaNodes());
42  std::vector<uint64_t> thread_offsets(tinfo_->GetMaxThreads());
43 
44  for (int tid = 0; tid < tinfo_->GetMaxThreads(); ++tid) {
45  auto* ctxt = all_exec_ctxts[tid];
46  int nid = tinfo_->GetNumaNode(tid);
47  thread_offsets[tid] = new_so_per_numa[nid];
48  new_so_per_numa[nid] += ctxt->new_sim_objects_.size();
49  }
50 
51  // reserve enough memory in ResourceManager
52  std::vector<uint64_t> numa_offsets(tinfo_->GetNumaNodes());
53  auto* rm = Simulation::GetActive()->GetResourceManager();
54  for (unsigned n = 0; n < new_so_per_numa.size(); n++) {
55  numa_offsets[n] = rm->GrowSoContainer(new_so_per_numa[n], n);
56  }
57 
58 // add new_sim_objects_ to the ResourceManager in parallel
59 #pragma omp parallel for schedule(static, 1)
60  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
61  auto* ctxt = all_exec_ctxts[i];
62  int nid = tinfo_->GetNumaNode(i);
63  uint64_t offset = thread_offsets[i] + numa_offsets[nid];
64  rm->AddNewSimObjects(nid, offset, ctxt->new_sim_objects_);
65  ctxt->new_sim_objects_.clear();
66  }
67 
68  // remove
69  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
70  auto* ctxt = all_exec_ctxts[i];
71  // removed sim objects
72  // remove them after adding new ones (maybe one has been removed
73  // that was in new_sim_objects_)
74  for (auto& uid : ctxt->remove_) {
75  rm->Remove(uid);
76  }
77  ctxt->remove_.clear();
78  }
79 }

87  std::lock_guard<decltype(mutex)> guard(mutex);
88  neighbor_cache_.clear();

119 void InPlaceExecutionContext::ForEachNeighbor(
120  const std::function<void(const SimObject*, double)>& lambda,
121  const SimObject& query) {
122  // use values in cache
123  if (neighbor_cache_.size() != 0) {
124  for (auto& pair : neighbor_cache_) {
125  lambda(pair.first, pair.second);
126  }
127  return;
128  }
129 
130  // forward call to grid and populate cache
131  auto* grid = Simulation::GetActive()->GetGrid();
132  auto* param = Simulation::GetActive()->GetParam();
133  auto for_each = [&, this](const SimObject* so, double squared_distance) {
134  if (param->cache_neighbors_) {
135  this->neighbor_cache_.push_back(make_pair(so, squared_distance));
136  }
137  lambda(so, squared_distance);
138  };
139  grid->ForEachNeighbor(for_each, query);
140 }

142 void InPlaceExecutionContext::ForEachNeighborWithinRadius(
143  const std::function<void(const SimObject*)>& lambda, const SimObject& query,
144  double squared_radius) {
145  // use values in cache
146  if (neighbor_cache_.size() != 0) {
147  for (auto& pair : neighbor_cache_) {
148  if (pair.second < squared_radius) {
149  lambda(pair.first);
150  }
151  }
152  return;
153  }

155  // forward call to grid and populate cache
156  auto* grid = Simulation::GetActive()->GetGrid();
157  auto* param = Simulation::GetActive()->GetParam();
158  auto for_each = [&, this](const SimObject* so, double squared_distance) {
159  if (param->cache_neighbors_) {

161  }
162  if (squared_distance < squared_radius) {
163  lambda(so);
164  }
165  };

169 SimObject* InPlaceExecutionContext::GetSimObject(SoUid uid) {

172  return so;
