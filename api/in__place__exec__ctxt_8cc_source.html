44 void InPlaceExecutionContext::ThreadSafeAgentUidMap::Insert(
45  const AgentUid& uid,
46  const typename InPlaceExecutionContext::ThreadSafeAgentUidMap::value_type&
47  value) {
48  auto index = uid.GetIndex();
49  auto bidx = index / kBatchSize;
50  if (bidx >= num_batches_) {
51  auto new_size =
52  std::max(index, static_cast<uint32_t>(num_batches_ * kBatchSize * 1.1));
53  Resize(new_size);
54  }
55 
56  auto el_idx = index % kBatchSize;
57  (*batches_.load()[bidx])[el_idx] = value;
58 }

82 void InPlaceExecutionContext::ThreadSafeAgentUidMap::Resize(uint64_t new_size) {
83  using Batch = InPlaceExecutionContext::ThreadSafeAgentUidMap::Batch;
84  auto new_num_batches = new_size / kBatchSize + 1;
85  std::lock_guard<Spinlock> guard(lock_);
86  if (new_num_batches >= num_batches_) {
87  auto bcopy = new Batch*[new_num_batches];
88  auto** non_atomic_batches = batches_.load();
89  for (uint64_t i = 0; i < new_num_batches; ++i) {
90  if (i < num_batches_) {
91  bcopy[i] = non_atomic_batches[i];
92  } else {
93  bcopy[i] = new Batch();
94  bcopy[i]->reserve(kBatchSize);
95  }
96  }
97  batches_.exchange(bcopy);
98  old_copies_.push_back(non_atomic_batches);
99  num_batches_ = new_num_batches;
100  }
101 }

131 void InPlaceExecutionContext::TearDownIterationAll(
132  const std::vector<InPlaceExecutionContext*>& all_exec_ctxts) {
133  // group execution contexts by numa domain
134  std::vector<uint64_t> new_agent_per_numa(tinfo_->GetNumaNodes());
135  std::vector<uint64_t> thread_offsets(tinfo_->GetMaxThreads());
136 
137  for (int tid = 0; tid < tinfo_->GetMaxThreads(); ++tid) {
138  auto* ctxt = all_exec_ctxts[tid];
139  int nid = tinfo_->GetNumaNode(tid);
140  thread_offsets[tid] = new_agent_per_numa[nid];
141  new_agent_per_numa[nid] += ctxt->new_agents_.size();
142  }
143 
144  // reserve enough memory in ResourceManager
145  std::vector<uint64_t> numa_offsets(tinfo_->GetNumaNodes());
146  auto* rm = Simulation::GetActive()->GetResourceManager();
147  rm->ResizeAgentUidMap();
148  for (unsigned n = 0; n < new_agent_per_numa.size(); n++) {
149  numa_offsets[n] = rm->GrowAgentContainer(new_agent_per_numa[n], n);
150  }
151 
152 // add new_agents_ to the ResourceManager in parallel
153 #pragma omp parallel for schedule(static, 1)
154  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
155  auto* ctxt = all_exec_ctxts[i];
156  int nid = tinfo_->GetNumaNode(i);
157  uint64_t offset = thread_offsets[i] + numa_offsets[nid];
158  rm->AddAgents(nid, offset, ctxt->new_agents_);
159  ctxt->new_agents_.clear();
160  }
161 
162  // remove
163  std::vector<decltype(remove_)*> all_remove(tinfo_->GetMaxThreads());
164 
165  auto num_removals = 0;
166  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
167  auto* ctxt = all_exec_ctxts[i];
168  all_remove[i] = &ctxt->remove_;
169  num_removals += ctxt->remove_.size();
170  }
171 
172  if (num_removals != 0) {
173  rm->RemoveAgents(all_remove);
174 
175  for (int i = 0; i < tinfo_->GetMaxThreads(); i++) {
176  all_exec_ctxts[i]->remove_.clear();
177  }
178  }
179 
180  rm->EndOfIteration();
181 
182  new_agent_map_->DeleteOldCopies();
183  if (rm->GetNumAgents() > new_agent_map_->Size()) {
184  new_agent_map_->Resize(rm->GetNumAgents() * 1.5);
185  }
186 }

188 void InPlaceExecutionContext::Execute(
189  Agent* agent, const std::vector<Operation*>& operations) {
190  auto* env = Simulation::GetActive()->GetEnvironment();
191  auto* param = Simulation::GetActive()->GetParam();
192 
193  if (param->thread_safety_mechanism ==
194  Param::ThreadSafetyMechanism::kUserSpecified) {
195  while (true) {
196  critical_region_.clear();
197  critical_region_2_.clear();
198  locks_.clear();
199  agent->CriticalRegion(&critical_region_);
200  std::sort(critical_region_.begin(), critical_region_.end());
201  for (auto uid : critical_region_) {
202  locks_.push_back(GetAgent(uid)->GetLock());
203  }
204  for (auto* l : locks_) {
205  l->lock();
206  }
207  agent->CriticalRegion(&critical_region_2_);
208  std::sort(critical_region_2_.begin(), critical_region_2_.end());
209  bool same = true;
210  if (critical_region_.size() == critical_region_2_.size()) {
211  for (size_t i = 0; i < critical_region_.size(); ++i) {
212  if (critical_region_[i] != critical_region_2_[i]) {
213  same = false;
214  break;
215  }
216  }
217  } else {
218  same = false;
219  }
220  if (same) {
221  break;
222  }
223  for (auto* l : locks_) {
224  l->unlock();
225  }
226  }
227  neighbor_cache_.clear();
228  cached_squared_search_radius_ = 0;
229  for (auto& op : operations) {
230  (*op)(agent);
231  }
232  for (auto* l : locks_) {
233  l->unlock();
234  }
235  } else if (param->thread_safety_mechanism ==
236  Param::ThreadSafetyMechanism::kAutomatic) {
237  auto* nb_mutex_builder = env->GetNeighborMutexBuilder();
238  auto* mutex = nb_mutex_builder->GetMutex(agent->GetBoxIdx());
239  std::lock_guard<decltype(*mutex)> guard(*mutex);
240  neighbor_cache_.clear();
241  cached_squared_search_radius_ = 0;
242  for (auto* op : operations) {
243  (*op)(agent);
244  }
245  } else if (param->thread_safety_mechanism ==
246  Param::ThreadSafetyMechanism::kNone) {
247  neighbor_cache_.clear();
248  cached_squared_search_radius_ = 0;
249  for (auto* op : operations) {
250  (*op)(agent);
251  }
252  } else {
253  Log::Fatal("InPlaceExecutionContext::Execute",
254  "Invalid value for parameter thread_safety_mechanism: ",
255  param->thread_safety_mechanism);
256  }
257 }

285 bool InPlaceExecutionContext::IsNeighborCacheValid(
286  double query_squared_radius) {
287  if (!cache_neighbors_) {
288  return false;
289  }
290 
291  // A neighbor cache is valid when the cached search radius was greater or
292  // equal than the current search radius (i.e. all the agents within the
293  // current neighbor cache are at least in the cache).
294  // The agents that are between the cached and the current search radius must
295  // be filtered out hereafter
296  return query_squared_radius <= cached_squared_search_radius_;
297 }

299 void InPlaceExecutionContext::ForEachNeighbor(
300  Functor<void, Agent*, double>& lambda, const Agent& query) {
301  // use values in cache
302  auto* env = Simulation::GetActive()->GetEnvironment();
303  auto default_squared_search_radius = env->GetLargestObjectSizeSquared();
304  if (IsNeighborCacheValid(default_squared_search_radius)) {
305  for (auto& pair : neighbor_cache_) {
306  if (pair.second < default_squared_search_radius) {
307  lambda(pair.first, pair.second);
308  }
309  }
310  return;
311  }
312 
313  // forward call to env and populate cache
314  cached_squared_search_radius_ = default_squared_search_radius;
315  ForEachNeighborFunctor for_each(lambda, neighbor_cache_);
316  env->ForEachNeighbor(for_each, query, default_squared_search_radius);
317 }

344 void InPlaceExecutionContext::ForEachNeighbor(
345  Functor<void, Agent*, double>& lambda, const Agent& query,
346  double squared_radius) {
347  // use values in cache
348  if (IsNeighborCacheValid(squared_radius)) {
349  for (auto& pair : neighbor_cache_) {
350  if (pair.second < squared_radius) {
