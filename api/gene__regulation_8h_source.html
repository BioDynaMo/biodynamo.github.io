86 
87  if (param->numerical_ode_solver == Param::NumericalODESolver::kEuler) {
88  // Euler
89  for (uint64_t i = 0; i < first_derivatives_.size(); i++) {
90  double slope = first_derivatives_[i](absolute_time, concentrations_[i]);
91  concentrations_[i] += slope * timestep;
92  }
93  } else if (param->numerical_ode_solver == Param::NumericalODESolver::kRK4) {
94  // Runge-Kutta 4
95  for (uint64_t i = 0; i < first_derivatives_.size(); i++) {
96  double interval_midpoint = absolute_time + timestep / 2.0;
97  double interval_endpoint = absolute_time + timestep;
98 
99  double k1 = first_derivatives_[i](absolute_time, concentrations_[i]);
100  double k2 = first_derivatives_[i](
101  interval_midpoint, concentrations_[i] + timestep * k1 / 2.0);
102  double k3 = first_derivatives_[i](
103  interval_midpoint, concentrations_[i] + timestep * k2 / 2.0);
104  double k4 = first_derivatives_[i](interval_endpoint,
105  concentrations_[i] + timestep * k3);
106 
107  concentrations_[i] += timestep / 6.0 * (k1 + 2 * k2 + 2 * k3 + k4);
108  }
109  }
110  }
