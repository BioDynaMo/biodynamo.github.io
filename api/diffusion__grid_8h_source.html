130  void Update(const std::array<int32_t, 2>& threshold_dimensions) {
131  // Update the grid dimensions such that each dimension ranges from
132  // {treshold_dimensions[0] - treshold_dimensions[1]}
133  auto min_gd = threshold_dimensions[0];
134  auto max_gd = threshold_dimensions[1];
135  grid_dimensions_ = {min_gd, max_gd, min_gd, max_gd, min_gd, max_gd};
136 
137  // If the grid is not perfectly divisible along each dimension by the
138  // box length, extend the grid so that it is
139  int dimension_length = max_gd - min_gd;
140  for (int i = 0; i < 3; i++) {
141  int r = fmod(dimension_length, box_length_);
142  if (r > 1e-9) {
143  // std::abs for the case that box_length_ > dimension_length
144  grid_dimensions_[2 * i + 1] += (box_length_ - r);
145  }
146  }
147 
148  // Calculate by how many boxes each dimension has grown
149  int new_dimension_length = grid_dimensions_[1] - grid_dimensions_[0];
150  int new_num_boxes = std::ceil(new_dimension_length / box_length_);
151  int growth = new_num_boxes - num_boxes_axis_[0];
152 
153  if (growth > 0) {
154  // Store the old number of boxes along each axis for comparison
155  std::array<size_t, 3> tmp_num_boxes_axis = num_boxes_axis_;
156 
157  // Increase number of boxes along axis accordingly
158  num_boxes_axis_[0] += growth;
159  num_boxes_axis_[1] += growth;
160  num_boxes_axis_[2] += growth;
161 
162  // We need to maintain the parity of the number of boxes along each
163  // dimension, otherwise copying of the substances to the increases grid
164  // will not be symmetrically done; resulting in shifting of boxes
165  // We add a box in the negative direction, because the only way the parity
166  // could have changed is because of adding a box in the positive direction
167  // (due to the grid not being perfectly divisible; see above)
168  if (num_boxes_axis_[0] % 2 != parity_) {
169  for (int i = 0; i < 3; i++) {
170  grid_dimensions_[2 * i] -= box_length_;
171  num_boxes_axis_[i]++;
172  }
173  }
174 
175  // Temporarily save previous grid data
176  std::vector<double> tmp_c1 = c1_;
177  std::vector<double> tmp_gradients = gradients_;
178 
179  c1_.clear();
180  c2_.clear();
181  gradients_.clear();
182 
183  total_num_boxes_ =
184  num_boxes_axis_[0] * num_boxes_axis_[1] * num_boxes_axis_[2];
185 
186  CopyOldData(tmp_c1, tmp_gradients, tmp_num_boxes_axis);
187 
188  assert(total_num_boxes_ >= tmp_num_boxes_axis[0] * tmp_num_boxes_axis[1] *
189  tmp_num_boxes_axis[2] &&
190  "The diffusion grid tried to shrink! It can only become larger");
191  }
192  }

254  for (int yy = 0; yy < ny; yy += YBF) {
255  for (int z = 0; z < nz; z++) {
256  // To let the edges bleed we set some diffusion coefficients
257  // to zero. This prevents substance building up at the edges
258  auto dc_2_ = dc_;

260  if (ymax >= ny) {
261  ymax = ny;
262  }

264  dc_2_ = dc_;

266  int c, n, s, b, t;
267  x = 0;
268  c = x + y * nx + z * nx * ny;
269  if (y == 0) {
270  n = c;
271  dc_2_[4] = 0;
272  } else {
273  n = c - nx;
274  }
275  if (y == (ny - 1)) {
276  s = c;
277  dc_2_[3] = 0;
278  } else {
279  s = c + nx;
280  }
281  if (z == 0) {
282  b = c;
283  dc_2_[5] = 0;

302  ++s;

535  void CalculateGradient() {
536  // check if gradient has been calculated once
537  // and if diffusion coefficient and decay constant are 0
538  // i.e. if we don't need to calculate gradient update
539  if (init_gradient_ && IsFixedSubstance()) {
540  return;
541  }
542 
543  double gd = 1 / (box_length_ * 2);
544 
545  auto nx = num_boxes_axis_[0];
546  auto ny = num_boxes_axis_[1];
547  auto nz = num_boxes_axis_[2];
548 
549 #pragma omp parallel for collapse(2)
550  for (size_t z = 0; z < nz; z++) {
551  for (size_t y = 0; y < ny; y++) {
552  for (size_t x = 0; x < nx; x++) {
553  int c, e, w, n, s, b, t;
554  c = x + y * nx + z * nx * ny;
555 
556  if (x == 0) {
557  e = c;
558  w = c + 2;
559  } else if (x == nx - 1) {
560  e = c - 2;
561  w = c;
562  } else {
563  e = c - 1;
564  w = c + 1;
565  }
566 
567  if (y == 0) {
568  n = c + 2 * nx;
569  s = c;
570  } else if (y == ny - 1) {
571  n = c;
572  s = c - 2 * nx;
573  } else {
574  n = c + nx;
575  s = c - nx;
576  }
577 
578  if (z == 0) {
579  t = c + 2 * nx * ny;
580  b = c;
581  } else if (z == nz - 1) {
582  t = c;
583  b = c - 2 * nx * ny;
584  } else {
585  t = c + nx * ny;
586  b = c - nx * ny;
587  }
588

590  gradients_[3 * c + 0] = (c1_[w] - c1_[e]) * gd;
591  gradients_[3 * c + 1] = (c1_[n] - c1_[s]) * gd;
592  gradients_[3 * c + 2] = (c1_[t] - c1_[b]) * gd;

594  }

597  init_gradient_ = true;
598  }
