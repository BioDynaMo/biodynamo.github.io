60  void Initialize(const std::array<int32_t, 6>& grid_dimensions) {
61  // Get grid properties from neighbor grid
62  grid_dimensions_ = grid_dimensions;
63  assert(resolution_ > 0 && "The resolution cannot be zero!");
64 
65  num_boxes_axis_[0] = resolution_;
66  num_boxes_axis_[1] = resolution_;
67  num_boxes_axis_[2] = resolution_;
68 
69  // Example: diffusion grid dimensions from 0-40 and resolution
70  // of 4. Resolution must be adjusted otherwise one data pointer will be
71  // missing.
72  // Without adjustment:
73  // box_length_: 10
74  // data points {0, 10, 20, 30} - 40 will be misssing!
75  // With adjustment
76  // box_length_: 13.3
77  // data points: {0, 13.3, 26.6, 39.9}
78  box_length_ = (grid_dimensions_[1] - grid_dimensions_[0]) /
79  static_cast<double>(resolution_ - 1);
80  ParametersCheck();
81 
82  box_volume_ = box_length_ * box_length_ * box_length_;
83 
84  assert(box_length_ > 0 &&
85  "Box length of diffusion grid must be greater than zero!");
86 
87  // Set the parity of the number of boxes along the dimensions (since all
88  // dimensions are the same, we just take the x-axis here)
89  parity_ = num_boxes_axis_[0] % 2;
90 
91  total_num_boxes_ =
92  num_boxes_axis_[0] * num_boxes_axis_[1] * num_boxes_axis_[2];
93 
94  // Allocate memory for the concentration and gradient arrays
95  c1_.resize(total_num_boxes_);
96  c2_.resize(total_num_boxes_);
97  gradients_.resize(3 * total_num_boxes_);
98 
99  // If we are utilising the Runge-Kutta method we need to resize an
100  // additional vector, this will be used in estimating the concentration
101  // between diffsuion steps.
102  auto* param = Simulation::GetActive()->GetParam();
103  if (param->diffusion_type_ == "RK") {
104  r1_.resize(total_num_boxes_);
105  }
106 
107  initialized_ = true;
108  }

110  void ParametersCheck() {
111  if (((1 - dc_[0]) * dt_) / (box_length_ * box_length_) >= (1.0 / 6)) {
112  Log::Fatal(
113  "DiffusionGrid",
114  "The specified parameters of the diffusion grid with substance [",
115  substance_name_,
116  "] will result in unphysical behavior (diffusion coefficient = ",
117  (1 - dc_[0]), ", resolution = ", resolution_,
118  "). Please refer to the user guide for more information.");
119  } else if (diffusion_step_ == 0) {
120  Log::Fatal("DiffusionGrid",
121  " The specified amount of diffusion steps for the grid with "
122  "substance [",
123  substance_name_,
124  "] is not greater than or equal to 1, "
125  "correct this and run the simulation again.");
126  }

129  void RunInitializers() {
130  assert(num_boxes_axis_[0] > 0 &&
131  "The number of boxes along an axis was found to be zero!");
132  if (initializers_.empty()) {
133  return;
134  }
135 
136  auto nx = num_boxes_axis_[0];
137  auto ny = num_boxes_axis_[1];
138  auto nz = num_boxes_axis_[2];
139 
140  // Apply all functions that initialize this diffusion grid
141  for (size_t f = 0; f < initializers_.size(); f++) {
142  for (size_t x = 0; x < nx; x++) {
143  double real_x = grid_dimensions_[0] + x * box_length_;
144  for (size_t y = 0; y < ny; y++) {
145  double real_y = grid_dimensions_[2] + y * box_length_;
146  for (size_t z = 0; z < nz; z++) {
147  double real_z = grid_dimensions_[4] + z * box_length_;
148  std::array<uint32_t, 3> box_coord;
149  box_coord[0] = x;
150  box_coord[1] = y;
151  box_coord[2] = z;
152  size_t idx = GetBoxIndex(box_coord);
153  IncreaseConcentrationBy(idx,
154  initializers_[f](real_x, real_y, real_z));
155  }
156  }
157  }
158  }
159 
160  // Clear the initializer to free up space
161  initializers_.clear();
162  initializers_.shrink_to_fit();
163  }

172  void Update(const std::array<int32_t, 2>& threshold_dimensions) {
173  // Update the grid dimensions such that each dimension ranges from
174  // {treshold_dimensions[0] - treshold_dimensions[1]}
175  auto min_gd = threshold_dimensions[0];
176  auto max_gd = threshold_dimensions[1];
177  grid_dimensions_ = {min_gd, max_gd, min_gd, max_gd, min_gd, max_gd};
178 
179  // If the grid is not perfectly divisible along each dimension by the
180  // box length, extend the grid so that it is
181  int dimension_length = max_gd - min_gd;
182  for (int i = 0; i < 3; i++) {
183  int r = fmod(dimension_length, box_length_);
184  if (r > 1e-9) {
185  // std::abs for the case that box_length_ > dimension_length
186  grid_dimensions_[2 * i + 1] += (box_length_ - r);
187  }
188  }
189 
190  // Calculate by how many boxes each dimension has grown
191  int new_dimension_length = grid_dimensions_[1] - grid_dimensions_[0];
192  int new_num_boxes = std::ceil(new_dimension_length / box_length_);
193  int growth = new_num_boxes - num_boxes_axis_[0];
194 
195  if (growth > 0) {
196  // Store the old number of boxes along each axis for comparison
197  std::array<size_t, 3> tmp_num_boxes_axis = num_boxes_axis_;
198 
199  // Increase number of boxes along axis accordingly
200  num_boxes_axis_[0] += growth;
201  num_boxes_axis_[1] += growth;
202  num_boxes_axis_[2] += growth;
203 
204  // We need to maintain the parity of the number of boxes along each
205  // dimension, otherwise copying of the substances to the increases grid
206  // will not be symmetrically done; resulting in shifting of boxes
207  // We add a box in the negative direction, because the only way the parity
208  // could have changed is because of adding a box in the positive direction
209  // (due to the grid not being perfectly divisible; see above)
210  if (num_boxes_axis_[0] % 2 != parity_) {
211  for (int i = 0; i < 3; i++) {
212  grid_dimensions_[2 * i] -= box_length_;
213  num_boxes_axis_[i]++;
214  }
215  }
216 
217  // Temporarily save previous grid data
218  auto tmp_c1 = c1_;
219  auto tmp_gradients = gradients_;
220 
221  c1_.clear();
222  c2_.clear();
223  gradients_.clear();
224 
225  total_num_boxes_ =
226  num_boxes_axis_[0] * num_boxes_axis_[1] * num_boxes_axis_[2];
227 
228  CopyOldData(tmp_c1, tmp_gradients, tmp_num_boxes_axis);
229 
230  assert(total_num_boxes_ >= tmp_num_boxes_axis[0] * tmp_num_boxes_axis[1] *
231  tmp_num_boxes_axis[2] &&
232  "The diffusion grid tried to shrink! It can only become larger");
233  }
234 
235  // If we are utilising the Runge-Kutta method we need to resize an
236  // additional vector, this will be used in estimating the concentration
237  // between diffsuion steps.
238  auto* param = Simulation::GetActive()->GetParam();
239  if (param->diffusion_type_ == "RK") {
240  r1_.resize(total_num_boxes_);
241  }
242  }

256  void CopyOldData(const ParallelResizeVector<double>& old_c1,
257  const ParallelResizeVector<double>& old_gradients,
258  const std::array<size_t, 3>& old_num_boxes_axis) {
259  // Allocate more memory for the grid data arrays
260  c1_.resize(total_num_boxes_);
261  c2_.resize(total_num_boxes_);
262  gradients_.resize(3 * total_num_boxes_);
263 
264  auto incr_dim_x = num_boxes_axis_[0] - old_num_boxes_axis[0];
265  auto incr_dim_y = num_boxes_axis_[1] - old_num_boxes_axis[1];
266  auto incr_dim_z = num_boxes_axis_[2] - old_num_boxes_axis[2];
267 
268  int off_x = incr_dim_x / 2;
269  int off_y = incr_dim_y / 2;
270  int off_z = incr_dim_z / 2;
271 
272  int num_box_xy = num_boxes_axis_[0] * num_boxes_axis_[1];
273  int old_box_xy = old_num_boxes_axis[0] * old_num_boxes_axis[1];
274  int new_origin = off_z * (num_boxes_axis_[0] * num_boxes_axis_[1]) +
275  off_y * num_boxes_axis_[0] + off_x;
276  for (size_t k = 0; k < old_num_boxes_axis[2]; k++) {
277  int offset = new_origin + k * num_box_xy;
278  for (size_t j = 0; j < old_num_boxes_axis[1]; j++) {
279  if (j != 0) {
280  offset += num_boxes_axis_[0];
281  }
282  for (size_t i = 0; i < old_num_boxes_axis[0]; i++) {
283  auto idx = k * old_box_xy + j * old_num_boxes_axis[0] + i;
284  c1_[offset + i] = old_c1[idx];
285  gradients_[3 * (offset + i)] = old_gradients[3 * idx];
286  gradients_[3 * (offset + i) + 1] = old_gradients[3 * idx + 1];
287  gradients_[3 * (offset + i) + 2] = old_gradients[3 * idx + 2];
288  }
289  }
290  }

297  void DiffuseWithLeakingEdge() {
298  int nx = num_boxes_axis_[0];
299  int ny = num_boxes_axis_[1];
300  int nz = num_boxes_axis_[2];
301 
302 #define YBF 16
303 #pragma omp parallel for collapse(2)
304  for (int yy = 0; yy < ny; yy += YBF) {
305  for (int z = 0; z < nz; z++) {
306  // To let the edges bleed we set some diffusion coefficients
307  // to zero. This prevents substance building up at the edges
308  auto dc_2_ = dc_;
309  int ymax = yy + YBF;
310  if (ymax >= ny) {
311  ymax = ny;
312  }
313  for (int y = yy; y < ymax; y++) {
314  dc_2_ = dc_;
315  int x;
316  int c, n, s, b, t;
317  x = 0;
318  c = x + y * nx + z * nx * ny;
319  if (y == 0) {
320  n = c;
321  dc_2_[4] = 0;
322  } else {
323  n = c - nx;
324  }
325  if (y == (ny - 1)) {
326  s = c;
327  dc_2_[3] = 0;
328  } else {
329  s = c + nx;
330  }
331  if (z == 0) {
332  b = c;
333  dc_2_[5] = 0;
334  } else {
335  b = c - nx * ny;
336  }
337  if (z == (nz - 1)) {
338  t = c;
339  dc_2_[6] = 0;
340  } else {
341  t = c + nx * ny;
342  }
343  // x = 0; we leak out substances past this edge (so multiply by 0)
344  c2_[c] = (dc_2_[0] * c1_[c] + 0 * c1_[c] + dc_2_[2] * c1_[c + 1] +
345  dc_2_[3] * c1_[s] + dc_2_[4] * c1_[n] + dc_2_[5] * c1_[b] +
346  dc_2_[6] * c1_[t]) *
347  (1 - mu_);
348 #pragma omp simd
349  for (x = 1; x < nx - 1; x++) {
350  ++c;
351  ++n;
352  ++s;
353  ++b;
354  ++t;
355  c2_[c] =
356  (dc_2_[0] * c1_[c] + dc_2_[1] * c1_[c - 1] +
357  dc_2_[2] * c1_[c + 1] + dc_2_[3] * c1_[s] + dc_2_[4] * c1_[n] +
358  dc_2_[5] * c1_[b] + dc_2_[6] * c1_[t]) *
359  (1 - mu_);
360  }
361  ++c;
362  ++n;
363  ++s;
364  ++b;
365  ++t;
366  // x = nx-1; we leak out substances past this edge (so multiply by 0)
367  c2_[c] = (dc_2_[0] * c1_[c] + dc_2_[1] * c1_[c - 1] + 0 * c1_[c] +
368  dc_2_[3] * c1_[s] + dc_2_[4] * c1_[n] + dc_2_[5] * c1_[b] +
369  dc_2_[6] * c1_[t]) *
370  (1 - mu_);
371  } // tile ny
372  } // tile nz
373  } // block ny
374  c1_.swap(c2_);
375  }

433  void DiffuseEuler() {
434  // check if diffusion coefficient and decay constant are 0
435  // i.e. if we don't need to calculate diffusion update
436  if (IsFixedSubstance()) {
437  return;
438  }
439 
440  const auto nx = num_boxes_axis_[0];
441  const auto ny = num_boxes_axis_[1];
442  const auto nz = num_boxes_axis_[2];
443 
444  const double ibl2 = 1 / (box_length_ * box_length_);
445  const double d = 1 - dc_[0];
446 
447 #define YBF 16
448 #pragma omp parallel for collapse(2)
449  for (size_t yy = 0; yy < ny; yy += YBF) {
450  for (size_t z = 0; z < nz; z++) {
451  size_t ymax = yy + YBF;
452  if (ymax >= ny) {
453  ymax = ny;
454  }
455  for (size_t y = yy; y < ymax; y++) {
456  size_t x = 0;
457  int c, n, s, b, t;
458  c = x + y * nx + z * nx * ny;
459 #pragma omp simd
460  for (x = 1; x < nx - 1; x++) {
461  ++c;
462  ++n;
463  ++s;
464  ++b;
465  ++t;
466 
467  if (y == 0 || y == (ny - 1) || z == 0 || z == (nz - 1)) {
468  continue;
469  }
470 
471  n = c - nx;
472  s = c + nx;
473  b = c - nx * ny;
474  t = c + nx * ny;
475  c2_[c] = (c1_[c] +
476  d * dt_ * (c1_[c - 1] - 2 * c1_[c] + c1_[c + 1]) * ibl2 +
477  d * dt_ * (c1_[s] - 2 * c1_[c] + c1_[n]) * ibl2 +
478  d * dt_ * (c1_[b] - 2 * c1_[c] + c1_[t]) * ibl2) *
479  (1 - mu_);
480  }
481  ++c;
482  ++n;
483  ++s;
484  ++b;
485  ++t;
486  } // tile ny
487  } // tile nz
488  } // block ny
489  c1_.swap(c2_);

584  void RK() {
585  // check if diffusion coefficient and decay constant are 0
586  // i.e. if we don't need to calculate diffusion update
587  if (IsFixedSubstance()) {
588  return;
589  }
590 
591  const auto nx = num_boxes_axis_[0];
592  const auto ny = num_boxes_axis_[1];
593  const auto nz = num_boxes_axis_[2];
594 
595  const double ibl2 = 1 / (box_length_ * box_length_);
596  const double d = 1 - dc_[0];
597  double step = diffusion_step_;
598  double h = dt_ / step;
599 #define YBF 16
600  for (size_t i = 0; i < step; i ++) {
601  for (size_t order = 0; order < 2; order++) {
602 #pragma omp parallel for collapse(2)
603  for (size_t yy = 0; yy < ny; yy += YBF) {
604  for (size_t z = 0; z < nz; z++) {
605  size_t ymax = yy + YBF;
606  if (ymax >= ny) {
607  ymax = ny;
608  }
609  for (size_t y = yy; y < ymax; y++) {
610  size_t x = 0;
611  int c, n, s, b, t;
612  c = x + y * nx + z * nx * ny;
613 #pragma omp simd
614  for (x = 1; x < nx - 1; x++) {
615  ++c;
616  ++n;
617  ++s;
618  ++b;
619  ++t;
620 
621  if (y == 0 || y == (ny - 1) || z == 0 || z == (nz - 1)) {
622  continue;
623  }
624 
625  n = c - nx;
626  s = c + nx;
627  b = c - nx * ny;
628  t = c + nx * ny;
629 
630  double h2 = h / 2.0;
631 
632  if (order == 0) {
633  k_[0] = (d * (c1_[c - 1] - 2 * c1_[c] + c1_[c + 1]) * ibl2 +
634  d * (c1_[s] - 2 * c1_[c] + c1_[n]) * ibl2 +
635  d * (c1_[b] - 2 * c1_[c] + c1_[t]) * ibl2);
636  r1_[c] = c1_[c] + (k_[0] * h2);
637  } else if (order == 1) {
638  k_[1] = (d * (c1_[c - 1] - 2 * c1_[c] + c1_[c + 1]) * ibl2 +
639  d * (c1_[s] - 2 * c1_[c] + c1_[n]) * ibl2 +
640  d * (c1_[b] - 2 * c1_[c] + c1_[t]) * ibl2);
641 
642  c2_[c] = c1_[c] + (k_[1] * h);
643  }
644  }
645  ++c;
646  ++n;
647  ++s;
648  ++b;
649  ++t;
650  } // tile ny
651  } // tile nz

658  void RKLeaking() {
659  // check if diffusion coefficient and decay constant are 0
660  // i.e. if we don't need to calculate diffusion update
661  if (IsFixedSubstance()) {
662  return;
663  }
664 
665  const auto nx = num_boxes_axis_[0];
666  const auto ny = num_boxes_axis_[1];
667  const auto nz = num_boxes_axis_[2];
668 
669  const double ibl2 = 1 / (box_length_ * box_length_);
670  const double d = 1 - dc_[0];
671  std::array<int, 6> l;
672 
673  double step = diffusion_step_;
674  double h = dt_ / step;
675 #define YBF 16
676  for (size_t i = 0; i < step; i ++) {
677  for (size_t order = 0; order < 2; order++) {
678 #pragma omp parallel for collapse(2)
679  for (size_t yy = 0; yy < ny; yy += YBF) {
680  for (size_t z = 0; z < nz; z++) {
681  size_t ymax = yy + YBF;
682  if (ymax >= ny) {
683  ymax = ny;
684  }
685  for (size_t y = yy; y < ymax; y++) {
686  size_t x = 0;
687  int c, cm, cp, n, s, b, t;
688  c = x + y * nx + z * nx * ny;
689 
690  l.fill(1);
691 
692  if (y == 0) {
693  n = c;
694  l[2] = 0;
695  } else {
696  n = c - nx;
697  }
698 
699  if (y == ny - 1) {
700  s = c;
701  l[3] = 0;
702  } else {
703  s = c + nx;
704  }
705 
706  if (z == 0) {
707  b = c;
708  l[4] = 0;
709  } else {
710  b = c - nx * ny;
711  }
712 
713  if (z == nz - 1) {
714  t = c;
715  l[5] = 0;
716  } else {
717  t = c + nx * ny;
718  }
719 
720 #pragma omp simd
721  for (x = 1; x < nx - 1; x++) {
722  ++c;
723  ++n;
724  ++s;
725  ++b;
726  ++t;
727 
728  if (x == 0) {
729  cm = c;
730  l[0] = 0;
731  } else {
732  cm = c - 1;
733  }
734 
735  if (y == ny - 1) {
736  cp = c;
737  l[1] = 0;
738  } else {
739  cp = c + 1;
740  }
741 
742  double h2 = h / 2.0;
743 
744  if (order == 0) {
745  k_[0] =
746  d * (l[0] * c1_[cm] - 2 * c1_[c] + l[1] * c1_[cp]) *
747  ibl2 +
748  d * (l[2] * c1_[s] - 2 * c1_[c] + l[3] * c1_[n]) * ibl2 +
749  d * (l[4] * c1_[b] - 2 * c1_[c] + l[5] * c1_[t]) * ibl2;
750  r1_[c] = c1_[c] + (k_[0] * h2);
751  } else if (order == 1) {
752  k_[1] =
753  d * (l[0] * c1_[cm] - 2 * c1_[c] + l[1] * c1_[cp]) *
754  ibl2 +
755  d * (l[2] * c1_[s] - 2 * c1_[c] + l[3] * c1_[n]) * ibl2 +
756  d * (l[4] * c1_[b] - 2 * c1_[c] + l[5] * c1_[t]) * ibl2;
757 
758  c2_[c] = c1_[c] + (k_[1] * h);
759  }
760  }
761  ++c;
762  ++n;
763  ++s;
764  ++b;
765  ++t;
766  } // tile ny
767  } // tile nz
768  } // block ny
769  }
770  c1_.swap(c2_);
771  }

782  void CalculateGradient() {
783  // check if gradient has been calculated once
784  // and if diffusion coefficient and decay constant are 0
785  // i.e. if we don't need to calculate gradient update
786  if (init_gradient_ && IsFixedSubstance()) {
787  return;
788  }
789 
790  double gd = 1 / (box_length_ * 2);
791 
792  auto nx = num_boxes_axis_[0];
793  auto ny = num_boxes_axis_[1];
794  auto nz = num_boxes_axis_[2];
795 
796 #pragma omp parallel for collapse(2)
797  for (size_t z = 0; z < nz; z++) {
798  for (size_t y = 0; y < ny; y++) {
799  for (size_t x = 0; x < nx; x++) {
800  int c, e, w, n, s, b, t;
801  c = x + y * nx + z * nx * ny;
802 
803  if (x == 0) {
804  e = c;
805  w = c + 2;
806  } else if (x == nx - 1) {
807  e = c - 2;
808  w = c;
809  } else {
810  e = c - 1;
811  w = c + 1;
812  }
813 
814  if (y == 0) {
815  n = c + 2 * nx;
816  s = c;
817  } else if (y == ny - 1) {
818  n = c;
819  s = c - 2 * nx;
820  } else {
821  n = c + nx;
822  s = c - nx;
823  }
824 
825  if (z == 0) {
826  t = c + 2 * nx * ny;
827  b = c;
828  } else if (z == nz - 1) {
829  t = c;
830  b = c - 2 * nx * ny;
831  } else {
832  t = c + nx * ny;
833  b = c - nx * ny;
834  }
835 
836  // Let the gradient point from low to high concentration
837  gradients_[3 * c + 0] = (c1_[w] - c1_[e]) * gd;
838  gradients_[3 * c + 1] = (c1_[n] - c1_[s]) * gd;
839  gradients_[3 * c + 2] = (c1_[t] - c1_[b]) * gd;
840  }
841  }
842  }
843  if (!init_gradient_) {
844  init_gradient_ = true;
845  }
846  }
