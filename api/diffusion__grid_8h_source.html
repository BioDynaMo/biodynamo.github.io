114  void Update(const std::array<int32_t, 2>& threshold_dimensions) {
115  // Update the grid dimensions such that each dimension ranges from
116  // {treshold_dimensions[0] - treshold_dimensions[1]}
117  auto min_gd = threshold_dimensions[0];
118  auto max_gd = threshold_dimensions[1];
119  grid_dimensions_ = {min_gd, max_gd, min_gd, max_gd, min_gd, max_gd};
120 
121  // If the grid is not perfectly divisible along each dimension by the
122  // box length, extend the grid so that it is
123  int dimension_length = max_gd - min_gd;
124  for (int i = 0; i < 3; i++) {
125  int r = fmod(dimension_length, box_length_);
126  if (r > 1e-9) {
127  // std::abs for the case that box_length_ > dimension_length
128  grid_dimensions_[2 * i + 1] += (box_length_ - r);
129  }
130  }
131 
132  // Calculate by how many boxes each dimension has grown
133  int new_dimension_length = grid_dimensions_[1] - grid_dimensions_[0];
134  int new_num_boxes = std::ceil(new_dimension_length / box_length_);
135  int growth = new_num_boxes - num_boxes_axis_[0];
136 
137  if (growth > 0) {
138  // Store the old number of boxes along each axis for comparison
139  array<size_t, 3> tmp_num_boxes_axis = num_boxes_axis_;
140 
141  // Increase number of boxes along axis accordingly
142  num_boxes_axis_[0] += growth;
143  num_boxes_axis_[1] += growth;
144  num_boxes_axis_[2] += growth;

147  // dimension, otherwise copying of the substances to the increases grid
148  // will not be symmetrically done; resulting in shifting of boxes
149  // We add a box in the negative direction, because the only way the parity

152  if (num_boxes_axis_[0] % 2 != parity_) {
153  for (int i = 0; i < 3; i++) {
