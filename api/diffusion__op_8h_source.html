46  void operator()() override {
47  // Get active simulation and related pointers
48  auto* sim = Simulation::GetActive();
49  auto* rm = sim->GetResourceManager();
50  auto* env = sim->GetEnvironment();
51  auto* param = sim->GetParam();
52 
53  // Compute the passed time to update the diffusion grid accordingly.
54  double current_time = sim->GetScheduler()->GetSimulatedTime();
55  delta_t_ = current_time - last_time_run_;
56  last_time_run_ = current_time;
57 
58  // Avoid computation if delta_t_ is zero
59  if (delta_t_ == 0.0) {
60  return;
61  }
62 
63  rm->ForEachDiffusionGrid([&](DiffusionGrid* dgrid) {
64  // Update the diffusion grid dimension if the environment dimensions
65  // have changed. If the space is bound, we do not need to update the
66  // dimensions, because these should not be changing anyway
67  if (env->HasGrown() &&
68  param->bound_space == Param::BoundSpaceMode::kOpen) {
69  dgrid->Update();
70  }
71  dgrid->Diffuse(delta_t_);
72  if (param->calculate_gradients) {
73  dgrid->CalculateGradient();
74  }
75  });
76  }
77 };
