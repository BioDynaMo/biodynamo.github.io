35  BDM_OP_HEADER(DiffusionOp);

37  void operator()() override {
38  auto* sim = Simulation::GetActive();
39  auto* rm = sim->GetResourceManager();
40  auto* env = sim->GetEnvironment();
41  auto* param = sim->GetParam();
42 
43  rm->ApplyOnAllDiffusionGrids([&](DiffusionGrid* dg) {
44  // Update the diffusion grid dimension if the environment dimensions
45  // have changed. If the space is bound, we do not need to update the
46  // dimensions, because these should not be changing anyway
47  if (env->HasGrown() && !param->bound_space_) {
48  Log::Info("DiffusionOp",
49  "Your simulation objects are getting near the edge of the "
50  "simulation space. Be aware of boundary conditions that may "
51  "come into play!");
52  dg->Update(env->GetDimensionThresholds());
53  }
54  if (param->diffusion_type_ == "RK") {
55  if (param->leaking_edges_) {
56  dg->RKLeaking();
57  } else {
58  dg->RK();
59  }
60 
61  } else {
62  if (param->leaking_edges_) {
63  dg->DiffuseEulerLeakingEdge();
64  } else {
65  dg->DiffuseEuler();
66  }
67  }
68  if (param->calculate_gradients_) {
69  dg->CalculateGradient();
70  }
71  });
72  }
73 };
