31 // Runs the given `simulation` for `iterations` amount of times` and computes
32 // the mean of the simulated results. If a real (experimental / analytical)
33 // dataset is presented (either as the argument or through a database), we
34 // compute the average error and return it
35 inline double Experiment(
36  Functor<void, Param*, TimeSeries*>& simulation, size_t iterations,
37  const Param* param, TimeSeries* real = nullptr,
38  Functor<void, const std::vector<TimeSeries>&, const TimeSeries&,
39  const TimeSeries&>* post_simulation = nullptr) {
40  // If no experimental / analytical data is given, we try to extract it from
41  // the database
42  bool use_real_data = true;
43  if (!real) {
44  real = &(Database::GetInstance()->data_);
45  // If also no real data is present in the database, we just run the
46  // simulation
47  if (!real) {
48  use_real_data = false;
49  }
50  }
51 
52  // Run the simulation with the input parameters for N iterations
53  std::vector<TimeSeries> results(iterations);
54  for (size_t i = 0; i < iterations; i++) {
55  Param param_copy = *param;
56  simulation(&param_copy, &results[i]);
57  }
58 
59  // Compute the mean result values of the N iterations
60  TimeSeries simulated;
61  TimeSeries::Merge(&simulated, results,
62  [](const std::vector<double> all_y_values, double* y,
63  double* eh, double* el) {
64  *y =
65  TMath::Mean(all_y_values.begin(), all_y_values.end());
66  });
67 
68  // Execute post-simulation lambda (e.g. plotting or exporting of simulated
69  // data)
70  if (post_simulation) {
71  (*post_simulation)(results, simulated, *real);
72  }
73 
74  if (use_real_data) {
75  // Compute and return the error between the real and simulated data
76  double err = TimeSeries::ComputeError(*real, simulated);
77 
78  return err;
79  }
80  return 0.0;
81 }
