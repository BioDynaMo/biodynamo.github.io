38  template <typename TContainer>
39  typename std::enable_if<is_soa_sphere<TContainer>::value>::type operator()(
40  TContainer* cells, uint16_t type_idx) {
41  auto& grid = TGrid::GetInstance();
42 
43  std::vector<std::array<double, 3>> cell_movements(cells->size());
44  std::vector<double> mass(cells->size());
45  std::vector<uint32_t> starts;
46  std::vector<uint16_t> lengths;
47  std::vector<uint32_t> successors(cells->size());
48  uint32_t box_length;
49  uint32_t num_objects = cells->size();
50  std::array<uint32_t, 3> num_boxes_axis;
51  std::array<int32_t, 3> grid_dimensions;
52  double squared_radius =
53  grid.GetLargestObjectSize() * grid.GetLargestObjectSize();
54 
55  // We need to create a mass vector, because it is not stored by default in
56  // a cell container
57  cells->FillMassVector(&mass);
58  grid.GetSuccessors(&successors);
59  grid.GetBoxInfo(&starts, &lengths);
60  grid.GetGridInfo(&box_length, &num_boxes_axis, &grid_dimensions);
61 
62  // If this is the first time we perform physics on GPU using CUDA
63  if (cdo_ == nullptr) {
64  // Allocate 25% more memory so we don't need to reallocate GPU memory
65  // for every (small) change
66  uint32_t new_num_objects = static_cast<uint32_t>(1.25 * num_objects);
67  uint32_t new_num_boxes = static_cast<uint32_t>(1.25 * starts.size());
68 
69  // Store these extende buffer sizes for future reference
70  num_objects_ = new_num_objects;
71  num_boxes_ = new_num_boxes;
72 
73  // Allocate required GPU memory
74  cdo_ = new DisplacementOpCudaKernel(new_num_objects, new_num_boxes);
75  } else {
76  // If the number of simulation objects increased
77  if (num_objects >= num_objects_) {
78  Log::Info("DisplacementOpCuda",
79  "\nThe number of cells increased signficantly (from ",
80  num_objects_, " to ", num_objects,
81  "), so we allocate bigger GPU buffers\n");
82  uint32_t new_num_objects = static_cast<uint32_t>(1.25 * num_objects);
83  num_objects_ = new_num_objects;
84  cdo_->ResizeCellBuffers(new_num_objects);
85  }
86 
87  // If the neighbor grid size increased
88  if (starts.size() >= num_boxes_) {
89  Log::Info("DisplacementOpCuda",
90  "\nThe number of boxes increased signficantly (from ",
91  num_boxes_, " to ", "), so we allocate bigger GPU buffers\n");
92  uint32_t new_num_boxes = static_cast<uint32_t>(1.25 * starts.size());
93  num_boxes_ = new_num_boxes;
94  cdo_->ResizeGridBuffers(new_num_boxes);
95  }
96  }
97 
98  cdo_->LaunchDisplacementKernel(
99  cells->GetPositionPtr(), cells->GetDiameterPtr(),
100  cells->GetTractorForcePtr(), cells->GetAdherencePtr(),
101  cells->GetBoxIdPtr(), mass.data(), &(Param::simulation_time_step_),
102  &(Param::simulation_max_displacement_), &squared_radius, &num_objects,
103  starts.data(), lengths.data(), successors.data(), &box_length,
104  num_boxes_axis.data(), grid_dimensions.data(),
105  cell_movements.data()->data());
106 
107 // set new positions after all updates have been calculated
108 // otherwise some cells would see neighbors with already updated positions
109 // which would lead to inconsistencies
110 #pragma omp parallel for
111  for (size_t i = 0; i < cells->size(); i++) {
112  auto&& cell = (*cells)[i];
113  cell.UpdatePosition(cell_movements[i]);
114  if (Param::bound_space_) {
115  ApplyBoundingBox(&cell, Param::min_bound_, Param::max_bound_);
116  }
117  cell.SetPosition(cell.GetPosition());
118 
119  // Reset biological movement to 0.
120  cell.SetTractorForce({0, 0, 0});
121  }
122  }
