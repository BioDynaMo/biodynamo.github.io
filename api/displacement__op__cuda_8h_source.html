46  private:
47  struct InitializeGPUData;

51  DisplacementOpCuda() {}

54  void operator()() {
55  auto* sim = Simulation::GetActive();
56  auto* grid = dynamic_cast<UniformGridEnvironment*>(sim->GetEnvironment());
57  auto* param = sim->GetParam();
58  auto* rm = sim->GetResourceManager();
59 
60  if (!grid) {
61  Log::Fatal("DisplacementOpCuda::operator()",
62  "DisplacementOpCuda only works with UniformGridEnvironement.");
63  }
64 
65  auto num_numa_nodes = ThreadInfo::GetInstance()->GetNumaNodes();
66  std::vector<SoHandle::ElementIdx_t> offset(num_numa_nodes);
67  offset[0] = 0;
68  for (int nn = 1; nn < num_numa_nodes; nn++) {
69  offset[nn] = offset[nn - 1] + rm->GetNumSimObjects(nn - 1);
70  }
71 
72  uint32_t total_num_objects = rm->GetNumSimObjects();
73 
74  // Cannot use Double3 here, because the `data()` function returns a const
75  // pointer to the underlying array, whereas the CUDA kernel will cast it to
76  // a void pointer. The conversion of `const double *` to `void *` is
77  // illegal.
78  std::vector<std::array<double, 3>> cell_movements;
79  cell_movements.reserve(total_num_objects);
80  std::vector<SoHandle::ElementIdx_t> successors;
81  successors.reserve(total_num_objects);
82  std::vector<uint32_t> starts;
83  std::vector<uint16_t> lengths;
84  std::vector<uint64_t> timestamps;
85  uint32_t box_length;
86  std::array<uint32_t, 3> num_boxes_axis;
87  std::array<int32_t, 3> grid_dimensions;
88  double squared_radius =
89  grid->GetLargestObjectSize() * grid->GetLargestObjectSize();
90 
91  InitializeGPUData f(total_num_objects, offset);
92  rm->ApplyOnAllElementsParallelDynamic(1000, f);
93 
94  // Populate successor list
95  for (int i = 0; i < num_numa_nodes; i++) {
96  for (size_t j = 0; j < rm->GetNumSimObjects(i); j++) {
97  auto idx = offset[i] + j;
98  successors[idx] =
99  offset[i] + grid->successors_.data_[i][j].GetElementIdx();
100  }
101  }
102 
103  uint64_t current_timestamp = grid->timestamp_;
104  auto num_boxes = grid->boxes_.size();
105  starts.reserve(num_boxes);
106  lengths.reserve(num_boxes);
107  timestamps.reserve(num_boxes);
108  size_t i = 0;
109  for (auto& box : grid->boxes_) {
110  timestamps[i] = box.timestamp_;
111  if (box.timestamp_ == current_timestamp) {
112  lengths[i] = box.length_;
113  starts[i] =
114  offset[box.start_.GetNumaNode()] + box.start_.GetElementIdx();
115  }
116  i++;
117  }
118  grid->GetGridInfo(&box_length, &num_boxes_axis, &grid_dimensions);
119 
120  // If this is the first time we perform physics on GPU using CUDA
121  if (cdo_ == nullptr) {
122  // Allocate 25% more memory so we don't need to reallocate GPU memory
123  // for every (small) change
124  total_num_objects_ = static_cast<uint32_t>(1.25 * total_num_objects);
125  num_boxes_ = static_cast<uint32_t>(1.25 * num_boxes);
126 
127  // Allocate required GPU memory
128  cdo_ = new DisplacementOpCudaKernel(total_num_objects_, num_boxes_);
129  } else {
130  // If the number of simulation objects increased
131  if (total_num_objects >= total_num_objects_) {
132  Log::Info("DisplacementOpCuda",
133  "\nThe number of cells increased signficantly (from ",
134  total_num_objects_, " to ", total_num_objects,
135  "), so we allocate bigger GPU buffers\n");
136  total_num_objects_ = static_cast<uint32_t>(1.25 * total_num_objects);
137  cdo_->ResizeCellBuffers(total_num_objects_);
138  }
139 
140  // If the neighbor grid size increased
141  if (starts.size() >= num_boxes_) {
142  Log::Info("DisplacementOpCuda",
143  "\nThe number of boxes increased signficantly (from ",
144  num_boxes_, " to ", "), so we allocate bigger GPU buffers\n");
145  num_boxes_ = static_cast<uint32_t>(1.25 * num_boxes);
146  cdo_->ResizeGridBuffers(num_boxes_);
147  }
148  }
149 
150  cdo_->LaunchDisplacementKernel(
151  f.cell_positions.data()->data(), f.cell_diameters.data(),
152  f.cell_tractor_force.data()->data(), f.cell_adherence.data(),
153  f.cell_boxid.data(), f.mass.data(), &(param->simulation_time_step_),
154  &(param->simulation_max_displacement_), &squared_radius,
155  &total_num_objects, starts.data(), lengths.data(), timestamps.data(),
156  &current_timestamp, successors.data(), &box_length,
157  num_boxes_axis.data(), grid_dimensions.data(),
158  cell_movements.data()->data());
159 
160  // set new positions after all updates have been calculated
161  // otherwise some cells would see neighbors with already updated positions
162  // which would lead to inconsistencies
163 
164  UpdateCPUResults b(&cell_movements, offset);
165  rm->ApplyOnAllElementsParallelDynamic(1000, b);
166  }
167 
168  private:
169  DisplacementOpCudaKernel* cdo_ = nullptr;
170  uint32_t num_boxes_ = 0;
171  uint32_t total_num_objects_ = 0;
172 
173  struct UpdateCPUResults : public Functor<void, SimObject*, SoHandle> {
174  std::vector<std::array<double, 3>>* cell_movements = nullptr;
175  std::vector<SoHandle::ElementIdx_t> offset;
176 
177  UpdateCPUResults(std::vector<std::array<double, 3>>* cm,
178  const std::vector<SoHandle::ElementIdx_t>& offs) {
179  cell_movements = cm;
180  offset = offs;
181  }
182 
183  void operator()(SimObject* so, SoHandle soh) override {
184  auto* param = Simulation::GetActive()->GetParam();
185  auto* cell = dynamic_cast<Cell*>(so);
186  auto idx = offset[soh.GetNumaNode()] + soh.GetElementIdx();
187  Double3 new_pos;
188  new_pos[0] = (*cell_movements)[idx][0];
189  new_pos[1] = (*cell_movements)[idx][1];
190  new_pos[2] = (*cell_movements)[idx][2];
191  cell->UpdatePosition(new_pos);
192  if (param->bound_space_) {
193  ApplyBoundingBox(so, param->min_bound_, param->max_bound_);
194  }
195  }
196  };

198  struct InitializeGPUData : public Functor<void, SimObject*, SoHandle> {
199  bool is_non_spherical_object = false;
200  std::vector<Double3> cell_positions;
201  std::vector<double> cell_diameters;
202  std::vector<double> cell_adherence;
203  std::vector<Double3> cell_tractor_force;

205  std::vector<double> mass;
206  std::vector<SoHandle::ElementIdx_t> offset;
207 
208  InitializeGPUData(uint32_t num_objects,
209  const std::vector<SoHandle::ElementIdx_t>& offs) {
210  cell_positions.reserve(num_objects);
211  cell_diameters.reserve(num_objects);
212  cell_adherence.reserve(num_objects);
213  cell_tractor_force.reserve(num_objects);
214  cell_boxid.reserve(num_objects);
215  mass.reserve(num_objects);
216  offset = offs;
217  }
218 
219  void operator()(SimObject* so, SoHandle soh) override {
220  // Check if there are any non-spherical objects in our simulation, because
221  // GPU accelerations currently supports only sphere-sphere interactions
222  IsNonSphericalObjectPresent(so, &is_non_spherical_object);
223  if (is_non_spherical_object) {
224  Log::Fatal("DisplacementOpCuda",
225  "\nWe detected a non-spherical object during the GPU "
226  "execution. This is currently not supported.");
227  return;
228  }
229  auto* cell = bdm_static_cast<Cell*>(so);
230  auto idx = offset[soh.GetNumaNode()] + soh.GetElementIdx();
231  mass[idx] = cell->GetMass();
232  cell_diameters[idx] = cell->GetDiameter();
233  cell_adherence[idx] = cell->GetAdherence();
234  cell_tractor_force[idx] = cell->GetTractorForce();
235  cell_positions[idx] = cell->GetPosition();
236  cell_boxid[idx] = cell->GetBoxIdx();
237  }
238  };
