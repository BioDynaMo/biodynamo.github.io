20 void ResourceManager::ApplyOnAllElementsParallel(
21  Functor<void, SimObject*, SoHandle>& function) {
22 #pragma omp parallel
23  {
24  auto tid = omp_get_thread_num();
25  auto nid = thread_info_->GetNumaNode(tid);
26  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(nid);
27  auto& numa_sos = sim_objects_[nid];
28  assert(thread_info_->GetNumaNode(tid) == numa_node_of_cpu(sched_getcpu()));
29 
30  // use static scheduling for now
31  auto correction = numa_sos.size() % threads_in_numa == 0 ? 0 : 1;
32  auto chunk = numa_sos.size() / threads_in_numa + correction;
33  auto start = thread_info_->GetNumaThreadId(tid) * chunk;
34  auto end = std::min(numa_sos.size(), start + chunk);
35 
36  for (uint64_t i = start; i < end; ++i) {
37  function(numa_sos[i], SoHandle(nid, i));
38  }
39  }
40 }

56 void ResourceManager::ApplyOnAllElementsParallelDynamic(
57  uint64_t chunk, Functor<void, SimObject*, SoHandle>& function) {
58  // adapt chunk size
59  auto num_so = GetNumSimObjects();
60  uint64_t factor = (num_so / thread_info_->GetMaxThreads()) / chunk;
61  chunk = (num_so / thread_info_->GetMaxThreads()) / (factor + 1);
62  chunk = chunk >= 1 ? chunk : 1;
63 
64  // use dynamic scheduling
65  // Unfortunately openmp's built in functionality can't be used, since
66  // threads belong to different numa domains and thus operate on
67  // different containers
68  auto numa_nodes = thread_info_->GetNumaNodes();
69  auto max_threads = omp_get_max_threads();
70  std::vector<uint64_t> num_chunks_per_numa(numa_nodes);
71  for (int n = 0; n < numa_nodes; n++) {
72  auto correction = sim_objects_[n].size() % chunk == 0 ? 0 : 1;
73  num_chunks_per_numa[n] = sim_objects_[n].size() / chunk + correction;
74  }
75 
76  std::vector<std::atomic<uint64_t>*> counters(max_threads, nullptr);
77  std::vector<uint64_t> max_counters(max_threads);
78  for (int thread_cnt = 0; thread_cnt < max_threads; thread_cnt++) {
79  uint64_t current_nid = thread_info_->GetNumaNode(thread_cnt);
80 
81  auto correction =
82  num_chunks_per_numa[current_nid] %
83  thread_info_->GetThreadsInNumaNode(current_nid) ==
84  0
85  ? 0
86  : 1;
87  uint64_t num_chunks_per_thread =
88  num_chunks_per_numa[current_nid] /
89  thread_info_->GetThreadsInNumaNode(current_nid) +
90  correction;
91  auto start =
92  num_chunks_per_thread * thread_info_->GetNumaThreadId(thread_cnt);
93  auto end = std::min(num_chunks_per_numa[current_nid],
94  start + num_chunks_per_thread);
95 
96  counters[thread_cnt] = new std::atomic<uint64_t>(start);
97  max_counters[thread_cnt] = end;
98  }
99 
100 #pragma omp parallel
101  {
102  auto tid = omp_get_thread_num();
103  auto nid = thread_info_->GetNumaNode(tid);
104 
105  // thread private variables (compilation error with
106  // firstprivate(chunk, numa_node_) with some openmp versions clause)
107  auto p_numa_nodes = thread_info_->GetNumaNodes();
108  auto p_max_threads = omp_get_max_threads();
109  auto p_chunk = chunk;
110  assert(thread_info_->GetNumaNode(tid) == numa_node_of_cpu(sched_getcpu()));
111 
112  // dynamic scheduling
113  uint64_t start = 0;
114  uint64_t end = 0;
115 
116  // this loop implements work stealing from other NUMA nodes if there
117  // are imbalances. Each thread starts with its NUMA domain. Once, it
118  // is finished the thread looks for tasks on other domains
119  for (int n = 0; n < p_numa_nodes; n++) {
120  int current_nid = (nid + n) % p_numa_nodes;
121  for (int thread_cnt = 0; thread_cnt < p_max_threads; thread_cnt++) {
122  uint64_t current_tid = (tid + thread_cnt) % p_max_threads;
123  if (current_nid != thread_info_->GetNumaNode(current_tid)) {
124  continue;
125  }
126 
127  auto& numa_sos = sim_objects_[current_nid];
128  uint64_t old_count = (*(counters[current_tid]))++;
129  while (old_count < max_counters[current_tid]) {
130  start = old_count * p_chunk;
131  end =
132  std::min(static_cast<uint64_t>(numa_sos.size()), start + p_chunk);
133 
134  for (uint64_t i = start; i < end; ++i) {
135  function(numa_sos[i], SoHandle(current_nid, i));
136  }
137 
138  old_count = (*(counters[current_tid]))++;
139  }
140  } // work stealing loop numa_nodes_
141  } // work stealing loop threads
142  }
143 
144  for (auto* counter : counters) {
145  delete counter;
146  }
147 }

166 struct RearrangeFunctor : public Functor<void, const SoHandle&> {
167  std::vector<std::vector<SoHandle>>& sorted_so_handles;
168  const std::vector<uint64_t>& so_per_numa;
169  uint64_t cnt = 0;
170  uint64_t current_numa = 0;
171 
172  RearrangeFunctor(std::vector<std::vector<SoHandle>>& sorted_so_handles,
173  const std::vector<uint64_t>& so_per_numa)
174  : sorted_so_handles(sorted_so_handles), so_per_numa(so_per_numa) {}
175 
176  void operator()(const SoHandle& handle) {
177  if (cnt == so_per_numa[current_numa]) {
178  cnt = 0;
179  current_numa++;
180  }
181 
182  sorted_so_handles[current_numa].push_back(handle);
183  cnt++;
184  }
185 };

187 void ResourceManager::SortAndBalanceNumaNodes() {
188  // balance simulation objects per numa node according to the number of
189  // threads associated with each numa domain
190  auto numa_nodes = thread_info_->GetNumaNodes();
191  std::vector<uint64_t> so_per_numa(numa_nodes);
192  uint64_t cummulative = 0;
193  auto max_threads = thread_info_->GetMaxThreads();
194  for (int n = 1; n < numa_nodes; ++n) {
195  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(n);
196  uint64_t num_so = GetNumSimObjects() * threads_in_numa / max_threads;
197  so_per_numa[n] = num_so;
198  cummulative += num_so;
199  }
200  so_per_numa[0] = GetNumSimObjects() - cummulative;
201 
202  // using first touch policy - page will be allocated to the numa domain of
203  // the thread that accesses it first.
204  // alternative, use numa_alloc_onnode.
205  int ret = numa_run_on_node(0);
206  if (ret != 0) {
207  Log::Fatal("ResourceManager", "Run on numa node failed. Return code: ",
208  ret);
209  }
210 
211  // new data structure for rearranged SimObject*
212  decltype(sim_objects_) so_rearranged;
213  so_rearranged.resize(numa_nodes);
214 
215  // numa node -> vector of SoHandles
216  std::vector<std::vector<SoHandle>> sorted_so_handles;
217  sorted_so_handles.resize(numa_nodes);
218 #pragma omp parallel for
219  for (int n = 0; n < numa_nodes; ++n) {
220  if (thread_info_->GetMyNumaNode() == n &&
221  thread_info_->GetMyNumaThreadId() == 0) {
222  sorted_so_handles[n].reserve(so_per_numa[n]);
223  }
224  }
225 
226  auto* env = Simulation::GetActive()->GetEnvironment();
227  RearrangeFunctor rearrange(sorted_so_handles, so_per_numa);
228  env->IterateZOrder(rearrange);
229 
230  auto* param = Simulation::GetActive()->GetParam();
231  const bool minimize_memory = param->minimize_memory_while_rebalancing_;
232 
233 // create new objects
234 #pragma omp parallel
235  {
236  auto tid = thread_info_->GetMyThreadId();
237  auto nid = thread_info_->GetNumaNode(tid);
238 
239  for (int n = 0; n < numa_nodes; n++) {
240  if (nid != n) {
241  continue;
242  }
243  auto& dest = so_rearranged[n];
244 
245  if (thread_info_->GetNumaThreadId(tid) == 0) {
246  dest.resize(sorted_so_handles[n].size());
247  }
248 
249 #pragma omp barrier
250 
251  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(nid);
252  auto& sohandles = sorted_so_handles[n];
253  assert(thread_info_->GetNumaNode(tid) ==
254  numa_node_of_cpu(sched_getcpu()));
255 
256  // use static scheduling
257  auto correction = sohandles.size() % threads_in_numa == 0 ? 0 : 1;
258  auto chunk = sohandles.size() / threads_in_numa + correction;
259  auto start = thread_info_->GetNumaThreadId(tid) * chunk;
260  auto end = std::min(sohandles.size(), start + chunk);
261 
262  for (uint64_t e = start; e < end; e++) {
263  auto& handle = sohandles[e];
264  auto* so = sim_objects_[handle.GetNumaNode()][handle.GetElementIdx()];
265  dest[e] = so->GetCopy();
266  if (minimize_memory) {
267  delete so;
268  }
269  }
270  }
271  }
272 
273  // delete old objects. This approach has a high chance that a thread
274  // in the right numa node will delete the object, thus minimizing thread
275  // synchronization overheads. The bdm memory allocator does not have this
276  // issue.
277  if (!minimize_memory) {
278  DeleteSimObjectsFunctor delete_functor;
279  ApplyOnAllElementsParallel(delete_functor);
280  }
281 
282  for (int n = 0; n < numa_nodes; n++) {
283  sim_objects_[n].swap(so_rearranged[n]);
284  }
285 
286  // update uid_soh_map_
287  UpdateUidSoHMapFunctor functor(uid_soh_map_);
288  ApplyOnAllElementsParallel(functor);
289 
290  if (Simulation::GetActive()->GetParam()->debug_numa_) {
291  DebugNuma();
292  }
293 }
