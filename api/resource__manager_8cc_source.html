35 void ResourceManager::ForEachAgentParallel(
36  Functor<void, Agent*, AgentHandle>& function) {
37 #pragma omp parallel
38  {
39  auto tid = omp_get_thread_num();
40  auto nid = thread_info_->GetNumaNode(tid);
41  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(nid);
42  auto& numa_agents = agents_[nid];
43  assert(thread_info_->GetNumaNode(tid) == numa_node_of_cpu(sched_getcpu()));
44 
45  // use static scheduling for now
46  auto correction = numa_agents.size() % threads_in_numa == 0 ? 0 : 1;
47  auto chunk = numa_agents.size() / threads_in_numa + correction;
48  auto start = thread_info_->GetNumaThreadId(tid) * chunk;
49  auto end = std::min(numa_agents.size(), start + chunk);
50 
51  for (uint64_t i = start; i < end; ++i) {
52  function(numa_agents[i], AgentHandle(nid, i));
53  }
54  }
55 }

74 void ResourceManager::ForEachAgentParallel(
75  uint64_t chunk, Functor<void, Agent*, AgentHandle>& function) {
76  // adapt chunk size
77  auto num_agents = GetNumAgents();
78  uint64_t factor = (num_agents / thread_info_->GetMaxThreads()) / chunk;
79  chunk = (num_agents / thread_info_->GetMaxThreads()) / (factor + 1);
80  chunk = chunk >= 1 ? chunk : 1;
81 
82  // use dynamic scheduling
83  // Unfortunately openmp's built in functionality can't be used, since
84  // threads belong to different numa domains and thus operate on
85  // different containers
86  auto numa_nodes = thread_info_->GetNumaNodes();
87  auto max_threads = omp_get_max_threads();
88  std::vector<uint64_t> num_chunks_per_numa(numa_nodes);
89  for (int n = 0; n < numa_nodes; n++) {
90  auto correction = agents_[n].size() % chunk == 0 ? 0 : 1;
91  num_chunks_per_numa[n] = agents_[n].size() / chunk + correction;
92  }
93 
94  std::vector<std::atomic<uint64_t>*> counters(max_threads, nullptr);
95  std::vector<uint64_t> max_counters(max_threads);
96  for (int thread_cnt = 0; thread_cnt < max_threads; thread_cnt++) {
97  uint64_t current_nid = thread_info_->GetNumaNode(thread_cnt);
98 
99  auto correction =
100  num_chunks_per_numa[current_nid] %
101  thread_info_->GetThreadsInNumaNode(current_nid) ==
102  0
103  ? 0
104  : 1;
105  uint64_t num_chunks_per_thread =
106  num_chunks_per_numa[current_nid] /
107  thread_info_->GetThreadsInNumaNode(current_nid) +
108  correction;
109  auto start =
110  num_chunks_per_thread * thread_info_->GetNumaThreadId(thread_cnt);
111  auto end = std::min(num_chunks_per_numa[current_nid],
112  start + num_chunks_per_thread);
113 
114  counters[thread_cnt] = new std::atomic<uint64_t>(start);
115  max_counters[thread_cnt] = end;
116  }
117 
118 #pragma omp parallel
119  {
120  auto tid = omp_get_thread_num();
121  auto nid = thread_info_->GetNumaNode(tid);
122 
123  // thread private variables (compilation error with
124  // firstprivate(chunk, numa_node_) with some openmp versions clause)
125  auto p_numa_nodes = thread_info_->GetNumaNodes();
126  auto p_max_threads = omp_get_max_threads();
127  auto p_chunk = chunk;
128  assert(thread_info_->GetNumaNode(tid) == numa_node_of_cpu(sched_getcpu()));
129 
130  // dynamic scheduling
131  uint64_t start = 0;
132  uint64_t end = 0;
133 
134  // this loop implements work stealing from other NUMA nodes if there
135  // are imbalances. Each thread starts with its NUMA domain. Once, it
136  // is finished the thread looks for tasks on other domains
137  for (int n = 0; n < p_numa_nodes; n++) {
138  int current_nid = (nid + n) % p_numa_nodes;
139  for (int thread_cnt = 0; thread_cnt < p_max_threads; thread_cnt++) {
140  uint64_t current_tid = (tid + thread_cnt) % p_max_threads;
141  if (current_nid != thread_info_->GetNumaNode(current_tid)) {
142  continue;
143  }
144 
145  auto& numa_agents = agents_[current_nid];
146  uint64_t old_count = (*(counters[current_tid]))++;
147  while (old_count < max_counters[current_tid]) {
148  start = old_count * p_chunk;
149  end = std::min(static_cast<uint64_t>(numa_agents.size()),
150  start + p_chunk);
151 
152  for (uint64_t i = start; i < end; ++i) {
153  function(numa_agents[i], AgentHandle(current_nid, i));
154  }
155 
156  old_count = (*(counters[current_tid]))++;
157  }
158  } // work stealing loop numa_nodes_
159  } // work stealing loop threads
160  }
161 
162  for (auto* counter : counters) {
163  delete counter;
164  }
165 }

185 struct RearrangeFunctor : public Functor<void, const AgentHandle&> {
186  std::vector<std::vector<AgentHandle>>& sorted_agent_handles;
187  const std::vector<uint64_t>& agent_per_numa;
188  uint64_t cnt = 0;
189  uint64_t current_numa = 0;
190 
191  RearrangeFunctor(std::vector<std::vector<AgentHandle>>& sorted_agent_handles,
192  const std::vector<uint64_t>& agent_per_numa)
193  : sorted_agent_handles(sorted_agent_handles),
194  agent_per_numa(agent_per_numa) {}
195 
196  void operator()(const AgentHandle& handle) {
197  if (cnt == agent_per_numa[current_numa]) {
198  cnt = 0;
199  current_numa++;
200  }
201 
202  sorted_agent_handles[current_numa].push_back(handle);
203  cnt++;
204  }
205 };

207 void ResourceManager::LoadBalance() {
208  // balance agents per numa node according to the number of
209  // threads associated with each numa domain
210  auto numa_nodes = thread_info_->GetNumaNodes();
211  std::vector<uint64_t> agent_per_numa(numa_nodes);
212  uint64_t cummulative = 0;
213  auto max_threads = thread_info_->GetMaxThreads();
214  for (int n = 1; n < numa_nodes; ++n) {
215  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(n);
216  uint64_t num_agents = GetNumAgents() * threads_in_numa / max_threads;
217  agent_per_numa[n] = num_agents;
218  cummulative += num_agents;
219  }
220  agent_per_numa[0] = GetNumAgents() - cummulative;
221 
222  // using first touch policy - page will be allocated to the numa domain of
223  // the thread that accesses it first.
224  // alternative, use numa_alloc_onnode.
225  int ret = numa_run_on_node(0);
226  if (ret != 0) {
227  Log::Fatal("ResourceManager",
228  "Run on numa node failed. Return code: ", ret);
229  }
230 
231  // new data structure for rearranged Agent*
232  decltype(agents_) agent_rearranged;
233  agent_rearranged.resize(numa_nodes);
234 
235  // numa node -> vector of AgentHandles
236  std::vector<std::vector<AgentHandle>> sorted_agent_handles;
237  sorted_agent_handles.resize(numa_nodes);
238 #pragma omp parallel for
239  for (int n = 0; n < numa_nodes; ++n) {
240  if (thread_info_->GetMyNumaNode() == n &&
241  thread_info_->GetMyNumaThreadId() == 0) {
242  sorted_agent_handles[n].reserve(agent_per_numa[n]);
243  }
244  }
245 
246  auto* env = Simulation::GetActive()->GetEnvironment();
247  RearrangeFunctor rearrange(sorted_agent_handles, agent_per_numa);
248  env->IterateZOrder(rearrange);
249 
250  auto* param = Simulation::GetActive()->GetParam();
251  const bool minimize_memory = param->minimize_memory_while_rebalancing;
252 
253 // create new objects
254 #pragma omp parallel
255  {
256  auto tid = thread_info_->GetMyThreadId();
257  auto nid = thread_info_->GetNumaNode(tid);
258 
259  for (int n = 0; n < numa_nodes; n++) {
260  if (nid != n) {
261  continue;
262  }
263  auto& dest = agent_rearranged[n];
264 
265  if (thread_info_->GetNumaThreadId(tid) == 0) {
266  dest.resize(sorted_agent_handles[n].size());
267  }
268 
269 #pragma omp barrier
270 
271  auto threads_in_numa = thread_info_->GetThreadsInNumaNode(nid);
272  auto& agent_handles = sorted_agent_handles[n];
273  assert(thread_info_->GetNumaNode(tid) ==
274  numa_node_of_cpu(sched_getcpu()));
275 
276  // use static scheduling
277  auto correction = agent_handles.size() % threads_in_numa == 0 ? 0 : 1;
278  auto chunk = agent_handles.size() / threads_in_numa + correction;
279  auto start = thread_info_->GetNumaThreadId(tid) * chunk;
280  auto end = std::min(agent_handles.size(), start + chunk);
281 
282  for (uint64_t e = start; e < end; e++) {
283  auto& handle = agent_handles[e];
284  auto* agent = agents_[handle.GetNumaNode()][handle.GetElementIdx()];
285  dest[e] = agent->NewCopy();
286  if (type_index_) {
287  type_index_->Update(dest[e]);
288  }
289  if (minimize_memory) {
290  delete agent;
291  }
292  }
293  }
294  }
295 
296  // delete old objects. This approach has a high chance that a thread
297  // in the right numa node will delete the object, thus minimizing thread
298  // synchronization overheads. The bdm memory allocator does not have this
299  // issue.
300  if (!minimize_memory) {
301  DeleteAgentsFunctor delete_functor;
302  ForEachAgentParallel(delete_functor);
303  }
304 
305  for (int n = 0; n < numa_nodes; n++) {
306  agents_[n].swap(agent_rearranged[n]);
307  }
308 
309  // update uid_ah_map_
310  UpdateUidAgentHandleMapFunctor functor(uid_ah_map_);
311  ForEachAgentParallel(functor);
312 
313  if (Simulation::GetActive()->GetParam()->debug_numa) {
314  std::cout << *this << std::endl;
315  }
316 }
