105 void InitializeGPUData::Initialize(
106  uint64_t num_agents, uint64_t num_boxes,
107  const std::vector<AgentHandle::ElementIdx_t>& offs,
108  UniformGridEnvironment* g) {
109  if (current_timestamp == nullptr) {
110  CudaAllocPinned(&current_timestamp, 1);
111  CudaAllocPinned(&num_boxes_axis, 3);
112  }
113 
114  if (allocated_num_agents < num_agents) {
115  if (allocated_num_agents != 0) {
116  FreeAgentBuffers();
117  }
118  allocated_num_agents = num_agents * 1.25;
119  CudaAllocPinned(&cell_movements, allocated_num_agents * 3);
120  CudaAllocPinned(&cell_positions, allocated_num_agents * 3);
121  CudaAllocPinned(&cell_diameters, allocated_num_agents);
122  CudaAllocPinned(&cell_adherence, allocated_num_agents);
123  CudaAllocPinned(&cell_tractor_force, allocated_num_agents * 3);
124  CudaAllocPinned(&cell_boxid, allocated_num_agents);
125  CudaAllocPinned(&mass, allocated_num_agents);
126  CudaAllocPinned(&successors, allocated_num_agents);
127  }
128 
129  if (allocated_num_boxes < num_boxes) {
130  if (allocated_num_boxes != 0) {
131  FreeGridBuffers();
132  }
133  allocated_num_boxes = num_boxes * 1.25;
134  CudaAllocPinned(&starts, allocated_num_boxes);
135  CudaAllocPinned(&lengths, allocated_num_boxes);
136  CudaAllocPinned(&timestamps, allocated_num_boxes);
137  }
138 
139  offset = offs;
140  grid = g;
141 }

163 void InitializeGPUData::operator()(Agent* agent, AgentHandle ah) {
164  // Check if there are any non-spherical objects in our simulation, because
165  // GPU accelerations currently supports only sphere-sphere interactions
166  IsNonSphericalObjectPresent(agent, &is_non_spherical_object);
167  if (is_non_spherical_object) {
168  Log::Fatal("MechanicalForcesOpCuda",
169  "\nWe detected a non-spherical object during the GPU "
170  "execution. This is currently not supported.");
171  return;
172  }
173  auto* cell = bdm_static_cast<Cell*>(agent);
174  auto idx = offset[ah.GetNumaNode()] + ah.GetElementIdx();
175  auto idxt3 = idx * 3;
176  mass[idx] = cell->GetMass();
177  cell_diameters[idx] = cell->GetDiameter();
178  cell_adherence[idx] = cell->GetAdherence();
179  const auto& tf = cell->GetTractorForce();
180  const auto& pos = cell->GetPosition();
181 
182  for (uint64_t i = 0; i < 3; ++i) {
183  cell_tractor_force[idxt3 + i] = tf[i];
184  cell_positions[idxt3 + i] = pos[i];
185  }
186 
187  cell_boxid[idx] = cell->GetBoxIdx();
188 
189  // populate successor list
190  auto nid = ah.GetNumaNode();
191  auto el = ah.GetElementIdx();
192  successors[idx] = offset[grid->successors_.data_[nid][el].GetNumaNode()] +
193  grid->successors_.data_[nid][el].GetElementIdx();
194 }

221 void UpdateCPUResults::operator()(Agent* agent, AgentHandle ah) {
222  auto* param = Simulation::GetActive()->GetParam();
223  auto* cell = bdm_static_cast<Cell*>(agent);
224  auto idx = offset[ah.GetNumaNode()] + ah.GetElementIdx();
225  idx *= 3;
226  Double3 new_pos = {cell_movements[idx], cell_movements[idx + 1],
227  cell_movements[idx + 2]};
228  cell->UpdatePosition(new_pos);
229  if (param->bound_space) {
230  ApplyBoundingBox(agent, param->min_bound, param->max_bound);
231  }
232 }

235 void MechanicalForcesOpCuda::SetUp() {
236  // Timing timer("MechanicalForcesOpCuda::SetUp");
237  auto* sim = Simulation::GetActive();
238  auto* grid = dynamic_cast<UniformGridEnvironment*>(sim->GetEnvironment());
239  auto* rm = sim->GetResourceManager();
240 
241  if (!grid) {
242  Log::Fatal(
243  "MechanicalForcesOpCuda::operator()",
244  "MechanicalForcesOpCuda only works with UniformGridEnvironement.");
245  }
246 
247  auto num_numa_nodes = ThreadInfo::GetInstance()->GetNumaNodes();
248  std::vector<AgentHandle::ElementIdx_t> offset(num_numa_nodes);
249  offset[0] = 0;
250  for (int nn = 1; nn < num_numa_nodes; nn++) {
251  offset[nn] = offset[nn - 1] + rm->GetNumAgents(nn - 1);
252  }
253 
254  auto total_num_agents = rm->GetNumAgents();
255  auto num_boxes = grid->boxes_.size();
256 
257  if (i_ == nullptr) {
258  i_ = new detail::InitializeGPUData();
259  }
260  i_->Initialize(total_num_agents, num_boxes, offset, grid);
261  {
262  // Timing timer("MechanicalForcesOpCuda::toColumnar");
263  rm->ForEachAgentParallel(1000, *i_);
264  }
265 
266  *i_->current_timestamp = grid->timestamp_;
267 #pragma omp parallel for
268  for (uint64_t i = 0; i < grid->boxes_.size(); ++i) {
269  auto& box = grid->boxes_[i];
270  i_->timestamps[i] = box.timestamp_;
271  if (box.timestamp_ == *i_->current_timestamp) {
272  i_->lengths[i] = box.length_;
273  i_->starts[i] =
274  offset[box.start_.GetNumaNode()] + box.start_.GetElementIdx();
275  }
276  }
277  grid->GetNumBoxesAxis(i_->num_boxes_axis);
278 }

281 void MechanicalForcesOpCuda::operator()() {
282  auto* sim = Simulation::GetActive();
283  auto* grid = dynamic_cast<UniformGridEnvironment*>(sim->GetEnvironment());
284  auto* param = sim->GetParam();
285  auto* rm = sim->GetResourceManager();
286 
287  uint32_t total_num_agents = rm->GetNumAgents();
288  auto num_boxes = grid->boxes_.size();
289  // If this is the first time we perform physics on GPU using CUDA
290  if (cdo_ == nullptr) {
291  // Allocate 25% more memory so we don't need to reallocate GPU memory
292  // for every (small) change
293  total_num_agents_ = static_cast<uint32_t>(1.25 * total_num_agents);
294  num_boxes_ = static_cast<uint32_t>(1.25 * num_boxes);
295 
296  // Allocate required GPU memory
297  cdo_ = new MechanicalForcesOpCudaKernel(total_num_agents_, num_boxes_);
298  } else {
299  // If the number of agents increased
300  if (total_num_agents >= total_num_agents_) {
301  Log::Info("MechanicalForcesOpCuda",
302  "\nThe number of cells increased signficantly (from ",
303  total_num_agents_, " to ", total_num_agents,
304  "), agent we allocate bigger GPU buffers\n");
305  total_num_agents_ = static_cast<uint32_t>(1.25 * total_num_agents);
306  cdo_->ResizeCellBuffers(total_num_agents_);
307  }
308 
309  // If the neighbor grid size increased
310  if (num_boxes >= num_boxes_) {
311  Log::Info("MechanicalForcesOpCuda",
312  "\nThe number of boxes increased signficantly (from ",
313  num_boxes_, " to ", "), so we allocate bigger GPU buffers\n");
314  num_boxes_ = static_cast<uint32_t>(1.25 * num_boxes);
315  cdo_->ResizeGridBuffers(num_boxes_);
316  }
317  }
318 
319  double squared_radius =
320  grid->GetLargestObjectSize() * grid->GetLargestObjectSize();
321 
322  // Timing timer("MechanicalForcesOpCuda::Kernel");
323  cdo_->LaunchMechanicalForcesKernel(
324  i_->cell_positions, i_->cell_diameters, i_->cell_tractor_force,
325  i_->cell_adherence, i_->cell_boxid, i_->mass, param->simulation_time_step,
326  param->simulation_max_displacement, squared_radius, total_num_agents,
327  i_->starts, i_->lengths, i_->timestamps, *(i_->current_timestamp),
328  i_->successors, i_->num_boxes_axis, i_->cell_movements);
329 }
