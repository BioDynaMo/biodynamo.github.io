40  void Visit(const std::string& dm_name, size_t type_hash_code,
41  const void* data) override {
42  if (type_hash_code == typeid(double).hash_code()) {
43  Double(dm_name, data);
44  } else if (type_hash_code == typeid(int).hash_code()) {
45  Int(dm_name, data);
46  } else if (type_hash_code == typeid(uint64_t).hash_code()) {
47  Uint64T(dm_name, data);
48  } else if (type_hash_code == typeid(Double3).hash_code()) {
49  MathArray3(dm_name, data);
50  } else if (type_hash_code == typeid(std::array<int, 3>).hash_code()) {
51  Int3(dm_name, data);
52  } else {
53  Log::Fatal("CatalystSoVisitor::Visit",
54  "This data member is not supported for visualization");
55  }
56  }

93  template <typename TDataArray>
94  TDataArray* GetDataArray(const std::string& dm_name,
95  int components = 1) const {
96  TDataArray* vtk_array = nullptr;
97  auto& data_arrays = so_grid_->data_arrays_;
98  auto search = data_arrays.find(dm_name);
99  if (search != data_arrays.end()) {
100  auto& da_wrapper = search->second;
101  vtk_array = static_cast<TDataArray*>(da_wrapper.data_);
102 
103  // reset
104  auto* scheduler = Simulation::GetActive()->GetScheduler();
105  if (da_wrapper.time_step_ != scheduler->GetSimulatedSteps()) {
106  vtk_array->Reset();
107  da_wrapper.time_step_ = scheduler->GetSimulatedSteps();
108  }
109 
110  } else {
111  // create
112  vtkNew<TDataArray> new_vtk_array;
113  new_vtk_array->SetName(dm_name.c_str());
114  vtk_array = new_vtk_array.GetPointer();
115  vtk_array->SetNumberOfComponents(components);
116  auto* point_data = so_grid_->data_->GetPointData();
117  point_data->AddArray(vtk_array);
118  data_arrays.insert({dm_name, VtkDataArrayWrapper(vtk_array)});
119  }
120 
121  return vtk_array;
122  }
