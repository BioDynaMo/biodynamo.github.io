24 std::string GenerateSimulationInfoJson(
25  const std::unordered_map<std::string, VtkAgents*>& vtk_agents,
26  const std::unordered_map<std::string, VtkDiffusionGrid*>& vtk_dgrids) {
27  auto* sim = Simulation::GetActive();
28  auto* param = sim->GetParam();
29  // agents
30  std::stringstream agents;
31  uint64_t num_agents = param->visualize_agents.size();
32  uint64_t counter = 0;
33  for (const auto& entry : param->visualize_agents) {
34  std::string agent_name = entry.first;
35 
36  auto search = vtk_agents.find(agent_name);
37  if (search == vtk_agents.end()) {
38  continue;
39  }
40  auto* agent_grid = search->second;
41 
42  agents << " {" << std::endl
43  << " \"name\":\"" << agent_name << "\"," << std::endl;
44  if (agent_grid->GetShape() == Shape::kSphere) {
45  agents << " \"glyph\":\"Glyph\"," << std::endl
46  << " \"shape\":\"Sphere\"," << std::endl
47  << " \"scaling_attribute\":\"diameter_\"" << std::endl;
48  } else if (agent_grid->GetShape() == kCylinder) {
49  agents << " \"glyph\":\"BDMGlyph\"," << std::endl
50  << " \"shape\":\"Cylinder\"," << std::endl
51  << " \"x_scaling_attribute\":\"diameter_\"," << std::endl
52  << " \"y_scaling_attribute\":\"actual_length_\"," << std::endl
53  << " \"z_scaling_attribute\":\"diameter_\"," << std::endl
54  << " \"Vectors\":\"spring_axis_\"," << std::endl
55  << " \"MassLocation\":\"mass_location_\"" << std::endl;
56  }
57  agents << " }";
58  if (counter != num_agents - 1) {
59  agents << ",";
60  }
61  agents << std::endl;
62  counter++;
63  }
64 
65  // extracellular substances
66  std::stringstream substances;
67  uint64_t num_substances = param->visualize_diffusion.size();
68  for (uint64_t i = 0; i < num_substances; i++) {
69  auto& name = param->visualize_diffusion[i].name;
70 
71  auto search = vtk_dgrids.find(name);
72  if (search == vtk_dgrids.end()) {
73  continue;
74  }
75  auto* dgrid = search->second;
76  // user wanted to export this substance, but it did not exist during
77  // the entire simulation
78  if (!dgrid->IsUsed()) {
79  Log::Warning(
80  "Visualize Diffusion Grids",
81  "You are trying to visualize diffusion grid ", name,
82  ", but it has not been created during the entire simulation. "
83  "Please make sure the names in the "
84  "configuration file match the ones in the simulation.");
85  num_substances--;
86  continue;
87  }
88  substances << " { \"name\":\"" << name << "\", ";
89  std::string has_gradient =
90  param->visualize_diffusion[i].gradient ? "true" : "false";
91  substances << "\"has_gradient\":\"" << has_gradient << "\" }";
92 
93  if (i != num_substances - 1) {
94  substances << "," << std::endl;
95  }
96  }
97 
98  std::stringstream str;
99  str << "{" << std::endl
100  << " \"simulation\": {" << std::endl
101  << " \"name\":\"" << sim->GetUniqueName() << "\"," << std::endl
102  << " \"result_dir\":\"" << sim->GetOutputDir() << "\"" << std::endl
103  << " }," << std::endl
104  << " \"agents\": [" << std::endl
105  << agents.str() << " ]," << std::endl
106  << " \"extracellular_substances\": [" << std::endl
107  << substances.str() << std::endl
108  << " ]," << std::endl
109  << " \"insitu_script_arguments\": \""
110  << param->pv_insitu_pipelinearguments << "\"" << std::endl
111  << "}" << std::endl;
112  return str.str();
113 }
