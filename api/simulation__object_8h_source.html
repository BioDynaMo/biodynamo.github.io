145  std::unordered_map<uint32_t, uint32_t> Commit() {
146  std::lock_guard<std::recursive_mutex> lock(mutex_);
147  std::unordered_map<uint32_t, uint32_t> updated_indices;
148  // commit delayed push backs
149  size_ = total_size_;
150  // commit delayed removes
151  // sort indices in descending order to prevent out of bounds accesses
152  auto descending = [](auto a, auto b) { return a > b; };
153  std::sort(to_be_removed_.begin(), to_be_removed_.end(), descending);
154  for (size_t idx : to_be_removed_) {
155  assert(idx < size_ && "Removed index outside array boundaries");
156  if (idx < size_ - 1) {
157  SwapAndPopBack(idx, size_);
158  updated_indices[size_ - 1] = idx;
159  } else {
160  PopBack();
161  }
162  size_--;
163  }
164  to_be_removed_.clear();
165 
166  // use implementation from TransactionalVector to avoid code duplication
167  TransactionalVector<int>::ShortcutUpdatedIndices(&updated_indices);
168 
169  total_size_ = size_;
170  return updated_indices;
171  }
