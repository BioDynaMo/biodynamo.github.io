137  std::unordered_map<uint32_t, uint32_t> Commit() {
138  std::lock_guard<std::recursive_mutex> lock(mutex_);
139  std::unordered_map<uint32_t, uint32_t> updated_indices;
140  // commit delayed push backs
141  size_ = total_size_;
142  // commit delayed removes
143  // sort indices in descending order to prevent out of bounds accesses
144  auto descending = [](auto a, auto b) { return a > b; };
145  std::sort(to_be_removed_.begin(), to_be_removed_.end(), descending);
146  for (size_t idx : to_be_removed_) {
147  assert(idx < size_ && "Removed index outside array boundaries");
148  if (idx < size_ - 1) {
149  SwapAndPopBack(idx, size_);
150  updated_indices[size_ - 1] = idx;
151  } else {
152  PopBack();
153  }
154  size_--;
155  }
156  to_be_removed_.clear();
157 
158  // use implementation from TransactionalVector to avoid code duplication
159  TransactionalVector<int>::ShortcutUpdatedIndices(&updated_indices);
160 
161  total_size_ = size_;
162  return updated_indices;
163  }
