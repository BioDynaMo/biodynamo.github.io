475  template <typename TEvent, typename TBiologyModules1,
476  typename... TBiologyModules>
477  void BiologyModuleEventHandler(const TEvent &event, TBiologyModules1 *current,
478  TBiologyModules *... bms) {
479  // call event handler for biology modules
480  uint64_t cnt = 0;
481  auto call_bm_event_handler = [&](auto &bm) {
482  using BiologyModuleType = raw_type<decltype(bm)>;
483 
485  auto extract = [](bool condition, auto *variant) -> BiologyModuleType * {
486  if (condition) {
487  return get_if<BiologyModuleType>(variant);
488  }
489  return nullptr;
490  };
491 
492  if (!bm.Remove(event.kEventId)) {
493  bool copy = bm.Copy(event.kEventId);
494  bm.EventHandler(event, extract(copy, &((*bms)[cnt]))...);
495  cnt += copy ? 1 : 0;
496  }
497  };
498  for (auto &el : *current) {
499  visit(call_bm_event_handler, el);
500  }
501 
502  // remove biology modules from current
503  bool remove;
504  auto remove_from_current = [&](auto &bm) {
505  remove = bm.Remove(event.kEventId);
506  };
507  for (auto it = current->begin(); it != current->end();) {
508  remove = false;
509  visit(remove_from_current, *it);
510  if (remove) {
511  it = current->erase(it);
512  } else {
513  ++it;
514  }
515  }
516  }
517 };
