154  std::unordered_map<uint32_t, uint32_t> Commit() {
155  std::lock_guard<std::recursive_mutex> lock(mutex_);
156  std::unordered_map<uint32_t, uint32_t> updated_indices;
157  // commit delayed push backs
158  size_ = total_size_;
159  // commit delayed removes
160  // sort indices in descending order to prevent out of bounds accesses
161  auto descending = [](auto a, auto b) { return a > b; };
162  std::sort(to_be_removed_.begin(), to_be_removed_.end(), descending);
163  for (size_t idx : to_be_removed_) {
164  assert(idx < size_ && "Removed index outside array boundaries");
165  if (idx < size_ - 1) {
166  SwapAndPopBack(idx, size_);
167  updated_indices[size_ - 1] = idx;
168  } else {
169  PopBack();
170  }
171  size_--;
172  }
173  to_be_removed_.clear();
174 
175  // use implementation from TransactionalVector to avoid code duplication
176  TransactionalVector<int>::ShortcutUpdatedIndices(&updated_indices);
177 
178  total_size_ = size_;
179  return updated_indices;
180  }

504  template <typename TEvent, typename TBiologyModules1,
505  typename... TBiologyModules>
506  void BiologyModuleEventHandler(const TEvent &event, TBiologyModules1 *current,
507  TBiologyModules *... bms) {
508  // call event handler for biology modules
509  uint64_t cnt = 0;
510  auto call_bm_event_handler = [&](auto &bm) {
511  using BiologyModuleType = raw_type<decltype(bm)>;
512 
514  auto extract = [](bool condition, auto *variant) -> BiologyModuleType * {
515  if (condition) {
516  return get_if<BiologyModuleType>(variant);
517  }
518  return nullptr;
519  };
520 
521  if (!bm.Remove(event.kEventId)) {
522  bool copy = bm.Copy(event.kEventId);
523  bm.EventHandler(event, extract(copy, &((*bms)[cnt]))...);
524  cnt += copy ? 1 : 0;
525  }
526  };
527  for (auto &el : *current) {
528  visit(call_bm_event_handler, el);
529  }
530 
531  // remove biology modules from current
532  bool remove;
533  auto remove_from_current = [&](auto &bm) {
534  remove = bm.Remove(event.kEventId);
535  };
536  for (auto it = current->begin(); it != current->end();) {
537  remove = false;
538  visit(remove_from_current, *it);
539  if (remove) {
540  it = current->erase(it);
541  } else {
542  ++it;
543  }
544  }
545  }
