174 void Scheduler::RunScheduledOps() {
175  auto* sim = Simulation::GetActive();
176  auto* rm = sim->GetResourceManager();
177  auto* param = sim->GetParam();
178  auto batch_size = param->scheduling_batch_size_;
179 
180  Timing::Time(update_environment_op_->name_,
181  [&]() { (*update_environment_op_)(); });
182 
183  // Run the sim object operations
184  std::vector<Operation*> sim_object_ops;
185  for (auto* op : scheduled_sim_object_ops_) {
186  if (total_steps_ % op->frequency_ == 0) {
187  sim_object_ops.push_back(op);
188  }
189  }
190  RunAllScheduledOps functor(sim_object_ops);
191 
192  Timing::Time("sim object ops", [&]() {
193  rm->ApplyOnAllElementsParallelDynamic(batch_size, functor);
194  });
195 
196  // Run the column-wise operations
197  for (auto* op : scheduled_standalone_ops_) {
198  if (total_steps_ % op->frequency_ == 0) {
199  Timing::Time(op->name_, [&]() { (*op)(); });
200  }
201  }
202 }

230 bool Scheduler::Restore(uint64_t* steps) {
231  if (backup_->RestoreEnabled() && restore_point_ > total_steps_ + *steps) {
232  total_steps_ += *steps;
233  // restore requested, but not last backup was not done during this call to
234  // Simualte. Therefore, we skip it.
235  return true;
236  } else if (backup_->RestoreEnabled() && restore_point_ > total_steps_ &&
237  restore_point_ < total_steps_ + *steps) {
238  // Restore
239  backup_->Restore();
240  *steps = total_steps_ + *steps - restore_point_;
241  total_steps_ = restore_point_;
242  }
243  return false;
244 }

249 void Scheduler::Initialize() {
250  auto* sim = Simulation::GetActive();
251  auto* env = sim->GetEnvironment();
252  auto* rm = sim->GetResourceManager();
253  auto* param = sim->GetParam();
254 
255  // commit all changes
256  const auto& all_exec_ctxts = sim->GetAllExecCtxts();
257  all_exec_ctxts[0]->TearDownIterationAll(all_exec_ctxts);
258 
259  if (param->bound_space_) {
260  auto* bound_space = NewOperation("bound space");
261  rm->ApplyOnAllElementsParallel(*bound_space);
262  delete bound_space;
263  }
264  env->Update();
265  int lbound = env->GetDimensionThresholds()[0];
266  int rbound = env->GetDimensionThresholds()[1];
267  rm->ApplyOnAllDiffusionGrids([&](DiffusionGrid* dgrid) {
268  // Create data structures, whose size depend on the env dimensions
269  dgrid->Initialize({lbound, rbound, lbound, rbound, lbound, rbound});
270  // Initialize data structures with user-defined values
271  dgrid->RunInitializers();
272  });
273 
274  ScheduleOps();

277 // Schedule the operations
278 void Scheduler::ScheduleOps() {
279  auto* param = Simulation::GetActive()->GetParam();
280  // Add requested operations
281  for (auto it = ops_to_add_.begin(); it != ops_to_add_.end();) {
282  auto* op = *it;
283  // Enable GPU operation implementations (if available) if CUDA or OpenCL
284  // flags are set
285  if (param->compute_target_ == "cuda" &&
286  op->IsComputeTargetSupported(kCuda)) {
287  op->SelectComputeTarget(kCuda);
288  } else if (param->compute_target_ == "opencl" &&
289  op->IsComputeTargetSupported(kOpenCl)) {
290  op->SelectComputeTarget(kOpenCl);
291  } else {
292  op->SelectComputeTarget(kCpu);
293  }
294 
295  if (op->IsStandalone()) {
296  scheduled_standalone_ops_.push_back(op);
297  } else {
298  scheduled_sim_object_ops_.push_back(op);
299  }
300 
301  // Remove operation from ops_to_add_
302  it = ops_to_add_.erase(it);
303  }
304 
305  // Remove requested operations
306  for (auto it = ops_to_remove_.begin(); it != ops_to_remove_.end();) {
307  auto* op = *it;
308  // Check scheduled row-wise operations list
309  for (auto it2 = scheduled_sim_object_ops_.begin();
310  it2 != scheduled_sim_object_ops_.end(); ++it2) {
311  if (op == (*it2)) {
312  // Add to list of unscheduled operations
313  unscheduled_ops_.push_back(op);
314  it2 = scheduled_sim_object_ops_.erase(it2);
315  goto label;
316  }
317  }
318 
319  // Check scheduled column-wise operations list
320  for (auto it2 = scheduled_standalone_ops_.begin();
321  it2 != scheduled_standalone_ops_.end(); ++it2) {
322  if (op == (*it2)) {
323  // Add to list of unscheduled operations
324  unscheduled_ops_.push_back(op);
325  it2 = scheduled_standalone_ops_.erase(it2);
326  goto label;
327  }
328  }
329  label:
330  it = ops_to_remove_.erase(it);
331  }
332 }
