50  auto* grid = sim->GetGrid();
51  auto* rm = sim->GetResourceManager();
52  auto* param = sim->GetParam();
53 
54  auto context = rm->GetOpenCLContext();
55  auto queue = rm->GetOpenCLCommandQueue();
56  auto programs = rm->GetOpenCLProgramList();
57 
58  std::vector<cl_double> mass(cells->size());

60  std::vector<cl_uint> gpu_starts;
61  std::vector<cl_ushort> gpu_lengths;
62  std::vector<cl_uint> successors(cells->size());
63  cl_uint box_length;
64  std::array<cl_uint, 3> num_boxes_axis;
65  std::array<cl_int, 3> grid_dimensions;
66  cl_double squared_radius =
67  grid->GetLargestObjectSize() * grid->GetLargestObjectSize();
68 
69  // We need to create a mass vector, because it is not stored by default in
70  // a cell container
71  cells->FillMassVector(&mass);
72  grid->GetSuccessors(&successors);
73  grid->GetBoxInfo(&gpu_starts, &gpu_lengths);
74  grid->GetGridInfo(&box_length, &num_boxes_axis, &grid_dimensions);
75 
76  // Allocate GPU buffers
77  cl::Buffer positions_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
78  cells->size() * 3 * sizeof(cl_double),
79  cells->GetPositionPtr());
80  cl::Buffer diameters_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
81  cells->size() * sizeof(cl_double),
82  cells->GetDiameterPtr());
83  cl::Buffer tractor_force_arg(
84  *context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
85  cells->size() * 3 * sizeof(cl_double), cells->GetTractorForcePtr());
86  cl::Buffer adherence_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,

93  cells->size() * sizeof(cl_double), mass.data());
94  cl::Buffer cell_movements_arg(
95  *context, CL_MEM_READ_WRITE | CL_MEM_USE_HOST_PTR,
96  cells->size() * 3 * sizeof(cl_double), cell_movements.data()->data());
97  cl::Buffer starts_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
98  gpu_starts.size() * sizeof(cl_uint),
99  gpu_starts.data());
100  cl::Buffer lengths_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
101  gpu_lengths.size() * sizeof(cl_short),
102  gpu_lengths.data());
103  cl::Buffer successors_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
104  successors.size() * sizeof(cl_uint),
105  successors.data());
106  cl::Buffer nba_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
107  3 * sizeof(cl_uint), num_boxes_axis.data());
108  cl::Buffer gd_arg(*context, CL_MEM_READ_ONLY | CL_MEM_USE_HOST_PTR,
109  3 * sizeof(cl_int), grid_dimensions.data());
110 
111  // Create the kernel object from our program
112  // TODO(ahmad): generalize the program selection, in case we have more than
113  // one. We can maintain an unordered map of programs maybe
114  cl::Kernel collide((*programs)[0], "collide");
115 
116  // Set kernel parameters
117  collide.setArg(0, positions_arg);
118  collide.setArg(1, diameters_arg);
119  collide.setArg(2, tractor_force_arg);
120  collide.setArg(3, adherence_arg);
121  collide.setArg(4, box_id_arg);
122  collide.setArg(5, mass_arg);
123  collide.setArg(6, param->simulation_time_step_);
124  collide.setArg(7, param->simulation_max_displacement_);
125  collide.setArg(8, squared_radius);
126 
127  collide.setArg(9, static_cast<cl_int>(cells->size()));
128  collide.setArg(10, starts_arg);
129  collide.setArg(11, lengths_arg);
130  collide.setArg(12, successors_arg);
131  collide.setArg(13, box_length);
132  collide.setArg(14, nba_arg);
133  collide.setArg(15, gd_arg);
134  collide.setArg(16, cell_movements_arg);
135 
136  // The amount of threads for each work group (analogous to CUDA thread
137  // block)
138  int block_size = 256;
139 
140  auto num_objects = cells->size();
141 
142  try {
143  // The global size determines the total number of threads that will be
144  // spawned on the GPU, in groups of local_size
145  cl::NDRange global_size =
146  cl::NDRange(num_objects + (block_size - (num_objects % block_size)));
147  cl::NDRange local_size = cl::NDRange(block_size);
148  queue->enqueueNDRangeKernel(collide, cl::NullRange, global_size,
149  local_size);
150  } catch (const cl::Error& err) {
151  Log::Error("DisplacementOpOpenCL", err.what(), "(", err.err(), ") = ",
152  GetErrorString(err.err()));
153  throw;
154  }
155 
156  try {
157  queue->enqueueReadBuffer(cell_movements_arg, CL_TRUE, 0,
158  cells->size() * 3 * sizeof(cl_double),
159  cell_movements.data()->data());
160  } catch (const cl::Error& err) {
161  Log::Error("DisplacementOpOpenCL", err.what(), "(", err.err(), ") = ",
162  GetErrorString(err.err()));
163  throw;
164  }
165 
166 // set new positions after all updates have been calculated
167 // otherwise some cells would see neighbors with already updated positions
168 // which would lead to inconsistencies
169 #pragma omp parallel for
170  for (size_t i = 0; i < cells->size(); i++) {
171  auto&& cell = (*cells)[i];
172  cell.UpdatePosition(cell_movements[i]);
173  if (param->bound_space_) {
174  ApplyBoundingBox(&cell, param->min_bound_, param->max_bound_);
175  }
176  cell.SetPosition(cell.GetPosition());
177 
178  // Reset biological movement to 0.
179  cell.SetTractorForce({0, 0, 0});
180  }
181 #endif
182  }
