205  NeighborIterator& ForwardToNonEmptyBox() {
206  // increment box id until non empty box has been found
207  while (++box_idx_ < neighbor_boxes_.size()) {
208  // box is empty or uninitialized (padding box) -> continue
209  if (neighbor_boxes_[box_idx_]->IsEmpty()) {
210  continue;
211  }
212  // a non-empty box has been found
213  box_iterator_ = neighbor_boxes_[box_idx_]->begin();

215  }

264  void UpdateGrid() {
265  auto* rm = TSimulation::GetActive()->GetResourceManager();
266 
267  if (rm->GetNumSimObjects() != 0) {
268  ClearGrid();
269 
270  auto inf = Math::kInfinity;
271  std::array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
272  CalculateGridDimensions(&tmp_dim);
273  RoundOffGridDimensions(tmp_dim);
274 
275  auto los = ceil(largest_object_size_);
276  assert(los > 0 &&
277  "The largest object size was found to be 0. Please check if your "
278  "cells are correctly initialized.");
279  box_length_ = los;
280 
281  for (int i = 0; i < 3; i++) {
282  int dimension_length =
283  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
284  int r = dimension_length % box_length_;
285  // If the grid is not perfectly divisible along each dimension by the
286  // resolution, extend the grid so that it is
287  if (r != 0) {
288  // std::abs for the case that box_length_ > dimension_length
289  grid_dimensions_[2 * i + 1] += (box_length_ - r);
290  } else {
291  // Else extend the grid dimension with one row, because the outmost
292  // object lies exactly on the border
293  grid_dimensions_[2 * i + 1] += box_length_;
294  }
295  }
296 
297  // Pad the grid to avoid out of bounds check when search neighbors
298  for (int i = 0; i < 3; i++) {
299  grid_dimensions_[2 * i] -= box_length_;
300  grid_dimensions_[2 * i + 1] += box_length_;
301  }
302 
303  // Calculate how many boxes fit along each dimension
304  for (int i = 0; i < 3; i++) {
305  int dimension_length =
306  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
307  assert((dimension_length % box_length_ == 0) &&
308  "The grid dimensions are not a multiple of its box length");
309  num_boxes_axis_[i] = dimension_length / box_length_;
310  }
311 
312  num_boxes_xy_ = num_boxes_axis_[0] * num_boxes_axis_[1];
313  auto total_num_boxes = num_boxes_xy_ * num_boxes_axis_[2];
314 
315  CheckGridGrowth();
316 
317  if (boxes_.size() != total_num_boxes) {
318  boxes_.resize(total_num_boxes, Box());
319  }
320 
321  successors_.Reserve();
322 
323  // Assign simulation objects to boxes
324  rm->ApplyOnAllElementsParallelDynamic(
325  1000, [this](auto&& sim_object,
326  SoHandle id) { // FIXME move back to parallel
327  const auto& position = sim_object.GetPosition();
328  auto idx = this->GetBoxIndex(position);
329  auto box = this->GetBoxPointer(idx);
330  box->AddObject(id, &successors_);
331  sim_object.SetBoxIdx(idx);
332  });
333  auto* param = TSimulation::GetActive()->GetParam();
334  if (param->bound_space_) {
335  int min = param->min_bound_;
336  int max = param->max_bound_;
337  threshold_dimensions_ = {min, max};
338  }
339 
340  if (nb_mutex_builder_ != nullptr) {
341  nb_mutex_builder_->Update();
342  }
343  } else {
344  // There are no sim objects in this simulation
345  auto* param = TSimulation::GetActive()->GetParam();
346 
347  bool uninitialized = boxes_.size() == 0;
348  if (uninitialized && param->bound_space_) {
349  // Simulation has never had any simulation objects
350  // Initialize grid dimensions with `Param::min_bound_` and
351  // `Param::max_bound_`
352  // This is required for the DiffusionGrid
353  int min = param->min_bound_;
354  int max = param->max_bound_;
355  grid_dimensions_ = {min, max, min, max, min, max};
356  threshold_dimensions_ = {min, max};
357  has_grown_ = true;
358  } else if (!uninitialized) {
359  // all simulation objects have been removed in the last iteration
360  // grid state remains the same, but we have to set has_grown_ to false
361  // otherwise the DiffusionGrid will attempt to resize
362  has_grown_ = false;
363  } else {
364  Log::Fatal(
365  "Grid",
366  "You tried to initialize an empty simulation without bound space. "
367  "Therefore we cannot determine the size of the simulation space. "
368  "Please add simulation objects, or set Param::bound_space_, "
369  "Param::min_bound_, and Param::max_bound_.");
370  }
371  }

382  inline double SquaredEuclideanDistance(
383  const std::array<double, 3>& pos1,
384  const std::array<double, 3>& pos2) const {
385  const double dx = pos2[0] - pos1[0];
386  const double dy = pos2[1] - pos1[1];
387  const double dz = pos2[2] - pos1[2];
388  return (dx * dx + dy * dy + dz * dz);
389  }

391  inline bool WithinSquaredEuclideanDistance(
392  double squared_radius, const std::array<double, 3>& pos1,
393  const std::array<double, 3>& pos2) const {
394  const double dx = pos2[0] - pos1[0];
395  const double dx2 = dx * dx;
396  if (dx2 > squared_radius) {
397  return false;
398  }
399 
400  const double dy = pos2[1] - pos1[1];
401  const double dy2_plus_dx2 = dy * dy + dx2;
402  if (dy2_plus_dx2 > squared_radius) {
403  return false;
404  }
405 
406  const double dz = pos2[2] - pos1[2];
407  const double distance = dz * dz + dy2_plus_dx2;
408  return distance < squared_radius;
409  }

411  void UpdateBoxZOrder() {
412  // iterate boxes in Z-order / morton order
413  // TODO(lukas) this is a very quick attempt to test an idea
414  // improve performance of this brute force solution
415  zorder_sorted_boxes_.resize(boxes_.size());
416 #pragma omp parallel for collapse(3)
417  for (uint32_t x = 0; x < num_boxes_axis_[0]; x++) {
418  for (uint32_t y = 0; y < num_boxes_axis_[1]; y++) {
419  for (uint32_t z = 0; z < num_boxes_axis_[2]; z++) {

546  template <typename TLambda>
547  void ForEachNeighborPairWithinRadius(const TLambda& lambda,
548  double squared_radius) const {
549  uint32_t z_start = 0, y_start = 0;
550  auto* rm = TSimulation::GetActive()->GetResourceManager();
551  // use special iteration pattern to avoid race conditions between neighbors
552  // main iteration will be done over rows of boxes. In order to avoid two
553  // threads accessing the same box, one has to use a margin reagion of two
554  // boxes in the y and z dimension.
555  for (uint16_t i = 0; i < 9; i++) {
556  switch (i) {
557  case 0:
558  z_start = 1;
559  y_start = 1;
560  break;
561  case 1:
562  z_start = 1;
563  y_start = 2;
564  break;
565  case 2:
566  z_start = 1;
567  y_start = 3;
568  break;
569  case 3:
570  z_start = 2;
571  y_start = 1;
572  break;
573  case 4:
574  z_start = 2;
575  y_start = 2;
576  break;
577  case 5:
578  z_start = 2;
579  y_start = 3;
580  break;
581  case 6:
582  z_start = 3;
583  y_start = 1;
584  break;
585  case 7:
586  z_start = 3;
587  y_start = 2;
588  break;
589  case 8:
590  z_start = 3;
591  y_start = 3;
592  break;
593  }
594 #pragma omp parallel
595  {
596  FixedSizeVector<size_t, 14> box_indices;
597  CircularBuffer<InlineVector<SoHandle, 4>, 14> cached_so_handles;
598  CircularBuffer<InlineVector<std::array<double, 3>, 4>, 14>
599  cached_positions;
600 
601 #pragma omp for collapse(2) schedule(dynamic, 1) firstprivate(z_start, y_start)
602  for (uint32_t z = z_start; z < num_boxes_axis_[2] - 1; z += 3) {
603  for (uint32_t y = y_start; y < num_boxes_axis_[1] - 1; y += 3) {
604  auto current_box_idx =
605  GetBoxIndex(std::array<uint32_t, 3>{1, y, z});
606 
607  box_indices.clear();
608  cached_so_handles.clear();
609  cached_positions.clear();
610 
611  // since we want to execute the lambda ONCE for each cell pair,
612  // it is sufficient to process half of the boxes and omit the
613  // opposite box
614  // Order: C BW FNW NW BNW B FN N BN E BE FNE NE BNE
615  GetHalfMooreBoxIndices(&box_indices, current_box_idx);
616  CacheSoHandles(box_indices, &cached_so_handles);
617  CachePositions(cached_so_handles, rm, &cached_positions);
618 
619  // first iteration peeled off
620  ForEachNeighborPairWithCenterBox(lambda, rm, cached_so_handles,
621  cached_positions, 0,
622  squared_radius);
623 
624  for (uint32_t x = 2; x < num_boxes_axis_[0] - 1; x++) {
625  // since every thread is iterating over a complete row, we can
626  // save time by updating only new boxes
627  ++box_indices;
628  UpdateSoHandles(box_indices, &cached_so_handles);
629  UpdateCachedPositions(cached_so_handles, rm, &cached_positions);
630  // updating `cached_so_handles` and `cached_positions` will change
631  // the
632  // position of the center box from 0 to 4
633  ForEachNeighborPairWithCenterBox(lambda, rm, cached_so_handles,
634  cached_positions, 4,
635  squared_radius);
636  }
637  }
638  }
639  }
640  }

892  void CalculateGridDimensions(std::array<double, 6>* ret_grid_dimensions) {
893  auto* rm = TSimulation::GetActive()->GetResourceManager();
894 
895  const auto max_threads = omp_get_max_threads();
896  // allocate version for each thread - avoid false sharing by padding them
897  // assumes 64 byte cache lines (8 * sizeof(double))
898  std::vector<std::array<double, 8>> xmin(max_threads, {{Math::kInfinity}});
899  std::vector<std::array<double, 8>> ymin(max_threads, {{Math::kInfinity}});
900  std::vector<std::array<double, 8>> zmin(max_threads, {{Math::kInfinity}});
901 
902  std::vector<std::array<double, 8>> xmax(max_threads, {{-Math::kInfinity}});
903  std::vector<std::array<double, 8>> ymax(max_threads, {{-Math::kInfinity}});
904  std::vector<std::array<double, 8>> zmax(max_threads, {{-Math::kInfinity}});
905 
906  std::vector<std::array<double, 8>> largest(max_threads, {{0}});
907 
908  rm->ApplyOnAllElementsParallelDynamic(1000, [&](auto&& so, const SoHandle) {
909  auto tid = omp_get_thread_num();
910  const auto& position = so.GetPosition();
911  // x
912  if (position[0] < xmin[tid][0]) {
913  xmin[tid][0] = position[0];
914  }
915  if (position[0] > xmax[tid][0]) {
916  xmax[tid][0] = position[0];
917  }
918  // y
919  if (position[1] < ymin[tid][0]) {
920  ymin[tid][0] = position[1];
921  }
922  if (position[1] > ymax[tid][0]) {
923  ymax[tid][0] = position[1];
924  }
925  // z
926  if (position[2] < zmin[tid][0]) {
927  zmin[tid][0] = position[2];
928  }
929  if (position[2] > zmax[tid][0]) {
930  zmax[tid][0] = position[2];
931  }
932  // larget object
933  auto diameter = so.GetDiameter();
934  if (diameter > largest[tid][0]) {
935  largest[tid][0] = diameter;
936  }
937  });
938 
939  // reduce partial results into global one
940  double& gxmin = (*ret_grid_dimensions)[0];
941  double& gxmax = (*ret_grid_dimensions)[1];
942  double& gymin = (*ret_grid_dimensions)[2];
943  double& gymax = (*ret_grid_dimensions)[3];
944  double& gzmin = (*ret_grid_dimensions)[4];
945  double& gzmax = (*ret_grid_dimensions)[5];
946  for (int tid = 0; tid < max_threads; tid++) {
947  // x
948  if (xmin[tid][0] < gxmin) {
949  gxmin = xmin[tid][0];
950  }
951  if (xmax[tid][0] > gxmax) {
952  gxmax = xmax[tid][0];
953  }
954  // y
955  if (ymin[tid][0] < gymin) {
956  gymin = ymin[tid][0];
957  }
958  if (ymax[tid][0] > gymax) {
959  gymax = ymax[tid][0];
960  }
961  // z
962  if (zmin[tid][0] < gzmin) {
963  gzmin = zmin[tid][0];
964  }
965  if (zmax[tid][0] > gzmax) {
966  gzmax = zmax[tid][0];
967  }
968  // larget object
969  if (largest[tid][0] > largest_object_size_) {
970  largest_object_size_ = largest[tid][0];
971  }
972  }
973  }
