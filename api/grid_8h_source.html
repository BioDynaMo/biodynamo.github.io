194  NeighborIterator& ForwardToNonEmptyBox() {
195  // increment box id until non empty box has been found
196  while (++box_idx_ < neighbor_boxes_.size()) {
197  // box is empty or uninitialized (padding box) -> continue
198  if (neighbor_boxes_[box_idx_]->IsEmpty()) {
199  continue;
200  }
201  // a non-empty box has been found
202  box_iterator_ = neighbor_boxes_[box_idx_]->begin();
203  return *this;
204  }
205  // all remaining boxes have been empty; reached end
206  is_end_ = true;
207  return *this;

255  void UpdateGrid() {
256  ClearGrid();
257 
258  auto inf = Constant::kInfinity;
259  std::array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
260  CalculateGridDimensions(&tmp_dim);
261  RoundOffGridDimensions(tmp_dim);
262 
263  auto los = ceil(largest_object_size_);
264  assert(los > 0 &&
265  "The largest object size was found to be 0. Please check if your "
266  "cells are correctly initialized.");
267  box_length_ = los;
268 
269  for (int i = 0; i < 3; i++) {
270  int dimension_length =
271  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
272  int r = dimension_length % box_length_;
273  // If the grid is not perfectly divisible along each dimension by the
274  // resolution, extend the grid so that it is
275  if (r != 0) {
276  // std::abs for the case that box_length_ > dimension_length
277  grid_dimensions_[2 * i + 1] += (box_length_ - r);
278  } else {
279  // Else extend the grid dimension with one row, because the outmost
280  // object lies exactly on the border
281  grid_dimensions_[2 * i + 1] += box_length_;
282  }
283  }
284 
285  // Pad the grid to avoid out of bounds check when search neighbors
286  for (int i = 0; i < 3; i++) {
287  grid_dimensions_[2 * i] -= box_length_;
288  grid_dimensions_[2 * i + 1] += box_length_;
289  }
290 
291  // Calculate how many boxes fit along each dimension
292  for (int i = 0; i < 3; i++) {
293  int dimension_length =
294  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
295  assert((dimension_length % box_length_ == 0) &&
296  "The grid dimensions are not a multiple of its box length");
297  num_boxes_axis_[i] = dimension_length / box_length_;
298  }
299 
300  num_boxes_xy_ = num_boxes_axis_[0] * num_boxes_axis_[1];
301  auto total_num_boxes = num_boxes_xy_ * num_boxes_axis_[2];
302 
303  CheckGridGrowth();
304 
305  if (boxes_.size() != total_num_boxes) {
306  boxes_.resize(total_num_boxes, Box());
307  }
308 
309  successors_.Initialize();
310 
311  // Assign simulation objects to boxes
312  auto* rm = TSimulation::GetActive()->GetResourceManager();
313  rm->ApplyOnAllElementsParallel([this](auto&& sim_object, SoHandle id) {
314  const auto& position = sim_object.GetPosition();
315  auto idx = this->GetBoxIndex(position);
316  auto box = this->GetBoxPointer(idx);
317  box->AddObject(id, &successors_);
318  sim_object.SetBoxIdx(idx);
319  });
320  }

322  void CheckGridGrowth() {
323  // Determine if the grid dimensions have changed (changed in the sense that
324  // the grid has grown outwards)
325  auto min_gd =
326  *std::min_element(grid_dimensions_.begin(), grid_dimensions_.end());
327  auto max_gd =
328  *std::max_element(grid_dimensions_.begin(), grid_dimensions_.end());
329  if (min_gd < threshold_dimensions_[0]) {
330  threshold_dimensions_[0] = min_gd;
331  has_grown_ = true;
332  }
333  if (max_gd > threshold_dimensions_[1]) {
334  Log::Info("Grid",
335  "Your simulation objects are getting near the edge of "
336  "the simulation space. Be aware of boundary conditions that "
337  "may come into play!");
338  threshold_dimensions_[1] = max_gd;
339  has_grown_ = true;
340  }

345  void CalculateGridDimensions(std::array<double, 6>* ret_grid_dimensions) {
346  auto* rm = TSimulation::GetActive()->GetResourceManager();
347 
348  const auto max_threads = omp_get_max_threads();
349 
350  std::vector<std::array<double, 6>*> all_grid_dimensions(max_threads,
351  nullptr);
352  std::vector<double*> all_largest_object_size(max_threads, nullptr);
353 
354 #pragma omp parallel
355  {
356  auto thread_id = omp_get_thread_num();
357  auto* grid_dimensions = new std::array<double, 6>;
358  *grid_dimensions = {{Constant::kInfinity, -Constant::kInfinity,
359  Constant::kInfinity, -Constant::kInfinity,
360  Constant::kInfinity, -Constant::kInfinity}};
361  double* largest_object_size = new double;
362  *largest_object_size = 0;
363  all_grid_dimensions[thread_id] = grid_dimensions;
364  all_largest_object_size[thread_id] = largest_object_size;
365 
366  rm->ApplyOnAllTypes([&](auto* sim_objects, uint16_t type_idx) {
367 #pragma omp for
368  for (size_t i = 0; i < sim_objects->size(); i++) {
369  const auto& position = (*sim_objects)[i].GetPosition();
370  for (size_t j = 0; j < 3; j++) {
371  if (position[j] < (*grid_dimensions)[2 * j]) {
372  (*grid_dimensions)[2 * j] = position[j];
373  }
374  if (position[j] > (*grid_dimensions)[2 * j + 1]) {
375  (*grid_dimensions)[2 * j + 1] = position[j];
376  }
377  }
378  auto diameter = (*sim_objects)[i].GetDiameter();
379  if (diameter > *largest_object_size) {
380  *largest_object_size = diameter;
381  }
382  }
383  });
384 
385 #pragma omp master
386  {
387  for (int i = 0; i < max_threads; i++) {
388  for (size_t j = 0; j < 3; j++) {
389  if ((*all_grid_dimensions[i])[2 * j] <
390  (*ret_grid_dimensions)[2 * j]) {
391  (*ret_grid_dimensions)[2 * j] = (*all_grid_dimensions[i])[2 * j];
392  }
393  if ((*all_grid_dimensions[i])[2 * j + 1] >
394  (*ret_grid_dimensions)[2 * j + 1]) {
395  (*ret_grid_dimensions)[2 * j + 1] =
396  (*all_grid_dimensions[i])[2 * j + 1];
397  }
398  }
399  if ((*all_largest_object_size[i]) > largest_object_size_) {
400  largest_object_size_ = *(all_largest_object_size[i]);
401  }
402  }
403  }
404  }
405 
406  for (auto element : all_grid_dimensions) {
407  delete element;
408  }
409  for (auto element : all_largest_object_size) {
410  delete element;
411  }
412  }

437  inline double SquaredEuclideanDistance(
438  const std::array<double, 3>& pos1,
439  const std::array<double, 3>& pos2) const {
440  const double dx = pos2[0] - pos1[0];
441  const double dy = pos2[1] - pos1[1];
442  const double dz = pos2[2] - pos1[2];
443  return (dx * dx + dy * dy + dz * dz);
444  }

539  template <typename TLambda>
540  void ForEachNeighborPairWithinRadius(const TLambda& lambda,
541  double squared_radius) const {
542  uint32_t z_start = 0, y_start = 0;
543  auto* rm = TSimulation::GetActive()->GetResourceManager();
544  // use special iteration pattern to avoid race conditions between neighbors
545  // main iteration will be done over rows of boxes. In order to avoid two
546  // threads accessing the same box, one has to use a margin reagion of two
547  // boxes in the y and z dimension.
548  for (uint16_t i = 0; i < 9; i++) {
549  switch (i) {
550  case 0:
551  z_start = 1;
552  y_start = 1;
553  break;
554  case 1:
555  z_start = 1;
556  y_start = 2;
557  break;
558  case 2:
559  z_start = 1;
560  y_start = 3;
561  break;
562  case 3:
563  z_start = 2;
564  y_start = 1;
565  break;
566  case 4:
567  z_start = 2;
568  y_start = 2;
569  break;
570  case 5:
571  z_start = 2;
572  y_start = 3;
573  break;
574  case 6:
575  z_start = 3;
576  y_start = 1;
577  break;
578  case 7:
579  z_start = 3;
580  y_start = 2;
581  break;
582  case 8:
583  z_start = 3;
584  y_start = 3;
585  break;
586  }
587 #pragma omp parallel
588  {
589  FixedSizeVector<size_t, 14> box_indices;
590  CircularBuffer<InlineVector<SoHandle, 4>, 14> cached_so_handles;
591  CircularBuffer<InlineVector<std::array<double, 3>, 4>, 14>
592  cached_positions;
593 
594 #pragma omp for collapse(2) schedule(dynamic, 1) firstprivate(z_start, y_start)
595  for (uint32_t z = z_start; z < num_boxes_axis_[2] - 1; z += 3) {
596  for (uint32_t y = y_start; y < num_boxes_axis_[1] - 1; y += 3) {
597  auto current_box_idx =
598  GetBoxIndex(std::array<uint32_t, 3>{1, y, z});
599 
600  box_indices.clear();
601  cached_so_handles.clear();
602  cached_positions.clear();
603 
604  // since we want to execute the lambda ONCE for each cell pair,
605  // it is sufficient to process half of the boxes and omit the
606  // opposite box
607  // Order: C BW FNW NW BNW B FN N BN E BE FNE NE BNE
608  GetHalfMooreBoxIndices(&box_indices, current_box_idx);
609  CacheSoHandles(box_indices, &cached_so_handles);
610  CachePositions(cached_so_handles, rm, &cached_positions);
611 
612  // first iteration peeled off
613  ForEachNeighborPairWithCenterBox(lambda, rm, cached_so_handles,
614  cached_positions, 0,
615  squared_radius);
616 
617  for (uint32_t x = 2; x < num_boxes_axis_[0] - 1; x++) {
618  // since every thread is iterating over a complete row, we can
619  // save time by updating only new boxes
620  ++box_indices;
621  UpdateSoHandles(box_indices, &cached_so_handles);
622  UpdateCachedPositions(cached_so_handles, rm, &cached_positions);
623  // updating `cached_so_handles` and `cached_positions` will change
624  // the
625  // position of the center box from 0 to 4
626  ForEachNeighborPairWithCenterBox(lambda, rm, cached_so_handles,
627  cached_positions, 4,
628  squared_radius);
629  }
630  }
631  }
632  }
633  }

776  void GetMooreBoxes(FixedSizeVector<const Box*, 27>* neighbor_boxes,
777  size_t box_idx) const {
778  neighbor_boxes->push_back(GetBoxPointer(box_idx));
779 
780  // Adjacent 6 (top, down, left, right, front and back)
781  if (adjacency_ >= kLow) {
782  neighbor_boxes->push_back(GetBoxPointer(box_idx - num_boxes_xy_));
783  neighbor_boxes->push_back(GetBoxPointer(box_idx + num_boxes_xy_));
784  neighbor_boxes->push_back(GetBoxPointer(box_idx - num_boxes_axis_[0]));
785  neighbor_boxes->push_back(GetBoxPointer(box_idx + num_boxes_axis_[0]));
786  neighbor_boxes->push_back(GetBoxPointer(box_idx - 1));
787  neighbor_boxes->push_back(GetBoxPointer(box_idx + 1));
788  }
789 
790  // Adjacent 12
791  if (adjacency_ >= kMedium) {
792  neighbor_boxes->push_back(
793  GetBoxPointer(box_idx - num_boxes_xy_ - num_boxes_axis_[0]));
794  neighbor_boxes->push_back(GetBoxPointer(box_idx - num_boxes_xy_ - 1));
795  neighbor_boxes->push_back(
796  GetBoxPointer(box_idx - num_boxes_axis_[0] - 1));
797  neighbor_boxes->push_back(
798  GetBoxPointer(box_idx + num_boxes_xy_ - num_boxes_axis_[0]));
799  neighbor_boxes->push_back(GetBoxPointer(box_idx + num_boxes_xy_ - 1));
800  neighbor_boxes->push_back(
