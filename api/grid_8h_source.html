546  break;

549  y_start = 2;
550  break;
551  case 2:

553  y_start = 3;
554  break;
555  case 3:
556  z_start = 2;
557  y_start = 1;
558  break;
559  case 4:
560  z_start = 2;
561  y_start = 2;
562  break;
563  case 5:
564  z_start = 2;
565  y_start = 3;
566  break;
567  case 6:
568  z_start = 3;
569  y_start = 1;
570  break;
571  case 7:
572  z_start = 3;
573  y_start = 2;
574  break;
575  case 8:

577  y_start = 3;
578  break;

1032  const CircularBuffer<InlineVector<SoHandle, 4>, 14>& so_handles,
1033  const CircularBuffer<InlineVector<std::array<double, 3>, 4>, 14>&
1034  positions,
1035  uint64_t center_box_idx, double squared_radius) const {
1036  const auto& soh_center_box = so_handles[center_box_idx];
1037  const auto& pos_center_box = positions[center_box_idx];
1038  if (soh_center_box.size() == 0) {
1039  return;
1040  }
1041  if (soh_center_box.size() > 1) {
1042  // pairs within the center box
1043  for (size_t n = 0; n < soh_center_box.size(); n++) {
1044  rm->ApplyOnElement(soh_center_box[n], [&, this](auto&& element_n) {
1045  const auto& pos_n = pos_center_box[n];
1046  for (size_t c = n + 1; c < soh_center_box.size(); c++) {
1047  rm->ApplyOnElement(soh_center_box[c], [&, this](auto&& element_c) {
1048  const auto& pos_c = pos_center_box[c];
1049  if (this->SquaredEuclideanDistance(pos_c, pos_n) <
1050  squared_radius) {
1051  lambda(element_c, soh_center_box[c], element_n,
1052  soh_center_box[n]);
1053  }
1054  });
1055  }
1056  });
1057  }
1058  }
1059 
1060  // pairs with one cell in the center box and the other in a surrounding one
1061  for (size_t i = 0; i < 14; i++) {
1062  if (i == center_box_idx) {
1063  continue;
1064  }
1065  const auto& soh_box = so_handles[i];
1066  const auto& pos_box = positions[i];

1072  const auto& pos_c = pos_center_box[c];

1074  squared_radius) {
