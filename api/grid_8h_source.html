197  NeighborIterator& ForwardToNonEmptyBox() {
198  // increment box id until non empty box has been found
199  while (++box_idx_ < neighbor_boxes_.size()) {

256  void UpdateGrid() {
257  auto* rm = TSimulation::GetActive()->GetResourceManager();
258 
259  if (rm->GetNumSimObjects() != 0) {
260  ClearGrid();
261 
262  auto inf = Constant::kInfinity;
263  std::array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
264  CalculateGridDimensions(&tmp_dim);
265  RoundOffGridDimensions(tmp_dim);
266 
267  auto los = ceil(largest_object_size_);
268  assert(los > 0 &&
269  "The largest object size was found to be 0. Please check if your "
270  "cells are correctly initialized.");
271  box_length_ = los;
272 
273  for (int i = 0; i < 3; i++) {
274  int dimension_length =
275  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
276  int r = dimension_length % box_length_;
277  // If the grid is not perfectly divisible along each dimension by the
278  // resolution, extend the grid so that it is
279  if (r != 0) {
280  // std::abs for the case that box_length_ > dimension_length
281  grid_dimensions_[2 * i + 1] += (box_length_ - r);
282  } else {
283  // Else extend the grid dimension with one row, because the outmost
284  // object lies exactly on the border
285  grid_dimensions_[2 * i + 1] += box_length_;
286  }
287  }
288 
289  // Pad the grid to avoid out of bounds check when search neighbors
290  for (int i = 0; i < 3; i++) {
291  grid_dimensions_[2 * i] -= box_length_;
292  grid_dimensions_[2 * i + 1] += box_length_;
293  }
294 
295  // Calculate how many boxes fit along each dimension
296  for (int i = 0; i < 3; i++) {
297  int dimension_length =
298  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
299  assert((dimension_length % box_length_ == 0) &&
300  "The grid dimensions are not a multiple of its box length");
301  num_boxes_axis_[i] = dimension_length / box_length_;
302  }
303 
304  num_boxes_xy_ = num_boxes_axis_[0] * num_boxes_axis_[1];
305  auto total_num_boxes = num_boxes_xy_ * num_boxes_axis_[2];
306 
307  CheckGridGrowth();
308 
309  if (boxes_.size() != total_num_boxes) {
310  boxes_.resize(total_num_boxes, Box());
311  }
312 
313  successors_.Reserve();
314 
315  // Assign simulation objects to boxes
316  rm->ApplyOnAllElementsParallel([this](auto&& sim_object, SoHandle id) {
317  const auto& position = sim_object.GetPosition();
318  auto idx = this->GetBoxIndex(position);
319  auto box = this->GetBoxPointer(idx);
320  box->AddObject(id, &successors_);
321  sim_object.SetBoxIdx(idx);
322  });
323  auto* param = TSimulation::GetActive()->GetParam();
324  if (param->bound_space_) {
325  int min = param->min_bound_;
326  int max = param->max_bound_;
327  threshold_dimensions_ = {min, max};
328  }
329  } else {
330  // There are no sim objects in this simulation
331  auto* param = TSimulation::GetActive()->GetParam();
332 
333  bool uninitialized = boxes_.size() == 0;
334  if (uninitialized && param->bound_space_) {
335  // Simulation has never had any simulation objects
336  // Initialize grid dimensions with `Param::min_bound_` and
337  // `Param::max_bound_`
338  // This is required for the DiffusionGrid
339  int min = param->min_bound_;
340  int max = param->max_bound_;
341  grid_dimensions_ = {min, max, min, max, min, max};
342  threshold_dimensions_ = {min, max};
343  has_grown_ = true;
344  } else if (!uninitialized) {
345  // all simulation objects have been removed in the last iteration
346  // grid state remains the same, but we have to set has_grown_ to false
347  // otherwise the DiffusionGrid will attempt to resize
348  has_grown_ = false;
349  } else {
350  Log::Fatal(
351  "Grid",
352  "You tried to initialize an empty simulation without bound space. "
353  "Therefore we cannot determine the size of the simulation space. "
354  "Please add simulation objects, or set Param::bound_space_, "
355  "Param::min_bound_, and Param::max_bound_.");
356  }
357  }
358  }

368  inline double SquaredEuclideanDistance(
369  const std::array<double, 3>& pos1,
370  const std::array<double, 3>& pos2) const {
371  const double dx = pos2[0] - pos1[0];
372  const double dy = pos2[1] - pos1[1];
373  const double dz = pos2[2] - pos1[2];
374  return (dx * dx + dy * dy + dz * dz);
375  }

377  inline bool WithinSquaredEuclideanDistance(
378  double squared_radius, const std::array<double, 3>& pos1,
379  const std::array<double, 3>& pos2) const {
380  const double dx = pos2[0] - pos1[0];
381  const double dx2 = dx * dx;
382  if (dx2 > squared_radius)
383  return false;
384 
385  const double dy = pos2[1] - pos1[1];
386  const double dy2_plus_dx2 = dy * dy + dx2;
387  if (dy2_plus_dx2 > squared_radius)
388  return false;
389 
390  const double dz = pos2[2] - pos1[2];
391  const double distance = dz * dz + dy2_plus_dx2;
392  return distance < squared_radius;
393  }

488  template <typename TLambda>
489  void ForEachNeighborPairWithinRadius(const TLambda& lambda,
490  double squared_radius) const {
491  uint32_t z_start = 0, y_start = 0;
492  auto* rm = TSimulation::GetActive()->GetResourceManager();
493  // use special iteration pattern to avoid race conditions between neighbors
494  // main iteration will be done over rows of boxes. In order to avoid two
495  // threads accessing the same box, one has to use a margin reagion of two
496  // boxes in the y and z dimension.
497  for (uint16_t i = 0; i < 9; i++) {
498  switch (i) {
499  case 0:
500  z_start = 1;
501  y_start = 1;
502  break;
503  case 1:
504  z_start = 1;
505  y_start = 2;
506  break;
507  case 2:
508  z_start = 1;
509  y_start = 3;
510  break;
511  case 3:
512  z_start = 2;
513  y_start = 1;
514  break;
515  case 4:
516  z_start = 2;
517  y_start = 2;
518  break;
519  case 5:
520  z_start = 2;
521  y_start = 3;
522  break;
523  case 6:
524  z_start = 3;
525  y_start = 1;
526  break;
527  case 7:
528  z_start = 3;
529  y_start = 2;
530  break;
531  case 8:
532  z_start = 3;
533  y_start = 3;
534  break;
535  }
536 #pragma omp parallel
537  {
538  FixedSizeVector<size_t, 14> box_indices;
539  CircularBuffer<InlineVector<SoHandle, 4>, 14> cached_so_handles;
540  CircularBuffer<InlineVector<std::array<double, 3>, 4>, 14>
541  cached_positions;
542 
543 #pragma omp for collapse(2) schedule(dynamic, 1) firstprivate(z_start, y_start)
544  for (uint32_t z = z_start; z < num_boxes_axis_[2] - 1; z += 3) {
545  for (uint32_t y = y_start; y < num_boxes_axis_[1] - 1; y += 3) {
546  auto current_box_idx =
547  GetBoxIndex(std::array<uint32_t, 3>{1, y, z});
548 
549  box_indices.clear();
550  cached_so_handles.clear();
551  cached_positions.clear();
552 
553  // since we want to execute the lambda ONCE for each cell pair,
554  // it is sufficient to process half of the boxes and omit the
555  // opposite box
556  // Order: C BW FNW NW BNW B FN N BN E BE FNE NE BNE
557  GetHalfMooreBoxIndices(&box_indices, current_box_idx);
558  CacheSoHandles(box_indices, &cached_so_handles);
559  CachePositions(cached_so_handles, rm, &cached_positions);
560 
561  // first iteration peeled off
562  ForEachNeighborPairWithCenterBox(lambda, rm, cached_so_handles,
563  cached_positions, 0,
564  squared_radius);
565 
566  for (uint32_t x = 2; x < num_boxes_axis_[0] - 1; x++) {
567  // since every thread is iterating over a complete row, we can
568  // save time by updating only new boxes
569  ++box_indices;
570  UpdateSoHandles(box_indices, &cached_so_handles);
571  UpdateCachedPositions(cached_so_handles, rm, &cached_positions);
572  // updating `cached_so_handles` and `cached_positions` will change
573  // the
574  // position of the center box from 0 to 4
575  ForEachNeighborPairWithCenterBox(lambda, rm, cached_so_handles,
576  cached_positions, 4,
577  squared_radius);
578  }
579  }
580  }
581  }
582  }

742  void CalculateGridDimensions(std::array<double, 6>* ret_grid_dimensions) {
743  auto* rm = TSimulation::GetActive()->GetResourceManager();
744 
745  const auto max_threads = omp_get_max_threads();
746 
747  std::vector<std::array<double, 6>*> all_grid_dimensions(max_threads,
748  nullptr);
749  std::vector<double*> all_largest_object_size(max_threads, nullptr);
750 
751 #pragma omp parallel
752  {
753  auto thread_id = omp_get_thread_num();
754  auto* grid_dimensions = new std::array<double, 6>;
755  *grid_dimensions = {{Constant::kInfinity, -Constant::kInfinity,
756  Constant::kInfinity, -Constant::kInfinity,
757  Constant::kInfinity, -Constant::kInfinity}};
758  double* largest_object_size = new double;
759  *largest_object_size = 0;
760  all_grid_dimensions[thread_id] = grid_dimensions;
761  all_largest_object_size[thread_id] = largest_object_size;
762 
763  rm->ApplyOnAllTypes([&](auto* sim_objects, uint16_t type_idx) {
764 #pragma omp for
765  for (size_t i = 0; i < sim_objects->size(); i++) {
766  const auto& position = (*sim_objects)[i].GetPosition();
767  for (size_t j = 0; j < 3; j++) {
768  if (position[j] < (*grid_dimensions)[2 * j]) {
769  (*grid_dimensions)[2 * j] = position[j];
770  }
771  if (position[j] > (*grid_dimensions)[2 * j + 1]) {
772  (*grid_dimensions)[2 * j + 1] = position[j];
773  }
774  }
775  auto diameter = (*sim_objects)[i].GetDiameter();
776  if (diameter > *largest_object_size) {
777  *largest_object_size = diameter;
778  }
779  }
780  });
781 
782 #pragma omp master
783  {
784  for (int i = 0; i < max_threads; i++) {
785  for (size_t j = 0; j < 3; j++) {
786  if ((*all_grid_dimensions[i])[2 * j] <
787  (*ret_grid_dimensions)[2 * j]) {
788  (*ret_grid_dimensions)[2 * j] = (*all_grid_dimensions[i])[2 * j];
789  }
790  if ((*all_grid_dimensions[i])[2 * j + 1] >
791  (*ret_grid_dimensions)[2 * j + 1]) {
792  (*ret_grid_dimensions)[2 * j + 1] =
793  (*all_grid_dimensions[i])[2 * j + 1];
794  }
795  }
796  if ((*all_largest_object_size[i]) > largest_object_size_) {
797  largest_object_size_ = *(all_largest_object_size[i]);
798  }
799  }
800  }
801  }
802 
803  for (auto element : all_grid_dimensions) {
804  delete element;
805  }
806  for (auto element : all_largest_object_size) {
807  delete element;
808  }
809  }
