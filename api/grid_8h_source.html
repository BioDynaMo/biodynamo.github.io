197  NeighborIterator& ForwardToNonEmptyBox() {
198  // increment box id until non empty box has been found
199  while (++box_idx_ < neighbor_boxes_.size()) {
200  // box is empty or uninitialized (padding box) -> continue
201  if (neighbor_boxes_[box_idx_]->IsEmpty()) {

262  void UpdateGrid() {
263  ClearGrid();
264 
265  auto inf = Constant::kInfinity;
266  array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
267  CalculateGridDimensions(&tmp_dim);
268  RoundOffGridDimensions(tmp_dim);
269 
270  auto los = ceil(largest_object_size_);
271  assert(los > 0 &&
272  "The largest object size was found to be 0. Please check if your "
273  "cells are correctly initialized.");
274  box_length_ = los;
275 
276  for (int i = 0; i < 3; i++) {
277  int dimension_length =
278  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
279  int r = dimension_length % box_length_;
280  // If the grid is not perfectly divisible along each dimension by the
281  // resolution, extend the grid so that it is
282  if (r != 0) {
283  // std::abs for the case that box_length_ > dimension_length
284  grid_dimensions_[2 * i + 1] += (box_length_ - r);
285  } else {
286  // Else extend the grid dimension with one row, because the outmost
287  // object lies exactly on the border
288  grid_dimensions_[2 * i + 1] += box_length_;
289  }
290  }
291 
292  // Pad the grid to avoid out of bounds check when search neighbors
293  for (int i = 0; i < 3; i++) {
294  grid_dimensions_[2 * i] -= box_length_;
295  grid_dimensions_[2 * i + 1] += box_length_;
296  }
297 
298  // Calculate how many boxes fit along each dimension
299  for (int i = 0; i < 3; i++) {
300  int dimension_length =
301  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
302  assert((dimension_length % box_length_ == 0) &&
303  "The grid dimensions are not a multiple of its box length");
304  num_boxes_axis_[i] = dimension_length / box_length_;
305  }
306 
307  num_boxes_xy_ = num_boxes_axis_[0] * num_boxes_axis_[1];
308  auto total_num_boxes = num_boxes_xy_ * num_boxes_axis_[2];
309 
310  CheckGridGrowth();
311 
312  if (boxes_.size() != total_num_boxes) {
313  boxes_.resize(total_num_boxes, Box());
314  }
315 
316  successors_.Initialize();
317 
318  // Assign simulation objects to boxes
319  auto rm = TResourceManager::Get();
320  rm->ApplyOnAllElementsParallel([this](auto&& sim_object, SoHandle id) {
321  const auto& position = sim_object.GetPosition();
322  auto idx = this->GetBoxIndex(position);
323  auto box = this->GetBoxPointer(idx);
324  box->AddObject(id, &successors_);
325  sim_object.SetBoxIdx(idx);
326  });
327  }

329  void CheckGridGrowth() {
330  // Determine if the grid dimensions have changed (changed in the sense that
331  // the grid has grown outwards)
332  auto min_gd =
333  *std::min_element(grid_dimensions_.begin(), grid_dimensions_.end());
334  auto max_gd =
335  *std::max_element(grid_dimensions_.begin(), grid_dimensions_.end());
336  if (min_gd < threshold_dimensions_[0]) {
337  threshold_dimensions_[0] = min_gd;
338  has_grown_ = true;
339  }
340  if (max_gd > threshold_dimensions_[1]) {
341  Log::Info("Grid",
342  "Your simulation objects are getting near the edge of "
343  "the simulation space. Be aware of boundary conditions that "
344  "may come into play!");
345  threshold_dimensions_[1] = max_gd;
346  has_grown_ = true;
347  }

352  void CalculateGridDimensions(array<double, 6>* ret_grid_dimensions) {
353  auto rm = TResourceManager::Get();
354 
355  const auto max_threads = omp_get_max_threads();
356 
357  std::vector<std::array<double, 6>*> all_grid_dimensions(max_threads,
358  nullptr);
359  std::vector<double*> all_largest_object_size(max_threads, nullptr);
360 
361 #pragma omp parallel
362  {
363  auto thread_id = omp_get_thread_num();
364  auto* grid_dimensions = new std::array<double, 6>;
365  *grid_dimensions = {{Constant::kInfinity, -Constant::kInfinity,
366  Constant::kInfinity, -Constant::kInfinity,
367  Constant::kInfinity, -Constant::kInfinity}};
368  double* largest_object_size = new double;
369  *largest_object_size = 0;
370  all_grid_dimensions[thread_id] = grid_dimensions;
371  all_largest_object_size[thread_id] = largest_object_size;
372 
373  rm->ApplyOnAllTypes([&](auto* sim_objects, uint16_t type_idx) {
374 #pragma omp for
375  for (size_t i = 0; i < sim_objects->size(); i++) {
376  const auto& position = (*sim_objects)[i].GetPosition();
377  for (size_t j = 0; j < 3; j++) {
378  if (position[j] < (*grid_dimensions)[2 * j]) {
379  (*grid_dimensions)[2 * j] = position[j];
380  }
381  if (position[j] > (*grid_dimensions)[2 * j + 1]) {
382  (*grid_dimensions)[2 * j + 1] = position[j];
383  }
384  }
385  auto diameter = (*sim_objects)[i].GetDiameter();
386  if (diameter > *largest_object_size) {
387  *largest_object_size = diameter;
388  }
389  }
390  });
391 
392 #pragma omp master
393  {
394  for (int i = 0; i < max_threads; i++) {
395  for (size_t j = 0; j < 3; j++) {
396  if ((*all_grid_dimensions[i])[2 * j] <
397  (*ret_grid_dimensions)[2 * j]) {
398  (*ret_grid_dimensions)[2 * j] = (*all_grid_dimensions[i])[2 * j];
399  }
400  if ((*all_grid_dimensions[i])[2 * j + 1] >
401  (*ret_grid_dimensions)[2 * j + 1]) {
402  (*ret_grid_dimensions)[2 * j + 1] =
403  (*all_grid_dimensions[i])[2 * j + 1];
404  }
405  }
406  if ((*all_largest_object_size[i]) > largest_object_size_) {
407  largest_object_size_ = *(all_largest_object_size[i]);
408  }
409  }
410  }
411  }
412 
413  for (auto element : all_grid_dimensions) {
414  delete element;
415  }
416  for (auto element : all_largest_object_size) {
417  delete element;
418  }
419  }

444  inline double SquaredEuclideanDistance(
445  const std::array<double, 3>& pos1,
446  const std::array<double, 3>& pos2) const {
447  const double dx = pos2[0] - pos1[0];
448  const double dy = pos2[1] - pos1[1];

450  return (dx * dx + dy * dy + dz * dz);

546  template <typename TLambda>
547  void ForEachNeighborPairWithinRadius(const TLambda& lambda,
548  double squared_radius) const {
549  uint32_t z_start = 0, y_start = 0;
550  auto rm = TResourceManager::Get();
551  // use special iteration pattern to avoid race conditions between neighbors
552  // main iteration will be done over rows of boxes. In order to avoid two
553  // threads accessing the same box, one has to use a margin reagion of two
554  // boxes in the y and z dimension.
555  for (uint16_t i = 0; i < 9; i++) {
556  switch (i) {
557  case 0:
558  z_start = 1;
559  y_start = 1;
560  break;
561  case 1:
562  z_start = 1;
563  y_start = 2;
564  break;
565  case 2:
566  z_start = 1;
567  y_start = 3;
568  break;
569  case 3:
570  z_start = 2;
571  y_start = 1;
572  break;
573  case 4:
574  z_start = 2;
575  y_start = 2;
576  break;
577  case 5:
578  z_start = 2;
579  y_start = 3;
580  break;
581  case 6:
582  z_start = 3;
583  y_start = 1;
584  break;
585  case 7:
586  z_start = 3;
587  y_start = 2;
588  break;
589  case 8:
590  z_start = 3;
591  y_start = 3;
592  break;
593  }
594 #pragma omp parallel
595  {
596  FixedSizeVector<size_t, 14> box_indices;
597  CircularBuffer<InlineVector<SoHandle, 4>, 14> cached_so_handles;
598  CircularBuffer<InlineVector<std::array<double, 3>, 4>, 14>
599  cached_positions;
600 
601 #pragma omp for collapse(2) schedule(dynamic, 1) firstprivate(z_start, y_start)
602  for (uint32_t z = z_start; z < num_boxes_axis_[2] - 1; z += 3) {
603  for (uint32_t y = y_start; y < num_boxes_axis_[1] - 1; y += 3) {
604  auto current_box_idx = GetBoxIndex(array<uint32_t, 3>{1, y, z});
605 
606  box_indices.clear();
607  cached_so_handles.clear();
608  cached_positions.clear();
609 
610  // since we want to execute the lambda ONCE for each cell pair,
611  // it is sufficient to process half of the boxes and omit the
612  // opposite box
613  // Order: C BW FNW NW BNW B FN N BN E BE FNE NE BNE
614  GetHalfMooreBoxIndices(&box_indices, current_box_idx);
615  CacheSoHandles(box_indices, &cached_so_handles);
616  CachePositions(cached_so_handles, rm, &cached_positions);
617 
618  // first iteration peeled off
619  ForEachNeighborPairWithCenterBox(lambda, rm, cached_so_handles,
620  cached_positions, 0,
621  squared_radius);
622 
623  for (uint32_t x = 2; x < num_boxes_axis_[0] - 1; x++) {
624  // since every thread is iterating over a complete row, we can
625  // save time by updating only new boxes
626  ++box_indices;
627  UpdateSoHandles(box_indices, &cached_so_handles);
628  UpdateCachedPositions(cached_so_handles, rm, &cached_positions);
629  // updating `cached_so_handles` and `cached_positions` will change
630  // the
631  // position of the center box from 0 to 4
632  ForEachNeighborPairWithCenterBox(lambda, rm, cached_so_handles,
633  cached_positions, 4,
634  squared_radius);
635  }
636  }
637  }
638  }
639  }

780  void GetMooreBoxes(FixedSizeVector<const Box*, 27>* neighbor_boxes,
781  size_t box_idx) const {
782  neighbor_boxes->push_back(GetBoxPointer(box_idx));
783 
784  // Adjacent 6 (top, down, left, right, front and back)
785  if (adjacency_ >= kLow) {
786  neighbor_boxes->push_back(GetBoxPointer(box_idx - num_boxes_xy_));
787  neighbor_boxes->push_back(GetBoxPointer(box_idx + num_boxes_xy_));
788  neighbor_boxes->push_back(GetBoxPointer(box_idx - num_boxes_axis_[0]));
789  neighbor_boxes->push_back(GetBoxPointer(box_idx + num_boxes_axis_[0]));
790  neighbor_boxes->push_back(GetBoxPointer(box_idx - 1));
791  neighbor_boxes->push_back(GetBoxPointer(box_idx + 1));
792  }
793 
794  // Adjacent 12
795  if (adjacency_ >= kMedium) {
796  neighbor_boxes->push_back(
797  GetBoxPointer(box_idx - num_boxes_xy_ - num_boxes_axis_[0]));
798  neighbor_boxes->push_back(GetBoxPointer(box_idx - num_boxes_xy_ - 1));
799  neighbor_boxes->push_back(
800  GetBoxPointer(box_idx - num_boxes_axis_[0] - 1));
801  neighbor_boxes->push_back(
802  GetBoxPointer(box_idx + num_boxes_xy_ - num_boxes_axis_[0]));
803  neighbor_boxes->push_back(GetBoxPointer(box_idx + num_boxes_xy_ - 1));
804  neighbor_boxes->push_back(

1045  const TLambda& lambda, TResourceManager* rm,
1046  const CircularBuffer<InlineVector<SoHandle, 4>, 14>& so_handles,
1047  const CircularBuffer<InlineVector<std::array<double, 3>, 4>, 14>&
1048  positions,
1049  uint64_t center_box_idx, double squared_radius) const {
1050  const auto& soh_center_box = so_handles[center_box_idx];
1051  const auto& pos_center_box = positions[center_box_idx];
1052  if (soh_center_box.size() == 0) {
1053  return;
1054  }
1055  if (soh_center_box.size() > 1) {
1056  // pairs within the center box
1057  for (size_t n = 0; n < soh_center_box.size(); n++) {
1058  rm->ApplyOnElement(soh_center_box[n], [&, this](auto&& element_n) {
1059  const auto& pos_n = pos_center_box[n];
1060  for (size_t c = n + 1; c < soh_center_box.size(); c++) {
1061  rm->ApplyOnElement(soh_center_box[c], [&, this](auto&& element_c) {
1062  const auto& pos_c = pos_center_box[c];
1063  if (this->SquaredEuclideanDistance(pos_c, pos_n) <
1064  squared_radius) {
1065  lambda(element_c, soh_center_box[c], element_n,
1066  soh_center_box[n]);
1067  }
1068  });
1069  }
1070  });
1071  }
1072  }
1073 
1074  // pairs with one cell in the center box and the other in a surrounding one
1075  for (size_t i = 0; i < 14; i++) {
1076  if (i == center_box_idx) {
1077  continue;
1078  }
