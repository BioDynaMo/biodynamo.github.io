133  void Backup() {
134  using std::chrono::seconds;
135  using std::chrono::duration_cast;
136  auto* param = TSimulation::GetActive()->GetParam();
137  if (backup_->BackupEnabled() &&
138  duration_cast<seconds>(Clock::now() - last_backup_).count() >=

148  bool Restore(uint64_t* steps) {
149  if (backup_->RestoreEnabled() && restore_point_ > total_steps_ + *steps) {
150  total_steps_ += *steps;
151  // restore requested, but not last backup was not done during this call to
152  // Simualte. Therefore, we skip it.
153  return true;
154  } else if (backup_->RestoreEnabled() && restore_point_ > total_steps_ &&
155  restore_point_ < total_steps_ + *steps) {
156  // Restore
157  backup_->Restore();
158  *steps = total_steps_ + *steps - restore_point_;
159  total_steps_ = restore_point_;
160  }
161  return false;
162  }

164  void CommitChangesAndUpdateReferences() {

166  auto* rm = sim->GetResourceManager();
167  rm->ApplyOnAllTypesParallel(commit_);
168 
169  const auto& update_info = commit_->GetUpdateInfo();

171  uint16_t type_idx) {
172 #pragma omp parallel for
173  for (uint64_t i = 0; i < sim_objects->size(); i++) {
174  (*sim_objects)[i].UpdateReferences(update_info);
175  }
176  };
177  rm->ApplyOnAllTypes(update_references);
178  }
