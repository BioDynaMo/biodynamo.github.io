21 void DiffusionGrid::Initialize() {
22  if (resolution_ == 0) {
23  Log::Fatal("DiffusionGrid::Initialize",
24  "Resolution cannot be zero. (substance '", substance_name_,
25  "')");
26  }
27 
28  // Get neighbor grid dimensions
29  auto* env = Simulation::GetActive()->GetEnvironment();
30  auto bounds = env->GetDimensionThresholds();
31 
32  grid_dimensions_ = {bounds[0], bounds[1]};
33 
34  // Example: diffusion grid dimensions from 0-40 and resolution
35  // of 4. Resolution must be adjusted otherwise one data pointer will be
36  // missing.
37  // Without adjustment:
38  // box_length_: 10
39  // data points {0, 10, 20, 30} - 40 will be misssing!
40  // With adjustment
41  // box_length_: 13.3
42  // data points: {0, 13.3, 26.6, 39.9}
43  auto adjusted_res =
44  resolution_ == 1 ? 2 : resolution_; // avoid division by 0
45  box_length_ = (grid_dimensions_[1] - grid_dimensions_[0]) /
46  static_cast<double>(adjusted_res - 1);
47  // TODO(ahmad): parametrize the minimum box_length
48  if (box_length_ <= 1e-15) {
49  Log::Fatal("DiffusionGrid::Initialize",
50  "The box length was found to be (close to) zero. Please check "
51  "the parameters for substance '",
52  substance_name_, "'");
53  }
54 
55  ParametersCheck();
56  box_volume_ = box_length_ * box_length_ * box_length_;
57  parity_ = resolution_ % 2;
58  total_num_boxes_ = resolution_ * resolution_ * resolution_;
59 
60  // Allocate memory for the concentration and gradient arrays
61  c1_.resize(total_num_boxes_);
62  c2_.resize(total_num_boxes_);
63  gradients_.resize(total_num_boxes_);
64 }

86 void DiffusionGrid::Update() {
87  // Get neighbor grid dimensions
88  auto* env = Simulation::GetActive()->GetEnvironment();
89  auto bounds = env->GetDimensionThresholds();
90  // Update the grid dimensions such that each dimension ranges from
91  // {bounds[0] - bounds[1]}
92  grid_dimensions_ = {bounds[0], bounds[1]};
93 
94  // If the grid is not perfectly divisible along each dimension by the
95  // box length, extend the grid so that it is
96  int dimension_length = bounds[1] - bounds[0];
97  for (int i = 0; i < 1; i++) {
98  int r = fmod(dimension_length, box_length_);
99  if (r > 1e-9) {
100  // std::abs for the case that box_length_ > dimension_length
101  grid_dimensions_[2 * i + 1] += (box_length_ - r);
102  }
103  }
104 
105  // Calculate new_dimension_length and new_resolution
106  int new_dimension_length = grid_dimensions_[1] - grid_dimensions_[0];
107  size_t new_resolution = std::ceil(new_dimension_length / box_length_);
108 
109  if (new_resolution > resolution_) {
110  // Store the old number of boxes along each axis for comparison
111  size_t tmp_resolution = resolution_;
112 
113  // Set new resolution
114  resolution_ = new_resolution;
115 
116  // We need to maintain the parity of the number of boxes along each
117  // dimension, otherwise copying of the substances to the increases grid
118  // will not be symmetrically done; resulting in shifting of boxes
119  // We add a box in the negative direction, because the only way the parity
120  // could have changed is because of adding a box in the positive direction
121  // (due to the grid not being perfectly divisible; see above)
122  if (resolution_ % 2 != parity_) {
123  grid_dimensions_[0] -= box_length_;
124  resolution_++;
125  }
126 
127  // Temporarily save previous grid data
128  auto tmp_c1 = c1_;
129  auto tmp_gradients = gradients_;
130 
131  c1_.clear();
132  c2_.clear();
133  gradients_.clear();
134 
135  total_num_boxes_ = resolution_ * resolution_ * resolution_;
136 
137  CopyOldData(tmp_c1, tmp_gradients, tmp_resolution);
138  }
139 }

141 void DiffusionGrid::CopyOldData(
142  const ParallelResizeVector<double>& old_c1,
143  const ParallelResizeVector<Double3>& old_gradients, size_t old_resolution) {
144  // Allocate more memory for the grid data arrays
145  c1_.resize(total_num_boxes_);
146  c2_.resize(total_num_boxes_);
147  gradients_.resize(total_num_boxes_);
148 
149  auto incr_num_boxes = resolution_ - old_resolution;
150  int off_dim = incr_num_boxes / 2;
151 
152  int num_box_xy = resolution_ * resolution_;
153  int old_box_xy = old_resolution * old_resolution;
154  int new_origin = off_dim * (num_box_xy) + off_dim * resolution_ + off_dim;
155  for (size_t k = 0; k < old_resolution; k++) {
156  int offset = new_origin + k * num_box_xy;
157  for (size_t j = 0; j < old_resolution; j++) {
158  if (j != 0) {
159  offset += resolution_;
160  }
161  for (size_t i = 0; i < old_resolution; i++) {
162  auto idx = k * old_box_xy + j * old_resolution + i;
163  c1_[offset + i] = old_c1[idx];
164  gradients_[offset + i] = old_gradients[idx];
165  }
166  }
167  }
168 }

170 void DiffusionGrid::RunInitializers() {
171  if (initializers_.empty()) {
172  return;
173  }
174 
175  auto nx = resolution_;
176  auto ny = resolution_;
177  auto nz = resolution_;
178 
179  // Apply all functions that initialize this diffusion grid
180  for (size_t f = 0; f < initializers_.size(); f++) {
181  for (uint32_t x = 0; x < nx; x++) {
182  double real_x = grid_dimensions_[0] + x * box_length_;
183  for (uint32_t y = 0; y < ny; y++) {
184  double real_y = grid_dimensions_[0] + y * box_length_;
185  for (uint32_t z = 0; z < nz; z++) {
186  double real_z = grid_dimensions_[0] + z * box_length_;
187  std::array<uint32_t, 3> box_coord = {x, y, z};
188  size_t idx = GetBoxIndex(box_coord);
189  ChangeConcentrationBy(idx, initializers_[f](real_x, real_y, real_z));
190  }
191  }
192  }
193  }
194 
195  // Clear the initializer to free up space
196  initializers_.clear();
197  initializers_.shrink_to_fit();
198 }

200 void DiffusionGrid::CalculateGradient() {
201  // check if gradient has been calculated once
202  // and if diffusion coefficient and decay constant are 0
203  // i.e. if we don't need to calculate gradient update
204  if (init_gradient_ && IsFixedSubstance()) {
205  return;
206  }
207 
208  double gd = 1 / (box_length_ * 2);
209 
210  auto nx = resolution_;
211  auto ny = resolution_;
212  auto nz = resolution_;
213 
214 #pragma omp parallel for collapse(2)
215  for (size_t z = 0; z < nz; z++) {
216  for (size_t y = 0; y < ny; y++) {
217  for (size_t x = 0; x < nx; x++) {
218  int c, e, w, n, s, b, t;
219  c = x + y * nx + z * nx * ny;
220 
221  if (x == 0) {
222  e = c;
223  w = c + 2;
224  } else if (x == nx - 1) {
225  e = c - 2;
226  w = c;
227  } else {
228  e = c - 1;
229  w = c + 1;
230  }
231 
232  if (y == 0) {
233  n = c + 2 * nx;
234  s = c;
235  } else if (y == ny - 1) {
236  n = c;
237  s = c - 2 * nx;
238  } else {
239  n = c + nx;
240  s = c - nx;
241  }
242 
243  if (z == 0) {
244  t = c + 2 * nx * ny;
245  b = c;
246  } else if (z == nz - 1) {
247  t = c;
248  b = c - 2 * nx * ny;
249  } else {
250  t = c + nx * ny;
251  b = c - nx * ny;
252  }
253 
254  // Let the gradient point from low to high concentration
255  gradients_[c][0] = (c1_[w] - c1_[e]) * gd;
256  gradients_[c][1] = (c1_[n] - c1_[s]) * gd;
257  gradients_[c][2] = (c1_[t] - c1_[b]) * gd;
258  }
259  }
260  }
261  if (!init_gradient_) {
262  init_gradient_ = true;
263  }
264 }

273 void DiffusionGrid::ChangeConcentrationBy(size_t idx, double amount) {
274  if (idx >= total_num_boxes_) {
275  Log::Error("DiffusionGrid::ChangeConcentrationBy",
276  "You tried to change the concentration outside the bounds of "
277  "the diffusion grid! The change was ignored.");
278  return;
279  }
280  c1_[idx] += amount;
281  if (c1_[idx] > concentration_threshold_) {
282  c1_[idx] = concentration_threshold_;
283  }

292 void DiffusionGrid::GetGradient(const Double3& position,
293  Double3* gradient) const {
294  auto idx = GetBoxIndex(position);
295  if (idx >= total_num_boxes_) {
296  Log::Error("DiffusionGrid::GetGradient",
297  "You tried to get the gradient outside the bounds of "
298  "the diffusion grid! Returning zero gradient.");
299  return;
300  }
301  *gradient = gradients_[idx];

303  if (norm > 1e-10) {

308 std::array<uint32_t, 3> DiffusionGrid::GetBoxCoordinates(
309  const Double3& position) const {
310  std::array<uint32_t, 3> box_coord;
311  box_coord[0] = (floor(position[0]) - grid_dimensions_[0]) / box_length_;
312  box_coord[1] = (floor(position[1]) - grid_dimensions_[0]) / box_length_;
313  box_coord[2] = (floor(position[2]) - grid_dimensions_[0]) / box_length_;
