71 void DiffusionGrid::Initialize() {
72  if (resolution_ == 0) {
73  Log::Fatal("DiffusionGrid::Initialize",
74  "Resolution cannot be zero. (substance '", GetContinuumName(),
75  "')");
76  }
77 
78  // Get neighbor grid dimensions
79  auto* env = Simulation::GetActive()->GetEnvironment();
80  auto bounds = env->GetDimensionThresholds();
81 
82  grid_dimensions_ = {bounds[0], bounds[1]};
83  if (bounds[0] > bounds[1]) {
84  Log::Fatal("DiffusionGrid::Initialize",
85  "The grid dimensions are not correct. Lower bound is greater",
86  " than upper bound. (substance '", GetContinuumName(), "')");
87  }
88 
89  auto adjusted_res =
90  resolution_ == 1 ? 2 : resolution_; // avoid division by 0
91  box_length_ = (grid_dimensions_[1] - grid_dimensions_[0]) /
92  static_cast<real_t>(adjusted_res);
93 
94  // Check if box length is not too small
95  if (box_length_ <= 1e-13) {
96  Log::Fatal("DiffusionGrid::Initialize",
97  "The box length was found to be (close to) zero. Please check "
98  "the parameters for substance '",
99  GetContinuumName(), "'");
100  }
101 
102  box_volume_ = box_length_ * box_length_ * box_length_;
103  parity_ = resolution_ % 2;
104  total_num_boxes_ = resolution_ * resolution_ * resolution_;
105 
106  // Allocate memory for the concentration and gradient arrays
107  locks_.resize(total_num_boxes_);
108  c1_.resize(total_num_boxes_);
109  c2_.resize(total_num_boxes_);
110  gradients_.resize(total_num_boxes_);
111 
112  // Print Info
113  initialized_ = true;
114  if (print_info_with_initialization_) {
115  PrintInfo();
116  }

119 void DiffusionGrid::Diffuse(real_t dt) {
120  // check if diffusion coefficient and decay constant are 0
121  // i.e. if we don't need to calculate diffusion update
122  if (IsFixedSubstance()) {
123  return;
124  }
125 
126  // Note down last timestep
127  last_dt_ = dt;
128  // Set timestep for this iteration.
129  ParametersCheck(dt);
130 
131  auto* param = Simulation::GetActive()->GetParam();
132  if (bc_type_ == BoundaryConditionType::kClosedBoundaries) {
133  DiffuseWithClosedEdge(dt);
134  } else if (bc_type_ == BoundaryConditionType::kOpenBoundaries) {
135  DiffuseWithOpenEdge(dt);
136  } else if (bc_type_ == BoundaryConditionType::kDirichlet) {
137  DiffuseWithDirichlet(dt);
138  } else if (bc_type_ == BoundaryConditionType::kNeumann) {
139  DiffuseWithNeumann(dt);
140  } else {
141  Log::Error(
142  "EulerGrid::Diffuse", "Boundary condition of type '",
143  param->diffusion_boundary_condition,
144  "' is not implemented. Defaulting to 'closed' boundary condition");
145  }

148 void DiffusionGrid::Update() {
149  // Get neighbor grid dimensions
150  auto* env = Simulation::GetActive()->GetEnvironment();
151  auto bounds = env->GetDimensionThresholds();
152  // Update the grid dimensions such that each dimension ranges from
153  // {bounds[0] - bounds[1]}
154  grid_dimensions_ = {bounds[0], bounds[1]};
155 
156  // If the grid is not perfectly divisible along each dimension by the
157  // box length, extend the grid so that it is
158  int dimension_length = bounds[1] - bounds[0];
159  for (int i = 0; i < 1; i++) {
160  int r = fmod(dimension_length, box_length_);
161  if (r > 1e-9) {
162  // std::abs for the case that box_length_ > dimension_length
163  grid_dimensions_[2 * i + 1] += (box_length_ - r);
164  }
165  }
166 
167  // Calculate new_dimension_length and new_resolution
168  int new_dimension_length = grid_dimensions_[1] - grid_dimensions_[0];
169  size_t new_resolution = std::ceil(new_dimension_length / box_length_);
170 
171  if (new_resolution > resolution_) {
172  // Store the old number of boxes along each axis for comparison
173  size_t tmp_resolution = resolution_;
174 
175  // Set new resolution
176  resolution_ = new_resolution;
177 
178  // We need to maintain the parity of the number of boxes along each
179  // dimension, otherwise copying of the substances to the increases grid
180  // will not be symmetrically done; resulting in shifting of boxes
181  // We add a box in the negative direction, because the only way the parity
182  // could have changed is because of adding a box in the positive direction
183  // (due to the grid not being perfectly divisible; see above)
184  if (resolution_ % 2 != parity_) {
185  grid_dimensions_[0] -= box_length_;
186  resolution_++;
187  }
188 
189  // Temporarily save previous grid data
190  auto tmp_c1 = c1_;
191  auto tmp_gradients = gradients_;
192 
193  c1_.clear();
194  c2_.clear();
195  gradients_.clear();
196 
197  total_num_boxes_ = resolution_ * resolution_ * resolution_;
198 
199  CopyOldData(tmp_c1, tmp_gradients, tmp_resolution);
200  }

203 void DiffusionGrid::CopyOldData(
204  const ParallelResizeVector<real_t>& old_c1,
205  const ParallelResizeVector<Real3>& old_gradients, size_t old_resolution) {
206  // Allocate more memory for the grid data arrays
207  locks_.resize(total_num_boxes_);
208  c1_.resize(total_num_boxes_);
209  c2_.resize(total_num_boxes_);
210  gradients_.resize(total_num_boxes_);
211 
212  Log::Warning(
213  "DiffusionGrid::CopyOldData",
214  "The size of the diffusion grid "
215  "increased. BioDynaMo adds a halo around the domain filled with zeros. "
216  "Depending your use-case, this might or might not be what you want. If "
217  "you have non-zero concentrations / temperatures in the surrounding, "
218  "this is likely to cause unphysical effects at the boundary. But if your "
219  "grid values are mostly zero this is likely to work fine. Evaluate your "
220  "results carefully.");
221 
222  auto incr_num_boxes = resolution_ - old_resolution;
223  int off_dim = incr_num_boxes / 2;
224 
225  int num_box_xy = resolution_ * resolution_;
226  int old_box_xy = old_resolution * old_resolution;
227  int new_origin = off_dim * num_box_xy + off_dim * resolution_ + off_dim;
228  for (size_t k = 0; k < old_resolution; k++) {
229  int offset = new_origin + k * num_box_xy;
230  for (size_t j = 0; j < old_resolution; j++) {
231  if (j != 0) {
232  offset += resolution_;
233  }
234  for (size_t i = 0; i < old_resolution; i++) {
235  auto idx = k * old_box_xy + j * old_resolution + i;
236  c1_[offset + i] = old_c1[idx];
237  gradients_[offset + i] = old_gradients[idx];
238  }
239  }
240  }
241  // TODO: here we also need to copy c1_ into c2_ such that the boundaries are
242  // equivalent once we introduce non-zero halos.
243 }

245 void DiffusionGrid::RunInitializers() {
246  // If there are no initializers, we don't need to do anything and can return
247  // immediately
248  if (initializers_.empty()) {
249  return;
250  }
251 
252  // Define variables for loop bounds & boundary condition check
253  const auto kNumBoxes = total_num_boxes_;
254  const auto kGridSize = resolution_;
255  // For certain boudaries, we also need to copy the values to the c2_ array
256  const bool kCopyToC2 =
257  (bc_type_ == BoundaryConditionType::kDirichlet ||
258  bc_type_ == BoundaryConditionType::kClosedBoundaries ||
259  bc_type_ == BoundaryConditionType::kOpenBoundaries);
260 
261 // Apply all functions that initialize this diffusion grid
262 #pragma omp parallel for schedule(static)
263  for (size_t idx = 0; idx < kNumBoxes; idx++) {
264  // Determine the coordinates of the box
265  const std::array<uint32_t, 3> box_coord = GetBoxCoordinates(idx);
266 
267  // Determine the real coordinates of the box
268  std::array<real_t, 3> real_coord;
269 #pragma omp simd
270  for (size_t i = 0; i < 3; i++) {
271  real_coord[i] = grid_dimensions_[0] +
272  static_cast<real_t>(box_coord[i]) * box_length_ +
273  box_length_ / 2.0;
274  }
275 
276  // Calculate the value of the substance in the box
277  real_t value{0};
278  if (bc_type_ == BoundaryConditionType::kDirichlet &&
279  (box_coord[0] == 0 || box_coord[0] == kGridSize - 1 ||
280  box_coord[1] == 0 || box_coord[1] == kGridSize - 1 ||
281  box_coord[2] == 0 || box_coord[2] == kGridSize - 1)) {
282  // Evaluate the boundary condition in case of Dirichlet boundary
283  value = boundary_condition_->Evaluate(real_coord[0], real_coord[1],
284  real_coord[2], 0);
285  } else {
286  for (size_t f = 0; f < initializers_.size(); f++) {
287  value += initializers_[f](real_coord[0], real_coord[1], real_coord[2]);
288  }
289  }
290 
291  // Allow only lower_threshold_ <= value <= upper_threshold_
292  value = std::max(value, lower_threshold_);
293  value = std::min(value, upper_threshold_);
294 
295  // Set the value of the substance in the box
296  c1_[idx] = value;
297 
298  // For certain boundaries, we need to copy the value to c2_
299  if (kCopyToC2) {
300  c2_[idx] = value;
301  }
302  }
303 
304  // Clear the initializer to free up space
305  initializers_.clear();
306  initializers_.shrink_to_fit();
307 }

309 void DiffusionGrid::CalculateGradient() {
310  // check if gradient has been calculated once
311  // and if diffusion coefficient and decay constant are 0
312  // i.e. if we don't need to calculate gradient update
313  if (init_gradient_ && IsFixedSubstance()) {
314  return;
315  }
316  if (!precompute_gradients_) {
317  return;
318  }
319 
320 #pragma omp parallel for collapse(2)
321  for (uint32_t z = 0; z < resolution_; z++) {
322  for (uint32_t y = 0; y < resolution_; y++) {
323  for (uint32_t x = 0; x < resolution_; x++) {
324  size_t idx = x + y * resolution_ + z * resolution_ * resolution_;
325  const std::array<uint32_t, 3> box_coord = {x, y, z};
326  // Get the neighboring boxes
327  const auto neighbors = GetNeighboringBoxes(idx, box_coord);
328  std::array<int, 6> comparison; // array to determine discretization h
329  std::transform(neighbors.begin(), neighbors.end(), comparison.begin(),
330  [idx](size_t n) { return (n == idx) ? 0 : 1; });
331 
332  // Calculate the gradient (no thread safety issues here)
333  gradients_[idx][0] = (c1_[neighbors[1]] - c1_[neighbors[0]]) /
334  ((comparison[1] + comparison[0]) * box_length_);
335  gradients_[idx][1] = (c1_[neighbors[3]] - c1_[neighbors[2]]) /
336  ((comparison[3] + comparison[2]) * box_length_);
337  gradients_[idx][2] = (c1_[neighbors[5]] - c1_[neighbors[4]]) /
338  ((comparison[5] + comparison[4]) * box_length_);
339  }
340  }
341  }
342  if (!init_gradient_) {
343  init_gradient_ = true;
344  }
345 }

355 void DiffusionGrid::ChangeConcentrationBy(size_t idx, real_t amount,
356  InteractionMode mode,
357  bool scale_with_resolution) {
358  if (idx >= total_num_boxes_) {
359  Log::Error("DiffusionGrid::ChangeConcentrationBy",
360  "You tried to change the concentration outside the bounds of "
361  "the diffusion grid! The change was ignored.");
362  return;
363  }
364  if (scale_with_resolution) {
365  // Convert from amount to concentration of substance by dividing by
366  // volume of box
367  amount /= box_length_ * box_length_ * box_length_;
368  }
369  std::lock_guard<Spinlock> guard(locks_[idx]);
370  assert(idx < locks_.size());
371  switch (mode) {
372  case InteractionMode::kAdditive:
373  c1_[idx] += amount;
374  break;
375  case InteractionMode::kExponential:
376  c1_[idx] *= amount;
377  break;
378  case InteractionMode::kLogistic:
379  c1_[idx] += ((amount > 0) ? upper_threshold_ - c1_[idx]
380  : c1_[idx] - lower_threshold_) *
381  amount;
382  break;
383  default:
384  Log::Fatal("DiffusionGrid::ChangeConcentrationBy",
385  "Unknown interaction mode!");
386  }
387 
388  // Enforce upper and lower bounds. (use std::clamp() when moving to C++17)
389  if (c1_[idx] > upper_threshold_) {
390  c1_[idx] = upper_threshold_;
391  } else if (c1_[idx] < lower_threshold_) {
392  c1_[idx] = lower_threshold_;
393  } else {
394  // c1_[idx] is bounded by the thresholds and does not need to be modified
395  }
396 }

417 void DiffusionGrid::GetGradient(const Real3& position, Real3* gradient,
418  bool normalize) const {
419  assert(gradient != nullptr);
420  auto idx = GetBoxIndex(position);
421  if (idx >= total_num_boxes_) {
422  Log::Error("DiffusionGrid::GetGradient",
423  "You tried to get the gradient outside the bounds of "
424  "the diffusion grid! Returning zero gradient.");
425  return;
426  }
427  if (init_gradient_) {
428  *gradient = gradients_[idx];
429  } else {
430  // Get the neighboring boxes
431  const auto neighbors = GetNeighboringBoxes(idx);
432  std::array<int, 6> comparison; // array to determine discretization h
433  std::transform(neighbors.begin(), neighbors.end(), comparison.begin(),
434  [idx](size_t n) { return (n == idx) ? 0 : 1; });
435 
436  // Calculate the gradient (GetConcentration for thread safety)
437  const real_t x_minus = GetConcentration(neighbors[0]);
438  const real_t x_plus = GetConcentration(neighbors[1]);
439  const real_t y_minus = GetConcentration(neighbors[2]);
440  const real_t y_plus = GetConcentration(neighbors[3]);
441  const real_t z_minus = GetConcentration(neighbors[4]);
442  const real_t z_plus = GetConcentration(neighbors[5]);
443 
444  real_t grad_x =
445  (x_plus - x_minus) / ((comparison[1] + comparison[0]) * box_length_);
446  real_t grad_y =
447  (y_plus - y_minus) / ((comparison[3] + comparison[2]) * box_length_);
448  real_t grad_z =
449  (z_plus - z_minus) / ((comparison[5] + comparison[4]) * box_length_);
450 
451  *gradient = Real3({grad_x, grad_y, grad_z});
452  }
453  if (normalize) {
454  auto norm = gradient->Norm();
455  if (norm > 1e-10) {
456  gradient->Normalize(norm);
457  }
458  }
459 }

461 std::array<uint32_t, 3> DiffusionGrid::GetBoxCoordinates(
462  const Real3& position) const {
463  std::array<uint32_t, 3> box_coord;
464 
465  for (size_t i = 0; i < 3; i++) {
466 // Check if position is within boundaries
467 #ifndef NDEBUG
468  assert((position[i] >= grid_dimensions_[0]) &&
469  "You tried to get the box coordinates outside the bounds of the "
470  "diffusion grid!");
471  assert((position[i] <= grid_dimensions_[1]) &&
472  "You tried to get the box coordinates outside the bounds of the "
473  "diffusion grid!");
474 #endif // NDEBUG
475  // Get box coords (Note: conversion to uint32_t should be save for typical
476  // grid sizes)
477  box_coord[i] = static_cast<uint32_t>(
478  std::floor((position[i] - grid_dimensions_[0]) / box_length_));

480  return box_coord;

513 std::array<size_t, 6> DiffusionGrid::GetNeighboringBoxes(
514  size_t index, const std::array<uint32_t, 3>& box_coord) const {
515  std::array<size_t, 6> neighbors;
516  neighbors[0] = (box_coord[0] == 0) ? index : index - 1;
517  neighbors[1] = (box_coord[0] == resolution_ - 1) ? index : index + 1;
518  neighbors[2] = (box_coord[1] == 0) ? index : index - resolution_;
519  neighbors[3] =
520  (box_coord[1] == resolution_ - 1) ? index : index + resolution_;
521  neighbors[4] =
522  (box_coord[2] == 0) ? index : index - resolution_ * resolution_;
523  neighbors[5] = (box_coord[2] == resolution_ - 1)
524  ? index
525  : index + resolution_ * resolution_;
526  return neighbors;
527 }

591 void DiffusionGrid::ParametersCheck(real_t dt) {
592  // We evaluate a stability condition derived via a von Neumann stability
593  // analysis (https://en.wikipedia.org/wiki/Von_Neumann_stability_analysis,
594  // accessed 2022-10-27) of the diffusion equation. In comparison to the
595  // Wikipedia article, we use a 3D diffusion equation and also consider the
596  // decay. We end up with the following result:
597  const bool stability =
598  ((mu_ + 12.0 * (1 - dc_[0]) / (box_length_ * box_length_)) * dt <= 2.0);
599  if (!stability) {
600  const double max_dt =
601  2.0 / (mu_ + 12.0 * (1 - dc_[0])) * (box_length_ * box_length_);
602  Log::Fatal(
603  "DiffusionGrid", "Stability condition violated. ",
604  "The specified parameters of the diffusion grid with substance [",
605  GetContinuumName(),
606  "] will result in unphysical behavior (diffusion coefficient = ",
607  (1 - dc_[0]), ", resolution = ", resolution_,
608  ", decay constant = ", mu_, ", dt = ", dt,
609  "). For the given parameters, the time step must be smaller than ",
610  max_dt, ". Please refer to the user guide for more information.");
611  }
612 
613  // We also check if the decay constant is too large. This is not a stability
614  // condition, but it may result in unphysical behavior, e.g. negative values
615  // for the concentration. We obtain the condition by assuming that all
616  // concentration values of the previous time step are positive and demanding
617  // the same for the next time step. We arrive at the following condition:
618  const bool decay_safety =
619  (1 - (mu_ + 6 * (1 - dc_[0]) / (box_length_ * box_length_) * dt) >= 0);
620  if (!decay_safety) {
621  Log::Fatal(
622  "DiffusionGrid", "Decay may overstep into negative regime. ",
623  "The specified parameters of the diffusion grid with substance [",
624  GetContinuumName(),
625  "] may result in unphysical behavior (diffusion coefficient = ",
626  (1 - dc_[0]), ", resolution = ", resolution_,
627  ", decay constant = ", mu_, ", dt = ", dt,
628  "). Please refer to the user guide for more information.");
629  }
630 }
