33 Double4 InteractionForce::Calculate(const Agent* lhs, const Agent* rhs) const {
34  if (lhs->GetShape() == Shape::kSphere && rhs->GetShape() == Shape::kSphere) {
35  Double3 result;
36  ForceBetweenSpheres(lhs, rhs, &result);
37  return {result[0], result[1], result[2], 0};
38  } else if (lhs->GetShape() == Shape::kSphere &&
39  rhs->GetShape() == Shape::kCylinder) {
40  Double3 result;
41  ForceOnASphereFromACylinder(lhs, rhs, &result);
42  return {result[0], result[1], result[2], 0};
43  } else if (lhs->GetShape() == Shape::kCylinder &&
44  rhs->GetShape() == Shape::kSphere) {
45  Double4 result;
46  ForceOnACylinderFromASphere(lhs, rhs, &result);
47  return result;
48  } else if (lhs->GetShape() == Shape::kCylinder &&
49  rhs->GetShape() == Shape::kCylinder) {
50  Double4 result;
51  ForceBetweenCylinders(lhs, rhs, &result);
52  return result;
53  } else {
54  Log::Fatal("InteractionForce",
55  "InteractionForce only supports sphere or cylinder shapes");
56  return {0, 0, 0, 0};
57  }
58 }

60 void InteractionForce::ForceBetweenSpheres(const Agent* sphere_lhs,
61  const Agent* sphere_rhs,
62  Double3* result) const {
63  const Double3& ref_mass_location = sphere_lhs->GetPosition();
64  double ref_diameter = sphere_lhs->GetDiameter();
65  double ref_iof_coefficient = 0.15;
66  const Double3& nb_mass_location = sphere_rhs->GetPosition();
67  double nb_diameter = sphere_rhs->GetDiameter();
68  double nb_iof_coefficient = 0.15;
69 
70  auto c1 = ref_mass_location;
71  double r1 = 0.5 * ref_diameter;
72  auto c2 = nb_mass_location;
73  double r2 = 0.5 * nb_diameter;
74  // We take virtual bigger radii to have a distant interaction, to get a
75  // desired density.
76  double additional_radius =
77  10.0 * std::min(ref_iof_coefficient, nb_iof_coefficient);
78  r1 += additional_radius;
79  r2 += additional_radius;
80  // the 3 components of the vector c2 -> c1
81  double comp1 = c1[0] - c2[0];
82  double comp2 = c1[1] - c2[1];
83  double comp3 = c1[2] - c2[2];
84  double center_distance =
85  std::sqrt(comp1 * comp1 + comp2 * comp2 + comp3 * comp3);
86  // the overlap distance (how much one penetrates in the other)
87  double delta = r1 + r2 - center_distance;
88  // if no overlap : no force
89  if (delta < 0) {
90  *result = {0.0, 0.0, 0.0};
91  return;
92  }
93  // to avoid a division by 0 if the centers are (almost) at the same
94  // location
95  if (center_distance < 0.00000001) {
96  auto* random = Simulation::GetActive()->GetRandom();
97  auto force2on1 = random->template UniformArray<3>(-3.0, 3.0);
98  *result = force2on1;
99  return;
100  }
101  // the force itself
102  double r = (r1 * r2) / (r1 + r2);
103  double gamma = 1; // attraction coeff
104  double k = 2; // repulsion coeff
105  double f = k * delta - gamma * std::sqrt(r * delta);
106 
107  double module = f / center_distance;
108  Double3 force2on1({module * comp1, module * comp2, module * comp3});
109  *result = force2on1;
110 }

112 void InteractionForce::ForceOnACylinderFromASphere(const Agent* cylinder,
113  const Agent* sphere,
114  Double4* result) const {
115  auto* ne = bdm_static_cast<const NeuriteElement*>(cylinder);
116  auto proximal_end = ne->ProximalEnd();
117  auto distal_end = ne->DistalEnd();
118  auto axis = ne->GetSpringAxis();
119  // TODO(neurites) use cylinder.GetActualLength() ??
120  double actual_length = axis.Norm();
121  double d = ne->GetDiameter();
122  auto c = sphere->GetPosition();
123  double r = 0.5 * sphere->GetDiameter();
124 
125  // I. If the cylinder is small with respect to the sphere:
126  // we only consider the interaction between the sphere and the point mass
127  // (i.e. distal point) of the cylinder - that we treat as a sphere.
128  if (actual_length < r) {
129  // move back sphere center by 1 cylinder radius from distal_end
130  // vector_x = rc * (axis[0]/actual_length)
131  // vector_y = rc * (axis[1]/actual_length)
132  // vector_z = rc * (axis[2]/actual_length)
133  double rc = 0.5 * d;
134  Double3 dvec = (axis / actual_length) * rc; // displacement vector
135  Double3 npd = distal_end - dvec; // new sphere center
136  *result = ComputeForceOfASphereOnASphere(npd, rc, c, r);
137  return;
138  }
139 
140  // II. If the cylinder is of the same scale or bigger than the sphere,
141  // we look at the interaction between the sphere and the closest point
142  // (to the sphere center) on the cylinder. This interaction is distributed
143  // to
144  // the two ends of the cylinder: the distal (point mass of the segment) and
145  // the proximal (point mass of the mother of the segment).
146 
147  // 1) Finding cc : the closest point to c on the line proximal_end
148  // proximal_end
149  // ("line" and not "segment")
150  // It is the projection of the vector proximal_end->c onto the vector
151  // proximal_end->distal_end
152  // (=axis)
153  auto proximal_end_closest = c - proximal_end;
154 
155  // projection of proximal_end_closest onto axis =
156  // (proximal_end_closest.axis)/norm(axis)^2 * axis
157  // length of the projection = (proximal_end_closest.axis)/norm(axis)
158  double proximal_end_closest_axis =
159  proximal_end_closest.EntryWiseProduct(axis).Sum();
160  double k = proximal_end_closest_axis / (actual_length * actual_length);
161  // cc = proximal_end + k* axis
162  Double3 cc = proximal_end + (axis * k);
163 
164  // 2) Look if c -and hence cc- is (a) between proximal_end and distal_end,
165  // (b) before proximal_end or
166  // (c) after distal_end
167  double proportion_to_proximal_end;
168  if (k <= 1.0 && k >= 0.0) {
169  // a) if cc (the closest point to c on the line pPpD) is between
170  // proximal_end
171  // and distal_end
172  // the force is distributed to the two nodes
173  proportion_to_proximal_end = 1.0 - k;
174  } else if (k < 0) {
175  // b) if the closest point to c on the line pPpD is before
176  // proximal_end
177  // the force is only on the proximal end (the mother point mass)
178  proportion_to_proximal_end = 1.0;
179  cc = proximal_end;
180  } else { // if(k>1)
181  // c) if cc is after distal_end, the force is only on the distal end
182  // (the
183  // segment's point mass).
184  proportion_to_proximal_end = 0.0;
185  cc = distal_end;
186  }
187 
188  // 3) If the smallest distance between the cylinder and the center of the
189  // sphere
190  // is larger than the radius of the two objects , there is no
191  // interaction:
192  double penetration = d / 2 + r - Math::GetL2Distance(c, cc);
193  if (penetration <= 0) {
194  *result = Double4{0.0, 0.0, 0.0, 0.0};
195  return;
196  }
197  auto force = ComputeForceOfASphereOnASphere(cc, d * 0.5, c, r);
198  *result = {force[0], force[1], force[2], proportion_to_proximal_end};
199  return;
200 }

212 void InteractionForce::ForceBetweenCylinders(const Agent* cylinder1,
213  const Agent* cylinder2,
214  Double4* result) const {
215  auto* c1 = bdm_static_cast<const NeuriteElement*>(cylinder1);
216  auto* c2 = bdm_static_cast<const NeuriteElement*>(cylinder2);
217  auto a = c1->ProximalEnd();
218  auto b = c1->GetMassLocation();
219  double d1 = c1->GetDiameter();
220  auto c = c2->ProximalEnd();
221  auto d = c2->GetMassLocation();
222  double d2 = c2->GetDiameter();
223 
224  double k = 0.5; // part devoted to the distal node
225 
226  // looking for closest point on them
227  // (based on http://local.wasp.uwa.edu.au/~pbourke/geometry/lineline3d/)
228  double p13x = a[0] - c[0];
229  double p13y = a[1] - c[1];
230  double p13z = a[2] - c[2];
231  double p43x = d[0] - c[0];
232  double p43y = d[1] - c[1];
233  double p43z = d[2] - c[2];
234  double p21x = b[0] - a[0];
235  double p21y = b[1] - a[1];
236  double p21z = b[2] - a[2];
237 
238  double d1343 = p13x * p43x + p13y * p43y + p13z * p43z;
239  double d4321 = p21x * p43x + p21y * p43y + p21z * p43z;
240  double d1321 = p21x * p13x + p21y * p13y + p21z * p13z;
241  double d4343 = p43x * p43x + p43y * p43y + p43z * p43z;
242  double d2121 = p21x * p21x + p21y * p21y + p21z * p21z;
243 
244  Double3 p1, p2;
245 
246  double denom = d2121 * d4343 - d4321 * d4321;
247 
248  // if the two segments are not ABSOLUTLY parallel
249  if (denom > 0.000000000001) {
250  double numer = d1343 * d4321 - d1321 * d4343;
251 
252  double mua = numer / denom;
253  double mub = (d1343 + mua * d4321) / d4343;
254 
255  if (mua < 0) {
256  p1 = a;
257  k = 1;
258  } else if (mua > 1) {
259  p1 = b;
260  k = 0;
261  } else {
262  p1 = Double3{a[0] + mua * p21x, a[1] + mua * p21y, a[2] + mua * p21z};
263  k = 1 - mua;
264  }
265 
266  if (mub < 0) {
267  p2 = c;
268  } else if (mub > 1) {
269  p2 = d;
270  } else {
271  p2 = Double3{c[0] + mub * p43x, c[1] + mub * p43y, c[2] + mub * p43z};
272  }
273 
274  } else {
275  p1 = a + (b - a) * 0.5;
276  p2 = c + (d - c) * 0.5;
277  }
278 
279  // W put a virtual sphere on the two cylinders
280  auto force = ComputeForceOfASphereOnASphere(p1, d1 / 2.0, p2, d2 / 2.0) * 10;
281 
282  *result = {force[0], force[1], force[2], k};
283 }

285 Double4 InteractionForce::ComputeForceOfASphereOnASphere(const Double3& c1,
286  double r1,
287  const Double3& c2,
288  double r2) const {
289  double comp1 = c1[0] - c2[0];
290  double comp2 = c1[1] - c2[1];
291  double comp3 = c1[2] - c2[2];
292  double distance_between_centers =
293  std::sqrt(comp1 * comp1 + comp2 * comp2 + comp3 * comp3);
294  // the overlap distance (how much one penetrates in the other)
295  double a = r1 + r2 - distance_between_centers;
296  // if no overlap: no force
297  if (a < 0) {
298  return Double4{0.0, 0.0, 0.0, 0.0};
299  }
300  // to avoid a division by 0 if the centers are (almost) at the same location
301  if (distance_between_centers <
302  0.00000001) { // TODO(neurites) hard coded values
303  auto* random = Simulation::GetActive()->GetRandom();
304  auto force2on1 = random->template UniformArray<3>(-3.0, 3.0);
305  return Double4{force2on1[0], force2on1[1], force2on1[2], 0.0};
306  } else {
307  // the force is prop to the square of the interpentration distance and to
308  // the radii.
309  double module = a / distance_between_centers;
310  Double4 force2on1({module * comp1, module * comp2, module * comp3, 0.0});
311  return force2on1;
312  }
313 }
