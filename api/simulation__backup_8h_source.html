54  void Backup(size_t completed_simulation_steps) {
55  if (!backup_) {
56  Log::Fatal("SimulationBackup",
57  "Requested to backup data, but no backup file given.");
58  }
59 
60  // create temporary file
61  // if application crashes during backup; last backup is not corrupted
62  std::stringstream tmp_file;
63  tmp_file << "tmp_" << backup_file_;
64 
65  // Backup
66  {
67  TFileRaii f(tmp_file.str(), "UPDATE");
68  auto* simulation = TSimulation::GetActive();
69  f.Get()->WriteObject(simulation, kSimulationName.c_str());
70  IntegralTypeWrapper<size_t> wrapper(completed_simulation_steps);
71  f.Get()->WriteObject(&wrapper, kSimulationStepName.c_str());
72  RuntimeVariables rv;
73  f.Get()->WriteObject(&rv, kRuntimeVariableName.c_str());
74  // TODO(lukas) random number generator; all statics (e.g. Param)
75  }
76 
77  // remove last backup file
78  remove(backup_file_.c_str());
79  // rename temporary file
80  rename(tmp_file.str().c_str(), backup_file_.c_str());
81  }

84  void Restore() {
85  if (!restore_) {

87  "Requested to restore data, but no restore file given.");
88  }
89  after_restore_event_.clear();
90 
91  TFileRaii file(TFile::Open(restore_file_.c_str()));
92  RuntimeVariables* restored_rv;
93  file.Get()->GetObject(kRuntimeVariableName.c_str(), restored_rv);
94  // check if runtime variables are the same
95  if (!(RuntimeVariables() == *restored_rv)) {
96  Log::Warning("SimulationBackup",
97  "Restoring simulation executed on a different system!");
98  }
99  TSimulation* restored_simulation = nullptr;
100  file.Get()->GetObject(kSimulationName.c_str(), restored_simulation);
101  TSimulation::GetActive()->Restore(std::move(*restored_simulation));
102  Log::Info("Scheduler", "Restored simulation from ", restore_file_);
103 
104  // call all after restore events
105  for (auto&& event : after_restore_event_) {
106  event();
107  }
108  after_restore_event_.clear();
109  }
