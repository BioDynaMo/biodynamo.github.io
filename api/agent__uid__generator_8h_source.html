41  AgentUid GenerateUid() {
42  if (map_ != nullptr) {
43  // defragmentation mode
44  std::lock_guard<Spinlock> guard(lock_);
45  // repeat check, another thread might have disabled defragmentation
46  if (map_ != nullptr) {
47  // find unused element in map
48  while (search_index_ < map_->size() &&
49  map_->GetReused(search_index_) !=
50  std::numeric_limits<typename AgentUid::Reused_t>::max()) {
51  search_index_++;
52  }
53  if (search_index_ < map_->size()) {
54  auto* scheduler = Simulation::GetActive()->GetScheduler();
55  return AgentUid(search_index_++, scheduler->GetSimulatedSteps());
56  }
57  // didn't find any empty slots -> disable defragmentation mode
58  DisableDefragmentation();
59  }
60  }
61  return AgentUid(counter_++);
62  }
63 
64  // Returns the highest index that was used for an AgentUid
65  uint64_t GetHighestIndex() const { return counter_; }
66 
67  void EnableDefragmentation(const AgentUidMap<AgentHandle>* map) {
68  // check if already in defragmentation mode
69  if (map_ == nullptr) {
70  map_ = map;
71  search_index_ = 0;
72  }
73  }
74 
75  void DisableDefragmentation() {
76  if (map_ != nullptr) {
77  counter_ = map_->size();
78  }
79  map_ = nullptr;
80  }

82  bool IsInDefragmentationMode() const { return map_ != nullptr; }
