21 // Function to move <val> into the range [<min>, <max>]. If it is in the range
22 // already, it is returned unchanged. Else, the closest boundary is returned.
23 inline size_t Clamp(size_t val, size_t min, size_t max) {
24  if (val < min) {
25  return min;
26  }
27  if (val > max) {
28  return max;
29  }
30  return val;
31 }

33 void EulerGrid::DiffuseWithClosedEdge(real_t dt) {
34  const auto nx = resolution_;
35  const auto ny = resolution_;
36  const auto nz = resolution_;
37 
38  const real_t ibl2 = 1 / (box_length_ * box_length_);
39  const real_t d = 1 - dc_[0];
40 
41  constexpr size_t YBF = 16;
42 #pragma omp parallel for collapse(2)
43  for (size_t yy = 0; yy < ny; yy += YBF) {
44  for (size_t z = 0; z < nz; z++) {
45  size_t ymax = yy + YBF;
46  if (ymax >= ny) {
47  ymax = ny;
48  }
49  for (size_t y = yy; y < ymax; y++) {
50  size_t x{0};
51  size_t c{0};
52  size_t n{0};
53  size_t s{0};
54  size_t b{0};
55  size_t t{0};
56  c = x + y * nx + z * nx * ny;
57 #pragma omp simd
58  for (x = 1; x < nx - 1; x++) {
59  ++c;
60 
61  if (y == 0 || y == (ny - 1) || z == 0 || z == (nz - 1)) {
62  continue;
63  }
64 
65  n = c - nx;
66  s = c + nx;
67  b = c - nx * ny;
68  t = c + nx * ny;
69 
70  c2_[c] = c1_[c] * (1 - mu_ * dt) +
71  (d * dt * ibl2) *
72  (c1_[c - 1] - 2 * c1_[c] + c1_[c + 1] + c1_[s] -
73  2 * c1_[c] + c1_[n] + c1_[b] - 2 * c1_[c] + c1_[t]);
74  }
75  } // tile ny
76  } // tile nz
77  } // block ny
78  c1_.swap(c2_);
79 }

83  const auto ny = resolution_;
84  const auto nz = resolution_;
85 
86  const real_t ibl2 = 1 / (box_length_ * box_length_);
87  const real_t d = 1 - dc_[0];
88  std::array<int, 4> l;
89 
90  constexpr size_t YBF = 16;
91 #pragma omp parallel for collapse(2)
92  for (size_t yy = 0; yy < ny; yy += YBF) {
93  for (size_t z = 0; z < nz; z++) {
94  size_t ymax = yy + YBF;
95  if (ymax >= ny) {
96  ymax = ny;
97  }

99  size_t x{0};

101  size_t n{0};
102  size_t s{0};
103  size_t b{0};
104  size_t t{0};
105  c = x + y * nx + z * nx * ny;
106 
107  l.fill(1);
108 
109  if (y == 0) {
110  n = c;
111  l[0] = 0;
112  } else {
113  n = c - nx;
114  }
115 
116  if (y == ny - 1) {
117  s = c;
118  l[1] = 0;
119  } else {
120  s = c + nx;
121  }
122 
123  if (z == 0) {
124  b = c;
125  l[2] = 0;
126  } else {
127  b = c - nx * ny;
128  }
129 
130  if (z == nz - 1) {
131  t = c;
132  l[3] = 0;
133  } else {
134  t = c + nx * ny;
135  }
136 
137  c2_[c] = c1_[c] * (1 - mu_ * dt) +
138  (d * dt * ibl2) *
139  (0 - 2 * c1_[c] + c1_[c + 1] + c1_[s] - 2 * c1_[c] +
140  c1_[n] + c1_[b] - 2 * c1_[c] + c1_[t]);
141 #pragma omp simd
142  for (x = 1; x < nx - 1; x++) {
143  ++c;
144  ++n;
145  ++s;
146  ++b;
147  ++t;
148  c2_[c] =
149  c1_[c] * (1 - mu_ * dt) +
150  (d * dt * ibl2) * (c1_[c - 1] - 2 * c1_[c] + c1_[c + 1] +
151  l[0] * c1_[s] - 2 * c1_[c] + l[1] * c1_[n] +
152  l[2] * c1_[b] - 2 * c1_[c] + l[3] * c1_[t]);
153  }
154  ++c;
155  ++n;
156  ++s;
157  ++b;
158  ++t;
159  c2_[c] = c1_[c] * (1 - mu_ * dt) +
160  (d * dt * ibl2) *
161  (c1_[c - 1] - 2 * c1_[c] + 0 + c1_[s] - 2 * c1_[c] +
162  c1_[n] + c1_[b] - 2 * c1_[c] + c1_[t]);
163  } // tile ny
164  } // tile nz
165  } // block ny

193  size_t t{0};
194  c = x + y * nx + z * nx * ny;
195 #pragma omp simd
196  for (x = 0; x < nx; x++) {
197  if (x == 0 || x == (nx - 1) || y == 0 || y == (ny - 1) || z == 0 ||
198  z == (nz - 1)) {
199  // For all boxes on the boundary, we simply evaluate the boundary
200  real_t real_x = grid_dimensions_[0] + x * box_length_;
201  real_t real_y = grid_dimensions_[0] + y * box_length_;
202  real_t real_z = grid_dimensions_[0] + z * box_length_;
203  c2_[c] =
204  boundary_condition_->Evaluate(real_x, real_y, real_z, sim_time);
205  } else {
206  // For inner boxes, we compute the regular stencil update
207  n = c - nx;
208  s = c + nx;
209  b = c - nx * ny;
210  t = c + nx * ny;
211 
212  c2_[c] = c1_[c] * (1 - mu_ * dt) +
213  (d * dt * ibl2) *
214  (c1_[c - 1] - 2 * c1_[c] + c1_[c + 1] + c1_[s] -
215  2 * c1_[c] + c1_[n] + c1_[b] - 2 * c1_[c] + c1_[t]);
216  }
217  ++c;
218  }
219  } // tile ny
220  } // tile nz
221  } // block ny

225 void EulerGrid::DiffuseWithNeumann(real_t dt) {
226  const auto nx = resolution_;
227  const auto ny = resolution_;
228  const auto nz = resolution_;
229  const auto num_boxes = nx * ny * nz;
230 
231  const real_t ibl2 = 1 / (box_length_ * box_length_);
232  const real_t d = 1 - dc_[0];
233 
234  const auto sim_time = GetSimulatedTime();
235 
236  constexpr size_t YBF = 16;
237 #pragma omp parallel for collapse(2)
238  for (size_t yy = 0; yy < ny; yy += YBF) {
239  for (size_t z = 0; z < nz; z++) {
240  size_t ymax = yy + YBF;
241  if (ymax >= ny) {
242  ymax = ny;
243  }
244  for (size_t y = yy; y < ymax; y++) {
245  size_t x{0};
246  size_t c{0};
247  size_t n{0};
248  size_t s{0};
249  size_t b{0};
250  size_t t{0};
251  c = x + y * nx + z * nx * ny;
252 #pragma omp simd
253  for (x = 0; x < nx; x++) {
254  n = c - nx;
255  s = c + nx;
256  b = c - nx * ny;
257  t = c + nx * ny;
258 
259  // Clamp to avoid out of bounds access. Clamped values are initialized
260  // to a wrong value but will be overwritten by the boundary condition
261  // evaluation. All other values are correct.
262  real_t left{c1_[Clamp(c - 1, 0, num_boxes - 1)]};
263  real_t right{c1_[Clamp(c + 1, 0, num_boxes - 1)]};
264  real_t bottom{c1_[Clamp(b, 0, num_boxes - 1)]};
265  real_t top{c1_[Clamp(t, 0, num_boxes - 1)]};
266  real_t north{c1_[Clamp(n, 0, num_boxes - 1)]};
267  real_t south{c1_[Clamp(s, 0, num_boxes - 1)]};
268  real_t center_factor{6.0};
269 
270  if (x == 0 || x == (nx - 1) || y == 0 || y == (ny - 1) || z == 0 ||
271  z == (nz - 1)) {
272  real_t real_x = grid_dimensions_[0] + x * box_length_;
273  real_t real_y = grid_dimensions_[0] + y * box_length_;
274  real_t real_z = grid_dimensions_[0] + z * box_length_;
275  real_t boundary_value =
276  -box_length_ *
277  boundary_condition_->Evaluate(real_x, real_y, real_z, sim_time);
278 
279  if (x == 0) {
280  left = boundary_value;
281  center_factor -= 1.0;
282  } else if (x == (nx - 1)) {
283  right = boundary_value;
284  center_factor -= 1.0;
285  }
286 
287  if (y == 0) {
288  north = boundary_value;
289  center_factor -= 1.0;
290  } else if (y == (ny - 1)) {
291  south = boundary_value;
292  center_factor -= 1.0;
293  }
294 
295  if (z == 0) {
296  bottom = boundary_value;
297  center_factor -= 1.0;
298  } else if (z == (nz - 1)) {
299  top = boundary_value;
300  center_factor -= 1.0;
301  }
302  }
303 
304  c2_[c] = c1_[c] * (1 - mu_ * dt) +
305  (d * dt * ibl2) * (left + right + south + north + top +
306  bottom - center_factor * c1_[c]);
307 
308  ++c;
309  }
310  } // tile ny
311  } // tile nz
312  } // block ny
313  c1_.swap(c2_);
314 }
