170 template <typename T>
171 inline T Reduce(Simulation* sim, Functor<void, Agent*, T*>& agent_functor,
172  Functor<T, const SharedData<T>&>& reduce_partial_results,
173  Functor<bool, Agent*>* filter = nullptr) {
174  // The thread-local (partial) results
175  SharedData<T> tl_results;
176  // initialize thread local data
177  tl_results.resize(ThreadInfo::GetInstance()->GetMaxThreads());
178  for (auto& el : tl_results) {
179  el = T();
180  }
181 
182  // reduce
183  // execute agent functor in parallel
184  auto actual_agent_func = L2F([&](Agent* agent, AgentHandle) {
185  auto tid = ThreadInfo::GetInstance()->GetMyThreadId();
186  agent_functor(agent, &(tl_results[tid]));
187  });
188  auto* rm = sim->GetResourceManager();
189  rm->ForEachAgentParallel(actual_agent_func, filter);
190  // combine thread-local results
191  return reduce_partial_results(tl_results);
192 }
