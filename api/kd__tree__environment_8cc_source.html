49 void KDTreeEnvironment::Update() {
50  nf_adapter_->rm_ = Simulation::GetActive()->GetResourceManager();
51 
52  // Update the flattened indices map
53  nf_adapter_->flat_idx_map_.Update();
54  if (nf_adapter_->rm_->GetNumAgents() != 0) {
55  Clear();
56  auto inf = Math::kInfinity;
57  std::array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
58  CalcSimDimensionsAndLargestAgent(&tmp_dim);
59  RoundOffGridDimensions(tmp_dim);
60  CheckGridGrowth();
61  impl_->index_->buildIndex();
62  } else {
63  // There are no sim objects in this simulation
64  auto* param = Simulation::GetActive()->GetParam();
65 
66  bool uninitialized = impl_->index_->m_size == 0;
67  if (uninitialized && param->bound_space) {
68  // Simulation has never had any simulation objects
69  // Initialize grid dimensions with `Param::min_bound_` and
70  // `Param::max_bound_`
71  // This is required for the DiffusionGrid
72  int min = param->min_bound;
73  int max = param->max_bound;
74  grid_dimensions_ = {min, max, min, max, min, max};
75  threshold_dimensions_ = {min, max};
76  has_grown_ = true;
77  } else if (!uninitialized) {
78  // all simulation objects have been removed in the last iteration
79  // grid state remains the same, but we have to set has_grown_ to false
80  // otherwise the DiffusionGrid will attempt to resize
81  has_grown_ = false;
82  } else {
83  Log::Fatal(
84  "KDtreeEnvironment",
85  "You tried to initialize an empty simulation without bound space. "
86  "Therefore we cannot determine the size of the simulation space. "
87  "Please add simulation objects, or set Param::bound_space_, "
88  "Param::min_bound_, and Param::max_bound_.");
89  }
90  }
91 }

93 void KDTreeEnvironment::ForEachNeighbor(Functor<void, Agent*, double>& lambda,
94  const Agent& query,
95  double squared_radius) {
96  std::vector<std::pair<uint64_t, double>> neighbors;
97 
98  nanoflann::SearchParams params;
99  params.sorted = false;
100 
101  const auto& position = query.GetPosition();
102 
103  // calculate neighbors
104  impl_->index_->radiusSearch(&position[0], squared_radius, neighbors, params);
105 
106  auto* rm = Simulation::GetActive()->GetResourceManager();
107  for (auto& n : neighbors) {
108  Agent* nb_so =
109  rm->GetAgent(nf_adapter_->flat_idx_map_.GetAgentHandle(n.first));
110  if (nb_so != &query) {
111  lambda(nb_so, n.second);
112  }
113  }
114 }

152 void KDTreeEnvironment::CheckGridGrowth() {
153  // Determine if the grid dimensions have changed (changed in the sense that
154  // the grid has grown outwards)
155  auto min_gd =
156  *std::min_element(grid_dimensions_.begin(), grid_dimensions_.end());
157  auto max_gd =
158  *std::max_element(grid_dimensions_.begin(), grid_dimensions_.end());
159  if (min_gd < threshold_dimensions_[0]) {
160  threshold_dimensions_[0] = min_gd;
161  has_grown_ = true;
162  }
163  if (max_gd > threshold_dimensions_[1]) {
164  threshold_dimensions_[1] = max_gd;
165  has_grown_ = true;
166  }
167 }
