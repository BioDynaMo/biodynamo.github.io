64 void GitTracker::PrintGitInfo(const std::string& repository_path,
65  std::ostream& out) const {
66  // Absolute path to repository
67  std::string abs_repo_path = GetAbsolutePath(repository_path);
68 
69  // Use libgit2 to verify that the repository is valid
70  git_libgit2_init();
71  git_repository* repo = nullptr;
72  git_repository_open(&repo, abs_repo_path.c_str());
73  if (repo == nullptr) {
74  Log::Error("GitTracker", "Error: ", abs_repo_path,
75  " is not a valid git repository");
76  return;
77  }
78 
79  // Get the name of the repository via libgit2
80  const char* repo_name = git_repository_path(repo);
81 
82  // Get the commit hash
83  git_reference* head = nullptr;
84  git_reference_lookup(&head, repo, "HEAD");
85  git_object* head_commit = nullptr;
86  git_reference_peel(&head_commit, head, GIT_OBJ_COMMIT);
87  const git_oid* oid = git_commit_id((git_commit*)head_commit);
88  char oid_str[GIT_OID_HEXSZ + 1];
89  git_oid_tostr(oid_str, GIT_OID_HEXSZ + 1, oid);
90 
91  // Get the commit message
92  const char* commit_message = git_commit_message((git_commit*)head_commit);
93 
94  // Get the commit author
95  const git_signature* commit_author =
96  git_commit_author((git_commit*)head_commit);
97 
98  // Get the commit author name
99  const char* author_name = commit_author->name;
100 
101  // Get the commit author email
102  const char* author_email = commit_author->email;
103 
104  // Get the branch name
105  const char* branch_name = git_reference_shorthand(head);
106 
107  // Get reference target
108  git_reference* target = nullptr;
109  git_reference_resolve(&target, head);
110  const char* target_name = git_reference_name(target);
111 
112  // Print the git info
113  out << "Git repository : " << repo_name << "\n"
114  << "Git branch name : " << target_name << "\n"
115  << " : " << branch_name << "\n"
116  << "Git commit hash : " << oid_str << "\n"
117  << "Git author email : " << author_email << "\n"
118  << "Git author name : " << author_name << "\n"
119  << "Git commit message : " << commit_message << "\n"
120  << std::endl;
121 
122  // Cleanup
123  git_object_free(head_commit);
124  git_reference_free(head);
125  git_reference_free(target);
126  git_repository_free(repo);
127  git_libgit2_shutdown();
128 }

130 void GitTracker::SaveGitDiff(const std::string& file,
131  const std::string& repository_path) const {
132  // Absolute path to repository
133  std::string abs_repo_path = GetAbsolutePath(repository_path);
134 
135  // Use libgit2 to verify that the repository is valid
136  git_libgit2_init();
137  git_repository* repo = nullptr;
138  git_repository_open(&repo, abs_repo_path.c_str());
139  if (repo == nullptr) {
140  Log::Error("GitTracker", "Error: ", abs_repo_path,
141  " is not a valid git repository");
142  return;
143  }
144 
145  // Generate the git diff for the repository
146  git_diff* diff = nullptr;
147  git_diff_index_to_workdir(&diff, repo, nullptr, nullptr);
148 
149  // Open C style file
150  FILE* fp = fopen(file.c_str(), "w");
151 
152  // Print all the deltas to the file. If fopen fails, the diff will be printed
153  // to stdout
154  git_diff_print(diff, GIT_DIFF_FORMAT_PATCH, diff_output, fp);
155 
156  // Close C style file if the pointer is not null
157  if (fp) {
158  fclose(fp);
159  }
160 
161  // Cleanup
162  git_diff_free(diff);
163  git_repository_free(repo);
164  git_libgit2_shutdown();
165 }

189 // This function is taken from the libgit2 documentation. It is used to print
190 // the git diff to a file.
191 int diff_output(const git_diff_delta* d, const git_diff_hunk* h,
192  const git_diff_line* l, void* p) {
193  FILE* fp = (FILE*)p;
194 
195  (void)d;
196  (void)h;
197 
198  if (!fp)
199  fp = stdout;
200 
201  if (l->origin == GIT_DIFF_LINE_CONTEXT ||
202  l->origin == GIT_DIFF_LINE_ADDITION ||
203  l->origin == GIT_DIFF_LINE_DELETION)
204  fputc(l->origin, fp);
205 
206  fwrite(l->content, 1, l->content_len, fp);
207 
208  return 0;
209 }
