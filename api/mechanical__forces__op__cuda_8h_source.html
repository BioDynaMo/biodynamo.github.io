54  void SetUp() override {
55  auto* sim = Simulation::GetActive();
56  auto* grid = dynamic_cast<UniformGridEnvironment*>(sim->GetEnvironment());
57  auto* rm = sim->GetResourceManager();
58 
59  if (!grid) {
60  Log::Fatal(
61  "MechanicalForcesOpCuda::operator()",
62  "MechanicalForcesOpCuda only works with UniformGridEnvironement.");
63  }
64 
65  auto num_numa_nodes = ThreadInfo::GetInstance()->GetNumaNodes();
66  std::vector<AgentHandle::ElementIdx_t> offset(num_numa_nodes);
67  offset[0] = 0;
68  for (int nn = 1; nn < num_numa_nodes; nn++) {
69  offset[nn] = offset[nn - 1] + rm->GetNumAgents(nn - 1);
70  }
71 
72  uint32_t total_num_objects = rm->GetNumAgents();
73 
74  i_ = new InitializeGPUData(total_num_objects, offset);
75  rm->ForEachAgentParallel(1000, *i_);
76 
77  // Populate successor list
78  for (int i = 0; i < num_numa_nodes; i++) {
79  for (size_t j = 0; j < rm->GetNumAgents(i); j++) {
80  auto idx = offset[i] + j;
81  i_->successors[idx] =
82  offset[grid->successors_.data_[i][j].GetNumaNode()] +
83  grid->successors_.data_[i][j].GetElementIdx();
84  }
85  }
86 
87  auto num_boxes = grid->boxes_.size();
88  i_->current_timestamp = grid->timestamp_;
89  i_->starts.reserve(num_boxes);
90  i_->lengths.reserve(num_boxes);
91  i_->timestamps.reserve(num_boxes);
92  size_t i = 0;
93  for (auto& box : grid->boxes_) {
94  i_->timestamps[i] = box.timestamp_;
95  if (box.timestamp_ == i_->current_timestamp) {
96  i_->lengths[i] = box.length_;
97  i_->starts[i] =
98  offset[box.start_.GetNumaNode()] + box.start_.GetElementIdx();
99  }
100  i++;
101  }
102  grid->GetGridInfo(&(i_->box_length), &(i_->num_boxes_axis),
103  &(i_->grid_dimensions));
104  }
105 
106  void operator()() override {
107  auto* sim = Simulation::GetActive();
108  auto* grid = dynamic_cast<UniformGridEnvironment*>(sim->GetEnvironment());
109  auto* param = sim->GetParam();
110  auto* rm = sim->GetResourceManager();
111 
112  uint32_t total_num_objects = rm->GetNumAgents();
113  auto num_boxes = grid->boxes_.size();
114  // If this is the first time we perform physics on GPU using CUDA
115  if (cdo_ == nullptr) {
116  // Allocate 25% more memory so we don't need to reallocate GPU memory
117  // for every (small) change
118  total_num_objects_ = static_cast<uint32_t>(1.25 * total_num_objects);
119  num_boxes_ = static_cast<uint32_t>(1.25 * num_boxes);
120 
121  // Allocate required GPU memory
122  cdo_ = new MechanicalForcesOpCudaKernel(total_num_objects_, num_boxes_);
123  } else {
124  // If the number of agents increased
125  if (total_num_objects >= total_num_objects_) {
126  Log::Info("MechanicalForcesOpCuda",
127  "\nThe number of cells increased signficantly (from ",
128  total_num_objects_, " to ", total_num_objects,
129  "), agent we allocate bigger GPU buffers\n");
130  total_num_objects_ = static_cast<uint32_t>(1.25 * total_num_objects);
131  cdo_->ResizeCellBuffers(total_num_objects_);
132  }
133 
134  // If the neighbor grid size increased
135  if (num_boxes >= num_boxes_) {
136  Log::Info("MechanicalForcesOpCuda",
137  "\nThe number of boxes increased signficantly (from ",
138  num_boxes_, " to ", "), so we allocate bigger GPU buffers\n");
139  num_boxes_ = static_cast<uint32_t>(1.25 * num_boxes);
140  cdo_->ResizeGridBuffers(num_boxes_);
141  }
142  }
143 
144  double squared_radius =
145  grid->GetLargestObjectSize() * grid->GetLargestObjectSize();
146  cdo_->LaunchMechanicalForcesKernel(
147  i_->cell_positions.data()->data(), i_->cell_diameters.data(),
148  i_->cell_tractor_force.data()->data(), i_->cell_adherence.data(),
149  i_->cell_boxid.data(), i_->mass.data(), &(param->simulation_time_step),
150  &(param->simulation_max_displacement), &squared_radius,
151  &total_num_objects, i_->starts.data(), i_->lengths.data(),
152  i_->timestamps.data(), &(i_->current_timestamp), i_->successors.data(),
153  &(i_->box_length), i_->num_boxes_axis.data(),
154  i_->grid_dimensions.data(), i_->cell_movements.data()->data());
155  }
156 
157  void TearDown() override {

159  Simulation::GetActive()->GetResourceManager()->ForEachAgentParallel(1000,
160  u);
161  }

179  void operator()(Agent* agent, AgentHandle ah) override {
180  auto* param = Simulation::GetActive()->GetParam();
181  auto* cell = dynamic_cast<Cell*>(agent);
182  auto idx = offset[ah.GetNumaNode()] + ah.GetElementIdx();
183  Double3 new_pos;
184  new_pos[0] = (*cell_movements)[idx][0];
185  new_pos[1] = (*cell_movements)[idx][1];
186  new_pos[2] = (*cell_movements)[idx][2];
187  cell->UpdatePosition(new_pos);
188  if (param->bound_space) {
189  ApplyBoundingBox(agent, param->min_bound, param->max_bound);
190  }
191  }
192  };

194  struct InitializeGPUData : public Functor<void, Agent*, AgentHandle> {
195  bool is_non_spherical_object = false;
196  // Cannot use Double3 here, because the `data()` function returns a const
197  // pointer to the underlying array, whereas the CUDA kernel will cast it to
198  // a void pointer. The conversion of `const double *` to `void *` is
199  // illegal.
200  std::vector<std::array<double, 3>> cell_movements;
201  std::vector<Double3> cell_positions;
202  std::vector<double> cell_diameters;
203  std::vector<double> cell_adherence;
204  std::vector<Double3> cell_tractor_force;
205  std::vector<uint32_t> cell_boxid;
206  std::vector<double> mass;
207  std::vector<AgentHandle::ElementIdx_t> offset;
208  std::vector<AgentHandle::ElementIdx_t> successors;
209  std::vector<uint32_t> starts;
210  std::vector<uint16_t> lengths;
211  std::vector<uint64_t> timestamps;
212  uint64_t current_timestamp;
213  uint32_t box_length;
214  std::array<uint32_t, 3> num_boxes_axis;
215  std::array<int32_t, 3> grid_dimensions;
216 
217  InitializeGPUData(uint32_t num_objects,
218  const std::vector<AgentHandle::ElementIdx_t>& offs) {
219  cell_movements.reserve(num_objects);
220  cell_positions.reserve(num_objects);
221  successors.reserve(num_objects);
222  cell_diameters.reserve(num_objects);
223  cell_adherence.reserve(num_objects);
224  cell_tractor_force.reserve(num_objects);
225  cell_boxid.reserve(num_objects);
226  mass.reserve(num_objects);
227  offset = offs;
228  }
229 
230  void operator()(Agent* agent, AgentHandle ah) override {
231  // Check if there are any non-spherical objects in our simulation, because
232  // GPU accelerations currently supports only sphere-sphere interactions
233  IsNonSphericalObjectPresent(agent, &is_non_spherical_object);
234  if (is_non_spherical_object) {
235  Log::Fatal("MechanicalForcesOpCuda",
236  "\nWe detected a non-spherical object during the GPU "
237  "execution. This is currently not supported.");
238  return;
239  }
240  auto* cell = bdm_static_cast<Cell*>(agent);
241  auto idx = offset[ah.GetNumaNode()] + ah.GetElementIdx();
242  mass[idx] = cell->GetMass();
243  cell_diameters[idx] = cell->GetDiameter();
244  cell_adherence[idx] = cell->GetAdherence();
245  cell_tractor_force[idx] = cell->GetTractorForce();
246  cell_positions[idx] = cell->GetPosition();
247  cell_boxid[idx] = cell->GetBoxIdx();
248  }
249  };
