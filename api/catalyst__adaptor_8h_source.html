73  void ConstructDiffusionGrid(DiffusionGrid* dg, uint16_t idx) {
74  auto num_boxes = dg->GetNumBoxesArray();
75  auto grid_dimensions = dg->GetDimensions();
76  auto box_length = dg->GetBoxLength();
77 
78  double origin_x = grid_dimensions[0];
79  double origin_y = grid_dimensions[2];
80  double origin_z = grid_dimensions[4];
81  vtk_dgrids_[idx]->SetOrigin(origin_x, origin_y, origin_z);
82  vtk_dgrids_[idx]->SetDimensions(num_boxes[0], num_boxes[1], num_boxes[2]);
83  vtk_dgrids_[idx]->SetSpacing(box_length, box_length, box_length);
84  }

168  template <typename TContainer>
169  void BuildCellsVTKStructures(TContainer* sim_objects, uint16_t type_idx) {
170  auto& scalar_name = TContainer::GetScalarTypeName();
171  if (so_is_initialized_[type_idx] == false) {
172  vtk_so_grids_.push_back(vtkUnstructuredGrid::New());
173  so_is_initialized_[type_idx] = true;
174  shapes_[scalar_name] = sim_objects->GetShape();
175  so_scalar_names_.push_back(scalar_name);
176  }
177 
178  auto num_cells = sim_objects->size();
179 
180  auto required_dm = TContainer::GetRequiredVisDataMembers();
181  sim_objects->ForEachDataMemberIn(
182  required_dm, AddCellAttributeData(type_idx, num_cells, &vtk_so_grids_));
183 
184  auto& additional_dm = Param::visualize_sim_objects_[scalar_name];
185  if (!additional_dm.empty()) {
186  sim_objects->ForEachDataMemberIn(
187  additional_dm,
188  AddCellAttributeData(type_idx, num_cells, &vtk_so_grids_));
189  }
190  }

197  void BuildDiffusionGridVTKStructures(DiffusionGrid* dg, uint16_t idx,
198  const Param::VisualizeDiffusion& vd) {
199  if (dg_is_initialized_[idx] == false) {
200  vtk_dgrids_.push_back(vtkImageData::New());
201  dg_is_initialized_[idx] = true;
202  }
203 
204  // Create the diffusion grid
205  ConstructDiffusionGrid(dg, idx);
206  auto total_boxes = dg->GetNumBoxes();
207 
208  // Add attribute data
209  if (vd.concentration_) {
210  vtkNew<vtkDoubleArray> concentration_array;
211  concentration_array->SetName("Substance Concentration");
212  auto co_ptr = dg->GetAllConcentrations();
213  concentration_array->SetArray(co_ptr, static_cast<vtkIdType>(total_boxes),
214  1);
215  vtk_dgrids_[idx]->GetPointData()->AddArray(
216  concentration_array.GetPointer());
217  }
218  if (vd.gradient_) {
219  vtkNew<vtkDoubleArray> gradient_array;
220  gradient_array->SetName("Diffusion Gradient");
221  gradient_array->SetNumberOfComponents(3);
222  auto gr_ptr = dg->GetAllGradients();
223  gradient_array->SetArray(gr_ptr, static_cast<vtkIdType>(total_boxes * 3),
224  1);
225  vtk_dgrids_[idx]->GetPointData()->AddArray(gradient_array.GetPointer());
226  }
227  }

234  inline void Initialize(const std::string& script) {
235  if (Param::live_visualization_ || Param::export_visualization_) {
236  if (g_processor_ == nullptr) {
237  g_processor_ = vtkCPProcessor::New();
238  g_processor_->Initialize();
239 
240  auto rm = TResourceManager::Get();
241  so_is_initialized_.resize(rm->NumberOfTypes());
242  dg_is_initialized_.resize(rm->GetDiffusionGrids().size());
243  } else {
244  g_processor_->RemoveAllPipelines();
245  }
246 
247  if (Param::python_catalyst_pipeline_) {
248  vtkNew<vtkCPPythonScriptPipeline> pipeline;
249  pipeline->Initialize(script.c_str());
250  g_processor_->AddPipeline(pipeline.GetPointer());
251  } else {
252  pipeline_ = new InSituPipeline();
253  g_processor_->AddPipeline(pipeline_);
254  }
255  }

317  void CreateVtkObjects(
318  vtkNew<vtkCPDataDescription>& data_description) { // NOLINT
319  // Add simulation objects to the visualization if requested
320  auto rm = TResourceManager::Get();
321  rm->ApplyOnAllTypes([&, this](auto* sim_objects, uint16_t type_idx) {
322  auto so_name =
323  std::decay<decltype(*sim_objects)>::type::GetScalarTypeName().c_str();
324 
325  data_description->AddInput(so_name);
326 
327  // If we segfault at here it probably means that the pipeline was not
328  // initialized (with a python script)
329  if ((g_processor_->RequestDataDescription(
330  data_description.GetPointer())) != 0) {
331  this->BuildCellsVTKStructures(sim_objects, type_idx);
332  data_description->GetInputDescriptionByName(so_name)->SetGrid(
333  vtk_so_grids_[type_idx]);
334  }
335  });
336 
337  // Add all diffusion grids to the visualization if requested
338  if (!Param::visualize_diffusion_.empty()) {
339  uint16_t idx = 0;
340  for (auto& vd : Param::visualize_diffusion_) {
341  auto dg = rm->GetDiffusionGrid(vd.name_);
342  if (dg == nullptr) {
343  Log::Warning("Visualize Diffusion", "The substance with the name ",
344  vd.name_,
345  " was not found in the list of defined substances. "
346  "Did you spell the name correctly during "
347  "configuration?");
348  continue;
349  }
350  data_description->AddInput(dg->GetSubstanceName().c_str());
351  if (g_processor_->RequestDataDescription(
352  data_description.GetPointer()) != 0) {
353  this->BuildDiffusionGridVTKStructures(dg, idx, vd);
354  data_description
355  ->GetInputDescriptionByName(dg->GetSubstanceName().c_str())
356  ->SetGrid(vtk_dgrids_[idx]);
357  }
358  idx++;
359  }
360  }
361  }

381  inline void CoProcess(double time, size_t step, bool last_time_step) {
382  vtkNew<vtkCPDataDescription> data_description;
383  data_description->SetTimeData(time, step);
384 
385  CreateVtkObjects(data_description);
386 
387  if (last_time_step == true) {
388  data_description->ForceOutputOn();
389  }
390 
391  if (pipeline_ != nullptr) {
392  if (!(pipeline_->IsInitialized())) {
393  pipeline_->Initialize(shapes_);
394  }
395  }

449  static void GenerateSimulationInfoJson(
450  const std::unordered_map<std::string, Shape>& shapes) {
451  // simulation objects
452  std::stringstream sim_objects;
453  uint64_t num_sim_objects = Param::visualize_sim_objects_.size();
454  uint64_t counter = 0;
455  for (const auto& entry : Param::visualize_sim_objects_) {
456  std::string so_name = entry.first;
457 
458  sim_objects << " {" << std::endl
459  << " \"name\":\"" << so_name << "\"," << std::endl;
460  if (shapes.at(so_name) == Shape::kSphere) {
461  sim_objects << " \"glyph\":\"Glyph\"," << std::endl
462  << " \"shape\":\"Sphere\"," << std::endl
463  << " \"scaling_attribute\":\"diameter_\"" << std::endl;
464  } else if (shapes.at(so_name) == kCylinder) {
465  sim_objects << " \"glyph\":\"BDMGlyph\"," << std::endl
466  << " \"shape\":\"Cylinder\"," << std::endl
467  << " \"x_scaling_attribute\":\"diameter_\","
468  << std::endl
469  << " \"y_scaling_attribute\":\"actual_length_\","
470  << std::endl
471  << " \"z_scaling_attribute\":\"diameter_\","
472  << std::endl
473  << " \"Vectors\":\"spring_axis_\"," << std::endl
474  << " \"MassLocation\":\"mass_location_\"" << std::endl;
475  }
476  sim_objects << " }";
477  if (counter != num_sim_objects - 1) {
478  sim_objects << ",";
479  }
480  sim_objects << std::endl;
481  counter++;
482  }
483 
484  // extracellular substances
485  std::stringstream substances;
486  uint64_t num_substances = Param::visualize_diffusion_.size();
487  for (uint64_t i = 0; i < num_substances; i++) {
488  substances << " { \"name\":\"" << Param::visualize_diffusion_[i].name_
489  << "\", ";
490  std::string has_gradient =
491  Param::visualize_diffusion_[i].gradient_ ? "true" : "false";
492  substances << "\"has_gradient\":\"" << has_gradient << "\" }";
493 
494  if (i != num_substances - 1) {
495  substances << "," << std::endl;
496  }
497  }
498 
499  // write to file
500  std::ofstream ofstr;
501  ofstr.open(kSimulationInfoJson);
502  ofstr << "{" << std::endl
503  << " \"simulation\": {" << std::endl
504  << " \"name\":\"" << Param::executable_name_ << "\"," << std::endl
505  << " \"result_dir\":\""
506  << "."
507  << "\"" << std::endl
508  << " }," << std::endl
509  << " \"sim_objects\": [" << std::endl
510  << sim_objects.str() << " ]," << std::endl
511  << " \"extracellular_substances\": [" << std::endl
512  << substances.str() << std::endl
513  << " ]" << std::endl
514  << "}" << std::endl;
515  ofstr.close();
516  }
