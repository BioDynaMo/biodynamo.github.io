110  vtkSMProxy* GetSource(vtkSMProxy* glyph, std::string source_name) {
111  vtkSMProxyListDomain* pld = vtkSMProxyListDomain::SafeDownCast(
112  glyph->GetProperty("Source")->FindDomain("vtkSMProxyListDomain"));
113 
114  for (unsigned int cc = 0; cc < pld->GetNumberOfProxies(); cc++) {
115  if (pld->GetProxyName(cc) &&
116  strcmp(pld->GetProxyName(cc), source_name.c_str()) == 0) {

123  // FOR ALL VTK OBJECTS:
124  // 1. Get vtkObject + it's unique name
125  // 2. Create a producer for it
126  // 3. Create a proxy for the producer and register it to the session_manager
127  // 4. Create a "real_producer" that sets the vtkObject to its output
128  // 5. Add it to the map of producers
129  void CreatePipeline(vtkCPDataDescription* data_description) {
130  auto n_inputs = data_description->GetNumberOfInputDescriptions();
131  for (size_t i = 0; i < n_inputs; i++) {
132  auto vtk_object = data_description->GetInputDescription(i)->GetGrid();
133  auto object_name = data_description->GetInputDescriptionName(i);
134 
135  // Create a vtkPVTrivialProducer and set its output to be the input grid.
136  vtkSmartPointer<vtkSMSourceProxy> producer;
137  producer.TakeReference(vtkSMSourceProxy::SafeDownCast(
138  session_manager_->NewProxy("sources", "PVTrivialProducer")));
139  session_manager_->RegisterProxy("sources", object_name, producer);
140  producer->UpdateVTKObjects();
141  vtkObjectBase* client_side_object = producer->GetClientSideObject();
142  vtkPVTrivialProducer* real_producer =
143  vtkPVTrivialProducer::SafeDownCast(client_side_object);
144 
145  std::string source_name = "SphereSource";
146  std::string glyph_type = "Glyph";
147  int scale_mode = 0;
148 
149  if (vtk_object->IsA("vtkUnstructuredGrid")) {
150  real_producer->SetOutput(vtk_object);
151 
152  auto& shape = vtk_so_grids_[object_name]->shape_;
153 
154  if (shape == Shape::kCylinder) {
155  source_name = "CylinderSource";
156  glyph_type = "BDMGlyph";
157  scale_mode = 4;
158  } else if (shape != Shape::kSphere) {
159  Log::Warning("CreatePipeline",
160  "We currently support only spherical and cylindrical "
161  "shaped objects for visualization. Received value %d",
162  shape);
163  }
164 
165  // Create a Glyph filter
166  vtkSmartPointer<vtkSMSourceProxy> glyph;
167  glyph.TakeReference(vtkSMSourceProxy::SafeDownCast(
168  session_manager_->NewProxy("filters", glyph_type.c_str())));
169  controller_->PreInitializeProxy(glyph);
170  std::string object_name_str = object_name;
171  std::string glyph_name = object_name_str + "_Glyph";
172 
173  vtkSMPropertyHelper(glyph, "Input").Set(producer);
174  vtkSMPropertyHelper(glyph, "Source").Set(GetSource(glyph, source_name));
175  vtkSMPropertyHelper(glyph, "ScaleMode", true).Set(scale_mode);
176  vtkSMPropertyHelper(glyph, "ScaleFactor", true).Set(1.0);
177  vtkSMPropertyHelper(glyph, "GlyphMode", true).Set(0);
178 
179  if (scale_mode == 0) {
180  vtkSMPropertyHelper(glyph, "Scalars")
181  .SetInputArrayToProcess(vtkDataObject::POINT, "diameter_");
182  } else if (scale_mode == 4) {
183  vtkSMPropertyHelper(glyph, "X-Scaling")
184  .SetInputArrayToProcess(vtkDataObject::POINT, "diameter_");
185  vtkSMPropertyHelper(glyph, "Y-Scaling")
186  .SetInputArrayToProcess(vtkDataObject::POINT, "actual_length_");
187  vtkSMPropertyHelper(glyph, "Z-Scaling")
188  .SetInputArrayToProcess(vtkDataObject::POINT, "diameter_");
189  vtkSMPropertyHelper(glyph, "Vectors")
190  .SetInputArrayToProcess(vtkDataObject::POINT, "spring_axis_");
191  vtkSMPropertyHelper(glyph, "MassLocation")
192  .SetInputArrayToProcess(vtkDataObject::POINT, "mass_location_");
193  }
194 
195  glyph->UpdateVTKObjects();
196  producer->UpdateVTKObjects();
197  controller_->PostInitializeProxy(glyph);
198  controller_->RegisterPipelineProxy(glyph, glyph_name.c_str());
199 
200  glyph->UpdatePropertyInformation();
201  glyph->UpdatePipeline();
202  filter_map_[glyph_name] = glyph;
203 
204  } else if (vtk_object->IsA("vtkImageData")) {
205  real_producer->SetOutput(vtkImageData::SafeDownCast(vtk_object));
206  } else {
207  Log::Warning("vtkCPVTKPipeline", "Object of type ",
208  vtk_object->GetClassName(),
209  " is not supported for visualization");
210  }
211 
212  // Record the producer for updates
213  producer_map_[object_name] = producer;
214  }
215  }

217  // 1. Call CreatePipeline to create the producers (only once)
218  // 2. For each producer in the map, update the output
219  void UpdateProducers(vtkCPDataDescription* data_description) {
220  // Ensure that the producers are created
221  if (!pipeline_created_) {
222  this->CreatePipeline(data_description);
223  this->pipeline_created_ = true;
224  }
225 
226  // Update the output of each recorded producer
227  for (auto it = this->producer_map_.begin(); it != this->producer_map_.end();
228  it++) {
229  vtkObjectBase* client_side_object = it->second->GetClientSideObject();
230  vtkPVTrivialProducer* real_producer =
231  vtkPVTrivialProducer::SafeDownCast(client_side_object);
232  real_producer->SetOutput(
233  data_description->GetInputDescriptionByName(it->first.c_str())
234  ->GetGrid(),
235  data_description->GetTime());
236  }
237  }

249  void DoLiveVisualization(vtkCPDataDescription* data_description) {
250  // Initialize the insitu link, if not already done
251  if (!insitu_link_) {
252  insitu_link_ = vtkSmartPointer<vtkLiveInsituLink>::New();
253  insitu_link_->SetHostname("localhost");
254  insitu_link_->SetInsituPort(22222);
255  insitu_link_->Initialize(
256  vtkSMProxyManager::GetProxyManager()->GetActiveSessionProxyManager());
257  }
258 
259  // If there is not link with the ParaView client, then do not go trough the
260  // process of updating sources / filters
261  if (!insitu_link_->Initialize(vtkSMProxyManager::GetProxyManager()
262  ->GetActiveSessionProxyManager())) {
263  return;
264  }
265 
266  // Get the time information from the data
267  double time = data_description->GetTime();
268  vtkIdType time_step = data_description->GetTimeStep();
269 
270  // Go into the main loop for visualization
271  while (true) {
272  // Initialize the link and update data
273  insitu_link_->InsituUpdate(time, time_step);
274  for (auto it = producer_map_.begin(); it != producer_map_.end(); it++) {
275  // Update the pipeline for the current time_step
276  it->second->UpdatePipeline(time);
277  }
278  for (auto it = this->filter_map_.begin(); it != this->filter_map_.end();
279  it++) {
280  it->second->UpdatePipeline(time);
281  }
282  insitu_link_->InsituPostProcess(time, time_step);
283 
284  // Wait for changes on ParaView side (Pause ...)
285  if (insitu_link_->GetSimulationPaused()) {
286  if (insitu_link_->WaitForLiveChange()) {
287  break;
288  }
289  } else {
290  break;
291  }
292  }
293  }

295  int CoProcess(vtkCPDataDescription* data_description) override {
296  if (!data_description) {
297  vtkWarningMacro(
298  "The data description is empty. There is nothing to visualize!");
299  return 0;
300  }
301  if (this->RequestDataDescription(data_description) == 0) {
302  return 1;

304 
305  // Update the producers and do live visualization
306  this->UpdateProducers(data_description);
307  this->UpdateFilters(data_description);
308  this->DoLiveVisualization(data_description);
309 
310  return 1;
