91  double interval_midpoint = absolute_time + timestep / 2.0;
92  double interval_endpoint = absolute_time + timestep;
93 
94  double k1 = first_derivatives_[i](absolute_time, concentrations_[i]);
95  double k2 = first_derivatives_[i](
96  interval_midpoint, concentrations_[i] + timestep * k1 / 2.0);
97  double k3 = first_derivatives_[i](
98  interval_midpoint, concentrations_[i] + timestep * k2 / 2.0);
99  double k4 = first_derivatives_[i](interval_endpoint,
100  concentrations_[i] + timestep * k3);
101 
102  concentrations_[i] += timestep / 6.0 * (k1 + 2 * k2 + 2 * k3 + k4);
103  }
