36 void ProgressBar::PrintProgressBar(std::ostream &out) {
37  // Do not write to file because it does not handle "\r" well.
38  if (write_to_file_) {
39  return;
40  }
41 
42  // Print empty line at first iteration to compensate for "\r" below.
43  if (first_iter_) {
44  // Determine if we write to file or terminal.
45  if (!isatty(fileno(stdout))) {
46  std::cout << "<ProgressBar::PrintProgressBar> omits output because you "
47  "write to file."
48  << std::endl;
49  write_to_file_ = true;
50  return;
51  }
52  out << "ET = Elapsed Time; TR = Time Remaining" << std::endl;
53  first_iter_ = false;
54  }
55 
56  // If total_steps_ is not set, we fall back to simple step printing. This can
57  // be the case if someone uses SimulateUntil where we do not know the number
58  // of steps in advance.
59  if (total_steps_ == 0) {
60  out << "Time step: " << total_steps_ << std::endl;
61  }
62 
63  // 1. Get current and compute elapsed time
64  int64_t current_time = Timing::Timestamp();
65  double elapsed_time = static_cast<double>(current_time - start_time_);
66 
67  // 2. Compute ETA
68  double fraction_computed =
69  static_cast<double>(executed_steps_) / static_cast<double>(total_steps_);
70  assert(fraction_computed <= 1);
71  double remaining_time = elapsed_time / fraction_computed - elapsed_time;
72 
73  // 3. Print progress bar
74  size_t n_steps_computed = std::floor(fraction_computed / 0.02);
75  int n_digits_steps = std::ceil(std::log10(total_steps_));
76  n_digits_time_ =
77  std::max({static_cast<int>(std::ceil(std::log10(elapsed_time))),
78  static_cast<int>(std::ceil(std::log10(remaining_time))),
79  n_digits_time_});
80 
81  // Progress bar
82  out << "[" << std::string(n_steps_computed, '#')
83  << std::string(50 - n_steps_computed, ' ') << "] ";
84  // Print number of executed steps
85  out << std::setw(n_digits_steps) << executed_steps_ << " / "
86  << std::setw(n_digits_steps) << total_steps_ << " ";
87  // Print remaining time
88  out << "( ET: " << std::setw(n_digits_time_) << elapsed_time
89  << ", TR:" << std::setw(n_digits_time_) << std::ceil(remaining_time)
90  << ")[ms]";
91  // Override lines in next step
92  out << "\r";
93 
94  // 3. Go to new line once we've reached the last step
95  if (executed_steps_ == total_steps_) {
96  out << "\n";
97  }
98 }
