48 void MortonOrder::Update(const std::array<uint64_t, 3>& num_boxes_axis) {
49  num_boxes_ = num_boxes_axis[0] * num_boxes_axis[1] * num_boxes_axis[2];
50  std::array<uint64_t, 3> max_dimensions{
51  num_boxes_axis[0] - 1, num_boxes_axis[1] - 1, num_boxes_axis[2] - 1};
52  auto max_dim = std::max(num_boxes_axis[0],
53  std::max(num_boxes_axis[1], num_boxes_axis[2]));
54  auto max_depth =
55  static_cast<uint64_t>(std::ceil(std::log(max_dim) / std::log(2)));
56 
57  offset_index_.clear();
58  offset_index_.reserve(2048);
59  offset_index_.push_back({0, 0});
60 
61  auto length = static_cast<uint64_t>(std::pow(2, max_depth - 1));
62 
63  if (length == 0) {
64  return;
65  }
66 
67  Stack<Aux> s(max_depth + 1);
68  s.Push({0, length, length, length, length});
69 
70  uint64_t box_cnt = 0;
71  uint64_t offset = 0;
72  bool record = false;
73 
74  while (s.Size() != 0) {
75  auto& c = s.Top();
76  // determine borders
77  auto xmax = c.xpos;
78  if (c.index == 1 || c.index == 3 || c.index == 5 || c.index == 7) {
79  xmax += c.length;
80  }
81  auto ymax = c.ypos;
82  if ((c.index > 1 && c.index < 4) || c.index > 5) {
83  ymax += c.length;
84  }
85  auto zmax = c.zpos;
86  if (c.index > 3) {
87  zmax += c.length;
88  }
89  auto xmin = xmax - c.length;
90  auto ymin = ymax - c.length;
91  auto zmin = zmax - c.length;
92 
93  if (max_dimensions[0] >= xmax - 1 && max_dimensions[1] >= ymax - 1 &&
94  max_dimensions[2] >= zmax - 1) {
95  // full
96  if (record) {
97  offset_index_.push_back({box_cnt, offset});
98  record = false;
99  }
100  box_cnt += c.length * c.length * c.length;
101  } else if (max_dimensions[0] < xmin || max_dimensions[1] < ymin ||
102  max_dimensions[2] < zmin) {
103  // empty
104  record = true;
105  auto elements = c.length * c.length * c.length;
106  offset += elements;
107  } else {
108  auto next_length = c.length >> 1;
109  s.Push({0, next_length, xmin + next_length, ymin + next_length,
110  zmin + next_length});
111  continue;
112  }
113 
114  // evaluation was full or empty
115  while (s.Size() > 0 && ++(s.Top().index) == 8) {
116  s.Pop();
117  }
118  }
119 }

122 template <typename T>
123 std::pair<uint64_t, uint64_t> BinarySearch(uint64_t search_val,
124  const T& container, uint64_t from,
125  uint64_t to) {
126  if (container.size() == 0) {
127  return {0, 0};
128  }
129  if (to <= from) {
130  if (container[from].first > search_val && from > 0) {
131  from--;
132  }
133  return {container[from].second, from};
134  }
135 
136  auto m = (from + to) / 2;
137  if (container[m].first == search_val) {
138  return {container[m].second, m};
139  } else if (container[m].first > search_val) {
140  return BinarySearch(search_val, container, from, m);
141  } else {
142  return BinarySearch(search_val, container, m + 1, to);
143  }
144 }

154 class MortonIterator : public Iterator<uint64_t> {
155  public:
156  MortonIterator(uint64_t start_index, uint64_t last_index,
157  uint64_t start_morton_code, uint64_t offset_pos,
158  const std::vector<std::pair<uint64_t, uint64_t>>& offset_index)
159  : next_index_(start_index),
160  last_index_(last_index),
161  next_morton_code_(start_morton_code),
162  offset_pos_(offset_pos),
163  offset_index_(offset_index) {}
164  virtual ~MortonIterator() {}
165 
166  bool HasNext() const override { return next_index_ <= last_index_; }
167 
168  uint64_t Next() override {
169  if (!HasNext()) {
170  return 0;
171  }
172  auto ret = next_morton_code_;
173  next_index_++;
174  if (HasNext()) {
175  if (offset_pos_ + 1 < offset_index_.size() &&
176  offset_index_[offset_pos_ + 1].first == next_index_) {
177  offset_pos_++;
178  }
179  next_morton_code_ = next_index_ + offset_index_[offset_pos_].second;
180  }
181  return ret;
182  }
183 
184  private:
185  uint64_t next_index_;
186  uint64_t last_index_;
187  uint64_t next_morton_code_ = 0;
188  uint64_t offset_pos_ = 0;
189  const std::vector<std::pair<uint64_t, uint64_t>>& offset_index_;
190 };
