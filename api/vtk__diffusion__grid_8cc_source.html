33 VtkDiffusionGrid::VtkDiffusionGrid(const std::string& name,
34  vtkCPDataDescription* data_description) {
35  auto* param = Simulation::GetActive()->GetParam();
36  if (param->export_visualization) {
37  auto* tinfo = ThreadInfo::GetInstance();
38  data_.resize(tinfo->GetMaxThreads());
39  } else {
40  data_.resize(1);
41  }
42 
43 #pragma omp parallel for schedule(static, 1)
44  for (uint64_t i = 0; i < data_.size(); ++i) {
45  data_[i] = vtkImageData::New();
46  }
47  name_ = name;
48 
49  // get visualization config
50  const Param::VisualizeDiffusion* vd = nullptr;
51  for (auto& entry : param->visualize_diffusion) {
52  if (entry.name == name) {
53  vd = &entry;
54  break;
55  }
56  }
57 
58  for (uint64_t i = 0; i < data_.size(); ++i) {
59  // Add attribute data
60  if (vd->concentration) {
61  vtkNew<vtkDoubleArray> concentration;
62  concentration->SetName("Substance Concentration");
63  concentration_array_idx_ =
64  data_[i]->GetPointData()->AddArray(concentration.GetPointer());
65  }
66  if (vd->gradient) {
67  vtkNew<vtkDoubleArray> gradient;
68  gradient->SetName("Diffusion Gradient");
69  gradient->SetNumberOfComponents(3);
70  gradient_array_idx_ =
71  data_[i]->GetPointData()->AddArray(gradient.GetPointer());
72  }
73  }
74 
75  if (!param->export_visualization) {
76  data_description->AddInput(name.c_str());
77  data_description->GetInputDescriptionByName(name.c_str())
78  ->SetGrid(data_[0]);
79  }
80 }

95 void VtkDiffusionGrid::Update(const DiffusionGrid* grid) {
96  used_ = true;
97 
98  auto num_boxes = grid->GetNumBoxesArray();
99  auto grid_dimensions = grid->GetDimensions();
100  auto box_length = grid->GetBoxLength();
101  auto total_boxes = grid->GetNumBoxes();
102 
103  auto* tinfo = ThreadInfo::GetInstance();
104  whole_extent_ = {{0, std::max(static_cast<int>(num_boxes[0]) - 1, 0), 0,
105  std::max(static_cast<int>(num_boxes[1]) - 1, 0), 0,
106  std::max(static_cast<int>(num_boxes[2]) - 1, 0)}};
107  Dissect(num_boxes[2], tinfo->GetMaxThreads());
108  CalcPieceExtents(num_boxes);
109  uint64_t xy_num_boxes = num_boxes[0] * num_boxes[1];
110  double origin_x = grid_dimensions[0];
111  double origin_y = grid_dimensions[2];
112  double origin_z = grid_dimensions[4];
113 
114  // do not partition data for insit visualization
115  if (data_.size() == 1) {
116  data_[0]->SetOrigin(origin_x, origin_y, origin_z);
117  data_[0]->SetDimensions(num_boxes[0], num_boxes[1], num_boxes[2]);
118  data_[0]->SetSpacing(box_length, box_length, box_length);
119 
120  if (concentration_array_idx_ != -1) {
121  auto* co_ptr = const_cast<double*>(grid->GetAllConcentrations());
122  auto elements = static_cast<vtkIdType>(total_boxes);
123  auto* array = static_cast<vtkDoubleArray*>(
124  data_[0]->GetPointData()->GetArray(concentration_array_idx_));
125  array->SetArray(co_ptr, elements, 1);
126  }
127  if (gradient_array_idx_ != -1) {
128  auto gr_ptr = const_cast<double*>(grid->GetAllGradients());
129  auto elements = static_cast<vtkIdType>(total_boxes * 3);
130  auto* array = static_cast<vtkDoubleArray*>(
131  data_[0]->GetPointData()->GetArray(gradient_array_idx_));
132  array->SetArray(gr_ptr, elements, 1);
133  }
134  return;
135  }
136 
137 #pragma omp parallel for schedule(static, 1)
138  for (uint64_t i = 0; i < num_pieces_; ++i) {
139  uint64_t piece_elements;
140  auto* e = piece_extents_[i].data();
141  if (i < num_pieces_ - 1) {
142  piece_elements = piece_boxes_z_ * xy_num_boxes;
143  data_[i]->SetDimensions(num_boxes[0], num_boxes[1], piece_boxes_z_);
144  data_[i]->SetExtent(e[0], e[1], e[2], e[3], e[4],
145  e[4] + piece_boxes_z_ - 1);
146  } else {
147  piece_elements = piece_boxes_z_last_ * xy_num_boxes;
148  data_[i]->SetDimensions(num_boxes[0], num_boxes[1], piece_boxes_z_last_);
149  data_[i]->SetExtent(e[0], e[1], e[2], e[3], e[4],
150  e[4] + piece_boxes_z_last_ - 1);
151  }
152  int piece_origin_z = origin_z + box_length * piece_boxes_z_ * i;
153  data_[i]->SetOrigin(origin_x, origin_y, piece_origin_z);
154  data_[i]->SetSpacing(box_length, box_length, box_length);
155 
156  if (concentration_array_idx_ != -1) {
157  auto* co_ptr = const_cast<double*>(grid->GetAllConcentrations());
158  auto elements = static_cast<vtkIdType>(piece_elements);
159  auto* array = static_cast<vtkDoubleArray*>(
160  data_[i]->GetPointData()->GetArray(concentration_array_idx_));
161  if (i < num_pieces_ - 1) {
162  array->SetArray(co_ptr + (elements * i), elements, 1);
163  } else {
164  array->SetArray(co_ptr + total_boxes - elements, elements, 1);
165  }
166  }
167  if (gradient_array_idx_ != -1) {
168  auto gr_ptr = const_cast<double*>(grid->GetAllGradients());
169  auto elements = static_cast<vtkIdType>(piece_elements * 3);
170  auto* array = static_cast<vtkDoubleArray*>(
171  data_[i]->GetPointData()->GetArray(gradient_array_idx_));
172  if (i < num_pieces_ - 1) {
173  array->SetArray(gr_ptr + (elements * i), elements, 1);
174  } else {
175  array->SetArray(gr_ptr + total_boxes - elements, elements, 1);
176  }
177  }
178  }
179 }

192 void VtkDiffusionGrid::Dissect(uint64_t boxes_z, uint64_t num_pieces_target) {
193  if (num_pieces_target == 1) {
194  piece_boxes_z_last_ = 1;
195  piece_boxes_z_ = 1;
196  num_pieces_ = 1;
197  } else if (boxes_z <= num_pieces_target) {
198  piece_boxes_z_last_ = 2;
199  piece_boxes_z_ = 1;
200  num_pieces_ = std::max(1UL, boxes_z - 1);
201  } else {
202  auto boxes_per_piece = static_cast<double>(boxes_z) / num_pieces_target;
203  piece_boxes_z_ = static_cast<uint64_t>(std::ceil(boxes_per_piece));
204  num_pieces_ = boxes_z / piece_boxes_z_;
205  piece_boxes_z_last_ = boxes_z - (num_pieces_ - 1) * piece_boxes_z_;
206  }
207  assert(num_pieces_ > 0);
208  assert(piece_boxes_z_last_ >= 2);
209  assert((num_pieces_ - 1) * piece_boxes_z_ + piece_boxes_z_last_ == boxes_z);
210 }

213 void VtkDiffusionGrid::CalcPieceExtents(
214  const std::array<size_t, 3>& num_boxes) {
215  piece_extents_.resize(num_pieces_);
216  if (num_pieces_ == 1) {
217  piece_extents_[0] = whole_extent_;
218  return;
219  }
220  int c = piece_boxes_z_;
221  piece_extents_[0] = {{0, static_cast<int>(num_boxes[0]) - 1, 0,
222  static_cast<int>(num_boxes[1]) - 1, 0, c}};
223  for (uint64_t i = 1; i < num_pieces_ - 1; ++i) {
224  piece_extents_[i] = {{0, static_cast<int>(num_boxes[0]) - 1, 0,
225  static_cast<int>(num_boxes[1]) - 1, c,
226  c + static_cast<int>(piece_boxes_z_)}};
227  c += piece_boxes_z_;
228  }
229  piece_extents_.back() = {{0, static_cast<int>(num_boxes[0]) - 1, 0,
230  static_cast<int>(num_boxes[1]) - 1, c,
231  static_cast<int>(num_boxes[2]) - 1}};
232 }
