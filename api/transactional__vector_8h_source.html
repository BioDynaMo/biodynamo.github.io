50  static void ShortcutUpdatedIndices(
51  std::unordered_map<uint32_t, uint32_t>* all_updates) {
52  std::vector<uint32_t> delete_keys;
53  for (auto it = all_updates->begin(); it != all_updates->end(); ++it) {
54  uint32_t intermediate = it->second;
55 
56  while (true) {
57  auto search = all_updates->find(intermediate);
58  if (search != all_updates->end()) {
59  intermediate = search->second;
60  delete_keys.push_back(search->first);
61  } else {
62  break;
63  }
64  }
65  // intermediate contains the final new index
66  (*all_updates)[it->first] = intermediate;
67  }
68 
69  // delete all intermediate entries
70  for (auto key : delete_keys) {
71  all_updates->erase(all_updates->find(key));
72  }
73  }

124  std::unordered_map<uint32_t, uint32_t> Commit() {
125  std::lock_guard<std::recursive_mutex> lock(mutex_);
126  std::unordered_map<uint32_t, uint32_t> updated_indices;
127  // commit delayed push backs
128  size_ = data_.size();
129  // commit delayed removes
130  // sort indices in descending order to prevent out of bounds accesses
131  auto descending = [](auto a, auto b) { return a > b; };
132  std::sort(to_be_removed_.begin(), to_be_removed_.end(), descending);
133  for (size_t idx : to_be_removed_) {
134  assert(idx < data_.size() && "Removed index outside array boundaries");
135  if (idx < data_.size() - 1) { // idx does not point to last element
136  // invalidates pointer of last element
137  uint32_t old_index = data_.size() - 1;
138  std::swap(data_[idx], data_[old_index]);
139  data_[idx].SetElementIdx(idx);
140  data_.pop_back();
141  updated_indices[old_index] = idx;
142  } else { // idx points to last element
143  data_.pop_back();
144  }
145  size_--;
146  }
147  to_be_removed_.clear();
148 
149  ShortcutUpdatedIndices(&updated_indices);
150 
151  return updated_indices;
152  }
