50  static void ShortcutUpdatedIndices(
51  std::unordered_map<uint32_t, uint32_t>* all_updates) {
52  std::vector<uint32_t> delete_keys;
53  for (auto it = all_updates->begin(); it != all_updates->end(); ++it) {
54  uint32_t intermediate = it->second;
55 
56  while (true) {
57  auto search = all_updates->find(intermediate);
58  if (search != all_updates->end()) {
59  intermediate = search->second;
60  delete_keys.push_back(search->first);

67  }

72  }

116  std::unordered_map<uint32_t, uint32_t> Commit() {
117  std::lock_guard<std::recursive_mutex> lock(mutex_);
118  std::unordered_map<uint32_t, uint32_t> updated_indices;
119  // commit delayed push backs
120  size_ = data_.size();
121  // commit delayed removes
122  // sort indices in descending order to prevent out of bounds accesses
123  auto descending = [](auto a, auto b) { return a > b; };
124  std::sort(to_be_removed_.begin(), to_be_removed_.end(), descending);
125  for (size_t idx : to_be_removed_) {
126  assert(idx < data_.size() && "Removed index outside array boundaries");
127  if (idx < data_.size() - 1) { // idx does not point to last element
128  // invalidates pointer of last element
129  uint32_t old_index = data_.size() - 1;
130  std::swap(data_[idx], data_[old_index]);
131  data_[idx].SetElementIdx(idx);
132  data_.pop_back();
133  updated_indices[old_index] = idx;
134  } else { // idx points to last element
135  data_.pop_back();
136  }
137  size_--;
138  }
139  to_be_removed_.clear();
140 
141  ShortcutUpdatedIndices(&updated_indices);
