92 void Simulation::Restore(Simulation&& restored) {
93  // random_
94  if (random_.size() != restored.random_.size()) {
95  Log::Warning("Simulation", "The restore file (", param_->restore_file_,
96  ") was run with a different number of threads. Can't restore "
97  "complete random number generator state.");
98  uint64_t min = std::min(random_.size(), restored.random_.size());
99  for (uint64_t i = 0; i < min; i++) {
100  *(random_[i]) = *(restored.random_[i]);
101  }
102  } else {
103  for (uint64_t i = 0; i < random_.size(); i++) {
104  *(random_[i]) = *(restored.random_[i]);
105  }
106  }
107 
108  // param and rm
109  param_->Restore(std::move(*restored.param_));
110  restored.param_ = nullptr;
111  *rm_ = std::move(*restored.rm_);
112  restored.rm_ = nullptr;
113 
114  // name_ and unique_name_
115  InitializeUniqueName(restored.name_);
116  InitializeOutputDir();
117 }

128  if (active_ != this) {
129  tmp = active_;

131  active_ = this;

133  delete rm_;

136  if (so_uid_generator_ != nullptr) {

141  delete r;
142  }
143  for (auto* ectxt : exec_ctxt_) {
144  delete ectxt;
145  }
146  if (mem_mgr_) {
147  delete mem_mgr_;
148  }
149  active_ = tmp;

151 }

217 void Simulation::InitializeMembers() {
218  if (param_->use_bdm_mem_mgr_) {
219  mem_mgr_ = new MemoryManager(param_->mem_mgr_aligned_pages_shift_,
220  param_->mem_mgr_growth_rate_,
221  param_->mem_mgr_max_mem_per_thread_);
222  }
223  so_uid_generator_ = new SoUidGenerator();
224  if (param_->debug_numa_) {
225  std::cout << "ThreadInfo:\n" << *ThreadInfo::GetInstance() << std::endl;
226  }
227 
228  random_.resize(omp_get_max_threads());
229 #pragma omp parallel for schedule(static, 1)
230  for (uint64_t i = 0; i < random_.size(); i++) {
231  random_[i] = new Random();
232  }
233  exec_ctxt_.resize(omp_get_max_threads());
234  auto map =
235  std::make_shared<typename InPlaceExecutionContext::ThreadSafeSoUidMap>();
236 #pragma omp parallel for schedule(static, 1)
237  for (uint64_t i = 0; i < exec_ctxt_.size(); i++) {
238  exec_ctxt_[i] = new InPlaceExecutionContext(map);
239  }
240  rm_ = new ResourceManager();
241  environment_ = new UniformGridEnvironment();
242  scheduler_ = new Scheduler();
243 }

245 void Simulation::InitializeRuntimeParams(
246  CommandLineOptions* clo, const std::function<void(Param*)>& set_param,
247  const std::string& ctor_config) {
248  // Renew thread info just in case it has been initialised as static and a
249  // simulation calls e.g. `omp_set_num_threads()` within main.
250  ThreadInfo::GetInstance()->Renew();
251 
252  param_ = new Param();
253 
254  // detect if the biodynamo environment has been sourced
255  if (std::getenv("BDMSYS") == nullptr) {
256  Log::Fatal(
257  "Simulation::InitializeRuntimeParams",
258  "The BioDynaMo environment is not set up correctly. Please execute "
259  "'source <path-to-bdm-installation>/bin/thisbdm.sh' and retry this "
260  "command.");
261  }
262 
263  static bool read_env = false;
264  if (!read_env) {
265  // Read, only once, bdm.rootrc to set BioDynaMo-related settings for ROOT
266  std::stringstream ss;
267  ss << std::getenv("BDMSYS") << "/etc/bdm.rootrc";
268  gEnv->ReadFile(ss.str().c_str(), kEnvUser);
269  read_env = true;
270  }
271 
272  LoadConfigFile(ctor_config, clo->Get<std::string>("config"));
273  if (clo->Get<std::string>("backup") != "") {
274  param_->backup_file_ = clo->Get<std::string>("backup");
275  }
276  if (clo->Get<std::string>("restore") != "") {
277  param_->restore_file_ = clo->Get<std::string>("restore");
278  }
279 
280  // Handle "cuda" and "opencl" arguments
281  if (clo->Get<bool>("cuda")) {
282  param_->use_gpu_ = true;
283  }
284 
285  if (clo->Get<bool>("opencl")) {
286  param_->use_gpu_ = true;
287  param_->use_opencl_ = true;
288  }
289 
290  ocl_state_ = new OpenCLState();
291 
292  set_param(param_);
293 
294  if (!is_gpu_environment_initialized_ && param_->use_gpu_) {
295  GpuHelper::GetInstance()->InitializeGPUEnvironment();
296  is_gpu_environment_initialized_ = true;
297  }
298 
299  // Removing this line causes an unexplainable segfault due to setting the
300  // gErrorIngoreLevel global parameter of ROOT. We need to log at least one
301  // thing before setting that parameter.
302  Log::Info("", "Initialize new simulation using BioDynaMo ",
303  Version::String());
304 }

306 void Simulation::LoadConfigFile(const std::string& ctor_config,
307  const std::string& cli_config) {
308  constexpr auto kConfigFile = "bdm.toml";
309  constexpr auto kConfigFileParentDir = "../bdm.toml";
310  if (ctor_config != "") {
311  if (FileExists(ctor_config)) {
312  auto config = cpptoml::parse_file(ctor_config);
313  param_->AssignFromConfig(config);
314  } else {
315  Log::Fatal("Simulation::LoadConfigFile", "The config file ", ctor_config,
316  " specified in the constructor of bdm::Simulation "
317  "could not be found.");
318  }
319  } else if (cli_config != "") {
320  if (FileExists(cli_config)) {
321  auto config = cpptoml::parse_file(cli_config);
322  param_->AssignFromConfig(config);
323  } else {
324  Log::Fatal("Simulation::LoadConfigFile", "The config file ", cli_config,
325  " specified as command line argument "
326  "could not be found.");
327  }
328  } else if (FileExists(kConfigFile)) {
329  auto config = cpptoml::parse_file(kConfigFile);
330  param_->AssignFromConfig(config);
331  } else if (FileExists(kConfigFileParentDir)) {
332  auto config = cpptoml::parse_file(kConfigFileParentDir);
333  param_->AssignFromConfig(config);
334  } else {
335  Log::Warning("Simulation::LoadConfigFile", "Config file ", kConfigFile,
336  " not found in `.` or `..` directory.");
337  }
338 }
