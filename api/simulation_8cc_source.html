107 void Simulation::Restore(Simulation&& restored) {
108  // random_
109  if (random_.size() != restored.random_.size()) {
110  Log::Warning("Simulation", "The restore file (", param_->restore_file,
111  ") was run with a different number of threads. Can't restore "
112  "complete random number generator state.");
113  uint64_t min = std::min(random_.size(), restored.random_.size());
114  for (uint64_t i = 0; i < min; i++) {
115  *(random_[i]) = *(restored.random_[i]);
116  }
117  } else {
118  for (uint64_t i = 0; i < random_.size(); i++) {
119  *(random_[i]) = *(restored.random_[i]);
120  }
121  }
122 
123  // param and rm
124  param_->Restore(std::move(*restored.param_));
125  restored.param_ = nullptr;
126  *rm_ = std::move(*restored.rm_);
127  restored.rm_ = nullptr;
128 
129  *time_series_ = std::move(*restored.time_series_);
130 
131  // name_ and unique_name_
132  InitializeUniqueName(restored.name_);
133  InitializeOutputDir();
134 }

327 void Simulation::InitializeMembers() {
328  if (param_->use_bdm_mem_mgr) {
329  mem_mgr_ = new MemoryManager(param_->mem_mgr_aligned_pages_shift,
330  param_->mem_mgr_growth_rate,
331  param_->mem_mgr_max_mem_per_thread_factor);
332  }
333  agent_uid_generator_ = new AgentUidGenerator();
334  if (param_->debug_numa) {
335  std::cout << "ThreadInfo:\n" << *ThreadInfo::GetInstance() << std::endl;
336  }
337 
338  random_.resize(omp_get_max_threads());
339 #pragma omp parallel for schedule(static, 1)
340  for (uint64_t i = 0; i < random_.size(); i++) {
341  random_[i] = new Random();
342  random_[i]->SetSeed(param_->random_seed * (i + 1));
343  }
344  exec_ctxt_.resize(omp_get_max_threads());
345  auto map = std::make_shared<
346  typename InPlaceExecutionContext::ThreadSafeAgentUidMap>();
347 #pragma omp parallel for schedule(static, 1)
348  for (uint64_t i = 0; i < exec_ctxt_.size(); i++) {
349  exec_ctxt_[i] = new InPlaceExecutionContext(map);
350  }
351  rm_ = new ResourceManager();
352 
353  // Set the specified neighborhood search method
354  if (param_->environment == "kd_tree") {
355  environment_ = new KDTreeEnvironment();
356  } else if (param_->environment == "octree") {
357  environment_ = new OctreeEnvironment();
358  } else if (param_->environment == "uniform_grid") {
359  environment_ = new UniformGridEnvironment();
360  } else {
361  Log::Error("Simulation::Initialize", "No such neighboring method '",
362  param_->environment, "'. Defaulting to 'uniform_grid'");
363  environment_ = new UniformGridEnvironment();
364  }
365  scheduler_ = new Scheduler();
366  time_series_ = new experimental::TimeSeries();
367 }

393 void Simulation::InitializeRuntimeParams(
394  CommandLineOptions* clo, const std::function<void(Param*)>& set_param,
395  const std::vector<std::string>& ctor_config_files) {
396  // Renew thread info just in case it has been initialised as static and a
397  // simulation calls e.g. `omp_set_num_threads()` within main.
398  ThreadInfo::GetInstance()->Renew();
399 
400  std::stringstream sstr;
401  sstr << (*clo);
402  command_line_parameter_str_ = sstr.str();
403 
404  param_ = new Param();
405 
406  // detect if the biodynamo environment has been sourced
407  if (std::getenv("BDMSYS") == nullptr) {
408  Log::Fatal(
409  "Simulation::InitializeRuntimeParams",
410  "The BioDynaMo environment is not set up correctly. Please execute "
411  "'source <path-to-bdm-installation>/bin/thisbdm.sh' and retry this "
412  "command.");
413  }
414 
415  static bool read_env = false;
416  if (!read_env) {
417  // Read, only once, bdm.rootrc to set BioDynaMo-related settings for ROOT
418  std::stringstream os;
419  os << std::getenv("BDMSYS") << "/etc/bdm.rootrc";
420  gEnv->ReadFile(os.str().c_str(), kEnvUser);
421  read_env = true;
422  }
423 
424  LoadConfigFiles(ctor_config_files,
425  clo->Get<std::vector<std::string>>("config"));
426  auto inline_configs = clo->Get<std::vector<std::string>>("inline-config");
427  if (inline_configs.size()) {
428  for (auto& inline_config : inline_configs) {
429  param_->MergeJsonPatch(inline_config);
430  }
431  }
432 
433  if (clo->Get<std::string>("backup") != "") {
434  param_->backup_file = clo->Get<std::string>("backup");
435  }
436  if (clo->Get<std::string>("restore") != "") {
437  param_->restore_file = clo->Get<std::string>("restore");
438  }
439 
440  // Handle "cuda" and "opencl" arguments
441  if (clo->Get<bool>("cuda")) {
442  param_->compute_target = "cuda";
443  }
444 
445  if (clo->Get<bool>("opencl")) {
446  param_->compute_target = "opencl";
447  }
448 
449  ocl_state_ = new OpenCLState();
450 
451  set_param(param_);
452 
453  if (!is_gpu_environment_initialized_ && param_->compute_target != "cpu") {
454  GpuHelper::GetInstance()->InitializeGPUEnvironment();
455  is_gpu_environment_initialized_ = true;
456  }
457 
458  // Removing this line causes an unexplainable segfault due to setting the
459  // gErrorIngoreLevel global parameter of ROOT. We need to log at least one
460  // thing before setting that parameter.
461  Log::Info("", "Initialize new simulation using BioDynaMo ",
462  Version::String());
463 }

465 void Simulation::LoadConfigFiles(const std::vector<std::string>& ctor_configs,
466  const std::vector<std::string>& cli_configs) {
467  constexpr auto kTomlConfigFile = "bdm.toml";
468  constexpr auto kJsonConfigFile = "bdm.json";
469  constexpr auto kTomlConfigFileParentDir = "../bdm.toml";
470  constexpr auto kJsonConfigFileParentDir = "../bdm.json";
471  // find config file
472  std::vector<std::string> configs = {};
473  if (ctor_configs.size()) {
474  for (auto& ctor_config : ctor_configs) {
475  if (FileExists(ctor_config)) {
476  configs.push_back(ctor_config);
477  } else {
478  Log::Fatal("Simulation::LoadConfigFiles", "The config file ",
479  ctor_config,
480  " specified in the constructor of bdm::Simulation "
481  "could not be found.");
482  }
483  }
484  }
485  if (cli_configs.size()) {
486  for (auto& cli_config : cli_configs) {
487  if (FileExists(cli_config)) {
488  configs.push_back(cli_config);
489  } else {
490  Log::Fatal("Simulation::LoadConfigFiles", "The config file ",
491  cli_config,
492  " specified as command line argument "
493  "could not be found.");
494  }
495  }
496  }
497 
498  // no config file specified in ctor or cli -> look for default
499  if (!configs.size()) {
500  if (FileExists(kTomlConfigFile)) {
501  configs.push_back(kTomlConfigFile);
502  } else if (FileExists(kTomlConfigFileParentDir)) {
503  configs.push_back(kTomlConfigFileParentDir);
504  } else if (FileExists(kJsonConfigFile)) {
505  configs.push_back(kJsonConfigFile);
506  } else if (FileExists(kJsonConfigFileParentDir)) {
507  configs.push_back(kJsonConfigFileParentDir);
508  }
509  }
510 
511  // load config files
512  if (configs.size()) {
513  for (auto& config : configs) {
514  if (EndsWith(config, ".toml")) {
515  auto toml = cpptoml::parse_file(config);
516  param_->AssignFromConfig(toml);
517  } else if (EndsWith(config, ".json")) {
518  std::ifstream ifs(config);
519  std::stringstream buffer;
520  buffer << ifs.rdbuf();
521  param_->MergeJsonPatch(buffer.str());
522  }
523  Log::Info("Simulation::LoadConfigFiles",
524  "Processed config file: ", config);
525  }
526  } else {
527  Log::Info("Simulation::LoadConfigFiles", "Default config file ",
528  kTomlConfigFile, " or ", kJsonConfigFile,
529  " not found in `.` or `..` directory. No other config file was "
530  "specified as command line parameter or passed to the "
531  "constructor of bdm::Simulation.");
532  }
533 }

545 void Simulation::InitializeOutputDir() {
546  if (unique_name_ == "") {
547  output_dir_ = param_->output_dir;
548  } else {
549  output_dir_ = Concat(param_->output_dir, "/", unique_name_);
550  }
551  if (system(Concat("mkdir -p ", output_dir_).c_str())) {
552  Log::Fatal("Simulation::InitializeOutputDir",
553  "Failed to make output directory ", output_dir_);
554  }
555  if (!fs::is_empty(output_dir_)) {
556  if (param_->remove_output_dir_contents) {
557  RemoveDirectoryContents(output_dir_);
558  } else {
559  Log::Warning(
560  "Simulation::InitializeOutputDir", "Output dir (", output_dir_,
561  ") is not empty. Files will be overriden. This could cause "
562  "inconsistent state of (e.g. visualization files). Consider removing "
563  "all contents "
564  "prior to running a simulation. Have a look at "
565  "Param::remove_output_dir_contents to remove files automatically.");
566  }
567  }
