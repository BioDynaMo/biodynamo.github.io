145  void CalcSimDimensionsAndLargestAgent(
146  std::array<double, 6>* ret_grid_dimensions) {
147  auto* rm = Simulation::GetActive()->GetResourceManager();
148 
149  const auto max_threads = omp_get_max_threads();
150  // allocate version for each thread - avoid false sharing by padding them
151  // assumes 64 byte cache lines (8 * sizeof(double))
152  std::vector<std::array<double, 8>> xmin(max_threads, {{Math::kInfinity}});
153  std::vector<std::array<double, 8>> xmax(max_threads, {{-Math::kInfinity}});
154 
155  std::vector<std::array<double, 8>> ymin(max_threads, {{Math::kInfinity}});
156  std::vector<std::array<double, 8>> ymax(max_threads, {{-Math::kInfinity}});
157 
158  std::vector<std::array<double, 8>> zmin(max_threads, {{Math::kInfinity}});
159  std::vector<std::array<double, 8>> zmax(max_threads, {{-Math::kInfinity}});
160 
161  std::vector<std::array<double, 8>> largest(max_threads, {{0}});
162 
163  SimDimensionAndLargestAgentFunctor functor(xmin, xmax, ymin, ymax, zmin,
164  zmax, largest);
165  rm->ForEachAgentParallel(1000, functor);
166 
167  // reduce partial results into global one
168  double& gxmin = (*ret_grid_dimensions)[0];
169  double& gxmax = (*ret_grid_dimensions)[1];
170  double& gymin = (*ret_grid_dimensions)[2];
171  double& gymax = (*ret_grid_dimensions)[3];
172  double& gzmin = (*ret_grid_dimensions)[4];
173  double& gzmax = (*ret_grid_dimensions)[5];
174  for (int tid = 0; tid < max_threads; tid++) {
175  // x
176  if (xmin[tid][0] < gxmin) {
177  gxmin = xmin[tid][0];
178  }
179  if (xmax[tid][0] > gxmax) {
180  gxmax = xmax[tid][0];
181  }
182  // y
183  if (ymin[tid][0] < gymin) {
184  gymin = ymin[tid][0];
185  }
186  if (ymax[tid][0] > gymax) {
187  gymax = ymax[tid][0];
188  }
189  // z
190  if (zmin[tid][0] < gzmin) {
191  gzmin = zmin[tid][0];
192  }
193  if (zmax[tid][0] > gzmax) {
194  gzmax = zmax[tid][0];
195  }
196  // largest object
197  if (largest[tid][0] > largest_object_size_) {
198  largest_object_size_ = largest[tid][0];
199  }
200  }
201 
202  largest_object_size_squared_ = largest_object_size_ * largest_object_size_;
203  }
204 };
