131  void CalcSimDimensionsAndLargestAgent(
132  std::array<double, 6>* ret_grid_dimensions, double* largest_agent) {
133  auto* rm = Simulation::GetActive()->GetResourceManager();
134 
135  const auto max_threads = omp_get_max_threads();
136  // allocate version for each thread - avoid false sharing by padding them
137  // assumes 64 byte cache lines (8 * sizeof(double))
138  std::vector<std::array<double, 8>> xmin(max_threads, {{Math::kInfinity}});
139  std::vector<std::array<double, 8>> xmax(max_threads, {{-Math::kInfinity}});
140 
141  std::vector<std::array<double, 8>> ymin(max_threads, {{Math::kInfinity}});
142  std::vector<std::array<double, 8>> ymax(max_threads, {{-Math::kInfinity}});
143 
144  std::vector<std::array<double, 8>> zmin(max_threads, {{Math::kInfinity}});
145  std::vector<std::array<double, 8>> zmax(max_threads, {{-Math::kInfinity}});
146 
147  std::vector<std::array<double, 8>> largest(max_threads, {{0}});
148 
149  SimDimensionAndLargestAgentFunctor functor(xmin, xmax, ymin, ymax, zmin,
150  zmax, largest);
151  rm->ForEachAgentParallel(1000, functor);
152 
153  // reduce partial results into global one
154  double& gxmin = (*ret_grid_dimensions)[0];
155  double& gxmax = (*ret_grid_dimensions)[1];
156  double& gymin = (*ret_grid_dimensions)[2];
157  double& gymax = (*ret_grid_dimensions)[3];
158  double& gzmin = (*ret_grid_dimensions)[4];
159  double& gzmax = (*ret_grid_dimensions)[5];
160  for (int tid = 0; tid < max_threads; tid++) {
161  // x
162  if (xmin[tid][0] < gxmin) {
163  gxmin = xmin[tid][0];
164  }
165  if (xmax[tid][0] > gxmax) {
166  gxmax = xmax[tid][0];
167  }
168  // y
169  if (ymin[tid][0] < gymin) {
170  gymin = ymin[tid][0];
171  }
172  if (ymax[tid][0] > gymax) {
173  gymax = ymax[tid][0];
174  }
175  // z
176  if (zmin[tid][0] < gzmin) {
177  gzmin = zmin[tid][0];
178  }
179  if (zmax[tid][0] > gzmax) {
180  gzmax = zmax[tid][0];
181  }
182  // larget object
183  if (largest[tid][0] > (*largest_agent)) {
184  (*largest_agent) = largest[tid][0];
185  }
186  }
187  }
188 };
