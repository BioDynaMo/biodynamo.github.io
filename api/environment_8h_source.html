130  void CalcSimDimensionsAndLargestSimObject(
131  std::array<double, 6>* ret_grid_dimensions, double* largest_so) {
132  auto* rm = Simulation::GetActive()->GetResourceManager();
133 
134  const auto max_threads = omp_get_max_threads();
135  // allocate version for each thread - avoid false sharing by padding them
136  // assumes 64 byte cache lines (8 * sizeof(double))
137  std::vector<std::array<double, 8>> xmin(max_threads, {{Math::kInfinity}});
138  std::vector<std::array<double, 8>> xmax(max_threads, {{-Math::kInfinity}});
139 
140  std::vector<std::array<double, 8>> ymin(max_threads, {{Math::kInfinity}});
141  std::vector<std::array<double, 8>> ymax(max_threads, {{-Math::kInfinity}});
142 
143  std::vector<std::array<double, 8>> zmin(max_threads, {{Math::kInfinity}});
144  std::vector<std::array<double, 8>> zmax(max_threads, {{-Math::kInfinity}});
145 
146  std::vector<std::array<double, 8>> largest(max_threads, {{0}});
147 
148  SimDimensionAndLargestSimObjectFunctor functor(xmin, xmax, ymin, ymax, zmin,
149  zmax, largest);
150  rm->ApplyOnAllElementsParallelDynamic(1000, functor);
151 
152  // reduce partial results into global one
153  double& gxmin = (*ret_grid_dimensions)[0];
154  double& gxmax = (*ret_grid_dimensions)[1];
155  double& gymin = (*ret_grid_dimensions)[2];
156  double& gymax = (*ret_grid_dimensions)[3];
157  double& gzmin = (*ret_grid_dimensions)[4];
158  double& gzmax = (*ret_grid_dimensions)[5];
159  for (int tid = 0; tid < max_threads; tid++) {
160  // x
161  if (xmin[tid][0] < gxmin) {
162  gxmin = xmin[tid][0];
163  }
164  if (xmax[tid][0] > gxmax) {
165  gxmax = xmax[tid][0];
166  }
167  // y
168  if (ymin[tid][0] < gymin) {
169  gymin = ymin[tid][0];
170  }
171  if (ymax[tid][0] > gymax) {
172  gymax = ymax[tid][0];
173  }
174  // z
175  if (zmin[tid][0] < gzmin) {
176  gzmin = zmin[tid][0];
177  }
178  if (zmax[tid][0] > gzmax) {
179  gzmax = zmax[tid][0];
180  }
181  // larget object
182  if (largest[tid][0] > (*largest_so)) {
183  (*largest_so) = largest[tid][0];
184  }
185  }
186  }
187 };
