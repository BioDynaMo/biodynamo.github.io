428  void Remove(SoUid uid) {
429  // remove from map
430  auto it = this->so_storage_location_.find(uid);
431  if (it != this->so_storage_location_.end()) {
432  SoHandle soh = it->second;
433  auto type_idx = soh.GetTypeIdx();
434 
435  ::bdm::Apply(&data_, type_idx, [&, this](auto* container) {
436  auto element_idx = soh.GetElementIdx();
437  auto last = container->size() - 1;
438  if (element_idx == last) {
439  container->pop_back();
440  } else {
441  (*container)[element_idx] = (*container)[last];
442  (*container)[element_idx].SetElementIdx(element_idx);
443  container->pop_back();
444  SoUid changed_uid = (*container)[element_idx].GetUid();
445  this->so_storage_location_[changed_uid] =
446  SoHandle(type_idx, element_idx);
447  }
448  });
449  so_storage_location_.erase(it);
450  }
