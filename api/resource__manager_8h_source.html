371  void Remove(SoUid uid) {
372  // remove from map
373  auto it = uid_soh_map_.find(uid);
374  if (it != uid_soh_map_.end()) {
375  SoHandle soh = it->second;
376  uid_soh_map_.unsafe_erase(it);
377  // remove from vector
378  auto& numa_sos = sim_objects_[soh.GetNumaNode()];
379  if (soh.GetElementIdx() == numa_sos.size() - 1) {
380  delete numa_sos.back();
381  numa_sos.pop_back();
382  } else {
383  // swap
384  delete numa_sos[soh.GetElementIdx()];
385  auto* reordered = numa_sos.back();
386  numa_sos[soh.GetElementIdx()] = reordered;
387  numa_sos.pop_back();
388  uid_soh_map_[reordered->GetUid()] = soh;
389  }
390  }
