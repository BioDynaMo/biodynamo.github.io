347  auto* sim = Simulation::GetActive();

382  void RemoveAgent(const AgentUid& uid) {
383  // remove from map
384  if (uid_ah_map_.Contains(uid)) {
385  auto ah = uid_ah_map_[uid];
386  uid_ah_map_.Remove(uid);
387  // remove from vector
388  auto& numa_agents = agents_[ah.GetNumaNode()];
389  Agent* agent = nullptr;
390  if (ah.GetElementIdx() == numa_agents.size() - 1) {
391  agent = numa_agents.back();
392  numa_agents.pop_back();
393  } else {
394  // swap
395  agent = numa_agents[ah.GetElementIdx()];
396  auto* reordered = numa_agents.back();
397  numa_agents[ah.GetElementIdx()] = reordered;
398  numa_agents.pop_back();
399  uid_ah_map_.Insert(reordered->GetUid(), ah);
400  }
401  if (type_index_) {
402  type_index_->Remove(agent);
403  }
404  delete agent;
405  }
406  }
