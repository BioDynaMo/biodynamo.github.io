348  virtual void EndOfIteration() {
349  // Check if SoUiD defragmentation should be turned on or off
350  double utilization = static_cast<double>(GetNumAgents()) /
351  static_cast<double>(uid_ah_map_.size());
352  auto* sim = Simulation::GetActive();
353  auto* param = sim->GetParam();
354  if (utilization < param->agent_uid_defragmentation_low_watermark) {
355  sim->GetAgentUidGenerator()->EnableDefragmentation(&uid_ah_map_);
356  } else if (utilization > param->agent_uid_defragmentation_high_watermark) {
357  sim->GetAgentUidGenerator()->DisableDefragmentation();
358  }

365  virtual void AddAgents(typename AgentHandle::NumaNode_t numa_node,
366  uint64_t offset,
367  const std::vector<Agent*>& new_agents) {
368  uint64_t i = 0;
369  for (auto* agent : new_agents) {
370  auto uid = agent->GetUid();
371  uid_ah_map_.Insert(uid, AgentHandle(numa_node, offset + i));
372  agents_[numa_node][offset + i] = agent;
373  i++;
374  }
375  if (type_index_) {
376 #pragma omp critical
377  for (auto* agent : new_agents) {
378  type_index_->Add(agent);
379  }
380  }
381 #pragma omp single
382  if (new_agents.size() != 0) {
383  MarkEnvironmentOutOfSync();
384  }

391  void RemoveAgent(const AgentUid& uid) {
392  // remove from map
393  if (uid_ah_map_.Contains(uid)) {
394  auto ah = uid_ah_map_[uid];
395  uid_ah_map_.Remove(uid);
396  // remove from vector
397  auto& numa_agents = agents_[ah.GetNumaNode()];
398  Agent* agent = nullptr;
399  if (ah.GetElementIdx() == numa_agents.size() - 1) {
400  agent = numa_agents.back();
401  numa_agents.pop_back();
402  } else {
403  // swap
404  agent = numa_agents[ah.GetElementIdx()];
405  auto* reordered = numa_agents.back();
406  numa_agents[ah.GetElementIdx()] = reordered;
407  numa_agents.pop_back();
408  uid_ah_map_.Insert(reordered->GetUid(), ah);
409  }
410  if (type_index_) {
411  type_index_->Remove(agent);
412  }
413  delete agent;
414  MarkEnvironmentOutOfSync();
415  }
416  }
