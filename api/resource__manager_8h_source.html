317  void EndOfIteration() {
318  // Check if SoUiD defragmentation should be turned on or off
319  double utilization = static_cast<double>(uid_soh_map_.size()) /
320  static_cast<double>(GetNumSimObjects());
321  auto* sim = Simulation::GetActive();
322  auto* param = sim->GetParam();
323  if (utilization < param->souid_defragmentation_low_watermark_) {
324  sim->GetSoUidGenerator()->EnableDefragmentation(&uid_soh_map_);
325  } else if (utilization > param->souid_defragmentation_high_watermark_) {
326  sim->GetSoUidGenerator()->DisableDefragmentation();

350  void Remove(const SoUid& uid) {
351  // remove from map
352  if (uid_soh_map_.Contains(uid)) {
353  auto soh = uid_soh_map_[uid];
354  uid_soh_map_.Remove(uid);
355  // remove from vector
356  auto& numa_sos = sim_objects_[soh.GetNumaNode()];
357  if (soh.GetElementIdx() == numa_sos.size() - 1) {
358  delete numa_sos.back();
359  numa_sos.pop_back();
360  } else {
361  // swap
362  delete numa_sos[soh.GetElementIdx()];
363  auto* reordered = numa_sos.back();
364  numa_sos[soh.GetElementIdx()] = reordered;
365  numa_sos.pop_back();
366  uid_soh_map_.Insert(reordered->GetUid(), soh);
367  }
368  }
369  }
