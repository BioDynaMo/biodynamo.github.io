494  template <typename TFunction>
495  void ApplyOnAllElementsParallelDynamic(uint64_t chunk, TFunction&& function) {
496  for (uint16_t t = 0; t < NumberOfTypes(); ++t) {
497  // only needed to get the type of the container
498  ::bdm::Apply(&sim_objects_[0], t, [&](auto* container) {
499  // collect all containers of this type
500  std::vector<decltype(container)> so_containers;
501  so_containers.resize(numa_nodes_);
502  so_containers[0] = container;
503  for (uint16_t n = 1; n < numa_nodes_; n++) {
504  ::bdm::Apply(&sim_objects_[n], t, [&](auto* container) {
505  if (std::is_same<raw_type<decltype(so_containers[n])>,
506  raw_type<decltype(container)>>::value) {
507  auto* tmp =
508  reinterpret_cast<raw_type<decltype(so_containers[n])>*>(
509  container);
510  so_containers[n] = tmp;
511  }
512  });
513  }
514 
515  // use dynamic scheduling
516  // Unfortunately openmp's built in functionality can't be used, since
517  // threads belong to different numa domains and thus operate on
518  // different containers
519  std::vector<std::atomic<uint64_t>*> counters(numa_nodes_, nullptr);
520  std::vector<uint64_t> max_counters(numa_nodes_);
521  for (int n = 0; n < numa_nodes_; n++) {
522  counters[n] = new std::atomic<uint64_t>(0);
523  // calculate value max_counters for each numa domain
524  auto correction = so_containers[n]->size() % chunk == 0 ? 0 : 1;
525  max_counters[n] = so_containers[n]->size() / chunk + correction;
526  }
527 
528 #pragma omp parallel
529  {
530  auto tid = omp_get_thread_num();
531  auto nid = thread_info_.GetNumaNode(tid);
532  // thread private variables (compilation error with
533  // firstprivate(chunk, numa_node_) with some openmp versions clause)
534  auto p_numa_nodes = numa_nodes_;
535  auto p_chunk = chunk;
536  assert(thread_info_.GetNumaNode(tid) ==
537  numa_node_of_cpu(sched_getcpu()));
538 
539  // dynamic scheduling
540  uint64_t start = 0;
541  uint64_t end = 0;
542 
543  // this loop implements work stealing from other NUMA nodes if there
544  // are imbalances. Each thread starts with its NUMA domain. Once, it
545  // is finished the thread looks for tasks on other domains
546  for (int n = 0; n < p_numa_nodes; n++) {
547  uint64_t current_nid = (nid + n) % p_numa_nodes;
548 
549  auto* so_container = so_containers[current_nid];
550  uint64_t old_count = (*(counters[current_nid]))++;
551  while (old_count <= max_counters[current_nid]) {
552  start = old_count * p_chunk;
553  end = std::min(static_cast<uint64_t>(so_container->size()),
554  start + p_chunk);
555 
556  for (uint64_t i = start; i < end; ++i) {
557  function((*so_container)[i], SoHandle(current_nid, t, i));
558  }
559 
560  old_count = (*(counters[current_nid]))++;
561  }
562  }
563  }
564 
565  for (int n = 0; n < numa_nodes_; n++) {
566  delete counters[n];
567  }
568  });
569  }
570  }

718  auto max_threads = thread_info_.GetMaxThreads();
719  for (int n = 1; n < numa_nodes_; ++n) {
720  auto threads_in_numa = thread_info_.GetThreadsInNumaNode(n);
721  uint64_t num_so = GetNumSimObjects() * threads_in_numa / max_threads;
722  so_per_numa[n] = num_so;
723  cummulative += num_so;
724  }
725  so_per_numa[0] = GetNumSimObjects() - cummulative;
726 
727  // using first touch policy - page will be allocated to the numa domain of
728  // the thread that accesses it first.
729  // alternative, use numa_alloc_onnode.
730  int ret = numa_run_on_node(0);
731  if (ret != 0) {
732  Log::Fatal("ResourceManager", "Run on numa node failed. Return code: ",
733  ret);
734  }
735 
736  TupleOfSOContainers* so_rearranged = new TupleOfSOContainers[numa_nodes_];
737  TupleOfSOContainers* tmp = sim_objects_;
738  sim_objects_ = so_rearranged;
739  Clear();
740  sim_objects_ = tmp;
741 
742  // numa node -> type idx -> vector of SoHandles
743  std::vector<std::vector<std::vector<SoHandle>>> sorted_so_handles;
744  sorted_so_handles.resize(numa_nodes_);
745  for (auto& e : sorted_so_handles) {
746  e.resize(NumberOfTypes());
747  }
748 
749  uint64_t cnt = 0;
750  uint64_t current_numa = 0;
751 
752  auto rearrange = [&](const SoHandle& handle) {
753  if (cnt == so_per_numa[current_numa]) {
754  cnt = 0;
755  current_numa++;
756  }
757 
758  sorted_so_handles[current_numa][handle.GetTypeIdx()].push_back(handle);
759  cnt++;
760  };
761 
762  auto* grid = Simulation<TCompileTimeParam>::GetActive()->GetGrid();
763  grid->IterateZOrder(rearrange);
764 
765  for (int n = 0; n < numa_nodes_; n++) {
766  for (uint16_t t = 0; t < NumberOfTypes(); t++) {
767  ::bdm::Apply(&so_rearranged[n], t, [&, this](auto* dest) {
768  std::atomic<bool> resized(false);
769 #pragma omp parallel
770  {
771  auto tid = omp_get_thread_num();
772  auto nid = thread_info_.GetNumaNode(tid);
773  if (nid == n) {
774  auto old = std::atomic_exchange(&resized, true);
775  if (!old) {
776  dest->resize(sorted_so_handles[n][t].size());
777  }
778  }
779  }
780 #pragma omp parallel
781  {
782  auto tid = omp_get_thread_num();
783  auto nid = thread_info_.GetNumaNode(tid);
784  if (nid == n) {
785  auto threads_in_numa = thread_info_.GetThreadsInNumaNode(nid);
786  auto& sohandles = sorted_so_handles[n][t];
787  assert(thread_info_.GetNumaNode(tid) ==
788  numa_node_of_cpu(sched_getcpu()));
789 
790  // use static scheduling
791  auto correction = sohandles.size() % threads_in_numa == 0 ? 0 : 1;
792  auto chunk = sohandles.size() / threads_in_numa + correction;
793  auto start = thread_info_.GetNumaThreadId(tid) * chunk;
794  auto end = std::min(sohandles.size(), start + chunk);
795 
796  for (uint64_t e = start; e < end; e++) {
797  auto& handle = sohandles[e];
798  this->ApplyOnElement(handle, [&](auto&& sim_object) {
799  using So = raw_type<decltype(sim_object)>;
800  using SoBackend = typename So::Backend;
801  using DestValueType =
802  typename raw_type<decltype(dest)>::value_type;
803  using DestScalarSoType =
804  typename DestValueType::template Self<Scalar>;
805  using ScalarSo = typename So::template Self<Scalar>;
806 
807  if (std::is_same<DestScalarSoType, ScalarSo>::value) {
808  auto&& tmp = reinterpret_cast<
809  typename DestValueType::template Self<SoBackend>&&>(
810  sim_object);
811  (*dest)[e] = tmp;
812  auto&& so = (*dest)[e];
813  so.SetNumaNode(n);
814  so.SetElementIdx(e);
815  }
816  });
817  }
818  }
819  }
820  });
821  }
822  }
823 
824  // TODO(lukas) for AOS deleting takes around 1s for 3M objects - think about
825  // delaying deletion - does not happen right away.
826  // auto* to_be_deleted = sim_objects_;
827  // std::async(std::launch::async, [=]{ delete[] to_be_deleted; });
828  delete[] sim_objects_;
829  sim_objects_ = so_rearranged;
830 
831  // update so_storage_location_
832  ApplyOnAllElements([this](auto&& so, SoHandle) {
833  this->so_storage_location_[so.GetUid()] = so.GetSoHandle();
834  });
835 
836  // TODO(lukas) do we need this? we don't change the scheduling anymore
837  thread_info_.Renew();
838  }

844  void push_back(const TSo& so) { // NOLINT
845  typename SoHandle::NumaNode_t numa_node = 0;
846  auto* container = GetContainer<TSo>(numa_node);
847  container->push_back(so);
848  auto el_idx = container->size() - 1;
849  auto&& inserted = (*container)[el_idx];
850  inserted.SetNumaNode(numa_node);
851  inserted.SetElementIdx(el_idx);
852  so_storage_location_[inserted.GetUid()] =
853  SoHandle(numa_node, GetTypeIndex<TSo>(), el_idx);
854  }

860  void Remove(SoUid uid) {
861  // remove from map
862  auto it = this->so_storage_location_.find(uid);
863  if (it != this->so_storage_location_.end()) {
864  SoHandle soh = it->second;
865  auto type_idx = soh.GetTypeIdx();
866 
867  ::bdm::Apply(&sim_objects_[soh.GetNumaNode()], type_idx,
868  [&, this](auto* container) {
869  auto element_idx = soh.GetElementIdx();
870  auto last = container->size() - 1;
871  if (element_idx == last) {
872  container->pop_back();
873  } else {
874  (*container)[element_idx] = (*container)[last];
875  (*container)[element_idx].SetElementIdx(element_idx);
876  container->pop_back();
877  SoUid changed_uid = (*container)[element_idx].GetUid();
878  this->so_storage_location_[changed_uid] =
879  SoHandle(type_idx, element_idx);
880  }
881  });
882  so_storage_location_.erase(it);
883  }
884  }
