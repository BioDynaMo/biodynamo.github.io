74 struct SetRunDisplacementForEachNeighbor
75  : public Functor<void, const SimObject*, double> {
76  SimObject* so_;
77  SetRunDisplacementForEachNeighbor(SimObject* so) : so_(so) {}
78 
79  void operator()(const SimObject* neighbor, double squared_distance) override {
80  double distance = so_->GetDiameter() + neighbor->GetDiameter();

122 void SimObject::RemoveBiologyModule(const BaseBiologyModule* remove_module) {
123  for (unsigned int i = 0; i < biology_modules_.size(); i++) {
124  if (biology_modules_[i] == remove_module) {

172 void SimObject::BiologyModuleEventHandler(const Event& event,
173  decltype(biology_modules_) * other1,
174  decltype(biology_modules_) * other2) {
175  // call event handler for biology modules
176  uint64_t cnt = 0;
177  for (auto* bm : biology_modules_) {
178  bool copy = bm->Copy(event.GetId());
179  if (!bm->Remove(event.GetId())) {
180  if (copy) {
181  auto* other1_bm = other1 != nullptr ? (*other1)[cnt] : nullptr;
182  auto* other2_bm = other2 != nullptr ? (*other2)[cnt] : nullptr;
183  bm->EventHandler(event, other1_bm, other2_bm);
184  } else {
185  bm->EventHandler(event, nullptr, nullptr);
186  }
187  }
188  cnt += copy ? 1 : 0;
189  }
190 
191  // remove biology modules from current
192  for (auto it = biology_modules_.begin(); it != biology_modules_.end();) {
193  auto* bm = *it;
194  if (bm->Remove(event.GetId())) {
195  delete *it;
196  it = biology_modules_.erase(it);
197  } else {
198  ++it;
199  }
200  }
201 }
