73 void SimObject::ApplyRunDisplacementForAllNextTs() {
74  if (!Simulation::GetActive()->GetParam()->detect_static_sim_objects_) {
75  run_displacement_next_ts_ = true;
76  return;
77  }
78 
79  if (!run_displacement_for_all_next_ts_) {
80  return;
81  }
82  run_displacement_for_all_next_ts_ = false;
83  run_displacement_next_ts_ = true;
84  auto* ctxt = Simulation::GetActive()->GetExecutionContext();
85  ctxt->ForEachNeighbor(
86  [this](const SimObject* neighbor, double squared_distance) {
87  double distance = this->GetDiameter() + neighbor->GetDiameter();
88  if (squared_distance < distance * distance) {
89  neighbor->SetRunDisplacementNextTimestep(true);
90  }
91  },
92  *this);
93 }

161 void SimObject::BiologyModuleEventHandler(const Event& event,
162  decltype(biology_modules_) * other1,
163  decltype(biology_modules_) * other2) {
164  // call event handler for biology modules
165  uint64_t cnt = 0;
166  for (auto* bm : biology_modules_) {
167  bool copy = bm->Copy(event.GetId());
168  if (!bm->Remove(event.GetId())) {
169  if (copy) {
170  auto* other1_bm = other1 != nullptr ? (*other1)[cnt] : nullptr;
171  auto* other2_bm = other2 != nullptr ? (*other2)[cnt] : nullptr;
172  bm->EventHandler(event, other1_bm, other2_bm);
173  } else {
174  bm->EventHandler(event, nullptr, nullptr);
175  }
176  }
177  cnt += copy ? 1 : 0;
178  }
179 
180  // remove biology modules from current
181  for (auto it = biology_modules_.begin(); it != biology_modules_.end();) {
182  auto* bm = *it;
183  if (bm->Remove(event.GetId())) {
184  delete *it;
185  it = biology_modules_.erase(it);
186  } else {
187  ++it;
188  }
189  }
190 }
