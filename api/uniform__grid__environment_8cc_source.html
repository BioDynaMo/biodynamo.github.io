53 void UniformGridEnvironment::LoadBalanceInfoUG::InitializeVectors() {
54 #pragma omp parallel
55  {
56  auto* ti = ThreadInfo::GetInstance();
57  auto num_threads = ti->GetMaxThreads();
58  auto tid = ti->GetMyThreadId();
59  // use static scheduling
60  auto correction = grid_->total_num_boxes_ % num_threads == 0 ? 0 : 1;
61  auto chunk = grid_->total_num_boxes_ / num_threads + correction;
62  auto start = tid * chunk;
63  auto end = std::min(grid_->total_num_boxes_, start + chunk);
64 
65  InitializeVectorFunctor f(grid_, start, sorted_boxes_, cummulated_agents_);
66  mo_.CallMortonIteratorConsumer(start, end - 1, f);
67  }
68 }

77  AgentHandleIterator(uint64_t start, uint64_t end, uint64_t box_index,
78  uint64_t discard, decltype(sorted_boxes) sorted_boxes)
79  : start(start),

83  sorted_boxes(sorted_boxes),

85  // discard elements
86  tid = ThreadInfo::GetInstance()->GetMyThreadId();

88  Next();

93  bool HasNext() const override { return start < end; }

95  AgentHandle Next() override {
96  while (box_it.IsAtEnd()) {
97  box_index++;
98  box_it = sorted_boxes[box_index]->begin();

100  auto ret = *box_it;
101  start++;
102  ++box_it;
103  return ret;
104  }

108 void UniformGridEnvironment::LoadBalanceInfoUG::CallHandleIteratorConsumer(
109  uint64_t start, uint64_t end,
110  Functor<void, Iterator<AgentHandle>*>& f) const {
111  if (grid_->total_num_boxes_ == 0 || end <= start) {
112  return;
113  }
114  auto index =
115  BinarySearch(start, cummulated_agents_, 0, grid_->total_num_boxes_ - 1) +
116  1;
117  AgentHandleIterator it(start, end, index,
118  start - cummulated_agents_[index - 1], sorted_boxes_);
119  f(&it);
120 }

137 void UniformGridEnvironment::LoadBalanceInfoUG::InitializeVectorFunctor::
138 operator()(Iterator<uint64_t>* it) {
139  while (it->HasNext()) {
140  auto morton_code = it->Next();
141  uint_fast32_t x, y, z;
142  libmorton::morton3D_64_decode(morton_code, x, y, z);
143  auto* box = grid->GetBoxPointer(grid->GetBoxIndex(std::array<uint64_t, 3>{
144  static_cast<uint64_t>(x), static_cast<uint64_t>(y),
145  static_cast<uint64_t>(z)}));
146  sorted_boxes[start] = box;
147  cummulated_agents[start] = box->Size(grid->timestamp_);
148  start++;
149  }
150 }
