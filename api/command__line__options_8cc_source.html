54 void CommandLineOptions::Parse() {
55  // Make a non-const deep copy of argv
56  char** argv_copy = (char**)malloc((argc_ + 1) * sizeof(char*));
57  int argc_copy = argc_;
58  for (int i = 0; i < argc_; ++i) {
59  size_t length = strlen(argv_[i]) + 1;
60  argv_copy[i] = (char*)malloc(length);
61  memcpy(argv_copy[i], argv_[i], length);
62  }
63  argv_copy[argc_] = NULL;
64 
65  // Perform parsing (consumes argc_copy and argv_copy)
66  if (parser_) {
67  delete parser_;
68  parser_ = nullptr;
69  }
70 
71  try {
72  parser_ = new cxxopts::ParseResult(options_.parse(argc_copy, argv_copy));
73  } catch (const cxxopts::option_not_exists_exception&) {
74  Log::Fatal("CommandLineOptions::ParseResult",
75  "Please add all your command line options before:\n 1) Using "
76  "the CommandLineOptions::Get() method.\n 2) Creating a "
77  "Simulation object.");
78  }
79 
80  if (first_parse_) {
81  // Perform operations on Core command line options
82  HandleCoreOptions();
83  first_parse_ = false;
84  }
85 
86  // free memory
87  for (int i = 0; i < argc_; ++i) {
88  free(argv_copy[i]);
89  }
90  free(argv_copy);
91 }

123 void CommandLineOptions::HandleCoreOptions() {
124  // Handle "help" argument
125  if (parser_->count("help")) {
126  auto groups = options_.groups();
127  auto it = std::find(groups.begin(), groups.end(), "Core");
128  std::rotate(it, it + 1, groups.end());
129  std::cout << options_.help(groups) << std::endl;
130  exit(0);
131  }
132 
133  if (parser_->count("version")) {
134  std::cout << "BioDynaMo Version: " << Version::String() << std::endl;
135  exit(0);
136  }
137 
138  // Handle "verbose" argument
139  // If set in etc/bdm.rootrc use that value, command line argument will
140  // override it
141  Int_t ll = kWarning;
142  TString slevel = "Warning";
143  TEnvRec* rec = gEnv->Lookup("Root.ErrorIgnoreLevel");
144  if (rec) {
145  if (rec->GetLevel() == kEnvUser)
146  slevel = rec->GetValue();
147  }
148  if (!slevel.CompareTo("Print", TString::kIgnoreCase))
149  ll = kPrint;
150  else if (!slevel.CompareTo("Info", TString::kIgnoreCase))
151  ll = kInfo;
152  else if (!slevel.CompareTo("Warning", TString::kIgnoreCase))
153  ll = kWarning;
154  else if (!slevel.CompareTo("Error", TString::kIgnoreCase))
155  ll = kError;
156 
157  if (parser_->count("verbose")) {
158  auto verbosity = parser_->count("verbose");
159 
160  switch (verbosity) {
161  // case 0 can never occur; we wouldn't go into this if statement
162  case 1:
163  ll = kWarning;
164  break;
165  case 2:
166  ll = kInfo;
167  break;
168  case 3:
169  ll = kPrint;
170  break;
171  default:
172  ll = kPrint;
173  break;
174  }
175  }
176  // Global variable of ROOT that determines verbosity of logging functions
177  gErrorIgnoreLevel = ll;
178 
179  if (parser_->count("output-default-json")) {
180  Param param;
181  std::cout << "Below you can find a JSON string with all available "
182  "parameters and their default values.\n"
183  << "Have a look at https://biodynamo.org/bioapi/ for more "
184  "details about each parameter."
185  << std::endl;
186  std::cout << param.ToJsonString() << std::endl;
187  exit(0);
188  }
189 
190  auto toml_file = (*parser_)["toml-to-json"].as<std::string>();
191  if (toml_file != "") {
192  if (!FileExists(toml_file)) {
193  Log::Fatal("CommandLineOptions::HandleCoreOptions",
194  "Specified TOML file (", toml_file, ") does not exist.");
195  }
196  auto toml = cpptoml::parse_file(toml_file);
197  Param param;
198  param.AssignFromConfig(toml);
199  std::cout << param.ToJsonString() << std::endl;
200  exit(0);
201  }
202 }
