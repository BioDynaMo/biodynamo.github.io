188  NeighborIterator& ForwardToNonEmptyBox(uint64_t grid_timestamp) {
189  // increment box id until non empty box has been found
190  while (++box_idx_ < neighbor_boxes_.size()) {
191  // box is empty or uninitialized (padding box) -> continue
192  if (neighbor_boxes_[box_idx_]->IsEmpty(grid_timestamp)) {
193  continue;
194  }
195  // a non-empty box has been found
196  box_iterator_ = neighbor_boxes_[box_idx_]->begin();
197  return *this;
198  }
199  // all remaining boxes have been empty; reached end
200  is_end_ = true;
201  return *this;
202  }

250  void Update() override {
251  auto* rm = Simulation::GetActive()->GetResourceManager();
252 
253  if (rm->GetNumSimObjects() != 0) {
254  Clear();
255  timestamp_++;
256 
257  auto inf = Math::kInfinity;
258  std::array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
259  CalcSimDimensionsAndLargestSimObject(&tmp_dim, &largest_object_size_);
260  RoundOffGridDimensions(tmp_dim);
261 
262  auto los = ceil(largest_object_size_);
263  assert(los > 0 &&
264  "The largest object size was found to be 0. Please check if your "
265  "cells are correctly initialized.");
266  box_length_ = los;
267 
268  for (int i = 0; i < 3; i++) {
269  int dimension_length =
270  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
271  int r = dimension_length % box_length_;
272  // If the grid is not perfectly divisible along each dimension by the
273  // resolution, extend the grid so that it is
274  if (r != 0) {
275  // std::abs for the case that box_length_ > dimension_length
276  grid_dimensions_[2 * i + 1] += (box_length_ - r);
277  } else {
278  // Else extend the grid dimension with one row, because the outmost
279  // object lies exactly on the border
280  grid_dimensions_[2 * i + 1] += box_length_;
281  }
282  }
283 
284  // Pad the grid to avoid out of bounds check when search neighbors
285  for (int i = 0; i < 3; i++) {
286  grid_dimensions_[2 * i] -= box_length_;
287  grid_dimensions_[2 * i + 1] += box_length_;
288  }
289 
290  // Calculate how many boxes fit along each dimension
291  for (int i = 0; i < 3; i++) {
292  int dimension_length =
293  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
294  assert((dimension_length % box_length_ == 0) &&
295  "The grid dimensions are not a multiple of its box length");
296  num_boxes_axis_[i] = dimension_length / box_length_;
297  }
298 
299  num_boxes_xy_ = num_boxes_axis_[0] * num_boxes_axis_[1];
300  auto total_num_boxes = num_boxes_xy_ * num_boxes_axis_[2];
301 
302  CheckGridGrowth();
303 
304  // resize boxes_
305  if (boxes_.size() != total_num_boxes) {
306  if (boxes_.capacity() < total_num_boxes) {
307  boxes_.reserve(total_num_boxes * 2);
308  }
309  boxes_.resize(total_num_boxes);
310  }
311 
312  successors_.reserve();
313 
314  // Assign simulation objects to boxes
315  AssignToBoxesFunctor functor(this);
316  rm->ApplyOnAllElementsParallelDynamic(1000, functor);
317  auto* param = Simulation::GetActive()->GetParam();
318  if (param->bound_space_) {
319  int min = param->min_bound_;
320  int max = param->max_bound_;
321  threshold_dimensions_ = {min, max};
322  }
323 
324  if (param->thread_safety_mechanism_ ==
325  Param::ThreadSafetyMechanism::kAutomatic) {
326  nb_mutex_builder_->Update();
327  }
328  } else {
329  // There are no sim objects in this simulation
330  auto* param = Simulation::GetActive()->GetParam();
331 
332  bool uninitialized = boxes_.size() == 0;
333  if (uninitialized && param->bound_space_) {
334  // Simulation has never had any simulation objects
335  // Initialize grid dimensions with `Param::min_bound_` and
336  // `Param::max_bound_`
337  // This is required for the DiffusionGrid
338  int min = param->min_bound_;
339  int max = param->max_bound_;
340  grid_dimensions_ = {min, max, min, max, min, max};
341  threshold_dimensions_ = {min, max};
342  has_grown_ = true;
343  } else if (!uninitialized) {
344  // all simulation objects have been removed in the last iteration
345  // grid state remains the same, but we have to set has_grown_ to false
346  // otherwise the DiffusionGrid will attempt to resize
347  has_grown_ = false;
348  } else {
349  Log::Fatal(
350  "UniformGridEnvironment",
351  "You tried to initialize an empty simulation without bound space. "
352  "Therefore we cannot determine the size of the simulation space. "
353  "Please add simulation objects, or set Param::bound_space_, "
354  "Param::min_bound_, and Param::max_bound_.");
355  }
356  }
357  }

367  inline double SquaredEuclideanDistance(const Double3& pos1,
368  const Double3& pos2) const {
369  const double dx = pos2[0] - pos1[0];
370  const double dy = pos2[1] - pos1[1];
371  const double dz = pos2[2] - pos1[2];
372  return (dx * dx + dy * dy + dz * dz);
373  }
374 
375  inline bool WithinSquaredEuclideanDistance(double squared_radius,
376  const Double3& pos1,
377  const Double3& pos2) const {
378  const double dx = pos2[0] - pos1[0];
379  const double dx2 = dx * dx;
380  if (dx2 > squared_radius) {
381  return false;
382  }
383 
384  const double dy = pos2[1] - pos1[1];
385  const double dy2_plus_dx2 = dy * dy + dx2;
386  if (dy2_plus_dx2 > squared_radius) {
387  return false;
388  }
389 
390  const double dz = pos2[2] - pos1[2];
391  const double distance = dz * dz + dy2_plus_dx2;
392  return distance < squared_radius;
393  }

395  void UpdateBoxZOrder() {
396  // iterate boxes in Z-order / morton order
397  // TODO(lukas) this is a very quick attempt to test an idea
398  // improve performance of this brute force solution
399  zorder_sorted_boxes_.resize(boxes_.size());
400  const uint32_t nx = num_boxes_axis_[0];
401  const uint32_t ny = num_boxes_axis_[1];
402  const uint32_t nz = num_boxes_axis_[2];
403 #pragma omp parallel for collapse(3)
404  for (uint32_t x = 0; x < nx; x++) {
405  for (uint32_t y = 0; y < ny; y++) {
406  for (uint32_t z = 0; z < nz; z++) {
407  auto box_idx = GetBoxIndex(std::array<uint32_t, 3>{x, y, z});
408  auto morton = libmorton::morton3D_64_encode(x, y, z);
409  zorder_sorted_boxes_[box_idx] =
410  std::pair<uint32_t, const Box*>{morton, &boxes_[box_idx]};
411  }
412  }
413  }
414 #ifdef LINUX
415  __gnu_parallel::sort(
416  zorder_sorted_boxes_.begin(), zorder_sorted_boxes_.end(),
417  [](const auto& lhs, const auto& rhs) { return lhs.first < rhs.first; });
418 #else
419  std::sort(
420  zorder_sorted_boxes_.begin(), zorder_sorted_boxes_.end(),
421  [](const auto& lhs, const auto& rhs) { return lhs.first < rhs.first; });
422 #endif // LINUX
423  }

470  void ForEachNeighbor(Functor<void, const SimObject*, double>& lambda,
471  const SimObject& query) override {
472  const auto& position = query.GetPosition();
473  auto idx = query.GetBoxIdx();
474 
475  FixedSizeVector<const Box*, 27> neighbor_boxes;
476  GetMooreBoxes(&neighbor_boxes, idx);
477 
478  auto* rm = Simulation::GetActive()->GetResourceManager();
479 
480  NeighborIterator ni(neighbor_boxes, timestamp_);
481  const unsigned batch_size = 64;
482  uint64_t size = 0;
483  SimObject* sim_objects[batch_size] __attribute__((aligned(64)));
484  double x[batch_size] __attribute__((aligned(64)));
485  double y[batch_size] __attribute__((aligned(64)));
486  double z[batch_size] __attribute__((aligned(64)));
487  double squared_distance[batch_size] __attribute__((aligned(64)));
488 
489  auto process_batch = [&]() {
490 #pragma omp simd
491  for (uint64_t i = 0; i < size; ++i) {
492  const double dx = x[i] - position[0];
493  const double dy = y[i] - position[1];
494  const double dz = z[i] - position[2];
495 
496  squared_distance[i] = dx * dx + dy * dy + dz * dz;
497  }
498 
499  for (uint64_t i = 0; i < size; ++i) {
500  lambda(sim_objects[i], squared_distance[i]);
501  }
502  size = 0;
503  };
504 
505  while (!ni.IsAtEnd()) {
506  auto soh = *ni;
507  // increment iterator already here to hide memory latency
508  ++ni;
509  auto* sim_object = rm->GetSimObjectWithSoHandle(soh);
510  if (sim_object != &query) {
511  sim_objects[size] = sim_object;
512  const auto& pos = sim_object->GetPosition();
513  x[size] = pos[0];
514  y[size] = pos[1];
515  z[size] = pos[2];
516  size++;
517  if (size == batch_size) {
518  process_batch();
519  }
520  }
521  }
522  process_batch();
523  }
