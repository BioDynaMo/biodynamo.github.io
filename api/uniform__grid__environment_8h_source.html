194  NeighborIterator& ForwardToNonEmptyBox(uint64_t grid_timestamp) {
195  // increment box id until non empty box has been found
196  while (++box_idx_ < neighbor_boxes_.size()) {
197  // box is empty or uninitialized (padding box) -> continue
198  if (neighbor_boxes_[box_idx_]->IsEmpty(grid_timestamp)) {
199  continue;
200  }
201  // a non-empty box has been found
202  box_iterator_ = neighbor_boxes_[box_idx_]->begin();
203  return *this;
204  }
205  // all remaining boxes have been empty; reached end
206  is_end_ = true;
207  return *this;

257  void Update() override {
258  auto* rm = Simulation::GetActive()->GetResourceManager();
259 
260  if (rm->GetNumAgents() != 0) {
261  Clear();
262  timestamp_++;
263 
264  auto inf = Math::kInfinity;
265  std::array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
266  CalcSimDimensionsAndLargestAgent(&tmp_dim, &largest_object_size_);
267  RoundOffGridDimensions(tmp_dim);
268 
269  auto los = ceil(largest_object_size_);
270  assert(los > 0 &&
271  "The largest object size was found to be 0. Please check if your "
272  "cells are correctly initialized.");
273  box_length_ = los;
274 
275  for (int i = 0; i < 3; i++) {
276  int dimension_length =
277  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
278  int r = dimension_length % box_length_;
279  // If the grid is not perfectly divisible along each dimension by the
280  // resolution, extend the grid so that it is
281  if (r != 0) {
282  // std::abs for the case that box_length_ > dimension_length
283  grid_dimensions_[2 * i + 1] += (box_length_ - r);
284  } else {
285  // Else extend the grid dimension with one row, because the outmost
286  // object lies exactly on the border
287  grid_dimensions_[2 * i + 1] += box_length_;
288  }
289  }
290 
291  // Pad the grid to avoid out of bounds check when search neighbors
292  for (int i = 0; i < 3; i++) {
293  grid_dimensions_[2 * i] -= box_length_;
294  grid_dimensions_[2 * i + 1] += box_length_;
295  }
296 
297  // Calculate how many boxes fit along each dimension
298  for (int i = 0; i < 3; i++) {
299  int dimension_length =
300  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
301  assert((dimension_length % box_length_ == 0) &&
302  "The grid dimensions are not a multiple of its box length");
303  num_boxes_axis_[i] = dimension_length / box_length_;
304  }
305 
306  num_boxes_xy_ = num_boxes_axis_[0] * num_boxes_axis_[1];
307  auto total_num_boxes = num_boxes_xy_ * num_boxes_axis_[2];
308 
309  CheckGridGrowth();
310 
311  // resize boxes_
312  if (boxes_.size() != total_num_boxes) {
313  if (boxes_.capacity() < total_num_boxes) {
314  boxes_.reserve(total_num_boxes * 2);
315  }
316  boxes_.resize(total_num_boxes);
317  }
318 
319  successors_.reserve();
320 
321  // Assign agents to boxes
322  AssignToBoxesFunctor functor(this);
323  rm->ForEachAgentParallel(1000, functor);
324  auto* param = Simulation::GetActive()->GetParam();
325  if (param->bound_space) {
326  int min = param->min_bound;
327  int max = param->max_bound;
328  threshold_dimensions_ = {min, max};
329  }
330 
331  if (param->thread_safety_mechanism ==
332  Param::ThreadSafetyMechanism::kAutomatic) {
333  nb_mutex_builder_->Update();
334  }
335  } else {
336  // There are no agents in this simulation
337  auto* param = Simulation::GetActive()->GetParam();
338 
339  bool uninitialized = boxes_.size() == 0;
340  if (uninitialized && param->bound_space) {
341  // Simulation has never had any agents
342  // Initialize grid dimensions with `Param::min_bound` and
343  // `Param::max_bound`
344  // This is required for the DiffusionGrid
345  int min = param->min_bound;
346  int max = param->max_bound;
347  grid_dimensions_ = {min, max, min, max, min, max};
348  threshold_dimensions_ = {min, max};
349  has_grown_ = true;
350  } else if (!uninitialized) {
351  // all agents have been removed in the last iteration
352  // grid state remains the same, but we have to set has_grown_ to false
353  // otherwise the DiffusionGrid will attempt to resize
354  has_grown_ = false;
355  } else {
356  Log::Fatal(
357  "UniformGridEnvironment",
358  "You tried to initialize an empty simulation without bound space. "
359  "Therefore we cannot determine the size of the simulation space. "
360  "Please add agents, or set Param::bound_space, "
361  "Param::min_bound, and Param::max_bound.");
362  }
363  }
364  }

374  inline double SquaredEuclideanDistance(const Double3& pos1,
375  const Double3& pos2) const {
376  const double dx = pos2[0] - pos1[0];
377  const double dy = pos2[1] - pos1[1];
378  const double dz = pos2[2] - pos1[2];
379  return (dx * dx + dy * dy + dz * dz);
380  }
381 
382  inline bool WithinSquaredEuclideanDistance(double squared_radius,
383  const Double3& pos1,
384  const Double3& pos2) const {
385  const double dx = pos2[0] - pos1[0];
386  const double dx2 = dx * dx;
387  if (dx2 > squared_radius) {
388  return false;
389  }
390 
391  const double dy = pos2[1] - pos1[1];
392  const double dy2_plus_dx2 = dy * dy + dx2;
393  if (dy2_plus_dx2 > squared_radius) {
394  return false;
395  }
396 
397  const double dz = pos2[2] - pos1[2];
398  const double distance = dz * dz + dy2_plus_dx2;
399  return distance < squared_radius;
400  }

402  void UpdateBoxZOrder() {
403  // iterate boxes in Z-order / morton order
404  // TODO(lukas) this is a very quick attempt to test an idea
405  // improve performance of this brute force solution
406  zorder_sorted_boxes_.resize(boxes_.size());
407  const uint32_t nx = num_boxes_axis_[0];
408  const uint32_t ny = num_boxes_axis_[1];
409  const uint32_t nz = num_boxes_axis_[2];
410 #pragma omp parallel for collapse(3)
411  for (uint32_t x = 0; x < nx; x++) {
412  for (uint32_t y = 0; y < ny; y++) {
413  for (uint32_t z = 0; z < nz; z++) {
414  auto box_idx = GetBoxIndex(std::array<uint32_t, 3>{x, y, z});
415  auto morton = libmorton::morton3D_64_encode(x, y, z);
416  zorder_sorted_boxes_[box_idx] =
417  std::pair<uint32_t, const Box*>{morton, &boxes_[box_idx]};
418  }
419  }
420  }
421 #ifdef LINUX
422  __gnu_parallel::sort(
423  zorder_sorted_boxes_.begin(), zorder_sorted_boxes_.end(),
424  [](const auto& lhs, const auto& rhs) { return lhs.first < rhs.first; });
425 #else
426  std::sort(
427  zorder_sorted_boxes_.begin(), zorder_sorted_boxes_.end(),
428  [](const auto& lhs, const auto& rhs) { return lhs.first < rhs.first; });
429 #endif // LINUX
430  }

477  void ForEachNeighbor(Functor<void, const Agent*, double>& lambda,
478  const Agent& query) override {
479  const auto& position = query.GetPosition();
480  auto idx = query.GetBoxIdx();
481 
482  FixedSizeVector<const Box*, 27> neighbor_boxes;
483  GetMooreBoxes(&neighbor_boxes, idx);
484 
485  auto* rm = Simulation::GetActive()->GetResourceManager();
486 
487  NeighborIterator ni(neighbor_boxes, timestamp_);
488  const unsigned batch_size = 64;
489  uint64_t size = 0;
490  Agent* agents[batch_size] __attribute__((aligned(64)));
491  double x[batch_size] __attribute__((aligned(64)));
492  double y[batch_size] __attribute__((aligned(64)));
493  double z[batch_size] __attribute__((aligned(64)));
494  double squared_distance[batch_size] __attribute__((aligned(64)));
495 
496  auto process_batch = [&]() {
497 #pragma omp simd
498  for (uint64_t i = 0; i < size; ++i) {
499  const double dx = x[i] - position[0];
500  const double dy = y[i] - position[1];
501  const double dz = z[i] - position[2];
502 
503  squared_distance[i] = dx * dx + dy * dy + dz * dz;
504  }
505 
506  for (uint64_t i = 0; i < size; ++i) {
507  lambda(agents[i], squared_distance[i]);
508  }
509  size = 0;
510  };
511 
512  while (!ni.IsAtEnd()) {
513  auto ah = *ni;
514  // increment iterator already here to hide memory latency
515  ++ni;
516  auto* agent = rm->GetAgent(ah);
517  if (agent != &query) {
518  agents[size] = agent;
519  const auto& pos = agent->GetPosition();
520  x[size] = pos[0];
521  y[size] = pos[1];
522  z[size] = pos[2];
523  size++;
524  if (size == batch_size) {
525  process_batch();
526  }
527  }
528  }
529  process_batch();
530  }

741  void CheckGridGrowth() {
742  // Determine if the grid dimensions have changed (changed in the sense that
743  // the grid has grown outwards)
744  auto min_gd =
745  *std::min_element(grid_dimensions_.begin(), grid_dimensions_.end());
746  auto max_gd =
747  *std::max_element(grid_dimensions_.begin(), grid_dimensions_.end());
748  if (min_gd < threshold_dimensions_[0]) {
749  threshold_dimensions_[0] = min_gd;
750  has_grown_ = true;
751  }
752  if (max_gd > threshold_dimensions_[1]) {
753  Log::Info("UniformGridEnvironment",
754  "Your agents are getting near the edge of "
755  "the simulation space. Be aware of boundary conditions that "
756  "may come into play!");
757  threshold_dimensions_[1] = max_gd;
758  has_grown_ = true;
759  }
