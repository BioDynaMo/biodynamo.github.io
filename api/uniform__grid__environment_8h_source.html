203  NeighborIterator& ForwardToNonEmptyBox(uint64_t grid_timestamp) {
204  // increment box id until non empty box has been found
205  while (++box_idx_ < neighbor_boxes_.size()) {
206  // box is empty or uninitialized (padding box) -> continue
207  if (neighbor_boxes_[box_idx_]->IsEmpty(grid_timestamp)) {
208  continue;
209  }
210  // a non-empty box has been found
211  box_iterator_ = neighbor_boxes_[box_idx_]->begin();
212  return *this;
213  }
214  // all remaining boxes have been empty; reached end
215  is_end_ = true;
216  return *this;
217  }

275  void Update() override {
276  auto* rm = Simulation::GetActive()->GetResourceManager();
277 
278  if (rm->GetNumAgents() != 0) {
279  Clear();
280  timestamp_++;
281 
282  auto inf = Math::kInfinity;
283  std::array<double, 6> tmp_dim = {{inf, -inf, inf, -inf, inf, -inf}};
284  CalcSimDimensionsAndLargestAgent(&tmp_dim);
285  RoundOffGridDimensions(tmp_dim);
286 
287  // If the box_length_ is not set manually, we set it to the largest agent
288  // size
289  if (!is_custom_box_length_) {
290  auto los = ceil(GetLargestAgentSize());
291  assert(
292  los > 0 &&
293  "The largest object size was found to be 0. Please check if your "
294  "cells are correctly initialized.");
295  box_length_ = los;
296  }
297  box_length_squared_ = box_length_ * box_length_;
298 
299  for (int i = 0; i < 3; i++) {
300  int dimension_length =
301  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
302  int r = dimension_length % box_length_;
303  // If the grid is not perfectly divisible along each dimension by the
304  // resolution, extend the grid so that it is
305  if (r != 0) {
306  // std::abs for the case that box_length_ > dimension_length
307  grid_dimensions_[2 * i + 1] += (box_length_ - r);
308  } else {
309  // Else extend the grid dimension with one row, because the outmost
310  // object lies exactly on the border
311  grid_dimensions_[2 * i + 1] += box_length_;
312  }
313  }
314 
315  // Pad the grid to avoid out of bounds check when search neighbors
316  for (int i = 0; i < 3; i++) {
317  grid_dimensions_[2 * i] -= box_length_;
318  grid_dimensions_[2 * i + 1] += box_length_;
319  }
320 
321  // Calculate how many boxes fit along each dimension
322  for (int i = 0; i < 3; i++) {
323  int dimension_length =
324  grid_dimensions_[2 * i + 1] - grid_dimensions_[2 * i];
325  assert((dimension_length % box_length_ == 0) &&
326  "The grid dimensions are not a multiple of its box length");
327  num_boxes_axis_[i] = dimension_length / box_length_;
328  }
329 
330  num_boxes_xy_ = num_boxes_axis_[0] * num_boxes_axis_[1];
331  total_num_boxes_ = num_boxes_xy_ * num_boxes_axis_[2];
332 
333  CheckGridGrowth();
334 
335  // resize boxes_
336  if (boxes_.size() != total_num_boxes_) {
337  if (boxes_.capacity() < total_num_boxes_) {
338  boxes_.reserve(total_num_boxes_ * 2);
339  }
340  boxes_.resize(total_num_boxes_);
341  }
342 
343  successors_.reserve();
344 
345  // Assign agents to boxes
346  auto* param = Simulation::GetActive()->GetParam();
347  AssignToBoxesFunctor functor(this);
348  rm->ForEachAgentParallel(param->scheduling_batch_size, functor);
349  if (param->bound_space) {
350  int min = param->min_bound;
351  int max = param->max_bound;
352  threshold_dimensions_ = {min, max};
353  }
354 
355  if (param->thread_safety_mechanism ==
356  Param::ThreadSafetyMechanism::kAutomatic) {
357  nb_mutex_builder_->Update();
358  }
359  } else {
360  // There are no agents in this simulation
361  auto* param = Simulation::GetActive()->GetParam();
362 
363  bool uninitialized = boxes_.size() == 0;
364  if (uninitialized && param->bound_space) {
365  // Simulation has never had any agents
366  // Initialize grid dimensions with `Param::min_bound` and
367  // `Param::max_bound`
368  // This is required for the DiffusionGrid
369  int min = param->min_bound;
370  int max = param->max_bound;
371  grid_dimensions_ = {min, max, min, max, min, max};
372  threshold_dimensions_ = {min, max};
373  has_grown_ = true;
374  } else if (!uninitialized) {
375  // all agents have been removed in the last iteration
376  // grid state remains the same, but we have to set has_grown_ to false
377  // otherwise the DiffusionGrid will attempt to resize
378  has_grown_ = false;
379  } else {
380  Log::Fatal(
381  "UniformGridEnvironment",
382  "You tried to initialize an empty simulation without bound space. "
383  "Therefore we cannot determine the size of the simulation space. "
384  "Please add agents, or set Param::bound_space, "
385  "Param::min_bound, and Param::max_bound.");
386  }
387  }
388  }

398  inline double SquaredEuclideanDistance(const Double3& pos1,
399  const Double3& pos2) const {
400  const double dx = pos2[0] - pos1[0];
401  const double dy = pos2[1] - pos1[1];
402  const double dz = pos2[2] - pos1[2];
403  return (dx * dx + dy * dy + dz * dz);
404  }

406  inline bool WithinSquaredEuclideanDistance(double squared_radius,
407  const Double3& pos1,
408  const Double3& pos2) const {
409  const double dx = pos2[0] - pos1[0];
410  const double dx2 = dx * dx;
411  if (dx2 > squared_radius) {
412  return false;
413  }
414 
415  const double dy = pos2[1] - pos1[1];
416  const double dy2_plus_dx2 = dy * dy + dx2;
417  if (dy2_plus_dx2 > squared_radius) {
418  return false;
419  }
420 
421  const double dz = pos2[2] - pos1[2];
422  const double distance = dz * dz + dy2_plus_dx2;
423  return distance < squared_radius;
424  }

441  void ForEachNeighbor(Functor<void, Agent*, double>& lambda,
442  const Agent& query, double squared_radius) override {
443  if (squared_radius > box_length_squared_) {
444  Log::Fatal(
445  "UniformGridEnvironment::ForEachNeighbor",
446  "The requested search radius (", std::sqrt(squared_radius), ")",
447  " of the neighborhood search exceeds the "
448  "box length (",
449  box_length_, "). The resulting neighborhood would be incomplete.");
450  }
451 
452  const auto& position = query.GetPosition();
453  auto idx = query.GetBoxIdx();
454 
455  FixedSizeVector<const Box*, 27> neighbor_boxes;
456  GetMooreBoxes(&neighbor_boxes, idx);
457 
458  auto* rm = Simulation::GetActive()->GetResourceManager();
459 
460  NeighborIterator ni(neighbor_boxes, timestamp_);
461  const unsigned batch_size = 64;
462  uint64_t size = 0;
463  Agent* agents[batch_size] __attribute__((aligned(64)));
464  double x[batch_size] __attribute__((aligned(64)));
465  double y[batch_size] __attribute__((aligned(64)));
466  double z[batch_size] __attribute__((aligned(64)));
467  double squared_distance[batch_size] __attribute__((aligned(64)));
468 
469  auto process_batch = [&]() {
470 #pragma omp simd
471  for (uint64_t i = 0; i < size; ++i) {
472  const double dx = x[i] - position[0];
473  const double dy = y[i] - position[1];
474  const double dz = z[i] - position[2];
475 
476  squared_distance[i] = dx * dx + dy * dy + dz * dz;
477  }
478 
479  for (uint64_t i = 0; i < size; ++i) {
480  if (squared_distance[i] < squared_radius) {
481  lambda(agents[i], squared_distance[i]);
482  }
483  }
484  size = 0;
485  };
486 
487  while (!ni.IsAtEnd()) {
488  auto ah = *ni;
489  // increment iterator already here to hide memory latency
490  ++ni;
491  auto* agent = rm->GetAgent(ah);
492  if (agent != &query) {
493  agents[size] = agent;
494  const auto& pos = agent->GetPosition();
495  x[size] = pos[0];
496  y[size] = pos[1];
497  z[size] = pos[2];
498  size++;
499  if (size == batch_size) {
500  process_batch();
501  }
502  }
503  }
504  process_batch();
505  }

703  void CheckGridGrowth() {
704  // Determine if the grid dimensions have changed (changed in the sense that
705  // the grid has grown outwards)
706  auto min_gd =
707  *std::min_element(grid_dimensions_.begin(), grid_dimensions_.end());
708  auto max_gd =
709  *std::max_element(grid_dimensions_.begin(), grid_dimensions_.end());
710  if (min_gd < threshold_dimensions_[0]) {
711  threshold_dimensions_[0] = min_gd;
712  has_grown_ = true;
713  }
714  if (max_gd > threshold_dimensions_[1]) {
715  Log::Info("UniformGridEnvironment",
716  "Your agents are getting near the edge of "
717  "the simulation space. Be aware of boundary conditions that "
718  "may come into play!");
719  threshold_dimensions_[1] = max_gd;
720  has_grown_ = true;
721  }
722  }

740  void GetMooreBoxes(FixedSizeVector<const Box*, 27>* neighbor_boxes,
741  size_t box_idx) const {
742  neighbor_boxes->push_back(GetBoxPointer(box_idx));
743 
744  // Adjacent 6 (top, down, left, right, front and back)
745  if (adjacency_ >= kLow) {
746  neighbor_boxes->push_back(GetBoxPointer(box_idx - num_boxes_xy_));
747  neighbor_boxes->push_back(GetBoxPointer(box_idx + num_boxes_xy_));
748  neighbor_boxes->push_back(GetBoxPointer(box_idx - num_boxes_axis_[0]));
749  neighbor_boxes->push_back(GetBoxPointer(box_idx + num_boxes_axis_[0]));
750  neighbor_boxes->push_back(GetBoxPointer(box_idx - 1));
751  neighbor_boxes->push_back(GetBoxPointer(box_idx + 1));
752  }
753 
754  // Adjacent 12
755  if (adjacency_ >= kMedium) {
756  neighbor_boxes->push_back(
757  GetBoxPointer(box_idx - num_boxes_xy_ - num_boxes_axis_[0]));
758  neighbor_boxes->push_back(GetBoxPointer(box_idx - num_boxes_xy_ - 1));
759  neighbor_boxes->push_back(
760  GetBoxPointer(box_idx - num_boxes_axis_[0] - 1));
761  neighbor_boxes->push_back(
762  GetBoxPointer(box_idx + num_boxes_xy_ - num_boxes_axis_[0]));
