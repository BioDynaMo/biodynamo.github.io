47 void NeuronSoma::Update(const NewAgentEvent& event) {
48  Base::Update(event);
49 
50  if (event.GetUid() == NewNeuriteExtensionEvent::kUid) {
51  const auto& ne_event = static_cast<const NewNeuriteExtensionEvent&>(event);
52  auto* neurite = bdm_static_cast<NeuriteElement*>(event.new_agents[0]);
53 
54  double theta = ne_event.theta;
55  double phi = ne_event.phi;
56  double x_coord = std::sin(theta) * std::cos(phi);
57  double y_coord = std::sin(theta) * std::sin(phi);
58  double z_coord = std::cos(theta);
59 
60  daughters_.push_back(neurite->GetAgentPtr<NeuriteElement>());
61  daughters_coord_[neurite->GetUid()] = {x_coord, y_coord, z_coord};
62  }
63 
64  // do nothing for CellDivisionEvent or others
65 }

95 Double3 NeuronSoma::OriginOf(const AgentUid& daughter_uid) const {
96  Double3 xyz = daughters_coord_.at(daughter_uid);
97 
98  double radius = GetDiameter() * .5;
99  xyz = xyz * radius;
100 
101  Double3 axis_0 = {Base::kXAxis[0], Base::kYAxis[0], Base::kZAxis[0]};
102  Double3 axis_1 = {Base::kXAxis[1], Base::kYAxis[1], Base::kZAxis[1]};
103  Double3 axis_2 = {Base::kXAxis[2], Base::kYAxis[2], Base::kZAxis[2]};
104 
105  Double3 result = {xyz * axis_0, xyz * axis_1, xyz * axis_2};
106  return GetPosition() + result;
107 }

111 void NeuronSoma::UpdateRelative(const NeuronOrNeurite& old_rel,
112  const NeuronOrNeurite& new_rel) {
113  auto old_rel_agent_ptr = bdm_static_cast<const NeuriteElement*>(&old_rel)
114  ->GetAgentPtr<NeuriteElement>();
115  auto new_rel_agent_ptr = bdm_static_cast<const NeuriteElement*>(&new_rel)
116  ->GetAgentPtr<NeuriteElement>();
117  auto coord = daughters_coord_[old_rel_agent_ptr->GetUid()];
118  auto it = std::find(std::begin(daughters_), std::end(daughters_),
119  old_rel_agent_ptr);
120  assert(it != std::end(daughters_) &&
121  "old_element_idx could not be found in daughters_ vector");
122  *it = new_rel_agent_ptr;
123  daughters_coord_[new_rel_agent_ptr->GetUid()] = coord;
124 }
