49 void NeuronSoma::Update(const NewAgentEvent& event) {
50  Base::Update(event);
51 
52  if (event.GetUid() == NewNeuriteExtensionEvent::kUid) {
53  const auto& ne_event = static_cast<const NewNeuriteExtensionEvent&>(event);
54  auto* neurite = bdm_static_cast<NeuriteElement*>(event.new_agents[0]);
55 
56  double theta = ne_event.theta;
57  double phi = ne_event.phi;
58  double x_coord = std::sin(theta) * std::cos(phi);
59  double y_coord = std::sin(theta) * std::sin(phi);
60  double z_coord = std::cos(theta);
61 
62  daughters_.push_back(neurite->GetAgentPtr<NeuriteElement>());
63  daughters_coord_[neurite->GetUid()] = {x_coord, y_coord, z_coord};
64  }
65 
66  // do nothing for CellDivisionEvent or others
67 }

105 Double3 NeuronSoma::OriginOf(const AgentUid& daughter_uid) const {
106  Double3 xyz = daughters_coord_.at(daughter_uid);
107 
108  double radius = GetDiameter() * .5;
109  xyz = xyz * radius;
110 
111  Double3 axis_0 = {Base::kXAxis[0], Base::kYAxis[0], Base::kZAxis[0]};
112  Double3 axis_1 = {Base::kXAxis[1], Base::kYAxis[1], Base::kZAxis[1]};
113  Double3 axis_2 = {Base::kXAxis[2], Base::kYAxis[2], Base::kZAxis[2]};
114 
115  Double3 result = {xyz * axis_0, xyz * axis_1, xyz * axis_2};
116  return GetPosition() + result;
117 }

147 void NeuronSoma::PrintSWC(std::ostream& out) const {
148  // 1. Write a header to the file such that users do not confuse it with real
149  // data.
150  out << "# This SWC file was created with BioDynaMo and is (very) likely "
151  "to\n# be the result of a simulation. Be careful not to confuse it\n"
152  "# with real data. \n\n";
153  auto datetime_raw = std::chrono::system_clock::now();
154  auto datetime = std::chrono::system_clock::to_time_t(datetime_raw);
155  out << "# Created at: " << std::ctime(&datetime) << "\n\n";
156 
157  // 2. Write the line for the soma at the beginning of the file
158  int element_id{1};
159  int previous_element_id{-1};
160  out << element_id++ << " " << GetIdentifierSWC();
161  PrintVector(out, GetPosition());
162  out << GetDiameter() / 2 << " " << previous_element_id << "\n";
163 
164  // 3. Iterate over all neuron branches that are attached to the soma
165  for (auto daughter : daughters_) {
166  // New branches need to be connected to soma
167  bool is_new_brach{true};
168  // Pointers to currently considered and next element
169  AgentPointer<NeuriteElement> current_element, next_element;
170  // Vector to track the bifurcation elements (right daughters that are
171  // neglected when going down the left path, see below)
172  std::vector<AgentPointer<NeuriteElement>> bifurcation_points;
173  // Track the labels that we need to connect the bifurcation points to the
174  // appropriate elements
175  std::vector<int> bifurcation_points_lables;
176  // Used to reconnect to previous bifurcation
177  bool connect_to_bifurcation{false};
178 
179  // The logic of the while loop is the following. The tree of dendrites /
180  // axons is made up of subsequent elements. Each element has a left daughter
181  // but only bifurcations points also have a right daughter. If a branch
182  // ends, the element has neither a left nor a right daughter. Thus, for each
183  // branch attached to the soma, we walk down the tree all the way down
184  // following the left path. On the way, we remember the bifurcation points.
185  // Once we've reached the end of a branch, we jump back to the last
186  // bifurcation point that we have visited and start to follow the left path
187  // again. With this method we can resolve the entire tree structure under
188  // the assumption, that each element has 0, 1, or 2 follow up elements and
189  // that different branches do not merge again.
190  current_element = daughter;
191  while (true) {
192  // Loop 1. Write down the SWC line corresponding to the current element.
193  // The file format per line is as follows:
194  // <element id> <type id> <x> <y> <z> <radius> <prev element id>
195  out << element_id << " " << current_element->GetIdentifierSWC();
196  PrintVector(out, current_element->GetPosition());
197  out << current_element->GetDiameter() / 2 << " ";
198  // This block establishes the connection via <prev element id>
199  if (is_new_brach) {
200  // New branches need to be attached to the soma
201  previous_element_id = 1;
202  is_new_brach = false;
203  } else {
204  if (!connect_to_bifurcation) {
205  // By default we connect to the previously considered element
206  previous_element_id = element_id - 1;
207  } else {
208  // At some points we need to reconnect the branches to the previous
209  // bifurcation point.
210  previous_element_id = bifurcation_points_lables.back();
211  bifurcation_points_lables.pop_back();
212  connect_to_bifurcation = false;
213  }
214  }
215  out << previous_element_id << "\n";
216 
217  // Loop 2. Walk down one side of the tree, take left option first, and
218  // remember bifurcation points.
219  next_element = current_element->GetDaughterLeft();
220  // Track the bifurcations
221  if (current_element->GetDaughterRight() != nullptr) {
222  bifurcation_points.push_back(current_element->GetDaughterRight());
223  bifurcation_points_lables.push_back(element_id);
224  }
225  element_id++; // Increase ID by 1
226  if (next_element == nullptr && bifurcation_points.empty()) {
227  // If we have resolved all bifurcations and don't have any next element,
228  // we exit the loop
229  break;
230  } else if (next_element == nullptr) {
231  // Jump to previous bifurcation
232  next_element = bifurcation_points.back();
233  bifurcation_points.pop_back();
234  connect_to_bifurcation = true;
235  }
236  std::swap(current_element, next_element); // Swap Pointers
237  }
238  }
239 }
