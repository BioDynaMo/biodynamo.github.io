34 NeuronSoma::NeuronSoma(const Event& event, SimObject* mother_so,

50 void NeuronSoma::EventHandler(const Event& event, SimObject* other1,
51  SimObject* other2) {
52  Base::EventHandler(event, other1, other2);
53 
54  const NewNeuriteExtensionEvent* ne_event =
55  dynamic_cast<const NewNeuriteExtensionEvent*>(&event);
56  NeuriteElement* neurite = dynamic_cast<NeuriteElement*>(other1);
57  if (ne_event && neurite) {
58  double theta = ne_event->theta_;
59  double phi = ne_event->phi_;
60  double x_coord = std::sin(theta) * std::cos(phi);
61  double y_coord = std::sin(theta) * std::sin(phi);
62  double z_coord = std::cos(theta);
63 
64  daughters_.push_back(neurite->GetSoPtr<NeuriteElement>());
65  daughters_coord_[neurite->GetUid()] = {x_coord, y_coord, z_coord};
66  }
67 
68  // do nothing for CellDivisionEvent or others
69 }

80 NeuriteElement* NeuronSoma::ExtendNewNeurite(double diameter, double phi,
81  double theta,
82  NeuriteElement* prototype) {
83  auto* ctxt = Simulation::GetActive()->GetExecutionContext();
84  NewNeuriteExtensionEvent event(diameter, phi, theta);
85  SimObject* neurite = nullptr;
86  if (!prototype) {
87  NeuriteElement ne_tmp;
88  neurite = ne_tmp.GetInstance(event, this);
89  } else {
90  neurite = prototype->GetInstance(event, this);
91  }
92  ctxt->push_back(neurite);
93  EventHandler(event, neurite);
94  return bdm_static_cast<NeuriteElement*>(neurite);

97 void NeuronSoma::RemoveDaughter(const SoPointer<NeuriteElement>& daughter) {

104 Double3 NeuronSoma::OriginOf(const SoUid& daughter_uid) const {
105  Double3 xyz = daughters_coord_.at(daughter_uid);
106 
107  double radius = Base::diameter_ * .5;
108  xyz = xyz * radius;
109 
110  Double3 axis_0 = {Base::kXAxis[0], Base::kYAxis[0], Base::kZAxis[0]};
111  Double3 axis_1 = {Base::kXAxis[1], Base::kYAxis[1], Base::kZAxis[1]};
112  Double3 axis_2 = {Base::kXAxis[2], Base::kYAxis[2], Base::kZAxis[2]};
113 
114  Double3 result = {xyz * axis_0, xyz * axis_1, xyz * axis_2};
115  return Base::position_ + result;
116 }

120 void NeuronSoma::UpdateRelative(const NeuronOrNeurite& old_rel,
121  const NeuronOrNeurite& new_rel) {

128  std::find(std::begin(daughters_), std::end(daughters_), old_rel_soptr);
129  assert(it != std::end(daughters_) &&
130  "old_element_idx could not be found in daughters_ vector");
131  *it = new_rel_soptr;
132  daughters_coord_[new_rel_soptr->GetUid()] = coord;
