203 // The following custom streamer should be visible to rootcling for dictionary

205 #if (!defined(__CLING__) || defined(__ROOTCLING__)) && defined(USE_DICT)
206 
207 // The custom streamer is needed because ROOT can't stream function pointers
208 // by default.
209 inline void TimeSeries::Data::Streamer(TBuffer& R__b) {
210  if (R__b.IsReading()) {
211  R__b.ReadClassBuffer(TimeSeries::Data::Class(), this);
212  Long64_t l;
213  R__b.ReadLong64(l);
214  this->ycollector = reinterpret_cast<real_t (*)(Simulation*)>(l);
215  R__b.ReadLong64(l);
216  this->xcollector = reinterpret_cast<real_t (*)(Simulation*)>(l);
217  } else {
218  R__b.WriteClassBuffer(TimeSeries::Data::Class(), this);
219  Long64_t l = reinterpret_cast<Long64_t>(this->ycollector);
220  R__b.WriteLong64(l);
221  l = reinterpret_cast<Long64_t>(this->xcollector);
222  R__b.WriteLong64(l);
223  }
