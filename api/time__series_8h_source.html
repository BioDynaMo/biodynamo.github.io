166 // The custom streamer is needed because ROOT can't stream function pointers

168 inline void TimeSeries::Data::Streamer(TBuffer& R__b) {
169  if (R__b.IsReading()) {
170  R__b.ReadClassBuffer(TimeSeries::Data::Class(), this);

176  } else {
177  R__b.WriteClassBuffer(TimeSeries::Data::Class(), this);
178  Long64_t l = reinterpret_cast<Long64_t>(this->ycollector);
