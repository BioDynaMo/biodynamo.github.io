198 // The custom streamer is needed because ROOT can't stream function pointers
199 // by default.
200 inline void TimeSeries::Data::Streamer(TBuffer& R__b) {
201  if (R__b.IsReading()) {
202  R__b.ReadClassBuffer(TimeSeries::Data::Class(), this);
203  Long64_t l;
204  R__b.ReadLong64(l);
205  this->ycollector = reinterpret_cast<double (*)(Simulation*)>(l);
206  R__b.ReadLong64(l);
207  this->xcollector = reinterpret_cast<double (*)(Simulation*)>(l);
208  } else {
209  R__b.WriteClassBuffer(TimeSeries::Data::Class(), this);
210  Long64_t l = reinterpret_cast<Long64_t>(this->ycollector);
211  R__b.WriteLong64(l);
212  l = reinterpret_cast<Long64_t>(this->xcollector);
213  R__b.WriteLong64(l);
