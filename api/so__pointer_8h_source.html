152 template <typename TSoSimBackend, typename TBackend>
153 struct ReadContainerFunctor {
155  template <typename TContainer, typename TTBackend = TBackend,
156  typename TSimulation = Simulation<>>
157  typename std::enable_if<std::is_same<
158  TTBackend, typename TSimulation::ResourceManager_t::Backend>::value>::type
159  operator()(TBuffer& R__b, TContainer** container, uint64_t) { // NOLINT
160  int state;
161  R__b >> state;
162  if (state == ContainerPointerState::kPointIntoRm) {
163  R__b >> *container;
164  // if a whole simulation is restored from a ROOT file, `TSimulation::Get`
165  // is
166  // not updated to the new `ResourceManager` yet. Therefore, we must delay
167  // this call. It will be executed after the restore operation has been
168  // completed.
169  SimulationBackup::after_restore_event_.push_back([=]() {
170  auto* rm = TSimulation::GetActive()->GetResourceManager();
171  *container = rm->template Get<TSoSimBackend>();
172  });
173  } else if (state == ContainerPointerState::kSeparate) {
174  R__b >> *container;
175  } else {
176  R__b >> *container;
177  }
178  }
179 
182  template <typename TContainer>
183  void operator()(TBuffer& R__b, TContainer** container, ...) { // NOLINT
184  int state;
185  R__b >> state;
186  if (state == ContainerPointerState::kSeparate) {
187  R__b >> *container;
188  } else {
189  // Read nullptr
190  R__b >> *container;
191  }
192  }
193 };

199 template <typename TSoSimBackend, typename TBackend>
200 struct WriteContainerFunctor {
202  template <typename TContainer, typename TTBackend = TBackend,
203  typename TSimulation = Simulation<>>
204  typename std::enable_if<std::is_same<
205  TTBackend, typename TSimulation::ResourceManager_t::Backend>::value>::type
206  operator()(TBuffer& R__b, const TContainer* container, uint64_t) { // NOLINT
207  auto* rm = TSimulation::GetActive()->GetResourceManager();
208  if (container == nullptr) {
209  // write nullptr
210  R__b << ContainerPointerState::kNullPtr;
211  R__b << container;
212  } else if (rm->template Get<TSoSimBackend>() == container) {
213  R__b << ContainerPointerState::kPointIntoRm;
214  // skip container inside ResourceManager and write nullptr instead
215  // ROOT does not recognise that the container points inside the
216  // ResourcManager and would duplicate the container.
217  TContainer* n = nullptr;
218  R__b << n;
219  } else {
220  // write separate container
221  R__b << ContainerPointerState::kSeparate;
222  R__b << container;
223  }
224  }
225 
228  template <typename TContainer>
229  void operator()(TBuffer& R__b, const TContainer* container, ...) { // NOLINT
230  if (container != nullptr) {
231  R__b << ContainerPointerState::kSeparate;
232  R__b << container;
233  } else {
234  // write nullptr
235  R__b << ContainerPointerState::kNullPtr;
236  R__b << container;
237  }
238  }
