72 void StencilGrid::DiffuseWithOpenEdge() {

77 #define YBF 16
78 #pragma omp parallel for collapse(2)
79  for (int yy = 0; yy < ny; yy += YBF) {
80  for (int z = 0; z < nz; z++) {
81  // To let the edges bleed we set some diffusion coefficients
82  // to zero. This prevents substance building up at the edges
83  auto dc_2_ = dc_;
84  int ymax = yy + YBF;
85  if (ymax >= ny) {
86  ymax = ny;
87  }
88  for (int y = yy; y < ymax; y++) {
89  dc_2_ = dc_;

91  int c, n, s, b, t;
92  x = 0;
93  c = x + y * nx + z * nx * ny;
94  if (y == 0) {
95  n = c;
96  dc_2_[4] = 0;
97  } else {
98  n = c - nx;
99  }
100  if (y == (ny - 1)) {
101  s = c;
102  dc_2_[3] = 0;
103  } else {
104  s = c + nx;
105  }
106  if (z == 0) {
107  b = c;
108  dc_2_[5] = 0;

111  }

113  t = c;
114  dc_2_[6] = 0;
115  } else {
116  t = c + nx * ny;
117  }
118  // x = 0; we leak out substances past this edge (so multiply by 0)
