69  template <typename TSimulation = Simulation<>>
70  void TearDownIteration() {
71  // new sim objects
72  auto* rm = TSimulation::GetActive()->GetResourceManager();
73  new_sim_objects_.ApplyOnAllElements(
74  [&](auto&& sim_object, SoHandle) { rm->push_back(sim_object); });
75  new_sim_objects_.Clear();
76 
77  // removed sim objects
78  // remove them after adding new ones (maybe one has been removed
79  // that was in new_sim_objects_)
80  for (auto& uid : remove_) {
81  rm->Remove(uid);
82  }
83  remove_.clear();
84  }

91  auto nb_mutex_builder = grid->GetNeighborMutexBuilder();
92  if (nb_mutex_builder != nullptr) {

94  std::lock_guard<decltype(mutex)> guard(mutex);
