24 void ModelInitializer::DefineSubstance(
25  int substance_id, const std::string& substance_name, real_t diffusion_coeff,
26  real_t decay_constant, int resolution,
27  const std::vector<real_t>& binding_coefficients,
28  const std::vector<int>& binding_substances) {
29  auto* sim = Simulation::GetActive();
30  auto* param = sim->GetParam();
31  auto* rm = sim->GetResourceManager();
32  DiffusionGrid* dgrid = nullptr;
33  if (decay_constant < 0.) {
34  Log::Fatal("ModelInitializer::DefineSubstance",
35  "The decay constant must be > 0 (", decay_constant, ").");
36  }
37  if (param->diffusion_method == "euler") {
38  if (!binding_substances.empty()) {
39  if (binding_coefficients.size() != binding_substances.size()) {
40  Log::Fatal("ModelInitializer::DefineSubstance",
41  "The binding coefficients and substances have different",
42  " sizes (", binding_coefficients.size(), " vs ",
43  binding_substances.size(), ").");
44  }
45  for (auto binding_coeff : binding_coefficients) {
46  if (binding_coeff < 0.) {
47  Log::Fatal("ModelInitializer::DefineSubstance",
48  "The binding coefficients must be > 0 (", binding_coeff,
49  ").");
50  }
51  }
52 
53  dgrid = new EulerDepletionGrid(
54  substance_id, substance_name, diffusion_coeff, decay_constant,
55  resolution, binding_coefficients, binding_substances);
56  } else {
57  dgrid = new EulerGrid(substance_id, substance_name, diffusion_coeff,
58  decay_constant, resolution);
59  }
60  } else if (param->diffusion_method == "runge-kutta") {
61  if (decay_constant != 0) {
62  Log::Warning(
63  "ModelInitializer::DefineSubstance",
64  "RungeKuttaGrid does not support a decay constant. Using 0.");
65  }
66  if (!binding_substances.empty()) {
67  Log::Warning("ModelInitializer::DefineSubstance",
68  "RungeKuttaGrid does not support depletion. Depleting "
69  "substances removed.");
70  }
71  dgrid = new RungeKuttaGrid(substance_id, substance_name, diffusion_coeff,
72  resolution);
73  } else {
74  Log::Error("ModelInitializer::DefineSubstance", "Diffusion method '",
75  param->diffusion_method,
76  "' does not exist. Defaulting to 'euler'");
77  dgrid = new EulerGrid(substance_id, substance_name, diffusion_coeff,
78  decay_constant, resolution);
79  }
80 
81  rm->AddContinuum(dgrid);
82 }
