26 inline void ApplyBoundingBox(Agent* agent, double lb, double rb) {
27  // Need to create a small distance from the positive edge of each dimension;
28  // otherwise it will fall out of the boundary of the simulation space
29  double eps = 1e-10;
30  auto pos = agent->GetPosition();
31  bool updated = false;
32  for (int i = 0; i < 3; i++) {
33  if (pos[i] < lb) {
34  pos[i] = lb;
35  updated = true;
36  } else if (pos[i] >= rb) {
37  pos[i] = rb - eps;
38  updated = true;
39  }
40  }
41  if (updated) {
42  agent->SetPosition(pos);
43  }
44 }
