78 double TimeSeries::ComputeError(const TimeSeries& ts1, const TimeSeries& ts2) {
79  // verify that all TimeSeries contain the same entries
80  if (ts1.data_.size() != ts2.data_.size()) {
81  Log::Warning("TimeSeries::ComputeError",
82  "The time series objects that should be merged do not "
83  "contain the same entries. Operation aborted.");
84  return std::numeric_limits<double>::infinity();
85  }
86  for (auto& p : ts1.data_) {
87  if (ts2.data_.find(p.first) == ts2.data_.end()) {
88  Log::Warning("TimeSeries::ComputeError",
89  "The time series objects that should be merged do not "
90  "contain the same entries. Operation aborted");
91  return std::numeric_limits<double>::infinity();
92  }
93  }
94 
95  for (auto& pair : ts1.data_) {
96  auto& key = pair.first;
97  // check that all time series objects have the same x values
98  auto& xref = ts1.data_.at(key).x_values;
99  auto& xcurrent = ts2.data_.at(key).x_values;
100  if (xref.size() != xcurrent.size()) {
101  Log::Warning("TimeSeries::ComputeError",
102  "The time series objects for entry (", key,
103  ") have different x_values. Operation aborted.");
104  return std::numeric_limits<double>::infinity();
105  } else {
106  for (uint64_t j = 0; j < xref.size(); ++j) {
107  if (std::abs(xref[j] - xcurrent[j]) > 1e-5) {
108  Log::Warning("TimeSeries::ComputeError",
109  "The time series objects for entry (", key,
110  ") have different x_values. Operation aborted.");
111  return std::numeric_limits<double>::infinity();
112  }
113  }
114  }
115  }
116 
117  double error = 0.0;
118  for (auto& pair : ts1.data_) {
119  auto& key = pair.first;
120  auto& yref = ts1.data_.at(key).y_values;
121  auto& ycurrent = ts2.data_.at(key).y_values;
122  error += Math::MSE(yref, ycurrent);
123  }
124  return error;
125 }

133 void TimeSeries::Merge(
134  TimeSeries* merged, const std::vector<TimeSeries>& time_series,
135  const std::function<void(const std::vector<double>&, double*, double*,
136  double*)>& merger) {
137  if (!merged) {
138  Log::Warning("TimeSeries::Merge",
139  "Parameter 'merged' is a nullptr. Operation aborted.");
140  return;
141  }
142 
143  // check that merged is empty
144  if (merged->data_.size() != 0) {
145  Log::Warning("TimeSeries::Merge",
146  "Parameter 'merged' is not empty. Operation aborted.");
147  return;
148  }
149 
150  if (time_series.size() == 0) {
151  Log::Warning("TimeSeries::Merge",
152  "The given time series vector is empty. Operation aborted.");
153  return;
154  } else if (time_series.size() == 1) {
155  *merged = time_series[0];
156  return;
157  }
158 
159  // verify that all TimeSeries contain the same entries
160  auto& ref = time_series[0];
161  for (uint64_t i = 1; i < time_series.size(); ++i) {
162  auto& current = time_series[i];
163  if (ref.data_.size() != current.data_.size()) {
164  Log::Warning("TimeSeries::Merge",
165  "The time series objects that should be merged do not "
166  "contain the same entries. Operation aborted.");
167  return;
168  }
169  for (auto& p : ref.data_) {
170  if (current.data_.find(p.first) == current.data_.end()) {
171  Log::Warning("TimeSeries::Merge",
172  "The time series objects that should be merged do not "
173  "contain the same entries. Operation aborted");
174  return;
175  }
176  }
177  }
178 
179  for (auto& pair : ref.data_) {
180  auto& key = pair.first;
181  // check that all time series objects have the same x values
182  auto& xref = ref.data_.at(key).x_values;
183  for (uint64_t i = 1; i < time_series.size(); ++i) {
184  auto& xcurrent = time_series[i].data_.at(key).x_values;
185  if (xref.size() != xcurrent.size()) {
186  Log::Warning("TimeSeries::Merge", "The time series objects for entry (",
187  key, ") have different x_values. Operation aborted.");
188  return;
189  } else {
190  for (uint64_t j = 0; j < xref.size(); ++j) {
191  if (std::abs(xref[j] - xcurrent[j]) > 1e-5) {
192  Log::Warning("TimeSeries::Merge",
193  "The time series objects for entry (", key,
194  ") have different x_values. Operation aborted.");
195  return;
196  }
197  }
198  }
199  }
200 
201  // merge
202  // initialize
203  merged->data_.emplace(key, Data());
204  auto& mdata = merged->data_[key];
205  mdata.x_values.resize(xref.size());
206  mdata.y_values.resize(xref.size());
207  mdata.y_error_low.resize(xref.size());
208  mdata.y_error_high.resize(xref.size());
209 
210 #pragma omp parallel for
211  for (uint64_t i = 0; i < xref.size(); ++i) {
212  mdata.x_values[i] = xref[i];
213  std::vector<double> all_y_values(time_series.size());
214  for (uint64_t j = 0; j < time_series.size(); ++j) {
215  all_y_values[j] = time_series[j].data_.at(key).y_values[i];
216  merger(all_y_values, &mdata.y_values[i], &mdata.y_error_low[i],
217  &mdata.y_error_high[i]);
218  }
219  }
220  }
221 }

270 void TimeSeries::Update() {
271  if (!data_.empty()) {
272  auto* sim = Simulation::GetActive();
273  auto* scheduler = sim->GetScheduler();
274  auto* param = sim->GetParam();
275  std::vector<std::pair<Reducer<double>*, const std::string>> reducers;
276  for (auto& entry : data_) {
277  auto& result_data = entry.second;
278  if (result_data.ycollector != nullptr) {
279  result_data.y_values.push_back(result_data.ycollector(sim));
280  }
281  if (result_data.y_reducer_collector != nullptr) {
282  result_data.y_reducer_collector->Reset();
283  reducers.push_back(
284  std::make_pair(result_data.y_reducer_collector, entry.first));
285  }
286  if (result_data.xcollector == nullptr) {
287  result_data.x_values.push_back(scheduler->GetSimulatedSteps() *
288  param->simulation_time_step);
289  } else {
290  result_data.x_values.push_back(result_data.xcollector(sim));
291  }
292  }
293 
294  // execute reducers
295  auto execute_reducers = L2F([&](Agent* agent) {
296  for (auto& el : reducers) {
297  (*el.first)(agent);
298  }
299  });
300  sim->GetResourceManager()->ForEachAgentParallel(execute_reducers);
301  for (auto& el : reducers) {
302  data_[el.second].y_values.push_back(el.first->GetResult());
303  }
304  }
305 }

308 void TimeSeries::Add(const TimeSeries& ts, const std::string& suffix) {
309  if (this == &ts) {
310  Log::Warning(
311  "TimeSeries::Add",
312  "Adding a TimeSeries to itself is not supported. Operation aborted.");
313  return;
314  }
315  for (auto& p : ts.data_) {
316  // verify that entry doesn't exist yet
317  auto id = Concat(p.first, "-", suffix);
318  if (data_.find(id) != data_.end()) {
319  Log::Warning("TimeSeries::Add", "TimeSeries with id (", id,
320  ") exists already. Consider changing the suffix parameter "
321  "to make it unique. Operation aborted.");
322  return;
323  }
324 
325  data_.emplace(id, p.second);
326  }
327 }
