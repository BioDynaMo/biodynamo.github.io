135 template <typename TEvent, typename TBiologyModules1,
136  typename... TBiologyModules>
137 void BiologyModuleEventHandler(const TEvent& event, TBiologyModules1* current,
138  TBiologyModules*... bms) {
139  // call event handler for biology modules
140  uint64_t cnt = 0;
141  auto call_bm_event_handler = [&](auto& bm) {
142  using BiologyModuleType = raw_type<decltype(bm)>;
143 
145  auto extract = [](bool condition, auto* variant) -> BiologyModuleType* {
146  if (condition) {
147  return get_if<BiologyModuleType>(variant);
148  }
149  return nullptr;
150  };
151 
152  if (!bm.Remove(event.kEventId)) {
153  bool copy = bm.Copy(event.kEventId);
154  bm.EventHandler(event, extract(copy, &((*bms)[cnt]))...);
155  cnt += copy ? 1 : 0;
156  }
157  };
158  for (auto& el : *current) {
159  visit(call_bm_event_handler, el);
160  }
161 
162  // remove biology modules from current
163  bool remove;
164  auto remove_from_current = [&](auto& bm) {
165  remove = bm.Remove(event.kEventId);
166  };
167  for (auto it = current->begin(); it != current->end();) {
168  remove = false;
169  visit(remove_from_current, *it);
170  if (remove) {
171  it = current->erase(it);
172  } else {
173  ++it;
174  }
175  }
176 }
