59 json FlattenModules(const json& j_document) {
60  json j_copy = j_document;
61  j_copy.erase("modules_");
62 
63  json j_new;
64  j_new["bdm::Param"] = j_copy;
65 
66  // iterator over all module parameters
67  auto j_modules = j_document["modules_"];
68  for (json::iterator it = j_modules.begin(); it != j_modules.end(); ++it) {
69  j_new[(*it)["second"]["_typename"].get<std::string>()] = (*it)["second"];
70  }
71  return j_new;
72 }

75 json UnflattenModules(const json& j_flattened, const json& j_original) {
76  json j_return = j_flattened["bdm::Param"];
77  j_return["modules_"] = {};
78  auto& j_modules = j_return["modules_"];
79 
80  auto j_original_modules = j_original["modules_"];
81  for (json::iterator it = j_original_modules.begin();
82  it != j_original_modules.end(); ++it) {
83  json j_module_param;
84  j_module_param["$pair"] = (*it)["$pair"];
85  j_module_param["first"] = (*it)["first"];
86  j_module_param["second"] =
87  j_flattened[(*it)["second"]["_typename"].get<std::string>()];
88  j_modules.push_back(j_module_param);
89  }
90  return j_return;
91 }

103 // -----------------------------------------------------------------------------
104 void Param::MergeJsonPatch(const std::string& patch) {
105  std::string json_str(
106  TBufferJSON::ToJSON(this, TBufferJSON::kMapAsObject).Data());
107  json j_param = json::parse(json_str);
108  auto j_flattened = FlattenModules(j_param);
109 
110  try {
111  auto j_patch = json::parse(patch);
112  j_flattened.merge_patch(j_patch);
113  } catch (std::exception& e) {
114  Log::Fatal("Param::RestoreFromJson",
115  Concat("Couldn't parse or merge the given json parameters.\n",
116  e.what(), "\n", patch));
117  }
118 
119  auto j_unflattened = UnflattenModules(j_flattened, j_param);
120  Param* restored = nullptr;
121  TBufferJSON::FromJSON(restored, j_unflattened.dump().c_str());
122  Restore(std::move(*restored));
123 }

126 void AssignThreadSafetyMechanism(const std::shared_ptr<cpptoml::table>& config,
127  Param* param) {
128  const std::string config_key = "simulation.thread_safety_mechanism";
129  if (config->contains_qualified(config_key)) {
130  auto value = config->get_qualified_as<std::string>(config_key);
131  if (!value) {
132  return;
133  }
134  auto str_value = *value;
135  if (str_value == "none") {
136  param->thread_safety_mechanism_ = Param::ThreadSafetyMechanism::kNone;
137  } else if (str_value == "user-specified") {
138  param->thread_safety_mechanism_ =
139  Param::ThreadSafetyMechanism::kUserSpecified;
140  } else if (str_value == "automatic") {
141  param->thread_safety_mechanism_ =
142  Param::ThreadSafetyMechanism::kAutomatic;
143  }
144  }
145 }
