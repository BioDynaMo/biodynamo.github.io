177  template <typename TNeurite>
178  void EventHandler(const NewNeuriteExtensionEvent& event, TNeurite* neurite) {
179  // call event handler for biology modules
180  auto* neurite_bms = &(neurite->biology_modules_[neurite->kIdx]);
181  BiologyModuleEventHandler(event, &(Base::biology_modules_[kIdx]),
182  neurite_bms);
183 
184  double theta = event.theta_;
185  double phi = event.phi_;
186  double x_coord = std::sin(theta) * std::cos(phi);
187  double y_coord = std::sin(theta) * std::sin(phi);
188  double z_coord = std::cos(theta);
189 
190  daughters_[kIdx].push_back(neurite->GetSoPtr());
191  daughters_coord_[kIdx][neurite->GetElementIdx()] = {x_coord, y_coord,
192  z_coord};
