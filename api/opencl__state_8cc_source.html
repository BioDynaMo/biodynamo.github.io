211 int OpenCLState::ClAssert(int const code, char const* const file,
212  int const line, bool const abort) {
213  if (code != CL_SUCCESS) {

217  err_str);
218 
219  if (abort) {
220  // stop profiling and reset device here if necessary
221  exit(code);
222  }
223  }
224
