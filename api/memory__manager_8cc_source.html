36 Node* List::PopFront() {
37  if (!Empty()) {
38  auto* ret = head_;
39  head_ = head_->next;
40  if (head_ != nullptr) {
41  __builtin_prefetch(head_->next);
42  }
43  if (tail_ == ret) {
44  tail_ = nullptr;
45  }
46  --size_;
47  --nodes_before_skip_list_;
48  if (skip_list_.back() == ret) {
49  skip_list_.pop_back();
50  nodes_before_skip_list_ = n_;
51  }
52  return ret;
53  }
54  return nullptr;
55 }

57 void List::PushFront(Node* head) {
58  assert(head != nullptr);
59 
60  auto* old_head = head_;
61  head_ = head;
62  head_->next = old_head;
63  if (old_head == nullptr) {
64  tail_ = head_;
65  }
66  ++size_;
67  ++nodes_before_skip_list_;
68 
69  if (nodes_before_skip_list_ >= n_ && size_ > n_) {
70  skip_list_.push_back(head_);
71  nodes_before_skip_list_ = 0;
72  }
73 }

80 void List::PushBackN(Node* head, Node* tail) {
81  assert(head != nullptr);
82  assert(tail != nullptr);
83 
84  if (head_ == nullptr) {

103 void List::PopBackN(Node** head, Node** tail) {
104  assert(head_ != nullptr);
105  assert(tail_ != nullptr);
106 
107  if (skip_list_.size() == 0) {
108  return;
109  }
110 
111  *head = skip_list_.front()->next;
112  skip_list_.front()->next = nullptr;
113  *tail = tail_;
114  tail_ = skip_list_.front();
115  skip_list_.pop_front();
116  size_ -= n_;
117 }

148 NumaPoolAllocator::NumaPoolAllocator(uint64_t size, int nid,
149  uint64_t size_n_pages, double growth_rate,
150  uint64_t max_mem_per_thread)
151  : size_n_pages_(size_n_pages),
152  growth_rate_(growth_rate),
153  max_nodes_per_thread_(max_mem_per_thread / size),
154  num_elements_per_n_pages_((size_n_pages_ - kMetadataSize) / size),
155  size_(size),
156  nid_(nid),
157  tinfo_(ThreadInfo::GetInstance()),

163  // To get one block of N aligned pages, at least 2 N pages must be allocated
164  AllocNewMemoryBlock(size_n_pages_ * 2);

174 void* NumaPoolAllocator::New(int tid) {
175  assert(static_cast<uint64_t>(tid) < free_lists_.size());
176  auto& tl_list = free_lists_[tid];
177  if (!tl_list.Empty()) {
178  auto* ret = tl_list.PopFront();
179  assert(ret != nullptr);
180  return ret;
181  } else if (central_.CanPopBackN()) {
182  Node *head = nullptr, *tail = nullptr;
183  central_.PopBackNThreadSafe(&head, &tail);
184  if (head == nullptr) {
185  return New(tid);
186  }
187  tl_list.PushBackN(head, tail);
188  auto* ret = tl_list.PopFront();
189  assert(ret != nullptr);
190  return ret;
191  } else {
192  lock_.lock();
193  if (memory_blocks_.back().IsFullyInitialized()) {
194  auto size = total_size_ * (growth_rate_ - 1.0);
195  size = RoundUpTo(size, size_n_pages_);
196  AllocNewMemoryBlock(size);
197  }
198  char* start_pointer;
199  uint64_t size;
200  memory_blocks_.back().GetNextPageBatch(size_n_pages_, &start_pointer,
201  &size);
202  lock_.unlock();
203  InitializeNPages(&tl_list, start_pointer, size);
204  auto* ret = tl_list.PopFront();
205  assert(ret != nullptr);
206  return ret;
207  }
208 }

210 void NumaPoolAllocator::Delete(void* p) {
211  auto* node = new (p) Node();
212  auto tid = tinfo_->GetMyThreadId();
213  auto& tl_list = free_lists_[tid];
214  tl_list.PushFront(node);
215  // migrate too much unused memory to the central list agent other threads
216  // can obtain it
217  if (tl_list.Size() > max_nodes_per_thread_ && tl_list.CanPopBackN()) {
218  Node* head = nullptr;
219  Node* tail = nullptr;
220  tl_list.PopBackN(&head, &tail);
221  central_.PushBackNThreadSafe(head, tail);
222  }

227 void NumaPoolAllocator::AllocNewMemoryBlock(std::size_t size) {
228  // check if size is multiple of N pages aligned
229  assert((size & (size_n_pages_ - 1)) == 0 &&
230  "Size must be a multiple of MemoryManager::kSizeNPages");
231  void* block = numa_alloc_onnode(size, nid_);
232  if (block == nullptr) {
233  Log::Fatal("NumaPoolAllocator::AllocNewMemoryBlock", "Allocation failed");
234  }
235  total_size_ += size;
236  auto n_pages_aligned =
237  RoundUpTo(reinterpret_cast<uint64_t>(block), size_n_pages_);
238  auto* start = reinterpret_cast<char*>(block);
239  char* end = start + size;
240  memory_blocks_.push_back(
241  {start, end, reinterpret_cast<char*>(n_pages_aligned)});
242 }

244 void NumaPoolAllocator::InitializeNPages(List* tl_list, char* block,
245  uint64_t mem_block_size) {
246  assert((reinterpret_cast<uint64_t>(block) & (size_n_pages_ - 1)) == 0 &&
247  "block is not N page aligned");
248  auto* block_npa = reinterpret_cast<NumaPoolAllocator**>(block);
249  *block_npa = this;
250 
251  auto* start_pointer = static_cast<char*>(block + kMetadataSize);
252  auto* pointer = start_pointer;
253  const uint64_t num_elements = (mem_block_size - kMetadataSize) / size_;
254 
255  if (tl_list->GetN() == num_elements) {
256  auto* head = new (pointer) Node();
257  assert(head->next == nullptr);
258  auto* tail = head;
259 
260  for (uint64_t i = 1; i < num_elements; ++i) {
261  assert(pointer >= static_cast<char*>(block));
262  assert(pointer <= start_pointer + mem_block_size - size_);
263  __builtin_prefetch(pointer + size_);
264  auto* node = new (pointer) Node();
265  tail->next = node;
266  tail = node;
267  pointer += size_;
268  }
269  tl_list->PushBackN(head, tail);
270  } else {
271  for (uint64_t i = 0; i < num_elements; ++i) {
272  assert(pointer >= static_cast<char*>(block));
273  assert(pointer <= start_pointer + mem_block_size - size_);
274  __builtin_prefetch(pointer + size_);
275  tl_list->PushFront(new (pointer) Node());
276  pointer += size_;
277  }
278  }
279 }

281 uint64_t NumaPoolAllocator::RoundUpTo(uint64_t number, uint64_t multiple) {
282  assert((multiple & (multiple - 1)) == 0 && multiple &&
283  "multiple must be a power of two and non-zero");
284  return (number + multiple - 1) & -multiple;

322 MemoryManager::MemoryManager(uint64_t aligned_pages_shift, double growth_rate,
323  uint64_t max_mem_per_thread)
324  : growth_rate_(growth_rate),
325  max_mem_per_thread_(max_mem_per_thread),
326  page_size_(sysconf(_SC_PAGESIZE)),
327  page_shift_(static_cast<uint64_t>(std::log2(page_size_))),
328  num_threads_(ThreadInfo::GetInstance()->GetMaxThreads()) {
329  aligned_pages_shift_ = aligned_pages_shift;
330  aligned_pages_ = (1 << aligned_pages_shift_);
331  size_n_pages_ = (1 << (page_shift_ + aligned_pages_shift_));
332 
333  if (max_mem_per_thread_ <= size_n_pages_) {
334  Log::Fatal(
335  "MemoryManager",
336  Concat("The parameter mem_mgr_max_mem_per_thread must be greater then "
337  "the size of N pages (PAGE_SIZE * 2 ^ mem_mgr_growth_rate)! "
338  "(max_mem_per_thread_ ",
339  max_mem_per_thread_, ", size_n_pages_ ", size_n_pages_, ")"));
340  }
341 
342  allocators_.reserve(num_threads_ * 2 + 100);
343 }

351 void* MemoryManager::New(std::size_t size) {
352  if (allocators_.Capacity() > num_threads_) {
353  auto it = allocators_.find(size);
354  if (it != allocators_.end()) {
355  return it->second->New(size);
356  } else {
357  std::lock_guard<Spinlock> guard(lock_);
358  // check again, another thread might have created it in between
359  if (allocators_.find(size) == allocators_.end()) {
360  allocators_.insert(std::make_pair(
361  size, new memory_manager_detail::PoolAllocator(
362  size, size_n_pages_, growth_rate_, max_mem_per_thread_)));
363  }
364  return New(size);
365  }
366  } else {
367  std::lock_guard<Spinlock> guard(lock_);
368  auto it = allocators_.find(size);
369  if (it != allocators_.end()) {
370  return it->second->New(size);
371  } else {
372  allocators_.insert(std::make_pair(
373  size, new memory_manager_detail::PoolAllocator(
374  size, size_n_pages_, growth_rate_, max_mem_per_thread_)));
375  return allocators_.find(size)->second;
376  }
