37 struct DisplacementOp : public SimObjectOperationImpl {
38  BDM_OP_HEADER(DisplacementOp);
39 
40  DisplacementOp() {
41  auto* tinfo = ThreadInfo::GetInstance();
42  last_iteration_.resize(tinfo->GetMaxThreads(),
43  std::numeric_limits<uint64_t>::max());
44  last_time_run_.resize(tinfo->GetMaxThreads(), 0);
45  delta_time_.resize(tinfo->GetMaxThreads(), 0);
46  }
47 
48  void operator()(SimObject* sim_object) override {
49  auto* sim = Simulation::GetActive();
50  auto* scheduler = sim->GetScheduler();
51  auto* param = sim->GetParam();
52 
53  if (!sim_object->RunDisplacement()) {
54  return;
55  }
56 
57  // Update search radius and delta_time_ at beginning of each iteration, and
58  // avoid updating them within an iteration
59  auto current_iteration = scheduler->GetSimulatedSteps();
60  auto tid = omp_get_thread_num();
61  if (last_iteration_[tid] != current_iteration) {
62  last_iteration_[tid] = current_iteration;
63 
64  auto* grid = sim->GetEnvironment();
65  auto search_radius = grid->GetLargestObjectSize();
66  squared_radius_ = search_radius * search_radius;
67  auto current_time =
68  (current_iteration + 1) * param->simulation_time_step_;
69  delta_time_[tid] = current_time - last_time_run_[tid];
70  last_time_run_[tid] = current_time;
71  }
72 
73  const auto& displacement =
74  sim_object->CalculateDisplacement(squared_radius_, delta_time_[tid]);
75  sim_object->ApplyDisplacement(displacement);
76  if (param->bound_space_) {
77  ApplyBoundingBox(sim_object, param->min_bound_, param->max_bound_);
78  }
79  }
80 
81  private:
82  double squared_radius_ = 0;
83  std::vector<double> last_time_run_;
84  std::vector<double> delta_time_;
85  std::vector<uint64_t> last_iteration_;
86 };
