76  void operator()(Agent* agent) override {
77  auto* sim = Simulation::GetActive();
78  auto* scheduler = sim->GetScheduler();
79  auto* param = sim->GetParam();
80 
81  if (!agent->RunDisplacement()) {
82  return;
83  }
84 
85  // Update search radius and delta_time_ at beginning of each iteration, and
86  // avoid updating them within an iteration
87  auto current_iteration = scheduler->GetSimulatedSteps();
88  auto tid = omp_get_thread_num();
89  if (last_iteration_[tid] != current_iteration) {
90  last_iteration_[tid] = current_iteration;
91 
92  auto* grid = sim->GetEnvironment();
93  auto search_radius = grid->GetLargestObjectSize();
94  squared_radius_ = search_radius * search_radius;
95  auto current_time = (current_iteration + 1) * param->simulation_time_step;
96  delta_time_[tid] = current_time - last_time_run_[tid];
97  last_time_run_[tid] = current_time;
98  }
99 
100  const auto& displacement =
101  agent->CalculateDisplacement(force_, squared_radius_, delta_time_[tid]);
102  agent->ApplyDisplacement(displacement);
103  if (param->bound_space) {
104  ApplyBoundingBox(agent, param->min_bound, param->max_bound);
105  }
106  }

113  std::vector<uint64_t> last_iteration_;
