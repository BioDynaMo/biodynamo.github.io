41  SoUid NewSoUid() {
42  if (map_ != nullptr) {
43  // defragmentation mode
44  std::lock_guard<Spinlock> guard(lock_);
45  while (search_index_ != map_->size() &&
46  map_->GetReused(search_index_) !=
47  std::numeric_limits<typename SoUid::Reused_t>::max()) {
48  search_index_++;
49  }
50  // find unused element in map
51  if (search_index_ < map_->size()) {
52  auto* scheduler = Simulation::GetActive()->GetScheduler();
53  return SoUid(search_index_++, scheduler->GetSimulatedSteps());
54  }
55  // didn't find any empty slots -> disable defragmentation mode
56  map_ = nullptr;
57  }
58  return SoUid(counter_++);
59  }
